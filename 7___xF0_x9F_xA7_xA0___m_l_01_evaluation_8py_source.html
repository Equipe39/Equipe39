71def compute_metrics(df_type: str, X_labels, y_label):
72 """
73 @brief Compute metrics for each data set. This is a wrapper around Logistic Regression algorithm to train and test the model
74 @param df_type Type of data set to evaluate
75 @param X_labels List of labels for X_type
76 @param y_label Label for y_type ( used for cross validation )
77 @return A dictionary with metrics for each data set. Key is dataframe name value is numpy array of shape ( n_samples
78 """
79 # Evaluate results for a given dataset.
80 if df_type in st.session_state:
81 # Evaluation results
82 X = st.session_state[df_type][X_labels]
83 y = st.session_state[df_type][y_label]
84 # Split data into training and testing sets
85 X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=1 / 3, random_state=0)
86 # Train model using Logistic Regression algorithm
87 model = LogisticRegression(random_state=0)
88 # Initiate k-fold cross-validation with k=10
89 kfold = model_selection.KFold(n_splits=10, random_state=0, shuffle=True)
90 predictions = model_selection.cross_val_predict(model, X_test, y_test, cv=kfold)
91 # Calculate performance metrics
92 accuracy = model_selection.cross_val_score(model, X_train, y_train, cv=kfold, scoring='accuracy')
93 neg_log_loss = model_selection.cross_val_score(model, X_train, y_train, cv=kfold, scoring='neg_log_loss')
94 roc_auc = model_selection.cross_val_score(model, X_train, y_train, cv=kfold, scoring='roc_auc')
95 f1_weighted = model_selection.cross_val_score(model, X_train, y_train, cv=kfold, scoring='f1_weighted')
96 precision_weighted = model_selection.cross_val_score(model, X_train, y_train, cv=kfold, scoring='precision_weighted')
97 recall_weighted = model_selection.cross_val_score(model, X_train, y_train, cv=kfold, scoring='recall_weighted')
98
99 return [
100 accuracy.mean(),
101 neg_log_loss.mean(),
102 roc_auc.mean(),
103 f1_weighted.mean(),
104 precision_weighted.mean(),
105 recall_weighted.mean()
106 ], confusion_matrix(y_test, y_pred=predictions), pd.DataFrame({y_label: predictions})

109def compare_metrics(unchanged_metrics: list, anonymized_metrics: list):
110 """
111 @brief Compare metrics to see if they are equal. This is used to calculate how much a metric was changed in the data set
112 @param unchanged_metrics List of metrics that were not in the data set
113 @param anonymized_metrics List of metrics that were anonymized
114 @return A list of metrics that have been changed in the data set or None if there are no differences in
115 """
116 # Returns true if the number of ununchanged metrics is equal to the number of anonymized metrics.
117 if len(unchanged_metrics) != len(anonymized_metrics):
118 return
119 deltas = []
120 # Add deltas to the list of deltas.
121 for i in range(len(unchanged_metrics)):
122 delta = unchanged_metrics[i] - anonymized_metrics[i]
123 deltas.append(delta)
124 return deltas

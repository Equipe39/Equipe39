24public abstract class AbstractPersistentMaskingProvider implements MaskingProvider {

26 private final boolean normalizeToLowerCase;

29 MEMORY,

35 public AbstractPersistentMaskingProvider(MaskingProvider maskingProvider, MaskingConfiguration configuration) {
36 this.maskingProvider = maskingProvider;
37 this.normalizeToLowerCase = configuration.getBooleanValue("persistence.normalize.toLower");
38 }

40 @Override
41 public final String mask(final String value) {
42 return mask(value, "");
43 }
44
45 @Override
46 public final String mask(String identifier, final String fieldName) throws IllegalArgumentException, NullPointerException {
47 identifier = transformIfRequired(identifier);
48 if (!isCached(identifier)) {
49 String maskedValue = maskingProvider.mask(identifier, fieldName);
50 cacheValue(identifier, maskedValue);
51 }
52
53 return getCachedValue(identifier);
54
55 }
56
57 private String transformIfRequired(String identifier) {
58 if (normalizeToLowerCase) {
59 return identifier.toLowerCase();
60 }
61 return identifier;
62 }

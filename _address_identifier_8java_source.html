92 public Address parseAddress(String data) {
93 //String key = removeDiacriticalMarks(data.trim()).toUpperCase();
94 String key = data.toUpperCase();
95
96 Address address = tryParsePOBOX(key);
97 if (address != null) {
98 return address;
99 }
100
101 Matcher roadtypeMatch = roadTypePattern.matcher(key);
102
103 int roadtypeMatchOffset = -1;
104 int roadtypeMatchEnd = -1;
105 String roadType = null;
106
107 while (roadtypeMatch.find()) {
108 roadtypeMatchOffset = roadtypeMatch.start();
109 roadtypeMatchEnd = roadtypeMatch.end();
110 roadType = roadtypeMatch.group("roadtype").trim();
111 }
112
113 if (roadtypeMatchOffset < 5) {
114 return null;
115 }
116
117
118 Matcher firstPartMatch = firstPartPattern.matcher(key.substring(0, roadtypeMatchOffset));
119 if (!firstPartMatch.find()) {
120 return null;
121 }
122
123 String number = firstPartMatch.group("number");
124 if (number == null) {
125 number = "";
126 }
127
128 String street = firstPartMatch.group("street").trim();
129
130 String cityOrState;
131 String postal;
132 String country;
133
134 Matcher secondPartMatch = secondPartPattern.matcher(key.substring(roadtypeMatchEnd));
135 if (!secondPartMatch.matches()) {
136 cityOrState = "";
137 postal = "";
138 country = "";
139 } else {
140 cityOrState = secondPartMatch.group("cityorstate").trim();
141
142 postal = secondPartMatch.group("postal");
143 if (postal == null) {
144 postal = "";
145 }
146
147 country = secondPartMatch.group("country");
148 if (country == null) {
149 country = "";
150 }
151 }
152
153 address = new Address();
154
155 address.setRoadType(roadType);
156 address.setNumber(number);
157 address.setName(street);
158 address.setCityOrState(cityOrState);
159 address.setPostalCode(postal.trim());
160 address.setCountry(country.trim());
161
162 return address;
163 }

199 public static List<PrivacyConstraint> privacyConstraintsFromJSON(JsonNode node) {
200 if (node.isNull() || !node.isArray()) {
201 throw new MisconfigurationException("Privacy constraints is not an array");
202 }
203
204 List<PrivacyConstraint> privacyConstraints = new ArrayList<>();
205
206 for (JsonNode element : node) {
207 if (element.isNull() || !element.isObject()) {
208 throw new MisconfigurationException("Element in privacy constraints array is null or not an object");
209 }
210
211 JsonNode nameNode = element.get("name");
212 if (nameNode == null || nameNode.isNull() || !nameNode.isTextual()) {
213 throw new MisconfigurationException("Name of privacy constraint missing or null or not a text");
214 }
215
216 String name = nameNode.asText();
217 switch (name) {
218 case "k":
219 JsonNode kNode = element.get("k");
220 if (kNode == null || kNode.isNull() || !kNode.isInt()) {
221 throw new MisconfigurationException("parameter for k is either missing, null or not an integer");
222 }
223
224 int kValue = kNode.intValue();
225
226 if (kValue <= 1) {
227 throw new MisconfigurationException("k value must be greater than 1");
228 }
229
230 privacyConstraints.add(new KAnonymity(kNode.intValue()));
231 break;
232 case "distinctL":
233 JsonNode lNode = element.get("l");
234 if (lNode == null || lNode.isNull() || !lNode.isInt()) {
235 throw new MisconfigurationException("parameter for distinctL is either missing, null or not an integer");
236 }
237
238 int lValue = lNode.intValue();
239
240 if (lValue < 1) {
241 throw new MisconfigurationException("l value must be >=1");
242 }
243
244 privacyConstraints.add(new DistinctLDiversity(lValue));
245 break;
246 case "entropyL":
247 JsonNode leNode = element.get("l");
248 if (leNode == null || leNode.isNull() || !leNode.isInt()) {
249 throw new MisconfigurationException("parameter for entropyL is either missing, null or not an integer");
250 }
251
252 privacyConstraints.add(new EntropyLDiversity(leNode.intValue()));
253 break;
254 case "recursiveCL":
255 JsonNode lcNode = element.get("l");
256 if (lcNode == null || lcNode.isNull() || !lcNode.isInt()) {
257 throw new MisconfigurationException("parameter for recursiveCL is either missing, null or not an integer");
258 }
259
260 JsonNode cNode = element.get("c");
261 if (cNode == null || cNode.isNull() || !cNode.isInt()) {
262 throw new MisconfigurationException("parameter for recursiveCL is either missing, null or not an integer");
263 }
264 privacyConstraints.add(new RecursiveCLDiversity(lcNode.intValue(), cNode.asDouble()));
265 break;
266 case "tCloseness":
267 JsonNode tNode = element.get("t");
268 if (tNode == null || tNode.isNull() || !tNode.isDouble()) {
269 throw new MisconfigurationException("parameter t for tCloseness is either missing, null or not a number");
270 }
271
272 privacyConstraints.add(new TCloseness(tNode.doubleValue()));
273 break;
274 default:
275 throw new MisconfigurationException("unknown privacy constraint name");
276 }
277 }
278
279 return privacyConstraints;
280 }

282 public static GeneralizationHierarchy getHierarchyFromJsonNode(JsonNode specification) {
283 GeneralizationHierarchy hierarchy;
284
285 switch (specification.getNodeType()) {
286 case ARRAY:
287 hierarchy = createMaterializedHierarchy(specification);
288 break;
289 case STRING:
290 hierarchy = GeneralizationHierarchyFactory.getDefaultHierarchy(specification.asText());
291
292 if (null == hierarchy) {
293 throw new MisconfigurationException(specification.asText() + " is not a well known hierarchy type");
294 }
295 break;
296 case OBJECT:
297 hierarchy = buildFromObjectSpecification(specification);
298 break;
299 default:
300 throw new MisconfigurationException("Hierarchy is not recognized");
301 }
302
303 return hierarchy;
304 }

306 public static Map<String, GeneralizationHierarchy> hierarchiesFromJSON(JsonNode hierarchies) {
307 if (hierarchies == null || hierarchies.isNull() || !hierarchies.isObject()) {
308 throw new MisconfigurationException("hierarchies key is either missing or null or not an object");
309 }
310
311 Map<String, GeneralizationHierarchy> hierarchyMap = new HashMap<>();
312
313 Iterator<Map.Entry<String, JsonNode>> hierarchyFields = hierarchies.fields();
314
315 while (hierarchyFields.hasNext()) {
316 Map.Entry<String, JsonNode> hiearchySpecification = hierarchyFields.next();
317
318 String hierarchyName = hiearchySpecification.getKey();
319 JsonNode specification = hiearchySpecification.getValue();
320
321 GeneralizationHierarchy hierarchy;
322
323 switch (specification.getNodeType()) {
324 case ARRAY:
325 hierarchy = createMaterializedHierarchy(specification);
326 break;
327 case STRING:
328 hierarchy = GeneralizationHierarchyFactory.getDefaultHierarchy(specification.asText());
329
330 if (null == hierarchy) {
331 throw new MisconfigurationException("Hierarchy " + specification.asText() + " is not a well known hierarchy type");
332 }
333 break;
334 case OBJECT:
335 hierarchy = buildFromObjectSpecification(specification);
336 break;
337 default:
338 throw new MisconfigurationException("Hierarchy " + hierarchyName + " is not recognized");
339 }
340
341 hierarchyMap.put(hierarchyName, hierarchy);
342 }
343
344 return hierarchyMap;
345 }

348 private static GeneralizationHierarchy buildFromObjectSpecification(JsonNode specification) {
349 if (!specification.has(CLASS_NAME_FIELD)) throw new MisconfigurationException("Missing hierarchy class name");
350
351 try {
352 Class<?> hierarchy = Class.forName(specification.get(CLASS_NAME_FIELD).asText());
353
354 Constructor<?> constructor = hierarchy.getConstructor(JsonNode.class);
355
356 return (GeneralizationHierarchy) constructor.newInstance(specification.get("options"));
357 } catch (ClassNotFoundException e) {
358 throw new MisconfigurationException("Hierarchy class " + specification.get(CLASS_NAME_FIELD).asText() + " not found");
359 } catch (NoSuchMethodException e) {
360 throw new MisconfigurationException("Hierarchy class " + specification.get(CLASS_NAME_FIELD).asText() + " missed constructor with one com.fasterxml.jackson.databind.JsonNode parameter");
361 } catch (IllegalAccessException | InstantiationException | InvocationTargetException e) {
362 throw new MisconfigurationException("Error instantiating new object of " + specification.get(CLASS_NAME_FIELD).asText());
363 }
364 }

366 private static GeneralizationHierarchy createMaterializedHierarchy(JsonNode hierarchyValues) {
367
368 MaterializedHierarchy materializedHierarchy = new MaterializedHierarchy();
369
370 for (JsonNode hierarchyEntry : hierarchyValues) {
371 Iterator<JsonNode> entryIterator = hierarchyEntry.iterator();
372 List<String> entries = new ArrayList<>();
373
374 while (entryIterator.hasNext()) {
375 String v = entryIterator.next().asText();
376 entries.add(v);
377 }
378
379 if (entries.isEmpty()) {
380 throw new MisconfigurationException("empty entries for hierarchy");
381 }
382
383 materializedHierarchy.add(entries);
384 }
385
386 return materializedHierarchy;
387 }

394 public static List<ColumnInformation> columnInformationFromJSON(JsonNode contents, Map<String, GeneralizationHierarchy> defaultHierarchyMap) {
395
396 List<ColumnInformation> columnInformationList = new ArrayList<>();
397
398 if (contents == null || !contents.isArray()) {
399 throw new MisconfigurationException("ColumnInformation is either missing or not an array");
400 }
401
402 for (JsonNode entry : contents) {
403 if (!entry.isObject()) {
404 throw new MisconfigurationException("Entry inside the columnInformation array is not an object");
405 }
406
407 JsonNode typeNode = entry.get("type");
408 if (typeNode == null || !typeNode.isTextual()) {
409 throw new MisconfigurationException("Type for column is either missing or not a text");
410 }
411
412 ColumnType columnType;
413 try {
414 columnType = ColumnType.valueOf(typeNode.asText());
415 } catch (IllegalArgumentException e) {
416 throw new MisconfigurationException("Invalid column type. Valid values are: " + listColumnTypes());
417 }
418
419 final boolean isForLinking = entry.has("isForLinking") && entry.get("isForLinking").asBoolean();
420
421
422 switch (columnType) {
423 case QUASI:
424 final double weight;
425
426 final JsonNode weightNode = entry.get("weight");
427
428 if (weightNode != null) {
429 if (!weightNode.isNumber()) {
430 throw new MisconfigurationException("Weight for column is not a number");
431 }
432
433 weight = weightNode.asDouble();
434
435 if (weight < 0.0) {
436 throw new MisconfigurationException("Weight for a column cannot be negative");
437 }
438 } else {
439 weight = 1.0;
440 }
441
442 JsonNode isCategoricalNode = entry.get("isCategorical");
443 if (isCategoricalNode == null || !isCategoricalNode.isBoolean()) {
444 throw new MisconfigurationException("isCategorical for column is either missing or not a boolean");
445 }
446
447 final boolean isCategorical = isCategoricalNode.asBoolean();
448
449 if (isCategorical) {
450 JsonNode hierarchyNode = entry.get("hierarchy");
451 if (hierarchyNode == null || !hierarchyNode.isTextual() || hierarchyNode.isNull()) {
452 throw new MisconfigurationException("Hierarchy element is either missing or not text");
453 }
454
455 final String hierarchyName = hierarchyNode.asText();
456 final GeneralizationHierarchy hierarchy = defaultHierarchyMap.get(hierarchyName);
457
458 if (hierarchy == null) {
459 throw new MisconfigurationException("Cannot find the hierarchy: " + hierarchyName);
460 }
461
462
463 int maximumLevel = -1;
464 final JsonNode maximumLevelNode = entry.get("maximumLevel");
465
466 if (maximumLevelNode != null) {
467 if (!maximumLevelNode.isInt()) {
468 throw new MisconfigurationException("MaximumLevel for column is not an integer");
469 }
470
471 maximumLevel = entry.get("maximumLevel").asInt();
472 if (maximumLevel < -1) {
473 throw new MisconfigurationException("MaximumLevel for column is invalid negative integer");
474 }
475
476 if (maximumLevel > hierarchy.getHeight()) {
477 maximumLevel = hierarchy.getHeight();
478 }
479 }
480
481 columnInformationList.add(new CategoricalInformation(hierarchy, ColumnType.QUASI, weight, maximumLevel, isForLinking));
482 } else {
483 List<Double> sortedValues = new ArrayList<>();
484 columnInformationList.add(new NumericalRange(sortedValues, ColumnType.QUASI, weight, isForLinking));
485 }
486 break;
487 case SENSITIVE:
488 columnInformationList.add(new SensitiveColumnInformation(isForLinking));
489 break;
490 default:
491 columnInformationList.add(new DefaultColumnInformation(isForLinking));
492 break;
493 }
494 }
495
496
497 return columnInformationList;
498 }

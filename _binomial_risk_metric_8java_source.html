95 public double reportLocalP() {
96 double risk = 0.0;
97
98 for (int k = 0; k < F.size(); ++k) {
99 final int Fk = extractFk(F.get(k));
100 final int tries = FastMath.min(10, f.get(k));
101
102 double p = (double) f.get(k) / (double) Fk;
103
104 double k_risk = risk(Fk, p, tries);
105
106 risk = FastMath.max(risk, k_risk);
107 }
108
109 return risk;
110 }

112 public double reportGlobalP() {
113 double risk = 0.0;
114
115 final double p = n / N;
116
117 for (int k = 0; k < F.size(); ++k) {
118 final int Fk = extractFk(F.get(k));
119 final int tries = FastMath.min(10, f.get(k));
120
121 risk = FastMath.max(risk, risk(Fk, p, tries));
122 }
123
124 return risk;
125 }

127 public static double risk(int Fk, double p, long tries) {
128 double k_risk = 0.0;
129 final BinomialDistribution binomialDistribution = new BinomialDistribution(Fk, p);
130
131 for (long i = 1L; i <= tries; ++i) {
132 k_risk += Math.pow(binomialDistribution.probability(1), i);
133 }
134
135 return k_risk;
136 }

140 List<Partition> partitions = AnonymizationUtils.generatePartitionsForLinking(anonymized, columnInformationList);
141 n = anonymized.getNumberOfRows();

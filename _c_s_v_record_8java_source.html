66 CsvMapper mapper = csvOptions.isTrimFields() ? CSVRecord.mapper.enable(CsvParser.Feature.TRIM_SPACES) : CSVRecord.mapper;
67 MappingIterator<String[]> reader = mapper.readerFor(String[].class).with(schema).readValues(input);

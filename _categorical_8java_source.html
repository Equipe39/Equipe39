67 @Override

83 @Override
84 public String randomise(String value) {
85 if (this.utilityFunction.size() == 0) {
86 initUtilityFunction();
87 }
88
89 value = value.toUpperCase();
90 double u = this.rnd.nextDouble() * this.normalisingConstant.get(value);
91 double sum = 0.0;
92
93 for (String target : this.hierarchyLeaves) {
94 sum += getProbOfValues(value, target);
95
96 if (u <= sum) {
97 return target;
98 }
99 }
100
101 return value;
102 }

122 private void initUtilityFunction() {
123 this.utilityFunction = new HashMap<>();
124
125 int hierarchyHeight = this.hierarchy.getHeight();
126
127 for (String base : this.hierarchyLeaves) {
128 List<String> baseGeneralizations = new ArrayList<>();
129
130 if (this.hierarchy.getNodeLevel(base) >= 0) {
131 for (int i = 0; i < hierarchyHeight; i++) {
132 baseGeneralizations.add(this.hierarchy.encode(base, i, false));
133 }
134 }
135
136 for (String target : this.hierarchyLeaves) {
137 int hash = getHashValue(base, target);
138
139 if (this.utilityFunction.containsKey(hash)) {
140 continue;
141 }
142
143 if (base.equals(target)) {
144 this.utilityFunction.put(hash, 0.0);
145 continue;
146 }
147
148 if (this.hierarchy.getNodeLevel(base) < 0 || this.hierarchy.getNodeLevel(target) < 0) {
149 this.utilityFunction.put(hash, (double) hierarchyHeight);
150 continue;
151 }
152
153 for (int i = 1; i < hierarchyHeight; i++) {
154 String targetGeneralised = this.hierarchy.encode(target, i, false);
155
156 if (targetGeneralised.equals(baseGeneralizations.get(i))) {
157 this.utilityFunction.put(hash, (double) i);
158 break;
159 }
160 }
161 }
162 }
163
164 initNormalisingConstant();
165 }
166
167 private void initNormalisingConstant() {
168 this.normalisingConstant = new HashMap<>();
169
170 for (String base : this.hierarchyLeaves) {
171 double tempNormalisingConstant = 0.0;
172
173 for (String target : this.hierarchyLeaves) {
174 tempNormalisingConstant += getProbOfValues(base, target);
175 }
176
177 this.normalisingConstant.put(base, tempNormalisingConstant);
178 }
179 }
180
181 private double getProbOfValues(String value1, String value2) {
182 return Math.exp(- this.epsilon * getUtility(value1, value2) / 2);
183 }
184
185 public double getUtility(String value1, String value2) {
186 return this.utilityFunction.get(getHashValue(value1, value2));
187 }

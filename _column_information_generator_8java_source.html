44 public static NumericalRange generateNumericalRange(IPVDataset dataset, int column, ColumnType columnType, double weight, boolean isForLinking) {
45 List<Double> doubles = new ArrayList<>();
46
47 int numberOfRows = dataset.getNumberOfRows();
48
49 for (int i = 0; i < numberOfRows; i++) {
50 List<String> row = dataset.getRow(i);
51 try {
52 Double v = Double.valueOf(row.get(column));
53 doubles.add(v);
54 } catch (NumberFormatException ignored) {
55 }
56 }
57
58 Collections.sort(doubles);
59 return new NumericalRange(doubles, columnType, weight, isForLinking);
60 }

109 private void precomputeNearest() {
110 this.latLonTree = new HashMap<>();
111
112 for (String key : continentListMap.keySet()) {
113 List<Continent> continentList = continentListMap.get(key);
114
115 int numberOfContinents = continentList.size();
116
117 try {
118 this.latLonTree.put(key, new LatLonKDTree<Continent>(continentList));
119 } catch (Exception e) {
120 e.printStackTrace();
121 throw new RuntimeException("error building KD-tree");
122 }
123
124 for (Location location : continentList) {
125 Continent continent = (Continent) location;
126 continent.setNeighbors(getClosestContinents(continent, key, numberOfContinents));
127 }
128 }
129 }

142 public String getClosestContinent(String identifier, int k) {
143 Continent continent = this.getKey(identifier);
144 if (continent == null) {
145 return getRandomKey();
146 }
147
148 List<Continent> neighbors = continent.getNeighbors();
149 if (neighbors == null) {
150 return getRandomKey();
151 }
152
153 if (k > neighbors.size()) {
154 k = neighbors.size();
155 }
156
157 return neighbors.get(random.nextInt(k)).getName();
158 }

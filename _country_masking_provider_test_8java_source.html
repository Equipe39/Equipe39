95 Set<String> greekValues = new HashSet<>();
96
97
98 for(ResourceEntry entry: entryCollection) {
99 InputStream inputStream = entry.createStream();
100 try (CSVParser reader = Readers.createCSVReaderFromStream(inputStream)) {
101 for (CSVRecord line : reader) {
102 String name = line.get(0);
103 greekValues.add(name.toUpperCase());
104 }
105 inputStream.close();
106 }
107 }
108
109 for(int i = 0; i < 100; i++) {
110 String maskedValue = maskingProvider.mask(greekOriginalValue);
111 assertTrue(greekValues.contains(maskedValue.toUpperCase()));
112 }
113 }

224 private static Dataset<Row> convertDatasetToOriginalSchema(final Dataset<Row> dataset,
225 final StructType originalSchema, final Map<String, String> timestampFormats) {
226 final StructField[] fields = originalSchema.fields();
227 final Column[] columns = new Column[fields.length];
228
229 for (int i = 0; i < fields.length; ++i) {
230 StructField field = fields[i];
231 Column column = col(field.name());
232
233 if (field.dataType().sameType(DataTypes.TimestampType)) {
234 String format = timestampFormats.get(field.name());
235 if (null != format) {
236 logger.info("will apply to_timestamp for " + field.name() + ", format is " + format);
237 column = unix_timestamp(column, format);
238 } else {
239 logger.info("will apply to_timestamp for " + field.name() + ", no format specified");
240 column = unix_timestamp(column);
241 }
242 column = column.cast(DataTypes.TimestampType);
243 } else if (field.dataType().sameType(DataTypes.DateType) && field.nullable()) {
244 column = to_date(column);
245 column = column.cast(DataTypes.DateType);
246 }
247
248 columns[i] = column.as(field.name());
249 }
250
251 return dataset.select(
252 columns
253 );
254 }

277 public static void checkNotNullableAndValue(Dataset<Row> dataset, StructType schema, String checkField, String value) {
278
279 boolean needsCheck = false;
280
281 for(StructField field: schema.fields()) {
282 if (!field.nullable()) {
283 needsCheck = true;
284 break;
285 }
286 }
287
288 if (Objects.nonNull(checkField) && Objects.nonNull(value)) {
289 needsCheck = true;
290 }
291
292 if (!needsCheck) {
293 return;
294 }
295
296 dataset.foreach((ForeachFunction<Row>) row -> {
297 for(StructField field: schema.fields()) {
298 if (!field.nullable()) {
299 int fieldIndex = row.fieldIndex(field.name());
300 Object v = row.get(fieldIndex);
301
302 if (v == null) {
303 throw new RuntimeException("field is null: " + field.name());
304 }
305
306 String s = v.toString();
307 if (s.isEmpty()) {
308 throw new RuntimeException("field is empty: " + field.name());
309 }
310 }
311 }
312 if (Objects.nonNull(checkField) && Objects.nonNull(value)) {
313 int fieldIndex = row.fieldIndex(checkField);
314 Object v = row.get(fieldIndex);
315
316 if (v == null) {
317 logger.error("Field to check is null: " + checkField);
318 throw new RuntimeException("field to check is null: " + checkField);
319 }
320 String s = v.toString();
321 if (!s.equals(value)) {
322 logger.error("Unexpected value in row " + checkField);
323 throw new RuntimeException("Unexpected value in row " + checkField);
324 }
325 }
326 });
327 }

371 private static DataType converToDataType(JsonNode type) {
372 switch(type.asText().split("\\(")[0].toLowerCase()) {
373 case "binary":
374 return DataTypes.BinaryType;
375
376 case "boolean":
377 return DataTypes.BooleanType;
378
379 case "byte":
380 return DataTypes.ByteType;
381
382 case "interval":
383 return DataTypes.CalendarIntervalType;
384
385 case "date":
386 return DataTypes.DateType;
387
388 case "numeric":
389 case "decimal":
390 return extractDecimalType(type.asText().toLowerCase());
391
392 case "double":
393 return DataTypes.DoubleType;
394
395 case "float":
396 return DataTypes.FloatType;
397
398 case "int":
399 case "int32":
400 return DataTypes.IntegerType;
401
402 case "int64":
403 case "long":
404 return DataTypes.LongType;
405
406 case "short":
407 return DataTypes.ShortType;
408
409 case "timestamp":
410 return DataTypes.TimestampType;
411
412 case "string":
413 case "varchar":
414 case "char":
415 return DataTypes.StringType;
416
417 case "null":
418 return DataTypes.NullType;
419
420 default:
421 throw new IllegalArgumentException("UnknownType: " + type.asText());
422 }

425 public static DataType extractDecimalType(String s) {
426 int idx = s.indexOf('(');
427
428 if (idx == -1) {
429 return new DecimalType();
430 }
431
432 int idx2 = s.indexOf(')', idx);
433
434 if (idx2 == -1) {
435 throw new IllegalArgumentException("Unclosed parenthesis in decimal type: " + s);
436 }
437
438 String internal = s.substring(idx + 1, idx2);
439 String[] toks = internal.split(",");
440
441 if (toks.length > 2) {
442 throw new IllegalArgumentException("Too many values in precision, scale part: " + s);
443 }
444
445 if (toks.length == 1) {
446 return new DecimalType(Integer.parseInt(toks[0].trim()));
447 }
448
449 return new DecimalType(Integer.parseInt(toks[0].trim()), Integer.parseInt(toks[1].trim()));
450 }

67 @Override
68 public Double call() throws Exception {
69 double itemsetsChecked = 0;
70
71 while (!seeds.isEmpty() || seeds.isThereAnActiveWorker()) {
72 ItemSet K = seeds.poll();
73 try {
74 if (null == K) continue;
75 seeds.isActive(this);
76
77 if (graphs.isNodePruned(K)) continue;
78
79 itemsetsChecked++;
80
81 final List<Collection<RowIDSet>> kPLIs = pliRepository.getAll(K);
82
83 if (hasUnique(kPLIs)) {
84 // mark all superset as unique
85 graphs.addUnique(K);
86 seeds.removeSuperSets(K);
87 } else {
88 // K is not unique
89 graphs.addNotUnique(K);
90 seeds.removeSubSets(K);
91 // iterate over all superset
92 for (Integer attribute : pliRepository.getAttributeNames()) {
93 if (K.canBeExtendWith(attribute)) {
94 // dumb version, need to verify that attribute can go to K
95 ItemSet kPrime = new ItemSet(K, attribute);
96 if (!graphs.isNodePruned(kPrime)) {
97 seeds.add(kPrime);
98 }
99 }
100 }
101 }
102 } finally {
103 seeds.done(this);
104 }
105 }
106
107 return itemsetsChecked;
108 }

110 static Iterator<RowIDSet> getCombinations(final List<Collection<RowIDSet>> lists) {
111 switch (lists.size()) {
112 case 0:
113 throw new IllegalArgumentException("List has to contain at least a set");
114 case 1:
115 return lists.get(0).iterator();
116 default:
117 final Iterator<RowIDSet> current = lists.get(0).iterator();
118
119 return new Iterator<RowIDSet>() {
120 RowIDSet element;
121 Iterator<RowIDSet> other;
122
123 @Override
124 public boolean hasNext() {
125 return current.hasNext() || (null != other && other.hasNext());
126 }
127
128 @Override
129 public RowIDSet next() {
130 if (!hasNext()) throw new NoSuchElementException();
131
132 if (null == other || !other.hasNext()) {
133 element = current.next();
134 other = getCombinations(lists.subList(1, lists.size()));
135 }
136
137 return element.intersect(other.next());
138 }
139
140 @Override
141 public void remove() {
142 throw new UnsupportedOperationException();
143 }
144 };
145 }
146 }

155 protected boolean hasUnique(List<Collection<RowIDSet>> uniqueValues) {
156 Iterator<RowIDSet> uniqueValueIterator = getCombinations(uniqueValues);
157
158 while (uniqueValueIterator.hasNext()) {
159 RowIDSet combination = uniqueValueIterator.next();
160
161 if (combination.isEmpty()) {
162 continue;
163 }
164
165 if (kValue > combination.size()) {
166 return true;
167 }
168 }
169
170 return false;
171 }

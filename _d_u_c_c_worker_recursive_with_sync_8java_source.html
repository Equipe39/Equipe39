56 private void process(ItemSet K, LocalPrunedGraphs graphs) {
57 if (graphs.isNodePruned(K) || hasUnique(pliRepository.getAll(K))) {
58 // mark all superset as unique
59 graphs.addUnique(K);
60 seeds.removeSuperSets(K);
61
62 if (K.size() > 2) {
63 for (Integer item : K.getItems()) {
64 ItemSet kPrime = new ItemSet(K);
65 kPrime.removeItem(item);
66
67 if (!graphs.isNodePruned(kPrime)) {
68 process(kPrime, graphs);
69 }
70 }
71 }
72 } else {
73 // K is not unique
74 graphs.addNotUnique(K);
75 seeds.removeSubSets(K);
76 // iterate over all superset
77 for (Integer attribute : validAttributes) {
78 if (K.canBeExtendWith(attribute)) {
79 ItemSet kPrime = new ItemSet(K, attribute);
80 if (!graphs.isNodePruned(kPrime)) {
81 process(kPrime, graphs);
82 }
83 }
84 }
85 }

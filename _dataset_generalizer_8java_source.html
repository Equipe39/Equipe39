50 public static List<String> generalizeRow(List<String> originalRow, List<ColumnInformation> columnInformationList, int[] levels) {
51 int numberOfColumns = originalRow.size();
52
53 List<String> anonymizedRow = new ArrayList<>();
54
55 int qidIndex = 0;
56
57 for (int j = 0; j < numberOfColumns; j++) {
58 String originalValue = originalRow.get(j);
59
60 if (columnInformationList.get(j).getColumnType() != ColumnType.QUASI) {
61 anonymizedRow.add(originalValue);
62 } else {
63 int level = levels[qidIndex];
64 CategoricalInformation categoricalInformation = (CategoricalInformation) columnInformationList.get(j);
65 String transformedValue = transformValue(originalValue, level, categoricalInformation);
66 anonymizedRow.add(transformedValue);
67 qidIndex++;
68 }
69 }
70
71 return anonymizedRow;
72 }

105 public static IPVDataset generalize(IPVDataset dataset, List<ColumnInformation> columnInformationList, int[] levels) {

119 public static Collection<Partition> generalizeAndPartition(IPVDataset dataset, List<ColumnInformation> columnInformationList,
120 int[] levels, int contentRequirements) {
121 List<Integer> quasiColumns = AnonymizationUtils.getColumnsByType(columnInformationList, ColumnType.QUASI);
122 Map<String, Partition> partitionMap = new HashMap<>(dataset.getNumberOfRows() / 10);
123
124 int numberOfRows = dataset.getNumberOfRows();
125
126 for (int i = 0; i < numberOfRows; i++) {
127 List<String> anonymizedRow = generalizeRow(dataset.getRow(i), columnInformationList, levels);
128
129 String key = AnonymizationUtils.generateEQKey(anonymizedRow, quasiColumns);
130
131 Partition p = partitionMap.get(key);
132 if (p == null) {
133 partitionMap.put(key, new InMemoryPartition(new IPVDataset(new ArrayList<>(), null, false)));
134 }
135
136 if (contentRequirements == ContentRequirements.NONE) {
137 partitionMap.get(key).getMember().addRow(Collections.emptyList());
138 } else {
139 partitionMap.get(key).getMember().addRow(anonymizedRow);
140 }
141 }
142
143 return partitionMap.values();
144 }

156 public static Map<String, Integer> generalizeCSVAndCountEQ(InputStream dataset, List<ColumnInformation> columnInformationList, int[] levels) throws IOException {

176 public static String generalizeAndGenerateKey(List<String> originalRow,
177 List<Integer> quasiColumns, List<ColumnInformation> columnInformationList, int[] levels) {
178 StringBuilder key = new StringBuilder();
179
180 for (int i = 0; i < quasiColumns.size(); i++) {
181 int level = levels[i];
182 int columnIndex = quasiColumns.get(i);
183
184 CategoricalInformation categoricalInformation = (CategoricalInformation) columnInformationList.get(columnIndex);
185 String transformedValue = transformValue(originalRow.get(columnIndex), level, categoricalInformation);
186
187 key.append(transformedValue);
188 key.append(":");
189 }
190
191 return key.toString();

194 public static Map<String, Integer> generalizeCSVAndCountEQ(IPVDataset dataset, List<ColumnInformation> columnInformationList, int[] levels) {
195 List<Integer> quasiColumns = AnonymizationUtils.getColumnsByType(columnInformationList, ColumnType.QUASI);
196
197 int numberOfRows = dataset.getNumberOfRows();
198 Map<String, Integer> counters = new HashMap<>(numberOfRows / 10);
199
200 for (int i = 0; i < numberOfRows; i++) {
201 String key = generalizeAndGenerateKey(dataset.getRow(i), quasiColumns, columnInformationList, levels);
202
203 Integer counter = counters.get(key);
204
205 if (counter == null) {
206 counters.put(key, 1);
207 } else {
208 counters.put(key, counter + 1);
209 }
210 }
211
212 return counters;
213 }

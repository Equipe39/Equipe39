110 static {
111 dateFormats = new DateTimeFormatter[patterns.length];
112 ampmDateFormats = new DateTimeFormatter[ampmPatterns.length];
113
114 for (int i = 0; i < patterns.length; i++) {
115 dateFormats[i] = new DateTimeFormatterBuilder()
116 .parseCaseInsensitive()
117 .appendPattern((patterns[i].getFirst()))
118 .parseDefaulting(ChronoField.HOUR_OF_DAY, 0)
119 .parseDefaulting(ChronoField.MINUTE_OF_HOUR, 0)
120 .parseDefaulting(ChronoField.SECOND_OF_MINUTE, 0)
121 .appendFraction(ChronoField.MICRO_OF_SECOND, 0, 6, true)
122 .toFormatter()
123 .withZone(ZoneOffset.systemDefault());
124 }
125
126 for (int i = 0; i < ampmPatterns.length; i++) {
127 ampmDateFormats[i] = new DateTimeFormatterBuilder()
128 .parseCaseInsensitive()
129 .appendPattern((ampmPatterns[i].getFirst()))
130 .parseDefaulting(ChronoField.HOUR_OF_AMPM, 0)

132 .parseDefaulting(ChronoField.SECOND_OF_MINUTE, 0)

149 public Tuple<DateTimeFormatter, TemporalAccessor> matchingFormat(String data) {
150
151 int digits = 0;
152
153 for (int i = 0; i < data.length(); i++) {
154 if (Character.isDigit(data.charAt(i))) {
155 digits++;
156 }
157 }
158
159 if (digits == 0) {
160 return null;
161 }
162
163 data = data.replaceAll("(?<=\\d)(st|nd|rd|th)", "").replaceAll("\\s+", " ");
164
165 if (data.endsWith("am")) {
166 data = data.substring(0, data.length() - 2) + "AM";
167 } else if (data.endsWith("pm")) {
168 data = data.substring(0, data.length() - 2) + "PM";
169 }
170
171 return checkAllPaterns(data);
172 }

174 private Tuple<DateTimeFormatter, TemporalAccessor> checkAllPaterns(String data) {
175 Tuple<DateTimeFormatter, TemporalAccessor> f = checkPatterns(data, patterns, dateFormats);
176 if (f != null) {
177 return f;
178 }
179
180 return checkPatterns(data, ampmPatterns, ampmDateFormats);
181 }

183 private Tuple<DateTimeFormatter, TemporalAccessor>
184 checkPatterns(String data, Tuple<String, Pattern>[] patterns, DateTimeFormatter[] dateFormats) {
185
186 for (int i = 0; i < patterns.length; i++) {
187 Pattern p = patterns[i].getSecond();
188 if (p.matcher(data).matches()) {
189 try {
190 DateTimeFormatter f = dateFormats[i];
191 ParsePosition position = new ParsePosition(0);
192 TemporalAccessor t = f.parse(data, position);
193 int index = position.getIndex();
194 if (index == 0 || index != data.length()) {
195 continue;
196 }
197 return new Tuple<>(f, t);
198 } catch (Exception ignored) {
199 }
200 }
201 }
202
203 return null;
204 }

215 public boolean isTemporal(String identifier) {
216 String p = identifier.toLowerCase();
217
218 for (String pattern : temporalSubstrings) {
219 if (p.contains(pattern)) {
220 return true;
221 }
222 }
223
224 return false;

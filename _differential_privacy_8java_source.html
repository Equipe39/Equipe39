56 private List<ColumnInformation> buildColumnInformationList(IPVDataset dataset, Collection<IPVVulnerability> vulnerabilities,
57 Collection<String> sensitiveFields, Map<String, ProviderType> fieldTypes) {
58 List<ColumnInformation> columnInformationList = new ArrayList<>(dataset.getNumberOfColumns());
59
60 IPVSchema schema = dataset.getSchema();
61 List<? extends IPVSchemaField> fields = schema.getFields();
62
63 IPVVulnerability vulnerability = AnonymizationUtils.mergeVulnerabilities(vulnerabilities);
64
65 for (int i = 0; i < dataset.getNumberOfColumns(); i++) {
66 String fieldName = fields.get(i).getName();
67
68 if (sensitiveFields.contains(fieldName)) {
69 columnInformationList.add(new SensitiveColumnInformation());
70 continue;
71 }
72
73 boolean isQuasi = vulnerability.contains(i);
74 if (!isQuasi) {
75 columnInformationList.add(new DefaultColumnInformation());
76 continue;
77 }
78
79 ProviderType fieldType = fieldTypes.get(fieldName);
80 if (fieldType == null) {
81 columnInformationList.add(ColumnInformationGenerator.generateCategoricalFromData(dataset, i, ColumnType.QUASI));
82 continue;
83 }
84
85 if (fieldType.getTypeClass() == TypeClass.NUMERICAL) {
86 GeneralizationHierarchy hierarchy = IntervalGenerator.generateHierarchy(dataset, i);
87 columnInformationList.add(new CategoricalInformation(hierarchy, ColumnType.QUASI));
88 } else {
89 GeneralizationHierarchy hierarchy = GeneralizationHierarchyFactory.getDefaultHierarchy(fieldType);
90 if (hierarchy == null) {
91 columnInformationList.add(ColumnInformationGenerator.generateCategoricalFromData(dataset, i, ColumnType.QUASI));
92 } else {
93 columnInformationList.add(new CategoricalInformation(hierarchy, ColumnType.QUASI));
94 }
95 }
96 }
97
98 return columnInformationList;
99 }

135 private AnonymizationAlgorithm initialize(List<ColumnInformation> columnInformationList, List<PrivacyConstraint> privacyConstraints, DifferentialPrivacyMechanismOptions options) {
136 this.numberOfColumns = columnInformationList.size();
137
138 this.columnIndex = -1;
139
140 for (int i = 0; i < this.numberOfColumns; i++) {
141 ColumnInformation col = columnInformationList.get(i);
142
143 if (col.getColumnType() == ColumnType.E_QUASI) {
144 if (this.columnIndex < 0) {
145 this.columnIndex = i;
146 } else {
147 throw new RuntimeException("Expected only one e-quasi");
148 }
149 }
150 }
151
152 this.perEquivalenceClass = options.isDPPerEquivalenceClass(this.perEquivalenceClass);
153 this.getBoundsFromData = options.isGetBoundsFromData();
154 this.mechanism = options.getMechanism();
155 this.mechanism.setOptions(options);
156
157 return this;
158 }

171 @Override
172 public IPVDataset apply() {
173 List<Integer> columnIndexes = IntStream.range(0, columnInformationList.size()).boxed().filter(i -> columnInformationList.get(i).getColumnType().equals(ColumnType.E_QUASI)).collect(Collectors.toList());
174 if (columnIndexes.size() != 1) throw new RuntimeException("More than one e-quasi identifier specified");
175
176 this.columnIndex = columnIndexes.get(0);
177
178 if (perEquivalenceClass) {
179 this.anonymizedPartitions = new ArrayList<>();
180 for (Partition equivalenceClass : AnonymizationUtils.generatePartitions(dataset, columnInformationList)) {
181 if (this.getBoundsFromData) {
182 this.mechanism.analyseForParams(Collections.singletonList(equivalenceClass), columnIndex);
183 }
184
185 Partition anonymizedPartition = applyDifferentialPrivacy(equivalenceClass.getMember(), this.columnIndex);
186 this.anonymizedPartitions.add(anonymizedPartition);
187 }
188
189 return consolidatePartitions(this.anonymizedPartitions);
190 } else {
191 this.anonymizedPartitions = new ArrayList<>();
192 if (this.getBoundsFromData) {
193 this.mechanism.analyseForParams(Collections.singletonList(new InMemoryPartition(dataset)), this.columnIndex);
194 }
195
196 Partition anonymizedPartition = applyDifferentialPrivacy(dataset, this.columnIndex);
197 this.anonymizedPartitions.add(anonymizedPartition);
198 return anonymizedPartition.getMember();
199 }
200 }

202 private Partition applyDifferentialPrivacy(IPVDataset dataset, int columnIndex) {
203 List<List<String>> noisyDataset = new ArrayList<>();
204
205 for (List<String> row : dataset) {
206 List<String> anonymisedRow = new ArrayList<>(row);
207
208 String value = row.get(columnIndex);
209 String noisyValue = this.mechanism.randomise(value);
210
211 anonymisedRow.set(columnIndex, noisyValue);
212 noisyDataset.add(anonymisedRow);
213 }
214
215 Partition finalDataset = new InMemoryPartition(noisyDataset);
216 finalDataset.setAnonymous(true);
217
218 return finalDataset;
219 }
220
221 private IPVDataset consolidatePartitions(List<Partition> equivalenceClasses) {
222 IPVDataset finalDataset = new IPVDataset(this.numberOfColumns);
223
224 for (Partition p : equivalenceClasses) {
225 IPVDataset member = p.getMember();
226 int memberRows = member.getNumberOfRows();
227
228 for (int i = 0; i < memberRows; i++) {
229 finalDataset.addRow(member.getRow(i));
230 }
231 }
232
233 return finalDataset;
234 }

43 public DifferentialPrivacyMaskingProvider(SecureRandom random, MaskingConfiguration configuration) {
44 DifferentialPrivacyMechanismOptions options = new DifferentialPrivacyMechanismOptions();
45 options.setEpsilon(configuration.getDoubleValue("differentialPrivacy.parameter.epsilon"));
46
47 Mechanism mechanism = Mechanism.valueOf(configuration.getStringValue("differentialPrivacy.mechanism"));
48 switch (mechanism) {
49 case BINARY:
50 String value1 = configuration.getStringValue("differentialPrivacy.binary.value1");
51 String value2 = configuration.getStringValue("differentialPrivacy.binary.value2");
52 options.setBinaryValues(value1, value2);
53 break;
54 case CATEGORICAL:
55 String hierarchyName = configuration.getStringValue("differentialPrivacy.categorical.hierarchyName");
56 JsonNode hierarchyMapNode = configuration.getJsonNodeValue("differentialPrivacy.categorical.hierarchyMap");
57
58 Map<String, GeneralizationHierarchy> hierarchyMap = AnonymizationOptions.hierarchiesFromJSON(hierarchyMapNode);
59 GeneralizationHierarchy hierarchy = hierarchyMap.get(hierarchyName);
60 if (hierarchy == null) {
61 String msg = "Unable to find the hierarchy with name: " + hierarchyName;
62 logger.error(msg);
63 throw new MisconfigurationException(msg);
64 }
65 options.setHierarchy(hierarchy);
66 break;
67 case LAPLACE_NATIVE:
68 case LAPLACE_BOUNDED:
69 case LAPLACE_TRUNCATED:
70 double lowerBound = configuration.getDoubleValue("differentialPrivacy.range.min");
71 double upperBound = configuration.getDoubleValue("differentialPrivacy.range.max");
72 options.setBounds(lowerBound, upperBound);
73 break;
74 case GEOMETRIC_NATIVE:
75 case GEOMETRIC_TRUNCATED:
76 throw new UnsupportedOperationException("Not supported at the moment");
77 }
78
79 this.mechanism = DPMechanismFactory.getMechanism(mechanism);
80 this.mechanism.setOptions(options);
81 }

84 this(new SecureRandom(), configuration);
85 }

89 return mechanism.randomise(value);

47 @Test
48 public void testDiscernibilityNumericNoSuppression() throws Exception {
49 int K = 2;
50
51 List<List<String>> values = new ArrayList<>();
52 values.add(toString(new Long[]{1L, 7L, 2L, 4L}));
53 values.add(toString(new Long[]{6L, 7L, 12L, 4L}));
54 values.add(toString(new Long[]{5L, 7L, 11L, 4L}));
55 values.add(toString(new Long[]{10L, 7L, 5L, 4L}));
56 values.add(toString(new Long[]{10L, 7L, 11L, 4L}));
57
58 IPVDataset dataset = new IPVDataset(values, null, false);
59
60 List<ColumnInformation> columnInformationList = new ArrayList<>(4);
61 columnInformationList.add(ColumnInformationGenerator.generateNumericalRange(dataset, 0, ColumnType.QUASI));
62 columnInformationList.add(new DefaultColumnInformation());
63 columnInformationList.add(ColumnInformationGenerator.generateNumericalRange(dataset, 2, ColumnType.QUASI));
64 columnInformationList.add(new DefaultColumnInformation());
65
66 IPVDataset anonymizedDataset = IPVDataset.load(this.getClass().getResourceAsStream("/discernibilityAnonymized.csv"), false, ',', '"', false);
67
68 Tuple<List<Partition>, List<Partition>> bothPartitions = OLAUtils.generatePartitions(dataset, anonymizedDataset, columnInformationList);
69 List<Partition> originalPartitions = bothPartitions.getFirst();
70 List<Partition> anonymizedPartitions = bothPartitions.getSecond();
71
72 for(Partition p: originalPartitions) p.setAnonymous(true);
73 for(Partition p: anonymizedPartitions) p.setAnonymous(true);
74
75 InformationMetric discernibility = new Discernibility().initialize(dataset, anonymizedDataset,
76 originalPartitions, anonymizedPartitions, columnInformationList, null);
77
78 double discernibilityValue = discernibility.report();
79
80 double expected = 2*2 + 3*3;
81 assertEquals(expected, discernibilityValue, 0.001);

84 @Test
85 public void testDiscernibilityWithSuppression() throws Exception {
86 int k = 3;
87 MaterializedHierarchy dateHierarchy = new MaterializedHierarchy();
88 dateHierarchy.add("01/01/2008", "Jan_2008", "2008");
89 dateHierarchy.add("02/01/2008", "Jan_2008", "2008");
90 dateHierarchy.add("03/01/2008", "Jan_2008", "2008");
91
92 MaterializedHierarchy genderHierarchy = new MaterializedHierarchy();
93 genderHierarchy.add("M", "Person");
94 genderHierarchy.add("F", "Person");
95
96 MaterializedHierarchy ageHierarchy = new MaterializedHierarchy();
97 ageHierarchy.add("13", "10-14", "10-19", "0-49", "0-99");
98 ageHierarchy.add("18", "15-19", "10-19", "0-49", "0-99");
99 ageHierarchy.add("19", "15-19", "10-19", "0-49", "0-99");
100 ageHierarchy.add("21", "20-24", "20-29", "0-49", "0-99");
101 ageHierarchy.add("22", "20-24", "20-29", "0-49", "0-99");
102 ageHierarchy.add("23", "20-24", "20-29", "0-49", "0-99");
103
104 List<ColumnInformation> columnInformationList = new ArrayList<>();
105 columnInformationList.add(new CategoricalInformation(dateHierarchy, ColumnType.QUASI));
106 columnInformationList.add(new CategoricalInformation(genderHierarchy, ColumnType.QUASI));
107 columnInformationList.add(new CategoricalInformation(ageHierarchy, ColumnType.QUASI));
108
109 IPVDataset original = IPVDataset.load(getClass().getResourceAsStream("/testOLA.csv"), false, ',', '"', false);
110
111 List<PrivacyConstraint> privacyConstraints = new ArrayList<>();
112 privacyConstraints.add(new KAnonymity(k));
113
114 OLAOptions olaOptions = new OLAOptions(20.0d);
115
116 OLA ola = new OLA();
117 ola.initialize(original, columnInformationList, privacyConstraints, olaOptions);
118
119 IPVDataset anonymized = ola.apply();
120
121 InformationMetric discernibility = new Discernibility().initialize(original, anonymized,
122 ola.getOriginalPartitions(), ola.getAnonymizedPartitions(), columnInformationList, null);
123
124 double discernibilityValue = discernibility.report();
125
126 // the anonymized dataset contains two clusters: one with 3 rows and another with 5 rows
127 // and also two suppressed partitions, each one with one row
128
129 double expected = 3.0*3.0 + 5.0*5.0 + 10.0*1.0*1.0 + 10.0*1.0*1.0;
130 assertEquals(discernibilityValue, expected);
131 }

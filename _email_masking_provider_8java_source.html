99 @Override
100 public String mask(String identifier) {
101 String domain;
102 String originalUsername;
103
104 if (!emailIdentifier.isOfThisType(identifier)) {
105 switch (failMode) {
106 case FailMode.RETURN_ORIGINAL:
107 return identifier;
108 case FailMode.GENERATE_RANDOM:
109 domain = RandomGenerators.randomUIDGenerator(8) + "." + RandomGenerators.getRandomTLD();
110 return buildEmail(RandomGenerators.randomUsernameAndDomain(), domain);
111 case FailMode.THROW_ERROR:
112 log.error("invalid numerical value");
113 throw new IllegalArgumentException("invalid numerical value");
114 case FailMode.RETURN_EMPTY:
115 default:
116 return "";
117 }
118
119 }
120
121 int index = identifier.indexOf('@');
122 originalUsername = identifier.substring(0, index);
123 domain = identifier.substring(index + 1);
124
125 String username = (this.usernameVirtualFieldMaskingProvider != null) ?
126 this.usernameVirtualFieldMaskingProvider.mask(originalUsername) : randomizeUsernamePart();
127
128 String maskedDomain;
129
130 if (this.domainVirtualFieldMaskingProvider != null) {
131 Tuple<String, String> domainParts = DomainUtils.splitDomain(domain, this.preserveDomains, this.preserveSubnets);
132 String hostname = domainParts.getFirst();
133 String tld = domainParts.getSecond();
134
135 maskedDomain = this.domainVirtualFieldMaskingProvider.mask(hostname);
136 if (!tld.isEmpty()) {
137 maskedDomain += "." + tld;
138 }
139 } else {
140 maskedDomain = RandomGenerators.randomHostnameGenerator(domain, this.preserveDomains);
141 }
142
143 return buildEmail(username, maskedDomain);
144 }

66 @Test
67 public void extractCardinalities() {
68 Dataset<Row> dataset = createDataset(Arrays.asList(
69 Arrays.asList( "A", false, 10 ),
70 Arrays.asList( "B", true, 1000 ),
71 Arrays.asList( "B", null, 200 ),
72 Arrays.asList( "D", false, 11 )
73 ), new StructType(new StructField[] {
74 new StructField("f1", DataTypes.StringType, false, Metadata.empty()),
75 new StructField("f2", DataTypes.BooleanType, true, Metadata.empty()),
76 new StructField("f3", DataTypes.IntegerType, false, Metadata.empty())
77 }));
78
79 Map<String, Long> cardinality = ExtractFieldsStatistics.extractCardinalities(dataset);
80
81 assertThat(cardinality.size(), is(3));
82 assertThat(cardinality.containsKey("f1"), is(true));
83 assertThat(cardinality.get("f1"), is(3L));
84 assertThat(cardinality.containsKey("f2"), is(true));
85 assertThat(cardinality.get("f2"), is(3L));
86 assertThat(cardinality.containsKey("f3"), is(true));
87 assertThat(cardinality.get("f3"), is(4L));
88 }

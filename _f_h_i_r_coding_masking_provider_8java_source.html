69 public JsonNode mask(JsonNode node) {
70 try {
71 FHIRCoding cc = JsonUtils.MAPPER.treeToValue(node, FHIRCoding.class);
72 FHIRCoding maskedCc = mask(cc);
73 return JsonUtils.MAPPER.valueToTree(maskedCc);
74 } catch (Exception e) {
75 return NullNode.getInstance();
76 }
77 }

79 public FHIRCoding mask(FHIRCoding coding) {
80 if (coding == null) {
81 return null;
82 }
83
84 if (this.maskDisplay && !isAlreadyMasked(DISPLAY_PATH)) {
85 String display = coding.getDisplay();
86 if (display != null) {
87 coding.setDisplay(displayMaskingProvider.mask(display));
88 }
89 }
90
91 if (this.maskVersion && !isAlreadyMasked(VERSION_PATH)) {
92 String version = coding.getVersion();
93 if (version != null) {
94 coding.setVersion(versionMaskingProvider.mask(version));
95 }
96 }
97
98 if (this.maskSystem && !isAlreadyMasked(SYSTEM_PATH)) {
99 String system = coding.getSystem();
100 if (system != null) {
101 coding.setSystem(systemMaskingProvider.mask(system));
102 }
103 }
104
105 if (this.maskCode && !isAlreadyMasked(CODE_PATH)) {
106 String code = coding.getCode();
107 if (code != null) {
108 coding.setCode(codeMaskingProvider.mask(code));
109 }
110 }
111
112 return coding;
113 }

88 public FHIRContactPoint mask(FHIRContactPoint contactPoint) {
89 if (contactPoint == null) {
90 return null;
91 }
92
93 if (this.removeSystem) {
94 contactPoint.setSystem(null);
95 }
96
97 String originalValue = contactPoint.getValue();
98
99 if (this.maskValue && !isAlreadyMasked(valuePath) && originalValue != null) {
100 String maskedValue;
101 String system = contactPoint.getSystem();
102
103 if (system == null) {
104 maskedValue = randomMaskingProvider.mask(originalValue);
105 } else {
106 /* https://www.hl7.org/fhir/valueset-contact-point-system.html */
107 switch (system) {
108 case "email":
109 maskedValue = emailMaskingProvider.mask(originalValue);
110 break;
111 case "other":
112 maskedValue = urlMaskingProvider.mask(originalValue);
113 break;
114 default:
115 maskedValue = phoneMaskingProvider.mask(originalValue);
116 break;
117 }
118 }
119
120 contactPoint.setValue(maskedValue);
121 }
122
123 if (this.randomizeUse && !isAlreadyMasked(usePath)) {
124 String randomUse = useValues[random.nextInt(useValues.length)];
125 contactPoint.setUse(randomUse);
126 }
127
128 if (this.maskPeriod && !isAlreadyMasked(periodPath)) {
129 contactPoint.setPeriod(periodMaskingProvider.mask(contactPoint.getPeriod()));
130 }
131
132 if (this.removeExtensions) {
133 contactPoint.setExtension(null);
134 }
135
136 return contactPoint;
137 }

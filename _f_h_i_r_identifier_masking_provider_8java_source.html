85 private String maskValue(String value) {

89 public JsonNode mask(JsonNode node) {
90 try {
91 FHIRIdentifier obj = JsonUtils.MAPPER.treeToValue(node, FHIRIdentifier.class);
92 FHIRIdentifier maskedObj = mask(obj);
93 return JsonUtils.MAPPER.valueToTree(maskedObj);
94 } catch (Exception e) {
95 return NullNode.getInstance();
96 }
97 }

99 public FHIRIdentifier mask(FHIRIdentifier identifier) {
100 if (identifier == null) {
101 return null;
102 }
103
104 if (this.maskPeriod && !isAlreadyMasked(PERIOD_PATH)) {
105 identifier.setPeriod(periodMaskingProvider.mask(identifier.getPeriod()));
106 }
107
108 if (this.maskType && !isAlreadyMasked(TYPE_PATH)) {
109 identifier.setType(typeMaskingProvider.mask(identifier.getType()));
110 }
111
112 if (this.maskSystem && !isAlreadyMasked(SYSTEM_PATH)) {
113 String system = identifier.getSystem();
114 if (system != null) {
115 identifier.setSystem(systemMaskingProvider.mask(system));
116 }
117 }
118
119 if (this.maskAssigner && !isAlreadyMasked(ASSIGNER_PATH)) {
120 identifier.setAssigner(assignerMaskingProvider.mask(identifier.getAssigner()));
121 }
122
123 if (this.maskValue && !isAlreadyMasked(VALUE_PATH)) {
124 String value = identifier.getValue();
125 if (value != null) {
126 identifier.setValue(maskValue(value));
127 }
128 }
129
130 if (this.removeExtensions) {
131 identifier.setExtension(null);
132 }
133
134 return identifier;
135 }

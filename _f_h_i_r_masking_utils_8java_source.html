34 public static String maskResourceId(String id, boolean preserveIdPrefix, MaskingProvider maskingProvider) {
35 if (!preserveIdPrefix) {
36 return maskingProvider.mask(id);
37 }
38
39 int slashIndex = id.lastIndexOf('/');
40 if (slashIndex == -1) {
41 return maskingProvider.mask(id);
42 }
43
44 String prefix = id.substring(0, slashIndex + 1);
45
46 return prefix + maskingProvider.mask(id.substring(slashIndex + 1));
47 }

49 public static Set<String> setFromString(String value, boolean toUppercase) {
50 Set<String> set = new HashSet<>();
51
52 if (value != null && !value.isEmpty()) {
53 for (String v : value.split(",")) {
54 if (toUppercase) {
55 set.add(v.toUpperCase());
56 } else {
57 set.add(v);
58 }
59 }
60 }
61
62 return set;
63 }

65 public static JsonNode preprocessFHIRObject(JsonNode node) {
66 JsonNode typeNode = node.get("resourceType");
67
68 if (typeNode == null) {
69 return null;
70 }
71
72 String resourceType = typeNode.asText();
73 String key = "/fhir/" + resourceType;
74
75 ObjectNode processedNode = new ObjectNode(JsonNodeFactory.instance);
76 processedNode.set(key, node);
77
78 return processedNode;

80 }

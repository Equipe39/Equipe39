44 public FHIRRangeMaskingProvider(MaskingConfiguration maskingConfiguration, Set<String> maskedFields, String fieldPath, MaskingProviderFactory factory) {

47 this.LOW_PATH = fieldPath + "/low";

58 maskedFields, HIGH_PATH, this.factory);
59 }

61 @Override
62 public JsonNode mask(JsonNode node) {
63 try {
64 FHIRRange obj = JsonUtils.MAPPER.treeToValue(node, FHIRRange.class);
65 FHIRRange maskedObj = mask(obj);
66 return JsonUtils.MAPPER.valueToTree(maskedObj);
67 } catch (Exception e) {
68 return NullNode.getInstance();
69 }
70 }
71
72 public FHIRRange mask(FHIRRange range) {
73 if (this.deleteHigh) {
74 range.setHigh(null);
75 } else if (this.maskHigh && !isAlreadyMasked(HIGH_PATH)) {
76 range.setHigh(highMaskingProvider.mask(range.getHigh()));
77 }
78
79
80 if (this.deleteLow) {
81 range.setLow(null);
82 } else if (this.maskLow && !isAlreadyMasked(LOW_PATH)) {
83 range.setLow(lowMaskingProvider.mask(range.getLow()));
84 }
85
86 return range;
87 }

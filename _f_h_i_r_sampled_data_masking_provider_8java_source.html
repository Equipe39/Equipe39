98 public FHIRSampledData mask(FHIRSampledData sampledData) {
99
100 if (this.maskOrigin && !isAlreadyMasked(ORIGIN_PATH)) {
101 sampledData.setOrigin(originMaskingProvider.mask(sampledData.getOrigin()));
102 }
103
104 if (this.maskData && !isAlreadyMasked(DATA_PATH)) {
105 String data = sampledData.getData();
106 if (data != null) {
107 sampledData.setData(dataMaskingProvider.mask(data));
108 }
109 }
110
111 if (this.maskDimensions && !isAlreadyMasked(DIMENSIONS_PATH)) {
112 String dimensions = sampledData.getDimensions();
113 if (dimensions != null) {
114 sampledData.setDimensions(dimensionsMaskingProvider.mask(dimensions));
115 }
116 }
117
118 if (this.maskPeriod && !isAlreadyMasked(PERIOD_PATH)) {
119 float period = sampledData.getPeriod();
120 sampledData.setPeriod(Float.valueOf(periodMaskingProvider.mask(Float.toString(period))));
121 }
122
123 if (this.maskLowerLimit && !isAlreadyMasked(LOWERLIMIT_PATH)) {
124 float lowerLimit = sampledData.getLowerLimit();
125 sampledData.setLowerLimit(Float.valueOf(lowerLimitMaskingProvider.mask(Float.toString(lowerLimit))));
126 }
127
128 if (this.maskUpperLimit && !isAlreadyMasked(UPPERLIMIT_PATH)) {
129 float upperLimit = sampledData.getUpperLimit();
130 sampledData.setUpperLimit(Float.valueOf(upperLimitMaskingProvider.mask(Float.toString(upperLimit))));
131 }
132
133 if (this.maskFactor && !isAlreadyMasked(FACTOR_PATH)) {
134 float factor = sampledData.getFactor();
135 sampledData.setFactor(Float.valueOf(factorMaskingProvider.mask(Float.toString(factor))));
136 }
137
138 return sampledData;
139 }

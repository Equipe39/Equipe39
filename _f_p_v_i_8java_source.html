81 private List<Set<Integer>> extractCandidates(int i, ItemSet itemSet) {
82 final List<Set<Integer>> candidates = new ArrayList<>(itemSet.size());

84 for (Integer j : itemSet.getItems()) {

91 private boolean isUnique(List<Set<Integer>> candidates) {
92 if (1 == candidates.size()) {
93 return candidates.get(0).size() < kValue;
94 }
95
96 candidates.sort(Comparator.comparingInt(Set::size));
97
98 final Set<Integer> res = new HashSet<>(candidates.get(0));
99
100 if (res.size() < kValue) return true;
101
102 for (int i = 1; i < candidates.size(); ++i) {
103 final Set<Integer> next = candidates.get(i);
104
105 if (next.size() < kValue) return true;
106
107 res.removeIf(n -> !next.contains(n));
108
109 if (res.size() < kValue) return true;
110 }
111
112 return false;
113 }

115 @Override
116 public void run() {
117 while (true) {
118 FPVITask task = tasks.poll();
119 FPVIResult result = null;
120
121 if (null == task) {
122 continue;
123 }
124
125 if (null == task.getItemSet() && -1 == task.getFrom() && -1 == task.getTo()) {
126 results.add(new FPVIResult(null, false));
127 return; // termination condition
128 }
129
130 for (int i = task.getFrom(); i < task.getTo(); ++i) {
131 final List<Set<Integer>> candidates = extractCandidates(i, task.getItemSet());
132
133 if (isUnique(candidates)) {
134 result = new FPVIResult(task, true);
135 break;
136 }
137 }
138
139 if (null == result) result = new FPVIResult(task, false);
140
141 results.add(result);
142 }
143 }
144 });
145
146 executors[i].start();
147 }
148
149 boolean terminated = false;
150
151 while (0 < counter) {
152 FPVIResult result = results.poll();
153
154 if (null == result) {
155 continue;
156 }
157
158 FPVITask taskCompleted = result.getTask();
159
160 if (null == taskCompleted) {
161 --counter;
162 continue;
163 }
164
165 if (result.isUnique()) {

167 }
168
169 if (taskGenerator.hasNext()) {
170 final FPVITask task = taskGenerator.next();
171 tasks.add(task);
172 } else if (!terminated) {
173 for (int i = 0; i < nThread; ++i) {
174 tasks.add(new FPVITask(null, -1, -1));
175 }
176
177 terminated = true;
178 }
179
180 }
181
182 for (Thread t : executors) {
183 try {
184 t.join();
185 } catch (InterruptedException ignored) {
186 }
187 }
188
189 final Collection<ItemSet> bannedItemSets = generator.getBanned();
190 final List<IPVVulnerability> vulnerabilities = new ArrayList<>(bannedItemSets.size());
191
192 for (final ItemSet banned : bannedItemSets)
193 vulnerabilities.add(new IPVVulnerability(banned));
194
195 return vulnerabilities;
196 }

198 @Override
199 public List<Integer> getRowIds(ItemSet itemSet) {
200 final List<Integer> rowIds = new LinkedList<>();
201
202 for (int i = 0; i < index.numberOfRows(); ++i) {
203 final Set<Integer> keySet = new HashSet<>();
204 for (final Integer item : itemSet.getItems()) {
205 if (keySet.isEmpty()) {
206 keySet.addAll(index.getKeySet(i, item));
207 } else {
208 Set<Integer> newKeySet = index.getKeySet(i, item);
209
210 keySet.removeIf(element -> !newKeySet.contains(element));
211 }
212
213 if (keySet.size() < kValue) {
214 rowIds.add(i);
215 break;
216 }
217 }
218 }
219
220 return rowIds;

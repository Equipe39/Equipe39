46 public FPVITaskGenerator(final int nParallelJobs, final int maxSize, final int batchSize, final IPVGenerator generator) {
47 this.nParallelJobs = nParallelJobs;
48 this.maxSize = maxSize;
49 this.batchSize = batchSize;
50 this.generator = generator;
51 nextTasks = new ArrayList<>(nParallelJobs);
52 idx = 0;
53 }
54
55 @Override
56 public boolean hasNext() {
57 return generator.hasNext() || 0 != nextTasks.size();
58 }
59
60 @Override
61 public FPVITask next() {
62 if (0 == nextTasks.size() && !generator.hasNext()) throw new NoSuchElementException();
63
64 if (generator.hasNext() && nextTasks.size() < nParallelJobs) {
65 nextTasks.add(new TaskEntry(generator.next()));
66 }
67
68 idx %= nextTasks.size();
69
70 final TaskEntry entry = nextTasks.get(idx);
71 final FPVITask next = new FPVITask(entry.getItemSet(), entry.getIdx(), Math.min(entry.getIdx() + batchSize, maxSize));
72
73 if (maxSize - batchSize <= entry.getIdx()) {
74 nextTasks.remove(idx);
75 } else {
76 entry.add(batchSize);
77 idx += 1;
78 }
79
80 return next;
81 }

88 public void ban(FPVITask task) {
89 // ban the itemset
90 ItemSet itemSet = task.getItemSet();
91
92 generator.ban(itemSet);
93
94 // clean up
95 nextTasks.removeIf(is -> itemSet.isSubSetOf(is.getItemSet()));
96 }

93 private Map<String, OriginalMaskedValuePair> extractOperandsNotToBeMasked(Record record, Map<String, FieldRelationship> predefinedRelationships, Map<String, DataMaskingTarget> fieldsToMask) {
94 final Map<String, OriginalMaskedValuePair> operandsNotToBeMasked = new HashMap<>();
95
96 if (Objects.nonNull(predefinedRelationships)) {
97 for (final FieldRelationship relationship : predefinedRelationships.values()) {
98 for (final RelationshipOperand operand : relationship.getOperands()) {
99 final String operandName = operand.getName();

104 operandsNotToBeMasked.put(operandName, new OriginalMaskedValuePair(operandValue, operandValue));

106 }

108 }

113 public Record maskRecord(Record record, MaskingProviderFactory maskingProvidersFactory, Set<String> alreadyMaskedFields, DataMaskingOptions dataMaskingOptions) {
114 Map<String, FieldRelationship> relationships = dataMaskingOptions.getPredefinedRelationships();
115 Map<String, DataMaskingTarget> fieldsToMask = dataMaskingOptions.getToBeMasked();
116 Queue<String> queuedFields = new LinkedList<>(fieldsToMask.keySet());
117
118 Map<String, OriginalMaskedValuePair> processedFields = extractOperandsNotToBeMasked(record, relationships, fieldsToMask);
119
120 fieldsLoop:
121 while (!queuedFields.isEmpty()) {
122 String fieldIdentifier = queuedFields.poll();
123
124 byte[] fieldValue = record.getFieldValue(fieldIdentifier);
125 if (fieldValue == null) {
126 processedFields.put(fieldIdentifier, new OriginalMaskedValuePair(null, null));
127 continue;
128 }
129
130 String originalValue = new String(fieldValue);
131 String maskedValue = originalValue;
132
133 if (!alreadyMaskedFields.contains(fieldIdentifier)) {
134 FieldRelationship fieldRelationship = null;
135
136 if (null != relationships && !relationships.isEmpty()) {
137 fieldRelationship = relationships.get(fieldIdentifier);
138
139 if (fieldRelationship != null) {
140 for (RelationshipOperand operand : fieldRelationship.getOperands()) {
141 if (!processedFields.containsKey(operand.getName())) {
142 queuedFields.add(fieldIdentifier);
143 continue fieldsLoop;
144 }
145 }
146 }
147 }
148
149 ProviderType fieldType = fieldsToMask.get(fieldIdentifier).getProviderType();
150 String targetPath = fieldsToMask.get(fieldIdentifier).getTargetPath();
151
152 MaskingProvider provider = maskingProvidersFactory.get(fieldIdentifier, fieldType);
153
154 maskedValue = maskValue(provider, originalValue, fieldIdentifier, fieldRelationship, processedFields);
155 record.setFieldValue(targetPath, (maskedValue != null) ? maskedValue.getBytes() : null);
156 }
157
158 processedFields.put(fieldIdentifier, new OriginalMaskedValuePair(originalValue, maskedValue));
159 }
160
161 return record;
162 }

164 String maskValue(MaskingProvider provider, String originalValue, String fieldIdentifier, FieldRelationship fieldRelationship, Map<String, OriginalMaskedValuePair> processedFields) {
165 if (null == fieldRelationship) {
166 return provider.mask(originalValue, fieldIdentifier);
167 } else {
168 switch (fieldRelationship.getRelationshipType()) {
169 case KEY:
170 return provider.maskWithKey(originalValue, processedFields.get(fieldRelationship.getOperands()[0].getName()).getOriginal());
171 case DISTANCE:
172 return provider.maskDistance(
173 originalValue,
174 processedFields.get(fieldRelationship.getOperands()[0].getName()).getOriginal(),
175 processedFields.get(fieldRelationship.getOperands()[0].getName()).getMasked());
176 case EQUALS:
177 return provider.maskEqual(originalValue, processedFields.get(fieldRelationship.getOperands()[0].getName()).getMasked());
178 case LINKED:
179 return provider.maskLinked(originalValue, processedFields.get(fieldRelationship.getOperands()[0].getName()).getMasked(), fieldRelationship.getOperands()[0].getType());
180 case LESS:
181 return provider.maskLess(originalValue, processedFields.get(fieldRelationship.getOperands()[0].getName()).getMasked(), processedFields.get(fieldRelationship.getOperands()[0].getName()).getOriginal());
182 case GREATER:
183 return provider.maskGreater(originalValue, processedFields.get(fieldRelationship.getOperands()[0].getName()).getMasked(), processedFields.get(fieldRelationship.getOperands()[0].getName()).getOriginal());
184 case RATIO:
185 return provider.maskWithRatio(originalValue, processedFields.get(fieldRelationship.getOperands()[0].getName()).getMasked(), processedFields.get(fieldRelationship.getOperands()[0].getName()).getOriginal());
186 case SUM:
187 case SUM_APPROXIMATE:
188 case PRODUCT:
189 case GREP_AND_MASK:
190 default:
191 throw new UnsupportedOperationException();
192 }
193 }

233 protected Map<String, Map<String, Counter>> checkFieldNames(Record record, Collection<Identifier> identifiers) {
234 Map<String, Map<String, Counter>> types = new HashMap<>();
235
236 for (final String fieldReference : record.getFieldReferences()) {
237 final Map<String, Counter> fieldCounters = new HashMap<>();
238 final String fieldName = extractFieldName(fieldReference);
239
240 for (Identifier identifier : identifiers) {
241 if (identifier.isAppropriateName(fieldName)) {
242 if (!fieldCounters.containsKey(identifier.getType().getName())) {
243 fieldCounters.put(identifier.getType().getName(), new Counter(0L));
244 }
245 fieldCounters.get(identifier.getType().getName()).counter += 1L;
246 }
247 }
248 if (!fieldCounters.isEmpty()) {
249 types.put(fieldReference, fieldCounters);
250 }
251 }
252
253 return types;
254 }

260 public IdentificationReport identifyTypesStream(InputStream input, DataTypeFormat inputFormatType,
261 DatasetOptions datasetOptions,
262 Collection<Identifier> identifiers, int firstN) throws IOException {
263 Map<String, Map<String, Counter>> allTypes = new HashMap<>();
264 Map<String, Map<String, Counter>> columnTypes = new HashMap<>();
265
266 long recordCount = 0;
267 for (final Record record : extractRecords(input, datasetOptions)) {
268 if (record.isHeader()) {
269 columnTypes = checkFieldNames(record, identifiers);
270
271 continue;
272 }
273
274 for (final String fieldReference : record.getFieldReferences()) {
275 final Map<String, Counter> fieldTypes = allTypes.computeIfAbsent(fieldReference, ignored -> new HashMap<>());
276
277 byte[] bValue = record.getFieldValue(fieldReference);
278
279 if (null == bValue) {
280 updateCounter(fieldTypes, ProviderType.EMPTY.getName());
281 continue;
282 }
283
284 final String value = new String(bValue);
285
286 if (value.isEmpty()) {
287 updateCounter(fieldTypes, ProviderType.EMPTY.getName());
288 continue;
289 }
290
291 int found = 0;
292
293 for (final Identifier identifier : identifiers) {
294 if (identifier.isOfThisType(value)) {
295 updateCounter(fieldTypes, identifier.getType().getName());
296 found += 1;
297 }
298 }
299
300 if (found == 0) {
301 updateCounter(fieldTypes, ProviderType.UNKNOWN.getName());
302 }
303 }
304
305 recordCount += 1;
306 }
307
308 return assembleReport(
309 allTypes,
310 columnTypes,
311 recordCount
312 );
313 }

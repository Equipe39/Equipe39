37 public static LatitudeLongitude xyzToLatlon(double x, double y, double z) {
38 double b = Math.sqrt(asq * (1 - esq));
39 double bsq = Math.pow(b, 2);
40 double ep = Math.sqrt((asq - bsq) / bsq);
41 double p = Math.sqrt(Math.pow(x, 2) + Math.pow(y, 2));
42 double th = Math.atan2(R * z, b * p);
43
44 double lon = Math.atan2(y, x);
45 double lat = Math.atan2((z + Math.pow(ep, 2) * b * Math.pow(Math.sin(th), 3)), (p - esq * R * Math.pow(Math.cos(th), 3)));
46 double N = R / (Math.sqrt(1 - esq * Math.pow(Math.sin(lat), 2)));
47 double alt = p / Math.cos(lat) - N;
48
49 // mod lat to 0-2pi
50 lon = lon % (2 * Math.PI);
51
52 // correction for altitude near poles left out.
53 return new LatitudeLongitude(Math.toDegrees(lat), Math.toDegrees(lon));
54 }

56 public static XYZ latlonToXYZ(double lat, double lon) {
57 double cosLat = Math.cos(lat * Math.PI / 180.0);
58 double sinLat = Math.sin(lat * Math.PI / 180.0);
59 double cosLon = Math.cos(lon * Math.PI / 180.0);
60 double sinLon = Math.sin(lon * Math.PI / 180.0);
61 double rad = R;
62 double f = 1.0 / 298.257224;
63 double C = 1.0 / Math.sqrt(cosLat * cosLat + (1 - f) * (1 - f) * sinLat * sinLat);
64 double S = (1.0 - f) * (1.0 - f) * C;
65 double h = 0.0;
66
67 double x = (rad * C + h) * cosLat * cosLon;
68 double y = (rad * C + h) * cosLat * sinLon;
69 double z = (rad * S + h) * sinLat;
70
71 return new XYZ(x, y, z);
72 }

95 public static double latitudeLongitudeDistance(Double lat1, Double lon1, Double lat2, Double lon2) {
96 double dLat = Math.toRadians(lat2 - lat1);
97 double dLng = Math.toRadians(lon2 - lon1);
98 double a = Math.sin(dLat / 2) * Math.sin(dLat / 2) +
99 Math.cos(Math.toRadians(lat1)) * Math.cos(Math.toRadians(lat2)) *
100 Math.sin(dLng / 2) * Math.sin(dLng / 2);
101 double c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
102 return R * c;
103 }

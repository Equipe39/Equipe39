34 private int numberOfQuasiColumns() {
35 int s = 0;
36
37 for (ColumnInformation c : columnInformationList) {
38 if (c.getColumnType() == ColumnType.QUASI) {
39 s += 1;
40 }
41 }
42
43 return s;
44 }
45
46 private double getNCP(Partition p) {
47 double gcp = 0.0;
48
49 for (int i = 0; i < columnInformationList.size(); i++) {
50 ColumnInformation columnInformation = columnInformationList.get(i);
51 if (columnInformation.getColumnType() == ColumnType.QUASI) {
52 gcp += p.getNormalizedWidth(i);
53 }
54 }
55
56 return gcp;
57 }

104 public double report() {
105 int totalRecords = anonymized.getNumberOfRows();
106 int d = numberOfQuasiColumns();
107
108 double gcp = 0.0;
109
110 for (Partition p : partitions) {
111 double ncp = getNCP(p);
112 gcp += p.size() * ncp;
113 }
114
115 return gcp / ((double) d * (double) totalRecords);
116 }
117
118 private InformationLossResult reportForColumn(int columnIndex) {
119 int totalRecords = anonymized.getNumberOfRows();
120 double gcp = 0.0;
121 for (Partition p : partitions) {
122 gcp += p.size() * p.getNormalizedWidth(columnIndex);
123 }
124
125 return new InformationLossResult(gcp / ((double) totalRecords), -1.0, -1.0);
126 }
127
128 @Override
129 public List<InformationLossResult> reportPerQuasiColumn() {
130
131 List<InformationLossResult> results = new ArrayList<>();
132 int columnIndex = 0;
133
134 for (ColumnInformation columnInformation : columnInformationList) {
135 if (columnInformation.getColumnType() != ColumnType.QUASI) {
136 columnIndex++;
137 continue;
138 }
139
140 InformationLossResult iloss = reportForColumn(columnIndex);
141
142 results.add(iloss);
143 columnIndex++;
144 }
145
146 return results;
147 }

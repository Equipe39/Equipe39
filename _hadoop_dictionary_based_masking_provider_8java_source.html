36public class HadoopDictionaryBasedMaskingProvider implements MaskingProvider {
37 private final List<String> dictionary;
38 private final SecureRandom random;
39
40 public HadoopDictionaryBasedMaskingProvider(SecureRandom random, MaskingConfiguration maskingConfiguration) throws IOException {
41 this.dictionary = loadDictionary(maskingConfiguration.getStringValue("hadoop.dictionary.path"));
42 this.random = random;
43 }
44
45 private List<String> loadDictionary(String path) throws IOException {
46 return new ArrayList<>(new HashSet<>(loadTerms(path)));
47 }
48
49 private Collection<? extends String> loadTerms(String path) throws IOException {
50 try (
51 InputStream inputStream = SparkUtils.createHDFSInputStream(path);
52 BufferedReader reader = new BufferedReader(new InputStreamReader(inputStream));
53 ) {
54 List<String> terms = new ArrayList<>();
55 for (String line = reader.readLine(); null != line; line = reader.readLine()) {
56 terms.add(line.trim());
57 }
58
59 return terms;
60 }
61 }

63 @Override
64 public String mask(String value) {
65 return dictionary.get(

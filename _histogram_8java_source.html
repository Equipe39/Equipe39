34 public void update(Histogram<T> other) {
35 for (Map.Entry<T, Long> pair : other.entrySet()) {
36 T key = pair.getKey();
37 Long value = pair.getValue();
38
39 if (!this.containsKey(key)) {
40 this.put(key, value);
41 } else {
42 Long v = this.get(key);
43 this.put(key, v + value);
44 }
45 }

48 public static <K> Histogram<K> createHistogram(List<K> values) {
49 Histogram<K> histogram = new Histogram();
50
51 for (K value : values) {
52
53 Long counter = histogram.get(value);
54 if (counter == null) {
55 histogram.put(value, 1L);
56 } else {
57 histogram.put(value, counter + 1);
58 }
59 }

62 public double report() {
63 double risk = 0.0;
64
65 final double p = n / _N;
66
67 for (int k = 0; k < F.size(); ++k) {
68 final int Fk = extractFk(k);
69
70 final HypergeometricDistribution distribution = new HypergeometricDistribution(Fk, f.get(k), f.get(k));
71
72 double k_risk = distribution.probability(1);
73
74 risk = FastMath.max(risk, k_risk);
75 }

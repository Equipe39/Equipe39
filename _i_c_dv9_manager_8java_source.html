67 private void readICDList(Collection<ResourceEntry> entries) {
68
69 for (ResourceEntry entry : entries) {
70 InputStream inputStream = entry.createStream();
71
72 try (CSVParser parser = Readers.createCSVReaderFromStream(inputStream, ';', '"')) {
73 for (CSVRecord record : parser) {
74 String code = record.get(0);
75 String shortName = record.get(1);
76 String fullName = record.get(2);
77 String chapterCode = record.get(3);
78 String chapterName = record.get(4);
79 String categoryCode = record.get(5);
80 String categoryName = record.get(6);
81
82 ICD ICDv9CodeRepr = new ICD(code, shortName, fullName,
83 chapterCode, chapterName, categoryCode, categoryName, ICDFormat.CODE);
84 ICD ICDv9NameRepr = new ICD(code, shortName, fullName,
85 chapterCode, chapterName, categoryCode, categoryName, ICDFormat.NAME);
86
87 this.icdByCodeMap.getMap().put(code, ICDv9CodeRepr);
88 this.icdByNameMap.getMap().put(shortName.toUpperCase(), ICDv9NameRepr);
89 this.icdByNameMap.getMap().put(fullName.toUpperCase(), ICDv9NameRepr);
90 }
91 inputStream.close();
92 } catch (IOException | NullPointerException e) {
93 e.printStackTrace();
94 }
95 }
96 }

109 public ICD lookupICD(String codeOrName) {
110 String key = codeOrName.toUpperCase();
111 ICD ICDv9;
112
113 if ((ICDv9 = this.icdByCodeMap.getMap().get(key)) != null) {
114 return ICDv9;
115 } else if ((ICDv9 = this.icdByNameMap.getMap().get(key)) != null) {
116 return ICDv9;
117 }
118
119 return null;
120 }

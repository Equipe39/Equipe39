69 @Override
70 public String mask(String identifier) {
71 String tac;
72 if (!IMEI_IDENTIFIER.isOfThisType(identifier) || !this.preserveTAC) {
73 tac = imeiManager.getRandomKey();
74 } else {
75 tac = identifier.substring(0, 8);
76 }
77
78 String body = tac + RandomGenerators.generateRandomDigitSequence(6);

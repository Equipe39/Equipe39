69 @Override
70 public String mask(String identifier) {
71 if (!imsiManager.isValidIMSI(identifier)) {
72 return RandomGenerators.generateRandomIMSI();
73 }
74
75 String mcc;
76 String mnc;
77
78 if (this.preserveMCC) {
79 mcc = identifier.substring(0, 3);
80 if (this.preserveMNC) {
81 mnc = identifier.substring(3, 6);
82 } else {
83 mnc = imsiManager.getRandomMNC(mcc);
84 }
85 } else {
86 mcc = imsiManager.getRandomMCC();
87 mnc = imsiManager.getRandomMNC(mcc);
88 }
89
90 String uid = RandomGenerators.generateRandomDigitSequence(15 - mcc.length() - mnc.length());
91
92 String builder = mcc +
93 mnc +
94 uid;
95
96 return builder;
97 }

28public class IPAddressIdentifier extends AbstractIdentifier {
29 private static final String ipv4AddressPattern = "^(\\d{1,2}|1\\d\\d|2[0-4]\\d|25[0-5])\\.(\\d{1,2}|1\\d\\d|2[0-4]\\d|25[0-5])\\.(\\d{1,2}|1\\d\\d|2[0-4]\\d|25[0-5])\\.(\\d{1,2}|1\\d\\d|2[0-4]\\d|25[0-5])$";
30 private static final Pattern ipv4matcher = Pattern.compile(ipv4AddressPattern);

32 private static final String[] appropriateNames = {"IP Address", "IPAddress"};

36 @Override

41 @Override
42 public boolean isOfThisType(String data) {
43 int matches = 0;
44
45 for (int i = 0; i < data.length(); i++) {
46 char c = data.charAt(i);
47 if (Character.isDigit(c) || c == '.' || c == ':') {
48 matches++;
49 }
50 }
51
52 if (matches == 0) {
53 return false;
54 }
55
56 return isIPv4(data) || isIPv6(data);
57 }
58
59 @Override
60 public String getDescription() {
61 return "IP address identification. Supports both IPv4 and IPv6 addresses";
62 }
63
70 public boolean isIPv4(String data) {
71 if (data.length() < 7 || data.length() > 15) {
72 return false;
73 }
74
75 return ipv4matcher.matcher(data).matches();
76 }
77
84 public boolean isIPv6(String data) {
85 if (data.length() > 45) {
86 return false;
87 }
88
89 int idx = data.indexOf('%');
90
91 if (idx > 0) {
92 data = data.substring(0, idx);
93 }
94
95 return INET_ADDRESS_VALIDATOR.isValidInet6Address(data);
96 //return ipv6matcher.matcher(data.toLowerCase()).matches();
97 }

112 public String toInlineXML() {
113 if (null == type || type.isEmpty()) {
114 return text;
115 } else {
116
117 if (type.size() == 1) {
118 String typeName = type.iterator().next().getType();
119 if (typeName.equals("UNKNOWN") || typeName.equals("O")) {
120 return text;
121 }
122 }
123
124 return "<ProviderType:" + concatTypes(",") + ">" + text + "</ProviderType>";

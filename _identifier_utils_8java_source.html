35 public static boolean looksLikeFreeText(String value) {
36 if (value.length() < 10) {
37 return false;
38 }
39
40 int whitespaceRegions = 0;
41
42 for (int i = 0; i < value.length() && whitespaceRegions < 2; i++) {
43 char c = value.charAt(i);
44 if (Character.isWhitespace(c)) {
45 whitespaceRegions++;
46 while (i < value.length() && Character.isWhitespace(value.charAt(i))) {
47 i++;
48 }
49 }
50 }
51
52 return whitespaceRegions >= 2;
53 }

55 public static int createCharacterProfile(String input) {
56 int mask = CharacterRequirements.NONE;
57
58 for (int i = 0; i < input.length(); i++) {
59 char c = input.charAt(i);
60
61 if (Character.isDigit(c)) {
62 mask |= CharacterRequirements.DIGIT;
63 } else if (Character.isAlphabetic(c)) {
64 mask |= CharacterRequirements.ALPHA;
65 } else if (Character.isWhitespace(c)) {
66 mask |= CharacterRequirements.SPACE;
67 } else if (c == '.') {
68 mask |= CharacterRequirements.DOT;
69 } else if (c == '@') {
70 mask |= CharacterRequirements.AT;
71 } else if (c == '-') {
72 mask |= CharacterRequirements.DASH;
73 } else if (c == ':') {
74 mask |= CharacterRequirements.COLUMN;
75 }
76 }
77
78 return mask;
79 }

81 public static int fillCharacterMap(String input, int[] counters) {
82 int nonASCII = 0;
83
84 for (int i = 0; i < input.length(); i++) {
85 char c = input.charAt(i);
86 if (c > 256) {
87 nonASCII++;
88 continue;
89 }
90
91 counters[c]++;
92 }
93
94 return nonASCII;

97 private static int calculateFrequency(long count, long numberOfIdentified, long numberOfEmpty,
98 long numberOfUnknown, boolean considerEmptyForConfidence) {
99 long totalCount = numberOfIdentified;
100 if (considerEmptyForConfidence) {
101 totalCount += numberOfEmpty;
102 }
103
104 if (totalCount == 0) {
105 return -1;
106 }
107
108 return (int) (100 * count / totalCount);
109 }

127 private static IdentifiedType findBestTypePriorityBased(Collection<IdentifiedType> identifiedTypes, Long rowsProcessed,
128 IdentificationConfiguration identificationConfiguration) {
129 IdentifiedType bestType = new IdentifiedType(ProviderType.UNKNOWN.name(), -1);
130 int bestTypePriority = -1;
131 final boolean considerEmptyForConfidence = identificationConfiguration.getConsiderEmptyForFrequency();
132
133 long numberOfEmpty = 0;
134 long numberOfUnknown = 0;
135
136 for (IdentifiedType type : identifiedTypes) {
137 if (type.getTypeName().equals(ProviderType.EMPTY.name())) {
138 numberOfEmpty += type.getCount();
139 } else if (type.getTypeName().equals(ProviderType.UNKNOWN.name())) {
140 numberOfUnknown += type.getCount();
141 }
142 }
143
144 long numberOfIdentified = rowsProcessed - numberOfEmpty;
145
146 for (IdentifiedType type : identifiedTypes) {
147 String typeName = type.getTypeName();
148 if (typeName.equals(ProviderType.EMPTY.name()) || typeName.equals(ProviderType.UNKNOWN.name())) {
149 continue;
150 }
151
152 int typeFrequency = calculateFrequency(type.getCount(), numberOfIdentified, numberOfEmpty,
153 numberOfUnknown, considerEmptyForConfidence);
154
155 if (typeIdentifiedLessThanRequiredFrequency(typeName, typeFrequency, identificationConfiguration)) {
156 continue;
157 }
158
159 int typePriority = identificationConfiguration.getPriorityForType(typeName);
160
161 if (typePriority > bestTypePriority || typePriority == bestTypePriority && type.getCount() > bestType.getCount()) {
162 bestType = type;
163 bestTypePriority = typePriority;
164 }
165 }
166
167 return bestType;
168 }

176 private static IdentifiedType findBestTypeFrequencyBased(Collection<IdentifiedType> identifiedTypes, Long rowsProcessed, IdentificationConfiguration identificationConfiguration) {
177 IdentifiedType bestType = new IdentifiedType(ProviderType.UNKNOWN.name(), -1);
178 int bestTypeFrequency = -1;
179 final boolean considerEmptyForConfidence = identificationConfiguration.getConsiderEmptyForFrequency();
180
181 long numberOfEmpty = 0;
182 long numberOfUnknown = 0;
183
184 for (IdentifiedType type : identifiedTypes) {
185 if (type.getTypeName().equals(ProviderType.EMPTY.name())) {
186 numberOfEmpty += type.getCount();
187 } else if (type.getTypeName().equals(ProviderType.UNKNOWN.name())) {
188 numberOfUnknown += type.getCount();
189 }
190 }
191
192 long numberOfIdentified = rowsProcessed - numberOfEmpty;
193
194 log.debug("empty: {} u: {} i: {}", numberOfEmpty, numberOfUnknown, numberOfIdentified);
195 for (IdentifiedType type : identifiedTypes) {
196 String typeName = type.getTypeName();
197 if (typeName.equals(ProviderType.EMPTY.name()) || typeName.equals(ProviderType.UNKNOWN.name())) {
198 continue;
199 }
200
201 log.debug("type: {} count: {}", typeName, type.getCount());
202
203 int typeFrequency = calculateFrequency(type.getCount(), numberOfIdentified, numberOfEmpty,
204 numberOfUnknown, considerEmptyForConfidence);
205
206 log.debug("frequency: {}", typeFrequency);
207
208 if (typeIdentifiedLessThanRequiredFrequency(typeName, typeFrequency, identificationConfiguration)) {
209 continue;
210 }
211
212 log.debug("frequency threshold exceeded");
213 if (bestType == null) {
214 bestType = type;
215 bestTypeFrequency = typeFrequency;
216 continue;
217 }
218
219 if (typeFrequency > bestTypeFrequency) {
220 bestType = type;
221 bestTypeFrequency = typeFrequency;
222 } else if (typeFrequency == bestTypeFrequency) {
223 if (identificationConfiguration.getPriorityForType(typeName) > identificationConfiguration.getPriorityForType(bestType.getTypeName())) {
224 bestType = type;
225 bestTypeFrequency = typeFrequency;
226 }
227 }
228 }
229
230 return bestType;
231 }

233 public static Map<String, IdentifiedType> getIdentifiedType(Map<String, List<IdentifiedType>> values, long rowsProcessed, IdentificationConfiguration identificationConfiguration) {
234 Map<String, IdentifiedType> identifiedTypes = new HashMap<>(values.size());
235
236 for (String fieldName : values.keySet()) {
237 IdentifiedType bestType = findBestType(values.get(fieldName), rowsProcessed, identificationConfiguration);
238
239 if (null != bestType) {

247 public static Map<ProviderType, Long> identifySingleValue(String value) {
248 Map<ProviderType, Long> results = new HashMap<>();
249
250 Collection<Identifier> identifiers = IdentifierFactory.defaultIdentifiers();
251
252 for (Identifier identifier : identifiers) {
253 if (identifier.isOfThisType(value)) {
254 ProviderType providerType = identifier.getType();
255 Long counter = results.get(providerType);
256
257 if (counter == null) {
258 counter = 1L;
259 } else {
260 counter += 1L;
261 }
262
263 results.put(providerType, counter);
264 }
265 }
266
267 return results;
268 }

270 public static Map<String, List<IdentifiedType>> organizeToCollection(Map<String, Map<String, Counter>> allTypes) {
271 Map<String, List<IdentifiedType>> results = new HashMap<>();
272 for (Map.Entry<String, Map<String, Counter>> entry : allTypes.entrySet()) {
273 String columnName = entry.getKey();
274
275 Map<String, Counter> counters = entry.getValue();
276
277 List<IdentifiedType> identifiedTypes = new ArrayList<>();
278
279 for (Map.Entry<String, Counter> counterEntry : counters.entrySet()) {
280 identifiedTypes.add(new IdentifiedType(counterEntry.getKey(), counterEntry.getValue().counter));
281 }
282
283 results.put(columnName, identifiedTypes);
284 }
285
286 return results;

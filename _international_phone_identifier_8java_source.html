64 public boolean isOfThisType(String data) {
65 int dataLength = data.length();
66
67 if (dataLength < getMinimumLength() || dataLength > getMaximumLength()) {
68 return false;
69 }
70
71 char firstChar = data.charAt(0);
72 if (firstChar != '+' && firstChar != '0') {
73 return false;
74 }
75
76 if (countSpacesAndDashes(data) == 0) {
77 long numberOfDigits = NumberUtils.countDigits(data);
78 if (numberOfDigits < 10 || numberOfDigits > 15) {
79 return false;
80 }
81
82 if (data.startsWith("+")) {
83 data = data.substring(1);
84 } else if (data.startsWith("00")) {
85 data = data.substring(2);
86 }
87
88 if (containsIllegalCharacters(data)) {

100 }

102 Matcher matcher = pattern.matcher(data);
103
104 boolean matches = matcher.matches();

107 }

111 if (!isValidCountryCode(countryCode)) {
112 return false;
113 }
114
115 String number = matcher.group("number");
116
117 return NumberUtils.countDigits(number) <= 15;
118 }

120 private boolean containsIllegalCharacters(String data) {
121 for (int i = 0; i < data.length(); i++) {
122 char ch = data.charAt(i);
123
124 if (Character.isDigit(ch) || Character.isWhitespace(ch) || ch == '-') {
125 continue;
126 }
127
128 return true;
129 }
130
131 return false;

141 counter++;
142 }
143 }

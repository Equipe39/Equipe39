132 private boolean isChecksumValid(String data) {
133 data = data.toUpperCase().trim();
134
135 int checksum = 0;
136
137 int position = 1;
138
139 for (int i = 0; i < data.length(); ++i) {
140 if (position == 16) break;
141 char character = data.charAt(i);
142
143 if (!Character.isLetterOrDigit(character)) continue;
144
145 Integer value;
146 if (0 == position % 2) {
147 value = evenCharacters.get(character);
148 } else {
149 value = oddCharacters.get(character);
150 }
151
152 if (null == value) {
153 return false; // invalid character which is still letter or digit
154 }
155 checksum += value;
156
157 position += 1;
158 }
159
160 // find last value
161 char checksumCharacter = data.charAt(data.length() - 1);
162
163 return checksumCharacter == (
164 (char) ('A' + (checksum % 26))
165 );
166 }

168 private boolean isDayValid(int day, String month) {
169 if (day > 40) {

171 }

173 return day > 0 && day < 31;

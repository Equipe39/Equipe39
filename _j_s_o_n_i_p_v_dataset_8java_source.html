53 private static List<JsonNode> readObjects(Reader reader) throws IOException {
54 final ObjectMapper mapper = new ObjectMapper();
55 final JsonParser parser = mapper.getFactory().createParser(reader);
56 final MappingIterator<JsonNode> nodes = mapper.readerFor(JsonNode.class).readValues(parser);
57
58 List<JsonNode> dataset = new ArrayList<>();
59
60 while (nodes.hasNext()) {
61 JsonNode node = nodes.next();
62 if (node.isArray()) {
63 node.elements().forEachRemaining(dataset::add);
64 } else {
65 dataset.add(node);
66 }
67 }
68
69 return dataset;
70 }

72 private static List<SimpleSchemaField> buildHeaders(String prefix, List<JsonNode> nodes) {
73 List<String> allFieldNames = nodes.stream().flatMap(
74 node -> StreamSupport.stream(
75 Spliterators.spliteratorUnknownSize(node.fieldNames(), Spliterator.ORDERED),
76 false
77 )
78 ).distinct().collect(Collectors.toList());
79
80 List<SimpleSchemaField> fields = new ArrayList<>(allFieldNames.size());
81
82 for (final JsonNode node : nodes) {
83 if (allFieldNames.isEmpty()) break;
84
85 Iterator<String> fieldNamesIterator = allFieldNames.iterator();
86 while (fieldNamesIterator.hasNext()) {
87 final String fieldName = fieldNamesIterator.next();
88 if (node.has(fieldName)) {
89 final JsonNode field = node.get(fieldName);
90
91 if (field.isValueNode()) {
92 final IPVSchemaFieldType fieldType = getFieldType(field);
93 fields.add(
94 new SimpleSchemaField(
95 prefix + fieldName,
96 fieldType
97 )
98 );
99 } else {
100 fields.addAll(
101 buildHeaders(
102 prefix + fieldName + "/",
103 Collections.singletonList(field)
104 )
105 );
106 }
107
108 fieldNamesIterator.remove();
109 }
110 }
111 }
112
113 return fields;
114 }

116 private static IPVSchemaFieldType getFieldType(JsonNode field) {
117 if (field.isBoolean()) return IPVSchemaFieldType.BOOLEAN;
118 else if (field.isDouble() || field.isFloat()) return IPVSchemaFieldType.FLOAT;
119 else if (field.isTextual()) return IPVSchemaFieldType.STRING;
120 else if (field.isIntegralNumber()) return IPVSchemaFieldType.INT;
121 else if (field.isNull()) return null;
122 else throw new UnsupportedOperationException("Type not supported");
123 }

132 }
133 return "";

106 private List<String> generatePaths(JsonNode node, List<String> parts, String head) {
107 if (node == null) {
108 return Collections.emptyList();
109 }
110
111 if (parts.isEmpty()) {
112 return Collections.singletonList(head);
113 }
114
115 String part = parts.get(0);
116
117 if (!part.equals("*")) {
118 if (node.isObject() || node.isNull()) {
119 return generatePaths(node.get(part), parts.subList(1, parts.size()), head + "/" + part);
120 } else if (node.isArray()) {
121 return generatePaths(node.get(Integer.parseInt(part)), parts.subList(1, parts.size()), head + "/" + part);
122 } else {
123 throw new IllegalArgumentException("Cannot access field " + part + " of a value node");
124 }
125 }
126
127 if (!node.isArray()) throw new UnsupportedOperationException("Unable to iterate over not array");
128
129 List<String> paths = new ArrayList<>();
130
131 for (int i = 0; i < node.size(); ++i) {
132 paths.addAll(generatePaths(node.get(i), parts.subList(1, parts.size()), head + "/" + i));
133 }
134
135 return paths;
136 }

138 private JsonNode createUpdatedValue(JsonNode originalNode, byte[] value) {
139 if (null == value) {
140 return NullNode.getInstance();
141 }
142 if (0 == value.length) {
143 return new TextNode("");
144 }
145 try {
146 switch (originalNode.getNodeType()) {
147 case STRING:
148 case MISSING:
149 return new TextNode(new String(value));
150 case BINARY:
151 return new BinaryNode(value);
152 case BOOLEAN:
153 return BooleanNode.valueOf(myParseBoolean(new String(value)));
154 case NULL:
155 return originalNode;
156 case NUMBER:
157 NumericNode numberValue = (NumericNode) originalNode;
158
159 if (numberValue.isShort()) return new ShortNode(Short.parseShort(new String(value)));
160 if (numberValue.isLong()) return new LongNode(Long.parseLong(new String(value)));
161 if (numberValue.isInt()) return new IntNode(Integer.parseInt(new String(value)));
162 if (numberValue.isFloat()) return new FloatNode(Float.parseFloat(new String(value)));
163 if (numberValue.isDouble()) return new DoubleNode(Double.parseDouble(new String(value)));
164
165 case ARRAY:
166 case OBJECT:
167 case POJO:
168 default:
169 logger.debug("Default behavior applied");
170 return new TextNode(new String(value));
171 }
172 } catch (IllegalArgumentException e) {
173 logger.debug("Incompatible type");
174 return new TextNode(new String(value));
175 }
176 }

194 private Iterable<String> listAllLeaves(JsonNode node, String pathSoFar) {
195 JsonNodeType nodeType = node.getNodeType();
196
197 switch (nodeType) {
198 case ARRAY:
199 return () -> IntStream.range(0, node.size()).boxed().flatMap(position -> StreamSupport.stream(
200 Spliterators.spliteratorUnknownSize(
201 listAllLeaves(
202 node.get(position),
203 pathSoFar + "/*"
204 ).iterator(),
205 Spliterator.IMMUTABLE
206 ), false)).distinct().iterator();
207 case OBJECT:
208 return () -> StreamSupport.stream(
209 Spliterators.spliteratorUnknownSize(
210 node.fieldNames(),
211 Spliterator.IMMUTABLE
212 ), false
213 ).flatMap(fieldName -> StreamSupport.stream(
214 Spliterators.spliteratorUnknownSize(
215 listAllLeaves(node.get(fieldName), pathSoFar + '/' + fieldName).iterator(),
216 Spliterator.IMMUTABLE
217 ), false)).iterator();
218 default:
219 return Collections.singletonList(pathSoFar);
220 }
221 }

223 private Iterable<String> getPaths(JsonNode node, String parent) {
224 JsonNodeType nodeType = node.getNodeType();
225 if (nodeType == JsonNodeType.ARRAY) {
226 return () -> IntStream.range(0, node.size()).boxed().flatMap(position -> StreamSupport.stream(
227 Spliterators.spliteratorUnknownSize(
228 getPaths(node.get(position), parent + "/" + position).iterator(),
229 Spliterator.IMMUTABLE
230 ), false)).iterator();
231 } else if (nodeType == JsonNodeType.OBJECT) {
232 return () -> StreamSupport.stream(
233 Spliterators.spliteratorUnknownSize(
234 node.fieldNames(),
235 Spliterator.IMMUTABLE
236 ), false
237 ).flatMap(fieldName -> StreamSupport.stream(
238 Spliterators.spliteratorUnknownSize(
239 getPaths(node.get(fieldName), parent + '/' + fieldName).iterator(),
240 Spliterator.IMMUTABLE
241 ), false)).iterator();
242 }
243 return Collections.singletonList(parent);
244 }

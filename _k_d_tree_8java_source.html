50 private KDTreeNode<T> createNode(List<T> list, int k, int depth) {
51 if (list == null || list.size() == 0)
52 return null;
53
54 int axis = depth % k;
55 if (axis == X_AXIS) {
56 list.sort(X_COMPARATOR);
57 } else if (axis == Y_AXIS) {
58 list.sort(Y_COMPARATOR);
59 } else {
60 list.sort(Z_COMPARATOR);
61 }
62
63 KDTreeNode<T> node = null;
64 final List<T> less = new ArrayList<>(list.size());
65 final List<T> more = new ArrayList<>(list.size());
66 if (list.size() > 0) {
67 int medianIndex = list.size() / 2;
68 node = new KDTreeNode<>(list.get(medianIndex), k, depth);
69 // Process list to see where each non-median point lies
70 for (int i = 0; i < list.size(); i++) {
71 if (i == medianIndex) continue;
72
73 T p = list.get(i);
74 if (KDTreeNode.compareTo(depth, k, p, node.point) <= 0) {
75 less.add(p);
76 } else {
77 more.add(p);
78 }
79 }
80
81 if ((medianIndex - 1 >= 0) && less.size() > 0) {
82 node.lesser = createNode(less, k, depth + 1);
83 node.lesser.parent = node;
84 }
85
86 if ((medianIndex <= list.size() - 1) && more.size() > 0) {
87 node.greater = createNode(more, k, depth + 1);
88 node.greater.parent = node;
89 }
90 }
91
92 return node;
93 }

103 private KDTreeNode<T> getNode(KDTree<T> tree, T value) {
104 if (tree == null || tree.root == null || value == null)
105 return null;
106
107 KDTreeNode<T> node = tree.root;
108 while (true) {
109 if (node.point.equals(value)) {
110 return node;
111 } else if (KDTreeNode.compareTo(node.depth, node.k, value, node.point) <= 0) {
112 // Lesser
113 if (node.lesser == null) {
114 return null;
115 }
116 node = node.lesser;
117 } else {
118 // Greater
119 if (node.greater == null) {
120 return null;
121 }
122 node = node.greater;
123 }
124 }
125 }

127 Collection<T> nearestNeighbourSearch(int k, T value) {
128 if (value == null || root == null) {
129 return Collections.emptyList();
130 }
131
132 TreeSet<KDTreeNode<T>> results = new TreeSet<>(new ComparatorAgainstFixedPoint<>(value));
133
134 // Find the closest leaf node
135 KDTreeNode<T> prev = null;
136 KDTreeNode<T> node = root;
137 while (node != null) {
138 if (KDTreeNode.compareTo(node.depth, node.k, value, node.point) <= 0) {
139 // Lesser
140 prev = node;
141 node = node.lesser;
142 } else {
143 // Greater
144 prev = node;
145 node = node.greater;
146 }
147 }
148 KDTreeNode<T> leaf = prev;
149
150 final Set<KDTreeNode<T>> examined = new HashSet<>();
151
152 node = leaf;
153 while (node != null) {
154 // Search node
155 searchNode(value, node, k, results, examined);
156 node = node.parent;
157 }
158
159 return results.stream().map(treeNode -> treeNode.point).collect(Collectors.toList());
160 }

162 private void searchNode(T value, KDTreeNode<T> node, int K, TreeSet<KDTreeNode<T>> results, Set<KDTreeNode<T>> examined) {
163 examined.add(node);
164
165 KDTreeNode<T> lastNode = null;
166 double lastDistance = Double.MAX_VALUE;
167 if (results.size() > 0) {
168 lastNode = results.last();
169 lastDistance = lastNode.point.euclideanDistance(value);
170 }
171 double nodeDistance = node.point.euclideanDistance(value);
172 if (nodeDistance< lastDistance) {
173 if (results.size() == K && lastNode != null)
174 results.remove(lastNode);
175 results.add(node);
176 } else if (nodeDistance == lastDistance) {
177 results.add(node);
178 } else if (results.size() < K) {
179 results.add(node);
180 }
181
182 lastNode = results.last();
183 lastDistance = lastNode.point.euclideanDistance(value);
184
185 int axis = node.depth % node.k;
186 KDTreeNode<T> lesser = node.lesser;
187 KDTreeNode<T> greater = node.greater;
188
189 if (lesser != null && !examined.contains(lesser)) {
190 examined.add(lesser);
191
192 double nodePoint;
193 double valuePlusDistance;
194
195 if (axis == X_AXIS) {
196 nodePoint = node.point.x;
197 valuePlusDistance = value.x - lastDistance;
198 } else if (axis == Y_AXIS) {
199 nodePoint = node.point.y;
200 valuePlusDistance = value.y - lastDistance;
201 } else {
202 nodePoint = node.point.z;
203 valuePlusDistance = value.z - lastDistance;
204 }
205
206 if (valuePlusDistance <= nodePoint) {
207 searchNode(value, lesser, K, results, examined);
208 }
209 }
210 if (greater != null && !examined.contains(greater)) {
211 examined.add(greater);
212
213 final double nodePoint;
214 final double valuePlusDistance;
215 if (axis == X_AXIS) {
216 nodePoint = node.point.x;
217 valuePlusDistance = value.x + lastDistance;
218 } else if (axis == Y_AXIS) {
219 nodePoint = node.point.y;
220 valuePlusDistance = value.y + lastDistance;
221 } else {
222 nodePoint = node.point.z;
223 valuePlusDistance = value.z + lastDistance;
224 }
225
226 if (valuePlusDistance >= nodePoint)
227 searchNode(value, greater, K, results, examined);
228 }

296 public static class CartesianPoint implements Comparable<CartesianPoint> {
297 final double x;
298 final double y;
299 final double z;
300
301 CartesianPoint(double x, double y) {
302 this.x = x;
303 this.y = y;
304 this.z = 0;
305 }
306
307 public CartesianPoint(Double latitude, Double longitude) {
308 x = cos(Math.toRadians(latitude)) * cos(Math.toRadians(longitude));
309 y = cos(Math.toRadians(latitude)) * sin(Math.toRadians(longitude));
310 z = sin(Math.toRadians(latitude));
311 }
312
313 public double euclideanDistance(CartesianPoint other) {
314 return Math.sqrt(Math.pow((other.x - this.x), 2) + Math.pow((other.y - this.y), 2) + Math.pow((other.z - this.z), 2));
315 }
316
317
318 @Override
319 public int hashCode() {
320 int result;
321 long temp;
322 temp = Double.doubleToLongBits(x);
323 result = (int) (temp ^ (temp >>> 32));
324 temp = Double.doubleToLongBits(y);
325 result = 31 * result + (int) (temp ^ (temp >>> 32));
326 temp = Double.doubleToLongBits(z);
327 result = 31 * result + (int) (temp ^ (temp >>> 32));
328 return result;
329 }
330
331 @Override
332 public boolean equals(Object o) {
333 if (this == o) return true;
334 if (!(o instanceof CartesianPoint)) return false;
335
336 CartesianPoint that = (CartesianPoint) o;
337
338 return Double.compare(that.x, x) == 0 && Double.compare(that.y, y) == 0 && Double.compare(that.z, z) == 0;
339 }
340
341 @Override
342 public int compareTo(CartesianPoint o) {
343 int xComp = X_COMPARATOR.compare(this, o);
344 if (xComp != 0)
345 return xComp;
346 int yComp = Y_COMPARATOR.compare(this, o);
347 if (yComp != 0)
348 return yComp;
349 return Z_COMPARATOR.compare(this, o);
350 }

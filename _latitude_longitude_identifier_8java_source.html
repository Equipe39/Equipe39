35public class LatitudeLongitudeIdentifier extends AbstractRegexBasedIdentifier {
36 private final static String[] appropriateNames = {"Latitude", "Longitude", "LatitudeLongitude"};
37 private final static Pattern latlonPattern = Pattern.compile("^-?([1-8]?[0-9]\\.{1}\\d{1,8}|90\\.{1}0{1,6}),\\s*-?((([1]?[0-7][0-9]|[1-9]?[0-9])\\.{1}\\d{1,8}$)|[1]?[1-8][0]\\.{1}0{1,6})$");
38 private final static Pattern compassPattern = Pattern.compile(
39 "^(?<ns>[NS])(((?<nsDegrees>[0-8][0-9])\\.(?<nsMinutes>[0-5]\\d)\\.(?<nsSeconds>[0-5]\\d))|((?<nsDegrees2>90)(?<nsMinutes2>\\.00)(?<nsSeconds2>\\.00)))[ |,](?<ew>[EW])(((?<ewDegrees>(0\\d\\d|1[0-7]\\d))\\.(?<ewMinutes>[0-5]\\d)\\.(?<ewSeconds>[0-5]\\d))|((?<ewDegrees2>180)\\.(?<ewMinutes2>00)\\.(?<ewSeconds2>00)))$");
40 private final static Pattern dmsCoordinatePattern = Pattern.compile("(?<nsDegrees>[0-9]{1,2})[:|°](?<nsMinutes>[0-9]{1,2})[:|'](?<nsSeconds>(?:\\b[0-9]+(?:\\.[0-9]*)?|\\.[0-9]+\\b))\"?(?<ns>[N|S])[ |,](?<ewDegrees>[0-9]{1,2})[:|°](?<ewMinutes>[0-9]{1,2})[:|'](?<ewSeconds>(?:\\b[0-9]+(?:\\.[0-9]*)?|\\.[0-9]+\\b))\"?(?<ew>[E|W])",
41 Pattern.UNICODE_CASE);

74 public Pattern getPatternFromFormat(LatitudeLongitudeFormat format) {

138 public static LatitudeLongitude parseGPSFormat(String identifier) throws NumberFormatException {
139 //GPS format is expected as "x,y"
140 String[] parts = identifier.split(",\\s*");
141 double latitude = Double.parseDouble(parts[0]);
142 double longitude = Double.parseDouble(parts[1]);
143 return new LatitudeLongitude(latitude, longitude, LatitudeLongitudeFormat.DECIMAL);
144 }

152 public LatitudeLongitude parseCompassFormat(String identifier) throws NumberFormatException {
153 double latitude;
154 double longitude;
155 double nsDegrees, ewDegrees;
156 double nsMinutes, ewMinutes;
157 double nsSeconds, ewSeconds;
158 String ns;
159 String ew;
160 LatitudeLongitudeFormat format = LatitudeLongitudeFormat.COMPASS;
161
162 Matcher m1 = getPatternFromFormat(LatitudeLongitudeFormat.DMS).matcher(identifier);
163 if (m1.matches()) {
164 nsDegrees = Double.parseDouble(m1.group("nsDegrees"));
165 nsMinutes = Double.parseDouble(m1.group("nsMinutes"));
166 nsSeconds = Double.parseDouble(m1.group("nsSeconds"));
167
168 ewDegrees = Double.parseDouble(m1.group("ewDegrees"));
169 ewMinutes = Double.parseDouble(m1.group("ewMinutes"));
170 ewSeconds = Double.parseDouble(m1.group("ewSeconds"));
171
172 ns = m1.group("ns");
173 ew = m1.group("ew");
174
175 format = LatitudeLongitudeFormat.DMS;
176 } else {
177 Matcher m2 = getPatternFromFormat(LatitudeLongitudeFormat.COMPASS).matcher(identifier);
178 if (!m2.matches()) {
179 return null;
180 }
181
182 try {
183 nsDegrees = Double.parseDouble(m2.group("nsDegrees"));
184 nsMinutes = Double.parseDouble(m2.group("nsMinutes"));
185 nsSeconds = Double.parseDouble(m2.group("nsSeconds"));
186 } catch (Exception e) {
187 nsDegrees = Double.parseDouble(m2.group("nsDegrees2"));
188 nsMinutes = Double.parseDouble(m2.group("nsMinutes2"));
189 nsSeconds = Double.parseDouble(m2.group("nsSeconds2"));
190 }
191
192 try {
193 ewDegrees = Double.parseDouble(m2.group("ewDegrees"));
194 ewMinutes = Double.parseDouble(m2.group("ewMinutes"));
195 ewSeconds = Double.parseDouble(m2.group("ewSeconds"));
196 } catch (Exception e) {
197 ewDegrees = Double.parseDouble(m2.group("ewDegrees2"));
198 ewMinutes = Double.parseDouble(m2.group("ewMinutes2"));
199 ewSeconds = Double.parseDouble(m2.group("ewSeconds2"));
200 }
201
202 ns = m2.group("ns");
203 ew = m2.group("ew");
204 }
205
206 latitude = GeoUtils.degreesToDecimal(nsDegrees, nsMinutes, nsSeconds);
207 longitude = GeoUtils.degreesToDecimal(ewDegrees, ewMinutes, ewSeconds);
208
209 if (ns.equals("S")) {
210 latitude = -latitude;
211 }
212 if (ew.equals("W")) {
213 longitude = -longitude;
214 }
215
216 return new LatitudeLongitude(latitude, longitude, format);
217 }

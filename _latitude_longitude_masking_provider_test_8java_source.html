34 @Test
35 public void testMask() throws Exception {
36 MaskingProvider latlonMaskingProvider = new LatitudeLongitudeMaskingProvider();
37 LatitudeLongitudeIdentifier latitudeLongitudeIdentifier = new LatitudeLongitudeIdentifier();
38
39 String gpsCoords = "10.0000,12.0000";
40 String maskedResult = latlonMaskingProvider.mask(gpsCoords);
41 assertNotEquals(maskedResult, gpsCoords);
42 assertTrue(latitudeLongitudeIdentifier.isOfThisType(maskedResult), maskedResult);
43 assertTrue(latitudeLongitudeIdentifier.isGPSFormat(maskedResult), maskedResult);
44
45 String compassCoords = "N90.00.00 E180.00.00";
46 maskedResult = latlonMaskingProvider.mask(compassCoords);
47 assertNotEquals(maskedResult, compassCoords);
48 assertTrue(latitudeLongitudeIdentifier.isOfThisType(maskedResult));
49 assertTrue(latitudeLongitudeIdentifier.isCompassFormat(maskedResult));
50
51 String dmsCoords = "12:30'23.256547S 12:30'23.256547E";
52 assertTrue(latitudeLongitudeIdentifier.isOfThisType(dmsCoords));
53 maskedResult = latlonMaskingProvider.mask(dmsCoords);
54 assertNotEquals(maskedResult, dmsCoords);
55 assertTrue(latitudeLongitudeIdentifier.isOfThisType(maskedResult));
56 assertTrue(latitudeLongitudeIdentifier.isDMSFormat(maskedResult));
57 }

56 private boolean increment() {
57 int idx = 0;
58
59 do {
60 int m = max(idx);
61 if (m > counters.get(idx)) {
62 counters.set(idx, counters.get(idx) + 1);
63 reset(idx);
64
65 return false;
66 } else {
67 ++idx;
68 }
69 } while (counters.size() > idx);
70
71 // larger than expected -> resize
72 counters.add(init(counters.size() - 1));
73 reset(counters.size() - 1);
74
75 return true;
76 }

78 private void reset(int idx) {

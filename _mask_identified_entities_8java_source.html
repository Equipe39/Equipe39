36public class MaskIdentifiedEntities {
37 private final MaskingProviderFactory factory;
38 private final ConfigurationManager configurationManager;

41
42 public MaskIdentifiedEntities(final ConfigurationManager configurationManager, final DataMaskingOptions dataMaskingOptions,
43 final MaskingProviderFactory factory) {
44 this.factory = factory;
45 this.configurationManager = configurationManager;
46 this.dataMaskingOptions = dataMaskingOptions;
47 this.cachedProviders = new HashMap<>();
48 }
49
50 public List<IdentifiedEntity> maskEntities(final List<IdentifiedEntity> entities) {
51 Map<String, DataMaskingTarget> toBeMasked = dataMaskingOptions.getToBeMasked();
52
53 return entities.parallelStream().map( entity -> {
54 final String type = entity.getType().iterator().next().getSubtype();
55
56 String maskedValue = (toBeMasked.containsKey(type)) ?
57 mask(entity.getText(), configurationManager.getFieldConfiguration(type),
58 toBeMasked.get(type).getProviderType(), type) : entity.getText();
59
60 return new IdentifiedEntity(maskedValue, entity.getStart(), entity.getEnd(), entity.getType(), entity.getPos());
61 }).collect(Collectors.toList());
62 }
63
64 private String mask(final String text, final MaskingConfiguration configuration, ProviderType providerType, String fieldName) {
65 return getMaskingProvider(providerType, configuration).mask(text, fieldName);
66 }
67
68 private MaskingProvider getMaskingProvider(ProviderType type, MaskingConfiguration configuration) {
69 String typeName = type.getName();
70
71 if (!cachedProviders.containsKey(typeName)) {
72 MaskingProvider maskingProvider = factory.get(type, configuration);
73 cachedProviders.put(typeName, maskingProvider);
74 }
75
76 return cachedProviders.get(typeName);
77 }

84 public void add(List<String> hierarchy) {
85 String leaf = hierarchy.get(0);
86 this.leaves.put(leaf.toLowerCase(), hierarchy);
87 this.terms.add(hierarchy);
88 this.indices.put(leaf.toLowerCase(), this.termsAdded);
89
90 if (this.termsAdded == 0) {
91 this.topTerm = hierarchy.get(hierarchy.size() - 1);
92 nodes.put(this.topTerm.toUpperCase(), new GeneralizationNode(this.topTerm, null, false, hierarchy.size()));
93 }
94
95 this.hierarchyHeight = Math.max(this.hierarchyHeight, hierarchy.size());
96
97 addNode(hierarchy);
98 this.termsAdded++;
99 }

112 private void addNode(List<String> listOriginal) {
113 List<String> list = new ArrayList<>(listOriginal);
114
115
116 /*
117 our input is a list of terms like :
118 Married, Coupled, *
119 Widowed, Alone, *
120 */
121 Collections.reverse(list);
122
123 for (int i = 0; i < list.size(); i++) {
124 String s = list.get(i).toUpperCase();
125 boolean isLeaf = i == list.size() - 1;
126
127 if (!nodes.containsKey(s)) {
128 String previous = list.get(i - 1).toUpperCase();
129 nodes.put(s, new GeneralizationNode(s, nodes.get(previous), isLeaf, list.size()));
130 }
131 }
132 }

170 @Override
171 public String encode(String value, int level, boolean randomizeOnFail) {
172 if (level == this.hierarchyHeight) {
173 return this.topTerm;
174 }
175
176 GeneralizationNode node = this.nodes.get(value.toUpperCase());
177
178 if (node == null) {
179 if (randomizeOnFail) {
180 return RandomGenerators.randomUIDGenerator(12);
181 }
182 return getTopTerm();
183 }
184
185 if (level == 0) {
186 return value.toUpperCase();
187 }
188
189 level--;
190
191 List<GeneralizationNode> hierarchy = node.getParents();
192
193 if (level >= hierarchy.size()) {
194 level = hierarchy.size() - 1;
195 }
196
197 return hierarchy.get(level).getValue().toUpperCase();
198 }

127 private List<PrivacyMetric> createMapToPair(String[] s, List<ColumnInformation> columnInformationList,
128 List<PrivacyMetric> privacyMetrics, List<Integer> quasiColumns,
129 List<Integer> sensitiveColumns, int privacyConstraintsContentRequirements) throws IOException {
130 List<String> sensitiveValues;
131
132 if (privacyConstraintsContentRequirements == ContentRequirements.NONE) {
133 sensitiveValues = Collections.emptyList();
134 }
135 else {
136 sensitiveValues = new ArrayList<>();
137 for (Integer sensitiveColumn : sensitiveColumns) {
138 sensitiveValues.add(s[sensitiveColumn]);
139 }
140 }
141
142 List<PrivacyMetric> metrics = new ArrayList<>();
143 for(PrivacyMetric metric: privacyMetrics) {
144 metrics.add(metric.getInstance(sensitiveValues));
145 }
146
147 return metrics;
148 }

150 private boolean checkPartitionForConstraints(JavaRDD<String[]> input) {
151 List<PrivacyMetric> metrics = input.map(s -> createMapToPair(s, columnInformationList, privacyMetrics, quasiColumns, sensitiveColumns, privacyConstraintsContentRequirements)).reduce((p1, p2) -> {
152 for(int i = 0; i < p1.size(); i++) {
153 PrivacyMetric m1 = p1.get(i);
154 PrivacyMetric m2 = p2.get(i);
155 m1.update(m2);
156 }
157
158 return p1;
159 });
160
161 for(int i = 0; i < privacyConstraints.size(); i++) {
162 PrivacyMetric metric = metrics.get(i);
163 if (!privacyConstraints.get(i).check(metric)) {
164 return false;
165 }
166 }
167
168 return true;
169 }

199 public List<MondrianSparkPartition> splitNumerical(int quasiIndex, int level) {
200 int columnIndex = this.quasiColumns.get(quasiIndex);
201
202 Interval partitionWidthInfo = width.get(columnIndex);
203
204 double median = partitionWidthInfo.getMedian();
205
206 middle.set(columnIndex, generateMiddleKey(partitionWidthInfo.getLow(), partitionWidthInfo.getHigh()));
207 width.set(columnIndex, new Interval(partitionWidthInfo.getLow(), partitionWidthInfo.getHigh(), median));
208
209 Tuple2<Tuple4<JavaRDD<String[]>, Double, Double, Double>, Tuple4<JavaRDD<String[]>, Double, Double, Double>> subPartitions =
210 MondrianSparkUtils.splitNumericalByMedian(this.member, columnIndex, median);
211
212 if (subPartitions._1._1().isEmpty() || subPartitions._2._1().isEmpty()) {
213 return Collections.emptyList();
214 }
215
216 if (!checkPartitionForConstraints(subPartitions._1._1()) || !checkPartitionForConstraints(subPartitions._2._1())) {
217 return Collections.emptyList();
218 }
219
220 subPartitions._1._1().cache();
221 subPartitions._2._1().cache();
222 this.member.unpersist();
223
224 double leftMin = subPartitions._1._2();
225 double leftMax = subPartitions._1._3();
226 double leftMedian = subPartitions._1._4();
227
228 double rightMin = subPartitions._2._2();
229 double rightMax = subPartitions._2._3();
230 double rightMedian = subPartitions._2._4();
231
232 List<String> leftMiddle = new ArrayList<>(middle);
233 leftMiddle.set(columnIndex, generateMiddleKey(leftMin, leftMax));
234
235 List<String> rightMiddle = new ArrayList<>(middle);
236 rightMiddle.set(columnIndex, generateMiddleKey(rightMin, rightMax));
237
238 List<Interval> leftWidth = copyList(width);
239 leftWidth.set(columnIndex, new Interval(leftMin, leftMax, leftMedian));
240
241 List<Interval> rightWidth = copyList(width);
242 rightWidth.set(columnIndex, new Interval(rightMin, rightMax, rightMedian));
243
244 MondrianSparkPartition lhs = new MondrianSparkPartition(subPartitions._1._1(), leftMiddle, leftWidth, columnInformationList, privacyConstraints, categoricalSplitStrategy);
245 MondrianSparkPartition rhs = new MondrianSparkPartition(subPartitions._2._1(), rightMiddle, rightWidth, columnInformationList, privacyConstraints, categoricalSplitStrategy);
246
247 return Arrays.asList(lhs, rhs);
248 }

275 private List<MondrianSparkPartition> splitCategoricalOrderBased(int quasiIndex, int level) {
276 int columnIndex = this.quasiColumns.get(quasiIndex);
277 MaterializedHierarchy materializedHierarchy = (MaterializedHierarchy)
278 ((CategoricalInformation)this.columnInformationList.get(columnIndex)).getHierarchy();
279
280 JavaRDD<Double> categoricalIndices = MondrianSparkUtils.extractCategoricalIndices(this.member, columnIndex, materializedHierarchy);
281 categoricalIndices = categoricalIndices.sortBy(new Function<Double, Double>() {
282 @Override
283 public Double call(Double aDouble) throws Exception {
284 return aDouble;
285 }
286 }, true, this.member.getNumPartitions());
287
288 double median = MondrianSparkUtils.findMedian(categoricalIndices)._3();
289
290 Tuple2<Tuple4<JavaRDD<String[]>, Double, Double, Double>, Tuple4<JavaRDD<String[]>, Double, Double, Double>> subPartitions =
291 MondrianSparkUtils.splitCategoricalByOrder(this.member, columnIndex, median, materializedHierarchy);
292
293 if (subPartitions._1._1().isEmpty() || subPartitions._2._1().isEmpty()) {
294 return Collections.emptyList();
295 }
296
297 if (!checkPartitionForConstraints(subPartitions._1._1()) || !checkPartitionForConstraints(subPartitions._2._1())) {
298 return Collections.emptyList();
299 }
300
301 double leftMin = subPartitions._1._2();
302 double leftMax = subPartitions._1._3();
303
304 double rightMin = subPartitions._2._2();
305 double rightMax = subPartitions._2._3();
306
307
308 List<String> leftMiddle = new ArrayList<>(middle);
309 leftMiddle.set(columnIndex, null);
310
311 List<String> rightMiddle = new ArrayList<>(middle);
312 rightMiddle.set(columnIndex, null);
313
314 List<Interval> leftWidth = copyList(width);
315 leftWidth.set(columnIndex, new Interval(leftMin, leftMax));
316
317 List<Interval> rightWidth = copyList(width);
318 rightWidth.set(columnIndex, new Interval(rightMin, rightMax));
319
320 MondrianSparkPartition lhs = new MondrianSparkPartition(subPartitions._1._1(), leftMiddle, leftWidth, columnInformationList, privacyConstraints, categoricalSplitStrategy);
321 MondrianSparkPartition rhs = new MondrianSparkPartition(subPartitions._2._1(), rightMiddle, rightWidth, columnInformationList, privacyConstraints, categoricalSplitStrategy);

323 return Arrays.asList(lhs, rhs);

341 public double getNormalizedWidth(int columnIndex) {
342 ColumnInformation columnInformation = columnInformationList.get(columnIndex);
343 double dividend;
344
345 if (columnInformation.isCategorical()) {
346 CategoricalInformation categoricalInformation = (CategoricalInformation) columnInformation;
347 String topTerm = categoricalInformation.getHierarchy().getTopTerm();
348 dividend = ((MaterializedHierarchy)categoricalInformation.getHierarchy()).getNode(topTerm).length();
349 }
350 else {
351 NumericalRange numericalInformation = (NumericalRange) columnInformation;
352 dividend = numericalInformation.getRange();
353 }
354
355 return width.get(columnIndex).getRange() / dividend;
356 }

363 public int chooseDimension() {
364 int maxWidth = -1;
365 int maxDim = -1;
366
367 for(int i = 0; i < qiLen; i++) {
368 if(allow[i] == 0) {
369 continue;
370 }
371
372 int normalizedWidth = (int)getNormalizedWidth(this.quasiColumns.get(i));
373 if (normalizedWidth > maxWidth) {
374 maxWidth = normalizedWidth;
375 maxDim = i;
376 }
377 }
378
379 if (maxWidth > 1) {
380 throw new RuntimeException("maxWidth > 1");
381 }
382
383 if (maxDim == -1) {
384 throw new RuntimeException("maxDim == -1");
385 }
386
387 return maxDim;
388 }

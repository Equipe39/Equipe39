50 @Test
51 public void testMask() {
52 NameMaskingProvider nameMaskingProvider = new NameMaskingProvider(factory);
53
54 int count1 = 0;
55 int count2 = 0;
56
57 for(int i = 0; i < 100; i++) {
58 String name = "John";
59 String res = nameMaskingProvider.mask(name);
60 assertThat(names.getGender(name) == names.getGender(res) || Gender.both == names.getGender(res), is(true));
61
62 //capitalization shouldn't matter
63 name = "JoHn";
64 res = nameMaskingProvider.mask(name);
65
66 if (!name.equals(res)) {
67 count1++;
68 }
69
70 assertThat(names.getGender(name) == names.getGender(res) || Gender.both == names.getGender(res), is(true));
71
72 //female detection
73 name = "Mary";
74 res = nameMaskingProvider.mask(name);
75 if(!name.equals(res)) {
76 count2++;
77 }
78
79 assertThat(names.getGender(name) == names.getGender(res) || Gender.both == names.getGender(res), is(true));
80 }
81
82 assertTrue(count1 > 0);
83 assertTrue(count2 > 0);
84
85 //surname detection
86 String name = "Smith";
87 int surnameCount = 0;
88 for(int i = 0; i < 100; i++) {
89 String res = nameMaskingProvider.mask(name);
90 assertTrue(names.isLastName(res));
91 if (!name.equals(res)) {
92 surnameCount++;
93 }
94 }
95 assertTrue(surnameCount > 0);
96
97 }

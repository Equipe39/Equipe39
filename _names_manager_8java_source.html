222 public String getRandomFirstName(Gender gender, boolean allowAnyGender, String countryCode) {
223
224 String name = null;
225
226 int currentTry = 0;
227 for (; currentTry < 50; ++currentTry) {
228 name = getRandomFirstName(gender, countryCode);
229 Gender newGender = getGender(name);
230
231 if (allowAnyGender ||
232 newGender.equals(gender) ||
233 newGender.equals(Gender.both)) {
234 break;
235 }
236 }
237
238 log.debug("First name identified after {}", currentTry);
239 return name;
240 }

248 public Gender getGender(String candidate) {

250 boolean isFemale = femaleNameManager.isValidKey(candidate);
251

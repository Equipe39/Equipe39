91 public static double entropyForOutlier(Partition originalPartition, Map<String, Double> originalFrequencies,
92 int columnIndex) {
93
94 IPVDataset members = originalPartition.getMember();
95 int numberOfRows = members.getNumberOfRows();
96
97 double result = 0.0;
98
99 for (int i = 0; i < numberOfRows; i++) {
100 String originalValue = members.get(i, columnIndex);
101 double proportion = originalFrequencies.get(originalValue);
102
103 result += -log2(proportion);
104 }
105
106 return result;
107 }

109 public static double entropyForAnonymous(Partition originalPartition, Map<String, Double> originalFrequencies,
110 Partition anonymizedPartition, Map<String, Double> anonymizedFrequencies,
111 int columnIndex) {
112
113 IPVDataset originalMembers = originalPartition.getMember();
114 IPVDataset anonymizedMembers = anonymizedPartition.getMember();
115
116 int numberOfRows = originalMembers.getNumberOfRows();
117
118 double result = 0.0;
119
120 for (int i = 0; i < numberOfRows; i++) {
121 String originalValue = originalMembers.get(i, columnIndex);
122 String anonymizedValue = anonymizedMembers.get(i, columnIndex);
123
124 double freqOrig = originalFrequencies.get(originalValue);
125 double freqAnon = anonymizedFrequencies.get(anonymizedValue);
126
127 result += -log2(freqOrig / freqAnon);
128 }
129
130
131 return result;
132 }

134 private InformationLossResult reportForColumn(int columnIndex) {
135 ColumnInformation columnInformation = columnInformationList.get(columnIndex);
136 double weight = columnInformation.getWeight();
137
138 if (columnInformation.getColumnType() != ColumnType.QUASI) {
139 return null;
140 }
141
142 double result = 0.0;
143
144 int numPartitions = originalPartitions.size();
145
146 Map<String, Double> originalFrequencies = this.proportionsOriginal.get(columnIndex);
147 Map<String, Double> anonymizedFrequencies = this.proprtionsAnonymized.get(columnIndex);
148
149 for (int i = 0; i < numPartitions; i++) {
150 Partition originalPartition = originalPartitions.get(i);
151
152 if (originalPartition.isAnonymous()) {
153 Partition anonymizedPartition = anonymizedPartitions.get(i);
154 result += entropyForAnonymous(originalPartition, originalFrequencies,
155 anonymizedPartition, anonymizedFrequencies, columnIndex);
156 } else {
157 result += entropyForOutlier(originalPartition, originalFrequencies, columnIndex);
158 }
159 }
160
161 double maxNUE = calculateMaxEntropy(originalFrequencies, columnIndex);
162
163 return new InformationLossResult(weight * result, 0.0, maxNUE);
164 }

166 private double calculateMaxEntropyForEntireDataset() {
167
168 double sum = 0.0;
169 for (int columnIndex = 0; columnIndex < this.columnInformationList.size(); columnIndex++) {
170 ColumnInformation columnInformation = columnInformationList.get(columnIndex);
171
172 if (columnInformation.getColumnType() != ColumnType.QUASI) {
173 continue;
174 }
175
176 Map<String, Double> originalFrequencies = this.proportionsOriginal.get(columnIndex);
177 double columnMax = calculateMaxEntropy(originalFrequencies, columnIndex);
178
179 sum += columnMax;
180 }
181
182 return sum;
183 }

185 private double calculateMaxEntropy(Map<String, Double> originalFrequencies, int columnIndex) {
186
187 double sum = 0.0;
188 int numPartitions = originalPartitions.size();
189
190 for (int i = 0; i < numPartitions; i++) {
191 Partition originalPartition = originalPartitions.get(i);
192 sum += entropyForOutlier(originalPartition, originalFrequencies, columnIndex);
193 }
194
195 return sum;
196 }

198 @Override
199 public double report() {
200 int numberOfColumns = anonymized.getNumberOfColumns();
201
202 double result = 0.0;
203
204 for (int j = 0; j < numberOfColumns; j++) {
205 if (columnInformationList.get(j).getColumnType() != ColumnType.QUASI) {
206 continue;
207 }
208
209 InformationLossResult columnResult = reportForColumn(j);
210 result += columnResult.getValue();
211 }
212
213 return result;

216 @Override
217 public List<InformationLossResult> reportPerQuasiColumn() {
218 List<InformationLossResult> results = new ArrayList<>();

220

222 if (columnInformation.getColumnType() != ColumnType.QUASI) {
223 columnIndex++;

228
229 results.add(iloss);
230
231 columnIndex++;
232 }

241 private Map<String, Double> calculateFrequenciesForColumn(IPVDataset dataset, int columnIndex) {
242 int numberOfRows = dataset.getNumberOfRows();
243
244 Map<String, Double> map = new HashMap<>();
245
246 for (int i = 0; i < numberOfRows; i++) {
247
248 String value = createKey(dataset.get(i, columnIndex));
249
250 Double counter = map.get(value);
251 if (counter == null) {
252 map.put(value, 1.0);
253 } else {
254 map.put(value, counter + 1.0);
255 }
256 }
257
258 for (Map.Entry<String, Double> entry : map.entrySet()) {
259 String key = entry.getKey();
260 Double value = entry.getValue();
261
262 map.put(key, value / (double) numberOfRows);
263 }
264 return map;
265 }
266
267 private List<Map<String, Double>> calculateFrequencies(IPVDataset dataset) {
268
269 List<Map<String, Double>> frequencies = new ArrayList<>();
270 int numberOfColumns = dataset.getNumberOfColumns();
271
272 for (int i = 0; i < numberOfColumns; i++) {
273 if (columnInformationList.get(i).getColumnType() != ColumnType.QUASI) {
274 frequencies.add(null);
275 } else {
276 frequencies.add(calculateFrequenciesForColumn(dataset, i));
277 }
278 }
279
280 return frequencies;
281 }

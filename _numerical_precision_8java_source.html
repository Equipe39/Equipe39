74 @Override
75 public double report() {
76 List<InformationLossResult> columnResults = reportPerQuasiColumn();
77
78 double sum = 0.0;
79 for (InformationLossResult lossResult : columnResults) {
80 sum += lossResult.getValue();
81 }
82
83 return sum / (double) columnResults.size();
84 }

86 private InformationLossResult reportForColumn(int columnIndex) {
87 ColumnInformation columnInformation = columnInformationList.get(columnIndex);
88
89 if (columnInformation.getColumnType() != ColumnType.QUASI) {
90 return null;
91 }
92
93 if (columnInformation.isCategorical()) {
94 return null;
95 }
96
97 int numberOfRows = anonymized.getNumberOfRows();
98 double globalRange = getGlobalRange(original, columnIndex);
99 double precision = 0.0;
100
101 for (int i = 0; i < numberOfRows; i++) {
102 String interval = anonymized.get(i, columnIndex);
103 double range = intervalToRange(interval);
104
105 precision += range;
106 }
107
108 double iloss = precision / numberOfRows / globalRange;
109 return new InformationLossResult(iloss, 0.0d, 1.0d);
110 }
111
112 private double intervalToRange(String interval) {
113 String[] extremes = interval.split("-");
114
115 if (extremes.length == 0) {
116 throw new IllegalArgumentException("Interval is empty");
117 }
118
119 if (extremes.length == 1) {
120 return 0.0d;
121 }
122
123 return Double.parseDouble(extremes[1]) - Double.parseDouble(extremes[0]);
124 }
125
126 private double getGlobalRange(IPVDataset original, int columnIndex) {
127 double globalMin = Double.POSITIVE_INFINITY;
128 double globalMax = Double.NEGATIVE_INFINITY;
129
130 for (List<String> row : original) {
131 double value = Double.parseDouble(row.get(columnIndex));
132
133 globalMax = Math.max(globalMax, value);
134 globalMin = Math.min(globalMin, value);
135 }
136
137 return globalMax - globalMin;
138 }

140 @Override
141 public List<InformationLossResult> reportPerQuasiColumn() {
142 List<InformationLossResult> results = new ArrayList<>();
143 int columnIndex = 0;
144
145 for (ColumnInformation columnInformation : columnInformationList) {
146 if (columnInformation.getColumnType() != ColumnType.QUASI) {
147 columnIndex++;
148 continue;
149 }
150
151 InformationLossResult iloss = reportForColumn(columnIndex);
152
153 results.add(iloss);
154 columnIndex++;
155 }
156
157 return results;
158 }

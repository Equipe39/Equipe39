55 private List<ColumnInformation> buildColumnInformationList(IPVDataset dataset, Collection<IPVVulnerability> vulnerabilities,
56 Collection<String> sensitiveFields, Map<String, ProviderType> fieldTypes) {
57 List<ColumnInformation> columnInformationList = new ArrayList<>(dataset.getNumberOfColumns());
58
59 IPVSchema schema = dataset.getSchema();
60 List<? extends IPVSchemaField> fields = schema.getFields();
61
62 IPVVulnerability vulnerability = AnonymizationUtils.mergeVulnerabilities(vulnerabilities);
63
64 for (int i = 0; i < dataset.getNumberOfColumns(); i++) {
65 String fieldName = fields.get(i).getName();
66
67 if (sensitiveFields.contains(fieldName)) {
68 columnInformationList.add(new SensitiveColumnInformation());
69 continue;
70 }
71
72 boolean isQuasi = vulnerability.contains(i);
73 if (!isQuasi) {
74 columnInformationList.add(new DefaultColumnInformation());
75 continue;
76 }
77
78 ProviderType fieldType = fieldTypes.get(fieldName);
79 if (fieldType == null) {
80 columnInformationList.add(ColumnInformationGenerator.generateCategoricalFromData(dataset, i, ColumnType.QUASI));
81 continue;
82 }
83
84 if (fieldType.getTypeClass() == TypeClass.NUMERICAL) {
85 GeneralizationHierarchy hierarchy = IntervalGenerator.generateHierarchy(dataset, i);
86 columnInformationList.add(new CategoricalInformation(hierarchy, ColumnType.QUASI));
87 } else {
88 GeneralizationHierarchy hierarchy = GeneralizationHierarchyFactory.getDefaultHierarchy(fieldType);
89 if (hierarchy == null) {
90 columnInformationList.add(ColumnInformationGenerator.generateCategoricalFromData(dataset, i, ColumnType.QUASI));
91 } else {
92 columnInformationList.add(new CategoricalInformation(hierarchy, ColumnType.QUASI));
93 }
94 }
95 }
96
97 return columnInformationList;
98 }

134 private boolean checkConstraints(Partition partition) {
135 for (PrivacyConstraint privacyConstraint : privacyConstraints) {
136 if (!privacyConstraint.check(partition, sensitiveColumns)) {
137 return false;
138 }
139 }
140
141 return true;
142 }

151 // bestLevel is a collection with the generalization levels for the quasi-identifiers
152 private IPVDataset createAnonymizedDataset(IPVDataset anonymized) {
153
154 IPVDataset finalDataset = new IPVDataset(new ArrayList<>(), anonymized.getSchema(), anonymized.hasColumnNames());
155
156 int numPartitions = this.originalPartitions.size();
157
158 for (int n = 0; n < numPartitions; n++) {
159 Partition p = this.originalPartitions.get(n);
160 Partition anonP = this.anonymizedPartitions.get(n);
161
162 if (!checkConstraints(p)) {
163 p.setAnonymous(false);
164 anonP.setAnonymous(false);
165 continue;
166 }
167
168 IPVDataset member = anonP.getMember();
169 int memberRows = member.getNumberOfRows();
170
171 for (int i = 0; i < memberRows; i++) {
172 finalDataset.addRow(member.getRow(i));
173 }
174
175 p.setAnonymous(true);
176 anonP.setAnonymous(true);
177 }
178
179 return finalDataset;
180 }

220 public static LatticeNode selectLowestLossOnLevel(List<LatticeNode> nodes) {
221
222 double lowestLoss = Double.MAX_VALUE;
223 LatticeNode matchNode = null;
224
225 for (LatticeNode n : nodes) {
226 Double informationLoss = n.getInformationLoss();
227 if (informationLoss == null) { //these came from predictive tagging
228 continue;
229 }
230
231 if (informationLoss < lowestLoss) {
232 lowestLoss = informationLoss;
233 matchNode = n;
234 }
235 }
236
237 return matchNode;
238 }

240 @Override
241 public IPVDataset apply() {
242 if (null == original
243 || null == columnInformationList
244 || null == privacyConstraints) throw new RuntimeException("Not initialized yet");
245
246 if (columnInformationList.stream().noneMatch(columnInformation -> columnInformation.getColumnType().equals(ColumnType.QUASI)))
247 return original;
248
249 AnonymityChecker anonymityChecker = new SimpleAnonymityChecker(original, this.columnInformationList, this.privacyConstraints);
250 lattice = new Lattice(anonymityChecker, this.columnInformationList, this.suppressionRate);
251 lattice.explore();
252
253
254 List<LatticeNode> kMinimalNodes = lattice.getKMinimal();
255
256 if (!kMinimalNodes.isEmpty()) {
257 bestNode = selectLowestLossOnLevel(kMinimalNodes);
258 } else {
259 throw new RuntimeException("unable to find a suitable generalization");
260 }
261
262 IPVDataset anonymized = DatasetGeneralizer.generalize(original, this.columnInformationList, bestNode.getValues());
263 Tuple<List<Partition>, List<Partition>> bothPartitions =
264 OLAUtils.generatePartitions(original, anonymized, this.columnInformationList);
265
266 this.originalPartitions = bothPartitions.getFirst();
267 this.anonymizedPartitions = bothPartitions.getSecond();
268
269 this.sensitiveColumns = AnonymizationUtils.getColumnsByType(this.columnInformationList, ColumnType.SENSITIVE);
270
271 //we need to suppress the non-k-anonymous values at this stage
272 return createAnonymizedDataset(anonymized);
273 }

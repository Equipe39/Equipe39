48 private static JavaRDD<String> generalizeDataset(JavaRDD<String> input,
49 final List<ColumnInformation> columnInformationList,
50 final int k, LatticeNode node,
51 final char delimiter,
52 final char quoteChar) {
53 final List<Integer> quasiColumns = AnonymizationUtils.getColumnsByType(columnInformationList, ColumnType.QUASI);
54 final int[] levels = node.getValues();
55
56 return input.mapToPair( s -> {
57 try (CSVParser parser = CSVParser.parse(s, CSVFormat.RFC4180.withDelimiter(delimiter).withQuote(quoteChar));) {
58 CSVRecord csvRecord = parser.getRecords().get(0);
59
60 List<String> originalRow = new ArrayList<>();
61 int numberOfTokens = csvRecord.size();
62
63 for (int i = 0; i < numberOfTokens; i++) {
64 originalRow.add(csvRecord.get(i));
65 }
66
67 List<String> anonymizedRow = DatasetGeneralizer.generalizeRow(originalRow, columnInformationList, levels);
68
69 StringBuilder stringBuffer = new StringBuilder();
70
71 for (Integer quasiColumn : quasiColumns) {
72 int columnIndex = quasiColumn;
73 String value = anonymizedRow.get(columnIndex);
74 stringBuffer.append(value);
75 stringBuffer.append(":");
76 }
77
78 String key = stringBuffer.toString();
79
80 StringWriter writer = new StringWriter();
81 CSVPrinter printer = new CSVPrinter(writer, CSVFormat.RFC4180.withDelimiter(delimiter).withQuoteMode(QuoteMode.MINIMAL));
82
83 printer.printRecord(anonymizedRow);
84
85 return new Tuple2<>(key, writer.toString().trim());
86 }
87 }).groupByKey().mapValues((Function<Iterable<String>, Iterable<String>>) rows -> {
88 Iterator<String> iterator = rows.iterator();
89 int counter = 0;
90
91 while (iterator.hasNext()) {
92 counter++;
93 iterator.next();
94 }
95
96 if (counter >= k) {
97 return rows;
98 } else {
99 List<String> suppressedRows = new ArrayList<>();
100
101 for (String row : rows) {
102 try (
103 CSVParser parser = CSVParser.parse(row, CSVFormat.RFC4180.withDelimiter(','));
104 ) {
105 CSVRecord csvRecord = parser.getRecords().get(0);
106
107 int numberOfTokens = csvRecord.size();
108
109 List<String> values = new ArrayList<>();
110 for (int i = 0; i < numberOfTokens; i++) {
111 ColumnInformation columnInformation = columnInformationList.get(i);
112
113 if (columnInformation.getColumnType() == ColumnType.QUASI) {
114 values.add("*");
115 } else {
116 String value = csvRecord.get(i);
117 values.add(value);
118 }
119 }
120
121 StringWriter writer = new StringWriter();
122 CSVPrinter printer = new CSVPrinter(writer, CSVFormat.RFC4180.withDelimiter(delimiter).withQuoteMode(QuoteMode.MINIMAL));
123
124 printer.printRecord(values);
125 suppressedRows.add(writer.toString().trim());
126 }
127 }
128
129 return suppressedRows;
130 }
131 }).values().flatMap(Iterable::iterator);

153 public static JavaRDD<String> run(final AnonymizationOptions anonymizationOptions, JavaRDD<String> rdd) throws MisconfigurationException {
154
155 List<ColumnInformation> columnInformationList = anonymizationOptions.getColumnInformation();
156 double suppressionRate = anonymizationOptions.getSuppressionRate();
157 List<PrivacyConstraint> privacyConstraints = anonymizationOptions.getPrivacyConstraints();
158
159 CSVDatasetOptions csvDatasetOptions = (CSVDatasetOptions) anonymizationOptions.getDatasetOptions();
160 char delimiter = csvDatasetOptions.getFieldDelimiter();
161 char quoteChar = csvDatasetOptions.getQuoteChar();
162
163 if (countQuasiColumns(columnInformationList) == 0) {
164 throw new RuntimeException("no quasi-identifiers are present");
165 }
166
167 long startMillis = System.currentTimeMillis();
168
169 final boolean hasHeader = DatasetUtils.checkForHeader(anonymizationOptions.getDatasetOptions());
170 if (hasHeader) {
171 rdd = rdd.zipWithIndex().filter( s -> (s._2() > 0)).map(Tuple2::_1);
172 }
173
174 int k = AnonymizationUtils.getK(privacyConstraints);
175
176 Long inputSize = rdd.count();
177 SparkAnonymityChecker sparkAnonymityChecker = new SparkAnonymityChecker(rdd, inputSize, columnInformationList, privacyConstraints, delimiter, quoteChar);
178
179 Lattice lattice = new Lattice(sparkAnonymityChecker, columnInformationList, suppressionRate);
180 lattice.explore();
181
182 List<LatticeNode> kMinimalNodes = lattice.getKMinimal();
183 if (kMinimalNodes == null || kMinimalNodes.size() == 0) {
184 throw new RuntimeException("cannot satisfy constraints");
185 }
186
187 LatticeNode bestNode = kMinimalNodes.get(0); //TODO: link with information loss metrics
188
189 JavaRDD<String> anonymized = generalizeDataset(rdd, columnInformationList, k, bestNode, delimiter, quoteChar);
190
191 long diff = System.currentTimeMillis() - startMillis;
192
193 System.out.println("==== Constraints ====");
194 for(PrivacyConstraint privacyConstraint: privacyConstraints) {
195 System.out.println(privacyConstraint.toString());
196 }
197
198 System.out.println(String.format("n=%d: OLA took %d milliseconds, suppression = %f (node: %s), checked %d out of %d",
199 inputSize, diff, bestNode.getSuppressionRate(), bestNode.toString(), lattice.getNodesChecked(), lattice.getTotalNodes()));
200
201
202 return anonymized;
203 }

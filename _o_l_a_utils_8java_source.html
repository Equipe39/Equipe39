30 private static String createRowKey(List<String> row, List<Integer> quasiColumns) {
31 List<String> quasiValues = new ArrayList<>(quasiColumns.size());
32 for (Integer quasiColumn : quasiColumns) {
33 int columnIndex = quasiColumn;
34 String value = row.get(columnIndex);
35 quasiValues.add(value);
36 }
37
38 return StringUtils.join(quasiValues, ':');
39 }

41 public static Tuple<List<Partition>, List<Partition>> generatePartitions(IPVDataset original, IPVDataset anonymized,
42 List<ColumnInformation> columnInformationList) {
43
44 Map<String, InMemoryPartition> anonymizedPartitionsMap = new HashMap<>();
45 Map<String, InMemoryPartition> originalPartitionsMap = new HashMap<>();
46
47 int n = anonymized.getNumberOfRows();
48 int numberOfColumns = anonymized.getNumberOfColumns();
49
50 List<Integer> quasiColumns = AnonymizationUtils.getColumnsByType(columnInformationList, ColumnType.QUASI);
51
52 for (int i = 0; i < n; i++) {
53 List<String> originalRow = original.getRow(i);
54 List<String> anonymizedRow = anonymized.getRow(i);
55
56 String key = createRowKey(anonymizedRow, quasiColumns);
57
58 InMemoryPartition anonPartition = anonymizedPartitionsMap.get(key);
59 InMemoryPartition originalPartition = originalPartitionsMap.get(key);
60
61 if (anonPartition == null) {
62 anonymizedPartitionsMap.put(key, new InMemoryPartition(numberOfColumns));
63 originalPartitionsMap.put(key, new InMemoryPartition(numberOfColumns));
64
65 anonPartition = anonymizedPartitionsMap.get(key);
66 originalPartition = originalPartitionsMap.get(key);
67 }
68
69 anonPartition.getMember().addRow(anonymizedRow);
70 originalPartition.getMember().addRow(originalRow);
71 }
72
73 Set<String> keys = anonymizedPartitionsMap.keySet();
74
75 List<Partition> originalPartitions = new ArrayList<>();
76 List<Partition> anonymizedPartitions = new ArrayList<>();
77
78 for (String key : keys) {
79 originalPartitions.add(originalPartitionsMap.get(key));
80 anonymizedPartitions.add(anonymizedPartitionsMap.get(key));
81 }
82
83 return new Tuple<>(originalPartitions, anonymizedPartitions);
84 }

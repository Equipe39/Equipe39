71 public OpenNLPAnnotator(JsonNode configuration) {
72 this.defaultLanguage = Language.valueOf(configuration.get("defaultLanguage").asText());

78 this.sentenceDetector = new HashMap<>();

80
81 Iterator<Map.Entry<String, JsonNode>> iterator = configuration.get("configuration").fields();
82 while (iterator.hasNext()) {
83 Map.Entry<String, JsonNode> entry = iterator.next();
84 JsonNode localConfiguration = entry.getValue();

87 this.dateFinder.put(language, new NameFinderME(loadNameModel(localConfiguration.get("dateFinder").asText())));

108 private List<IdentifiedEntity> processSpans(String text, Span[] tokens,
109 Span[] detected, int sentenceStart, ProviderType providerType) {
110 List<IdentifiedEntity> results = new ArrayList<>();
111
112 for(Span span: detected) {
113 int beginningOfFirstWord = tokens[span.getStart()].getStart();
114 int endOfLastWord = tokens[span.getEnd() - 1].getEnd();
115
116 int startOffsetInDoc = sentenceStart + beginningOfFirstWord;
117 int endOffsetInDoc = sentenceStart + endOfLastWord;
118
119 String nameInDocument = text.substring(startOffsetInDoc, endOffsetInDoc);
120
121 IdentifiedEntity entity = new IdentifiedEntity(nameInDocument, startOffsetInDoc, endOffsetInDoc,
122 Collections.singleton(new IdentifiedEntityType(providerType.getName(), providerType.getName(), annotatorName)), Collections.singleton(PartOfSpeechType.UNKNOWN));
123
124 results.add(entity);
125 }
126
127 return results;
128 }

130 @Override
131 public List<IdentifiedEntity> identify(String text, Language language) throws IOException {
132 List<IdentifiedEntity> results = new ArrayList<>();
133
134 if (language == Language.UNKNOWN) {
135 language = this.defaultLanguage;
136 }
137
138 SentenceDetectorME sentenceDetector = this.sentenceDetector.get(language);
139 if (sentenceDetector == null) {
140 throw new RuntimeException("configuration for language " + language.name() + " is missing");
141 }
142
143 final Span[] sentenceSpans = sentenceDetector.sentPosDetect(text);
144
145 for (Span sentenceSpan : sentenceSpans) {
146 String sentence = text.substring(sentenceSpan.getStart(), sentenceSpan.getEnd());
147
148 String[] tokens = tokenizer.get(language).tokenize(sentence);
149 Span[] tokenPositionsWithinSentence = tokenizer.get(language).tokenizePos(sentence);
150
151 Span[] nameSpans = nameFinder.get(language).find(tokens);
152 Span[] locationSpans = locationFinder.get(language).find(tokens);
153 Span[] orgSpans = organizationFinder.get(language).find(tokens);
154 Span[] timeSpans = timeFinder.get(language).find(tokens);
155 Span[] dateSpans = dateFinder.get(language).find(tokens);
156
157
158 int sentenceStart = sentenceSpan.getStart();
159 results.addAll(processSpans(text, tokenPositionsWithinSentence, nameSpans, sentenceStart, ProviderType.NAME));
160 results.addAll(processSpans(text, tokenPositionsWithinSentence, locationSpans, sentenceStart, ProviderType.ADDRESS));
161 results.addAll(processSpans(text, tokenPositionsWithinSentence, orgSpans, sentenceStart, ProviderType.ORGANIZATION));
162 results.addAll(processSpans(text, tokenPositionsWithinSentence, timeSpans, sentenceStart, ProviderType.DATETIME));
163 results.addAll(processSpans(text, tokenPositionsWithinSentence, dateSpans, sentenceStart, ProviderType.DATETIME));
164
165 }
166
167 nameFinder.get(language).clearAdaptiveData();
168 locationFinder.get(language).clearAdaptiveData();
169 timeFinder.get(language).clearAdaptiveData();
170 dateFinder.get(language).clearAdaptiveData();
171 organizationFinder.get(language).clearAdaptiveData();
172
173 return results;
174 }

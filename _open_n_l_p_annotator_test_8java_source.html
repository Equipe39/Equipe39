38 @Test
39 public void testIdentification() throws Exception {
40 //instantiate a new NLPAnnotator
41 try (InputStream inputStream = OpenNLPAnnotatorTest.class.getResourceAsStream("/opennlp.json")) {
42 NLPAnnotator identifier = new OpenNLPAnnotator(JsonUtils.MAPPER.readTree(inputStream));
43
44 //we invoke the NLPAnnotator. It will return a list of identified tokens per sentence.
45 final List<IdentifiedEntity> identifiedEntities = identifier.identify(note1, Language.UNKNOWN);
46
47 //NLPUtils.createString is a utility function to convert a list of identified tokens in a string with annotation information
48 //example print: <ProviderType:NAME>Peter</ProviderType> was not home at present
49 System.out.println("sentence:" + NLPUtils.createString(identifiedEntities));
50
51 //for each token you can get the value and the ProviderType (PRIMA data type class):
52 for (final IdentifiedEntity original : identifiedEntities) {
53 System.out.println("\ttoken:" + original.getText() + ", type:" + original.getType());
54 }

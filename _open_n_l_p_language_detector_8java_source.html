36 public OpenNLPLanguageDetector() {

38 LanguageDetectorModel m = new LanguageDetectorModel(Objects.requireNonNull(is));
39 this.myCategorizer = new LanguageDetectorME(m);

41 throw new RuntimeException("unable to initialize OpenNLPLanguageDetector");
42 }
43
44 this.languageMap = new HashMap<>();

47 this.languageMap.put("ita", Language.ITALIAN);
48 this.languageMap.put("gle", Language.IRISH);
49 this.languageMap.put("heb", Language.HEBREW);
50 }
51
52 @Override
53 public Language detectLanguage(String input) {

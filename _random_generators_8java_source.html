118 public static int luhnCheckDigit(String body) {
119 boolean evenPosition = true;
120 int sum = 0;
121
122 for (int i = body.length() - 1; i >= 0; i--) {
123 int n = body.charAt(i) - '0';
124 if (evenPosition) {
125 n *= 2;
126 if (n > 9) {
127 n = (n % 10) + 1;
128 }
129 }
130 sum += n;
131 evenPosition = !evenPosition;
132 }
133
134 int s = 10 - sum % 10;
135 return (s % 10 == 0) ? 0 : s;
136 }

242 private static String bytesToHex(byte[] bytes) {
243 char[] hexChars = new char[bytes.length * 2];
244 for (int j = 0; j < bytes.length; j++) {
245 int v = bytes[j] & 0xFF;
246 hexChars[j * 2] = hexArray[v >>> 4];
247 hexChars[j * 2 + 1] = hexArray[v & 0x0F];
248 }
249
250 return new String(hexChars);
251 }

259 public static String deterministicReplacement(String identifier) {
260
261 try {
262 MessageDigest salt = MessageDigest.getInstance("SHA-512");
263 salt.update(identifier.getBytes(StandardCharsets.UTF_8));
264 String digest = bytesToHex(salt.digest());
265
266 if (digest.length() >= identifier.length()) {
267 StringBuilder replacement = new StringBuilder();
268
269 while (replacement.length() < identifier.length()) {
270 replacement.append(digest);
271 }
272
273 digest = replacement.toString();
274 }
275
276 return digest.substring(0, identifier.length());
277 } catch (NoSuchAlgorithmException e) {
278 logger.error(e.getMessage(), e);
279 throw new RuntimeException(e);
280 }
281 }

355 public static LatitudeLongitude generateRandomCoordinateFromBearing(
356 double latitude, double longitude, double radian, int distance) {
357
358 double delta = (double) distance / GeoUtils.getR();
359 double theta = Math.toRadians(radian);
360
361 double f1 = Math.toRadians(latitude);
362 double l1 = Math.toRadians(longitude);
363
364 double f2 = Math.asin(Math.sin(f1) * Math.cos(delta) +
365 Math.cos(f1) * Math.sin(delta) * Math.cos(theta));
366
367 double l2 = l1 + Math.atan2(Math.sin(theta) * Math.sin(delta) * Math.cos(f1),
368 Math.cos(delta) - Math.sin(f1) * Math.sin(f2));
369 // normalise to -180..+180
370 l2 = (l2 + 3 * Math.PI) % (2 * Math.PI) - Math.PI;
371
372 return new LatitudeLongitude(Math.toDegrees(f2), Math.toDegrees(l2));
373
374 }

412 public static LatitudeLongitude generateRandomCoordinate(
413 Double latitude, Double longitude, int offsetRadius) {
414
415 double radiusInDegrees = offsetRadius / 111000f;
416
417 double u = random.nextDouble();
418 double v = random.nextDouble();
419 double w = radiusInDegrees * Math.sqrt(u);
420
421 double t = 2 * Math.PI * v;
422 double x = w * Math.cos(t);
423 double y = w * Math.sin(t);
424
425 // Adjust the x-coordinate for the shrinking of the east-west distances
426 double new_x = x / Math.cos(longitude);
427
428 double foundLatitude = y + latitude;
429 if (foundLatitude > 90.0) {
430 double diff = foundLatitude - 90.0;
431 foundLatitude = Math.abs(-90 + diff);
432 }
433
434 double foundLongitude = new_x + longitude;
435 if (foundLongitude > 180) {
436 double diff = foundLongitude - 180.0;
437 foundLongitude = Math.abs(-180 + diff);
438 }
439
440 return new LatitudeLongitude(foundLatitude, foundLongitude, LatitudeLongitudeFormat.DECIMAL);
441 }

502 public static String randomHostnameGenerator(String hostname, int preserveDomains) {
503
504 if (preserveDomains == -1) {
505 return hostname;
506 }
507
508 int idx;
509 int preserveSubdomains = Math.max(preserveDomains - 1, 0);
510
511 if (ipAddressIdentifier.isIPv4(hostname)) {
512 return ipAddressMaskingProvider.directMask(hostname, true);
513 } else if (ipAddressIdentifier.isIPv6(hostname)) {
514 return ipAddressMaskingProvider.directMask(hostname, false);
515 }
516
517
518 if (preserveDomains == 0) {
519 return generateRandomHost(hostname, preserveSubdomains) + '.' + tldManager.getRandomTLD();
520 }
521
522 String tld = tldManager.getTLD(hostname);
523 if (tld == null || (idx = hostname.indexOf(tld)) == 0) {
524 return generateRandomHost(hostname, preserveDomains);
525 }
526
527 hostname = hostname.substring(0, idx - 1);
528
529 return generateRandomHost(hostname, preserveSubdomains) + '.' + tld;
530 }

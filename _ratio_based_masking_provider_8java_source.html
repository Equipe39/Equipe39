73 private String maskWithRatioAsOperand(double value, FieldRelationship fieldRelationship,
74 Map<String, OriginalMaskedValuePair> values, String identifier) {
75 String operandName = fieldRelationship.getOperands()[0].getName();
76
77 String operandValueString = values.get(operandName).getOriginal();
78
79 try {
80 double operandValue = Double.parseDouble(operandValueString);
81 double masked = value * operandValue;
82 return formatResult(masked);
83 } catch (NumberFormatException e) {
84 switch (failMode) {
85 case FailMode.RETURN_ORIGINAL:
86 return identifier;
87 case FailMode.THROW_ERROR:
88 logger.error("invalid numerical value");
89 throw new IllegalArgumentException("invalid numerical value");
90 case FailMode.RETURN_EMPTY:
91 default:
92 return "";
93 }
94 }
95 }

117 @Override
118 public String maskWithRatio(String identifier, String operandMasked, String operandOriginal) {
119 double value;
120 try {
121 value = Double.parseDouble(identifier);
122 } catch (NumberFormatException e) {
123 switch (failMode) {
124 case FailMode.RETURN_ORIGINAL:
125 return identifier;
126 case FailMode.THROW_ERROR:
127 logger.error("invalid numerical value");
128 throw new IllegalArgumentException("invalid numerical value");
129 case FailMode.RETURN_EMPTY:
130 default:
131 return "";
132 }
133 }
134
135 try {
136 double originalBase = Double.parseDouble(operandOriginal);
137 double maskedBase = Double.parseDouble(operandMasked);
138
139 double ratio = originalBase / value;
140
141 double masked = maskedBase / ratio;
142 return formatResult(masked);
143 } catch (NumberFormatException e) {
144 switch (failMode) {
145 case FailMode.RETURN_ORIGINAL:
146 return identifier;
147 case FailMode.THROW_ERROR:
148 logger.error("invalid numerical value");
149 throw new IllegalArgumentException("invalid numerical value");
150 case FailMode.RETURN_EMPTY:
151 default:
152 return "";
153 }
154 }
155
156 }

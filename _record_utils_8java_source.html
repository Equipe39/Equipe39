60 public static Map<String,Integer> createFieldMap(List<String> fieldNames) {

65 Map<String, Integer> fieldMap = new HashMap<>();

74 public static Map<String,Integer> createFieldMap(StructType schema) {
75 if (schema == null || schema.size() == 0) {
76 return Collections.emptyMap();
77 }
78
79 Map<String, Integer> fieldMap = new HashMap<>();
80
81 final StructField[] fields = schema.fields();
82 for (int i = 0; i < fields.length; ++i) {
83 final StructField field = fields[i];
84 fieldMap.put(field.name(), i);
85 }
86
87 return fieldMap;

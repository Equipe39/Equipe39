63 protected void addToMapByLocale(Map<String, Map<String, K>> perLocaleMap, String countryCode, String key, K value) {
64 Map<String, K> localMap = perLocaleMap.get(countryCode);
65
66 if (localMap == null) {
67 perLocaleMap.put(countryCode, new HashMap<>());
68 listMap.put(countryCode, new ArrayList<>());
69 probMap.put(countryCode, new ArrayList<>());
70 localMap = perLocaleMap.get(countryCode);
71 }
72
73 localMap.put(key, value);
74 listMap.get(countryCode).add(key);
75
76 if (value instanceof ProbabilisticEntity) {
77 double probability = ((ProbabilisticEntity) value).getProbability();
78 probMap.get(countryCode).add(new Pair<>(key, probability));
79 }

96 protected Map<String, Map<String, K>> readResources(Collection<ResourceEntry> entries) {
97 Map<String, Map<String, K>> resources = new HashMap<>();
98
99 this.minimumLength = Integer.MAX_VALUE;
100 this.maximumLength = Integer.MIN_VALUE;
101
102 for (ResourceEntry entry : entries) {
103 try (InputStream inputStream = entry.createStream();
104 CSVParser reader = Readers.createCSVReaderFromStream(inputStream)) {
105 String countryCode = entry.getCountryCode();
106
107 for (CSVRecord line : reader) {
108 List<Tuple<String, K>> keyValuePairs = parseResourceRecord(line, countryCode);
109
110 for (Tuple<String, K> keyValue : keyValuePairs) {
111 String key = keyValue.getFirst();
112 K value = keyValue.getSecond();
113
114 if (!appliesToAllCountriesOnly()) {
115 addToMapByLocale(resources, countryCode, key, value);
116 }
117
118 addToMapByLocale(resources, getAllCountriesName(), key, value);
119
120 this.minimumLength = Math.min(minimumLength, key.length());
121 this.maximumLength = Math.max(maximumLength, key.length());
122 }
123 }
124
125 } catch (IOException | NullPointerException e) {
126 e.printStackTrace();
127 }
128 }
129
130 return resources;
131 }

148 public ResourceBasedManager() {
149 init();
150
151 this.resourceMap = new HashMap<>();
152 this.listMap = new HashMap<>();
153 this.probMap = new HashMap<>();
154 this.probDistMap = new HashMap<>();
155
156 Map<String, Map<String, K>> contents = readResources(getResources());
157
158 for (final Map.Entry<String, Map<String, K>> entry : contents.entrySet()) {
159 final String key = entry.getKey();
160 final Map<String, K> value = entry.getValue();
161 MapWithRandomPick<String, K> mapWithRandomPick = new MapWithRandomPick<>(value);
162 this.resourceMap.put(key, mapWithRandomPick);
163 this.resourceMap.get(key).setKeyList();
164 }
165
166 for (String key : this.probMap.keySet()) {
167 List<Pair<String, Double>> pmf = probMap.get(key);
168
169 if (!pmf.isEmpty()) {
170 this.probDistMap.put(key, new EnumeratedDistribution<>(probMap.get(key)));
171 }
172 }
173
174 postInit();
175 }

214 private String getPseudorandomElement(List<String> keys, String key) {
215 long hash = Math.abs(HashUtils.longFromHash(key));
216
217 if (keys == null || keys.size() == 0) {
218 return Long.toString(hash);
219 }
220
221 int position = (int) (hash % keys.size());
222 return keys.get(position);
223 }

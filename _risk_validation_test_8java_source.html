123 private double calculateRealRisk(String populationResourceName, IPVDataset anonymizedDataset,
124 List<ColumnInformation> columnInformation, int[] levels) throws IOException {
125
126 InputStream originalIS = this.getClass().getResourceAsStream(populationResourceName);
127
128 Map<String, Integer> anonEQCounters = AnonymizationUtils.generateEQCounters(anonymizedDataset, columnInformation);
129 Map<String, Integer> populationEQCounters = DatasetGeneralizer.generalizeCSVAndCountEQ(originalIS, columnInformation, levels);
130
131 int minimumLink = anonEQCounters.values().stream().mapToInt(Integer::intValue).min().orElse(Integer.MAX_VALUE);
132
133 for(String key: anonEQCounters.keySet()) {
134 Integer linkedWith = populationEQCounters.get(key);
135 minimumLink = Math.min(minimumLink, linkedWith);
136 }
137
138 return 1.0/(double)minimumLink;
139 }

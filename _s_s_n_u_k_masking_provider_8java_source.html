28public class SSNUKMaskingProvider implements MaskingProvider {
29 /*
30 The format of the number is two prefix letters, six digits, and one suffix letter.[5]
31 The example used is typically AB123456C. Often, the number is printed with spaces to pair off
32 the digits, like this: AB 12 34 56 C.
33 Neither of the first two letters can be D, F, I, Q, U or V. The second letter also cannot be O.
34 The prefixes BG, GB, NK, KN, TN, NT and ZZ are not allocated.[6]
35 Validation lists of issued two-letter prefixes are published from time to time.[7][8]
36 The suffix letter is either A, B, C, or D.[5] (although F, M, and P have been used for temporary numbers in the past)
37 */
38 private static final SSNUKIdentifier identifier = new SSNUKIdentifier();
39 private static final char[] allowedFirstLetters = "ABCEGHJKLMNOPRSTWXYZ".toCharArray();
40 private static final char[] allowedSecondLetters = "ABCEGHJKLMNPRSTWXYZ".toCharArray();
41 private static final char[] allowedSuffixLetters = "ABCD".toCharArray();
42 private final boolean preservePrefix;
43 private final SecureRandom random;

81 @Override
82 public String mask(String identifier) {
83 String prefix;
84
85 if (!SSNUKMaskingProvider.identifier.isOfThisType(identifier) || !this.preservePrefix) {
86 prefix = "" + allowedFirstLetters[random.nextInt(allowedFirstLetters.length)];
87 prefix += allowedSecondLetters[random.nextInt(allowedSecondLetters.length)];
88 } else {
89 prefix = identifier.substring(0, 2);
90 }
91
92 StringBuilder builder = new StringBuilder(prefix);
93 for (int i = 0; i < 6; i++) {
94 char randomDigit = RandomGenerators.randomDigit();
95 builder.append(randomDigit);
96 }
97
98 char suffix = allowedSuffixLetters[random.nextInt(allowedSuffixLetters.length)];
99
100 builder.append(suffix);
101
102 return builder.toString();
103 }

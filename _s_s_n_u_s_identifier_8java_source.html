53 public SSNUS parseSSNUS(String data) {
54 String[] toks = data.split("-");
55 String[] ssnParts;
56
57 if (toks.length == 3) {
58 if (toks[0].length() != 3 || toks[1].length() != 2 || toks[2].length() != 4) {
59 return null;
60 }
61
62 ssnParts = toks;
63 } else {
64 return null;
65 }
66
67 for (int i = 0; i < 3; i++) {
68 for (int j = 0; j < ssnParts[i].length(); j++) {
69 if (!Character.isDigit(ssnParts[i].charAt(j))) {
70 return null;
71 }
72 }
73 }
74
75 if (ssnParts[0].equals("000") || ssnParts[0].equals("666")) {
76 return null;
77 }
78
79
80 return new SSNUS(ssnParts[0], ssnParts[1], ssnParts[2]);
81 }

89 private int prefixMatch(String data) {
90 data = data.toUpperCase();
91
92 for (String prefix : prefixes) {
93 if (data.startsWith(prefix)) {
94 String ssn = data.substring(prefix.length()).trim();
95 if (parseSSNUS(ssn) != null) {
96 return firstOccurenceOfNumber(data, prefix.length());
97 }
98 }
99 }
100
101 return -1;
102 }

104 private int firstOccurenceOfNumber(String data, int start) {
105 for (int i = start; i < data.length(); i++) {
106 if (Character.isDigit(data.charAt(i))) {
107 return i;
108 }
109 }
110
111 return -1;
112 }

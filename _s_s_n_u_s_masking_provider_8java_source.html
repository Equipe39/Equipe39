66 public SSNUSMaskingProvider(SecureRandom random, MaskingConfiguration configuration) {
67 this.random = random;
68 this.preserveAreaNumber = configuration.getBooleanValue("ssnus.mask.preserveAreaNumber");
69 this.preserveGroup = configuration.getBooleanValue("ssnus.mask.preserveGroup");
70 }
71
72 /*
73 The Social Security number is a nine-digit number in the format "AAA-GG-SSSS".[27]
74 The number is divided into three parts.
75 The area number, the first three digits, is assigned by geographical region.
76 The middle two digits are the group number
77 The last four digits are serial numbers
78 */
79 @Override
80 public String mask(String identifier) {
81 SSNUS ssn = ssnUSIdentifier.parseSSNUS(identifier);
82 String areaNumber;
83 String group;
84 String serialNumber = String.format("%04d", random.nextInt(9999));
85
86 if (ssn != null) {
87 if (this.preserveAreaNumber) {
88 areaNumber = ssn.getAreaNumber();
89 } else {
90 while (true) {
91 int areaNumberInt = random.nextInt(999);
92 if (areaNumberInt == 0 || areaNumberInt == 666) {
93 continue;
94 }
95 areaNumber = String.format("%03d", areaNumberInt);
96 break;
97 }
98 }
99
100 if (this.preserveGroup) {
101 group = ssn.getGroup();
102 } else {
103 group = String.format("%02d", random.nextInt(99));
104 }
105 } else {
106 areaNumber = String.format("%03d", random.nextInt(999));
107 group = String.format("%02d", random.nextInt(99));
108 }
109
110 return (new SSNUS(areaNumber, group, serialNumber)).toString();
111
112 }

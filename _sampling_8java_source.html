90 @Override
91 public IPVDataset apply() {
92 int populationSize = this.original.getNumberOfRows();
93 int sampleSize = (int) (this.percentage * populationSize);
94
95 if (sampleSize == 0) {
96 return new IPVDataset(
97 new ArrayList<>(),
98 original.getSchema(),
99 original.hasColumnNames()
100 );
101 }
102
103 List<List<String>> finalRows = new ArrayList<>();
104 Random random = new SecureRandom();
105
106 Set<Integer> used = new HashSet<>();
107
108 Partition originalAnonymous = new InMemoryPartition(this.original.getNumberOfColumns());
109 Partition originalNonAnonymous = new InMemoryPartition(this.original.getNumberOfColumns());
110 Partition anonymizedNonAnonymous = new InMemoryPartition(this.original.getNumberOfColumns());
111
112 for (int i = 0; i < sampleSize; i++) {
113 while (true) {
114 int nextPos = random.nextInt(populationSize);
115 if (used.contains(nextPos)) {
116 continue;
117 }
118
119 List<String> row = this.original.getRow(nextPos);
120
121 finalRows.add(row);
122 originalAnonymous.getMember().addRow(row);
123
124 used.add(nextPos);
125 break;
126 }
127 }
128
129 for (int i = 0; i < this.original.getNumberOfRows(); i++) {
130 if (!used.contains(i)) {
131 List<String> row = this.original.getRow(i);
132 originalNonAnonymous.getMember().addRow(row);
133 anonymizedNonAnonymous.getMember().addRow(row);
134 }
135 }
136
137 Partition anonymizedAnonymous = new InMemoryPartition(finalRows);
138
139 originalAnonymous.setAnonymous(true);
140 originalNonAnonymous.setAnonymous(false);
141 this.originalPartitions = Arrays.asList(originalAnonymous, originalNonAnonymous);
142
143 anonymizedAnonymous.setAnonymous(true);
144 anonymizedNonAnonymous.setAnonymous(false);
145 this.anonymizedPartitions = Arrays.asList(anonymizedAnonymous, anonymizedNonAnonymous);
146
147 return new IPVDataset(
148 finalRows,
149 original.getSchema(),
150 original.hasColumnNames()
151 );
152 }

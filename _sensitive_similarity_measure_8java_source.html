84 private Double computeMSE() {
85 double maximumMSE = 0.0;
86
87 for (Map<String, Double> localDistribution : partitionDistributions) {
88 double MSE = 0.0;
89
90 for (Map.Entry<String, Double> knownValues : globalDistributions.entrySet()) {
91 final Double local = localDistribution.get(knownValues.getKey());
92
93 MSE += Math.abs(knownValues.getValue() - (null == local ? 0.0 : local));
94 }
95
96 MSE /= globalDistributions.size();
97
98 maximumMSE = Math.max(MSE, maximumMSE);
99 }
100
101 return maximumMSE;
102 }

135 private Map<String, Double> computeDistributionOfSensitiveValues(IPVDataset dataset) {
136 Map<String, Double> distribution = new HashMap<>();
137
138 for (List<String> row : dataset) {
139 String sensitiveValues = combineSensitiveFields(row);
140
141 Double oldCount = distribution.get(sensitiveValues);
142
143 distribution.put(sensitiveValues, (null == oldCount ? 0 : oldCount) + 1);
144 }
145
146 for (Map.Entry<String, Double> entry : distribution.entrySet()) {
147 entry.setValue(entry.getValue() / dataset.getNumberOfRows());
148 }
149
150 return distribution;
151 }

153 private Collection<Integer> extractSensitiveFields(List<ColumnInformation> columnInformationList) {
154 List<Integer> sensitiveFields = new ArrayList<>();
155
156 for (int i = 0; i < columnInformationList.size(); ++i) {
157 if (columnInformationList.get(i) instanceof SensitiveColumnInformation) sensitiveFields.add(i);
158 }
159
160 return sensitiveFields;
161 }

107 public static boolean checkConstraints(List<PrivacyMetric> privacyMetrics,
108 List<PrivacyConstraint> privacyConstraints) {
109
110 for(int i = 0; i < privacyConstraints.size(); i++) {
111 PrivacyConstraint constraint = privacyConstraints.get(i);
112 PrivacyMetric metric = privacyMetrics.get(i);
113
114 if(!constraint.check(metric)) {
115 return true;
116 }
117 }
118
119 return false;
120 }

122 @Override
123 public double calculateSuppressionRate(LatticeNode node) {
124 final int[] levels = node.getValues();
125
126 logger.info("Checking node: " + node);
127
128 JavaRDD<Long> counters = input.mapToPair((PairFunction<String, String, List<PrivacyMetric>>) s -> createMapToPair(s, delimiter, quoteChar, columnInformationList, privacyMetrics, quasiColumns,
129 sensitiveColumns, levels, privacyConstraintsContentRequirements)).reduceByKey(new Function2<List<PrivacyMetric>, List<PrivacyMetric>, List<PrivacyMetric>>() {
130 @Override
131 public List<PrivacyMetric> call(List<PrivacyMetric> p1, List<PrivacyMetric> p2) throws Exception {
132 for(int i = 0; i < p1.size(); i++) {
133 PrivacyMetric m1 = p1.get(i);
134 PrivacyMetric m2 = p2.get(i);
135 m1.update(m2);
136 }
137
138 return p1;
139 }
140 }).filter(new Function<Tuple2<String, List<PrivacyMetric>>, Boolean>() {
141 @Override
142 public Boolean call(Tuple2<String, List<PrivacyMetric>> s) throws Exception {
143 return checkConstraints(s._2(), privacyConstraints);
144 }
145 }).map(new Function<Tuple2<String, List<PrivacyMetric>>, Long>() {
146 @Override
147 public Long call(Tuple2<String, List<PrivacyMetric>> s) throws Exception {
148 for(PrivacyMetric metric: s._2()) {
149 if (metric instanceof KAnonymityMetric) {
150 return ((KAnonymityMetric)metric).getCount();
151 }
152 }
153
154 return 0L;
155 }
156 });
157
158 Long suppressed = 0L;
159
160 long countersCount = counters.count();
161
162 if (countersCount == 1) {
163 suppressed = counters.collect().get(0);
164 }
165 else if (countersCount > 0) {
166 suppressed = counters.reduce(new Function2<Long, Long, Long>() {
167 @Override
168 public Long call(Long a, Long b) throws Exception {
169 return a + b;
170 }
171 });
172 }
173
174 Logger.getLogger(SparkAnonymityChecker.class).info("Suppression for node: " + node.toString() + " is " + (100.0 * (double) suppressed / (double) inputSize));
175 return 100.0 * (double) suppressed / (double) inputSize;
176 }

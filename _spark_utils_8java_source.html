172 public static Dataset<Row> createDataset(SparkSession sparkSession, String input, DataTypeFormat inputFormat,
173 DatasetOptions datasetOptions, String basePath) {
174 DataFrameReader reader = sparkSession.read();
175
176 if (basePath != null) {
177 reader = reader.option("basePath", basePath);
178 }
179
180 switch (inputFormat) {
181 case CSV:
182 CSVDatasetOptions csvDatasetOptions = (CSVDatasetOptions) datasetOptions;
183 reader = reader.option("sep", Character.toString(csvDatasetOptions.getFieldDelimiter())).option("quote", Character.toString(csvDatasetOptions.getQuoteChar()));
184
185 if (csvDatasetOptions.isHasHeader()) {
186 reader = reader.option("header", true);
187 }
188
189 if (csvDatasetOptions.isTrimFields()) {
190 reader = reader.option("ignoreLeadingWhiteSpace", true).option("ignoreTrailingWhiteSpace", true);
191 }
192
193 Dataset<Row> dataset = reader.csv(input);
194
195 if (!csvDatasetOptions.isHasHeader()) {
196 int numberOfFields = dataset.schema().fields().length;
197 String[] columnNames = CSVFormatProcessor.generateColumnNames(numberOfFields);
198 return dataset.toDF(columnNames);
199 }
200
201 return dataset;
202
203 case PARQUET:
204 return reader.parquet(input);
205 default:
206 return reader.text(input);
207 }

211 public static List<String> createFieldNames(Dataset<Row> rdd, DataTypeFormat format, DatasetOptions datasetOptions) {
212
213 if (format != DataTypeFormat.CSV && format != DataTypeFormat.PARQUET) {
214 return null;
215 }
216
217 StructType schema = rdd.schema();
218 List<String> fieldNames = new ArrayList<>(schema.fields().length);
219
220 if (format == DataTypeFormat.CSV) {
221 CSVDatasetOptions csvDatasetOptions = (CSVDatasetOptions) datasetOptions;
222
223 if (csvDatasetOptions.isHasHeader()) {
224 for (StructField field : schema.fields()) {
225 fieldNames.add(field.name());
226 }
227 } else {
228 String[] columnNames = CSVFormatProcessor.generateColumnNames(schema.fields().length);
229 fieldNames = Arrays.asList(columnNames);
230 }
231 } else {
232 for (int i = 0; i < schema.fields().length; i++) {
233 fieldNames.add(null);
234 }
235
236 for (StructField field : schema.fields()) {
237 scala.Option<Object> option = schema.getFieldIndex(field.name());
238 if (!option.isDefined()) {
239 throw new RuntimeException("undefined field: " + field.name());
240 }
241 fieldNames.set((Integer) option.get(), field.name());
242 }
243 }
244
245 return fieldNames;
246 }

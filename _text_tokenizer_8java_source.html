66 public Iterable<String> split(final Reader reader) throws IOException {
67 final TokenStream tokenStream = analyzer.tokenStream("_garbage", reader);
68 tokenStream.reset();
69
70 return () -> new Iterator<>() {

78 next = tokenStream.getAttribute(CharTermAttribute.class).toString();
79 return true;
80 }
81 } else {

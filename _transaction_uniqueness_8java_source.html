155 private static JavaPairRDD<String, Iterable<String>> computeTransactionsToUsersWithCombinations(JavaRDD<Row> rdd,
156 String[] idColumns, String[] targetColumns,
157 DataTypeFormat inputFormat, DatasetOptions datasetOptions,
158 List<String> fieldNames, Map<String, Integer> fieldMap, List<String> fieldTypes,
159 DataMaskingOptions dataMaskingOptions, MaskingProviderFactory maskingProviderFactory) {
160
161 final FormatProcessor formatProcessor = FormatProcessorFactory.getProcessor(inputFormat);
162
163 return rdd.mapToPair( row -> {
164 Record record = RecordUtils.createRecord(row, inputFormat, datasetOptions, fieldNames, fieldMap, fieldTypes, false);
165
166 if (!dataMaskingOptions.getToBeMasked().isEmpty()) {
167 record = formatProcessor.maskRecord(record, maskingProviderFactory, Collections.emptySet(), dataMaskingOptions);
168 }
169
170 StringBuilder keyBuilder = new StringBuilder();
171 for(String idColumn: idColumns) {
172 String k = new String(record.getFieldValue(idColumn));
173 keyBuilder.append(k);
174 keyBuilder.append(":");
175 }
176
177 String key = keyBuilder.toString();
178
179 StringBuilder value = new StringBuilder();
180 for (String target : targetColumns) {
181 value.append(new String(record.getFieldValue(target)));
182 value.append(":");
183 }
184
185 return new Tuple2<>(key, value.toString());
186 }).groupByKey().flatMapToPair( perUserTransactions -> {
187 Set<String> uniques = new HashSet<>();
188
189 for (String v : perUserTransactions._2) {
190 uniques.add(v);
191 }
192
193 List<Tuple2<String, String>> perTransaction = new ArrayList<>();
194 String user = perUserTransactions._1;
195
196 for(String t: uniques) {
197 perTransaction.add(new Tuple2<>(t, user));
198 }
199
200 return perTransaction.iterator();
201 }).groupByKey();
202 }

204 public static Tuple2<String, Set<String>> transactionToUID(Row row, String[] idColumns, String[] targetColumns, FormatProcessor formatProcessor,
205 DataTypeFormat inputFormat, DatasetOptions datasetOptions,
206 List<String> fieldNames, Map<String,Integer> fieldMap,
207 List<String> fieldTypes, DataMaskingOptions dataMaskingOptions, MaskingProviderFactory maskingProviderFactory) throws Exception{
208
209 Record record = RecordUtils.createRecord(row, inputFormat, datasetOptions, fieldNames, fieldMap, fieldTypes, false);
210
211 for (String targetColumn : targetColumns) {
212 byte[] fieldValue = record.getFieldValue(targetColumn);
213 if (fieldValue == null) {
214 return null;
215 }
216 }
217
218 if (!dataMaskingOptions.getToBeMasked().isEmpty()) {
219 record = formatProcessor.maskRecord(record, maskingProviderFactory, Collections.emptySet(), dataMaskingOptions);
220 }
221
222 StringBuilder keyBuilder = new StringBuilder();
223
224 for(int i = 0; i < idColumns.length - 1; i++) {
225 String k = new String(record.getFieldValue(idColumns[i]));
226 keyBuilder.append(k);
227 keyBuilder.append(":");
228 }
229
230 String k = new String(record.getFieldValue(idColumns[idColumns.length - 1]));
231 keyBuilder.append(k);
232
233 String key = keyBuilder.toString();
234
235 StringBuilder value = new StringBuilder();
236 for(int i = 0; i < targetColumns.length - 1; i++) {
237 byte[] fieldValue = record.getFieldValue(targetColumns[i]);
238 value.append(new String(fieldValue));
239 value.append(":");
240 }
241
242 byte[] lastValue = record.getFieldValue(targetColumns[targetColumns.length - 1]);
243 value.append(new String(lastValue));
244
245 return new Tuple2<>(value.toString(), new HashSet<>(Arrays.asList(key))); // transaction -> uid
246 }

248 private static JavaPairRDD<String, Set<String>> computeTransactionsToUsersNoCombinations(JavaRDD<Row> rdd, String[] idColumns, String[] targetColumns,
249 DataTypeFormat inputFormat, DatasetOptions datasetOptions,
250 List<String> fieldNames, Map<String,Integer> fieldMap,
251 List<String> fieldTypes, DataMaskingOptions dataMaskingOptions,
252 MaskingProviderFactory maskingProviderFactory) {
253
254 final FormatProcessor formatProcessor = FormatProcessorFactory.getProcessor(inputFormat);
255
256 return rdd.mapToPair(
257 row -> transactionToUID(row, idColumns, targetColumns, formatProcessor, inputFormat, datasetOptions, fieldNames, fieldMap, fieldTypes, dataMaskingOptions, maskingProviderFactory)
258 ).filter(Objects::nonNull).
259 reduceByKey( (v1, v2) -> {
260 v1.addAll(v2);
261 return v1;
262 });
263 }

265 public static Map<String, Long> run(Dataset<Row> dataset,
266 String[] idColumns, String[] targetColumns,
267 DataTypeFormat inputFormat, DatasetOptions datasetOptions,
268 List<String> fieldNames, Map<String, Integer> fieldMap, List<String> fieldTypes,
269 DataMaskingOptions dataMaskingOptions, MaskingProviderFactory maskingProviderFactory,
270 int factor, int threshold) {
271 JavaRDD<Row> rdd = dataset.javaRDD();
272
273 long uniqueTransactionCombinationsCount;
274 long uniqueIDs;
275 long totalTransactions = rdd.count();
276 long totalIDs;
277
278 if (factor > 1) {
279 JavaPairRDD<String, Iterable<String>> transactionsToUsers;
280 transactionsToUsers = computeTransactionsToUsersWithCombinations(rdd, idColumns, targetColumns, inputFormat, datasetOptions, fieldNames, fieldMap, fieldTypes,
281 dataMaskingOptions, maskingProviderFactory);
282
283 totalIDs = transactionsToUsers.values().flatMap(Iterable::iterator).distinct().count();
284
285 JavaPairRDD<String, Iterable<String>> uniqueTransactionCombinations = transactionsToUsers.filter(r -> {
286 Set<String> s = new HashSet<>();
287 r._2().forEach(s::add);
288 return s.size() <= threshold;
289 });
290
291 uniqueTransactionCombinationsCount = uniqueTransactionCombinations.count();
292 uniqueIDs = uniqueTransactionCombinations.values().flatMap(Iterable::iterator).distinct().count();
293 } else {
294 JavaPairRDD<String, Set<String>> transactionsToUsers;
295 transactionsToUsers = computeTransactionsToUsersNoCombinations(rdd, idColumns, targetColumns, inputFormat, datasetOptions, fieldNames, fieldMap, fieldTypes,
296 dataMaskingOptions, maskingProviderFactory);
297
298 totalIDs = transactionsToUsers.values().flatMap(Set::iterator).distinct().count();
299
300 JavaPairRDD<String, Set<String>> uniqueTransactionCombinations = transactionsToUsers.filter(r -> r._2().size() <= threshold);
301
302 uniqueTransactionCombinationsCount = uniqueTransactionCombinations.count();
303 uniqueIDs = uniqueTransactionCombinations.values().flatMap(Set::iterator).distinct().count();
304 }
305
306
307 Map<String, Long> results = new HashMap<>();
308 results.put("unique_transactions", uniqueTransactionCombinationsCount);
309 results.put("unique_ids", uniqueIDs);
310 results.put("total_transactions", totalTransactions);
311 results.put("total_ids", totalIDs);
312
313 logger.info("Combinations leading to a unique ID: " + uniqueTransactionCombinationsCount);
314 logger.info("Unique IDs : " + uniqueIDs);
315
316 return results;
317 }

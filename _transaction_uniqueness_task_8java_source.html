98 @Override
99 public void processFile(InputStream input, OutputStream output) throws MisconfigurationException, IOException {
100 final IPVDataset dataset = readInputDataset(input);
101 final int threshold = getTaskOptions().getThreshold();
102
103 Map<Integer, Set<Integer>> idsByTransaction = groupTransactionIdByTargetValues(dataset, getTaskOptions().getIdentityFields(), getTaskOptions().getExternallyObservableFields());
104
105 List<Integer> totalIDs = countNumberOfTotalIDs(idsByTransaction);
106 List<Set<Integer>> uniqueTransactionCombinations = identifyUniqueTransactions(idsByTransaction, threshold);
107 List<Integer> uniqueIDs = extractIDsOfUniqueTransactions(uniqueTransactionCombinations);
108
109 if (this.getTaskOptions().isExploreExternallyObservableFields()) {
110 List<TransactionUniquenessReportColumnContribution> columnsContributions = new ArrayList<>(getTaskOptions().getExternallyObservableFields().size());
111 for(String column : getTaskOptions().getExternallyObservableFields()) {
112 Map<Integer, Set<Integer>> contribIdsByTransaction = groupTransactionIdByTargetValues(dataset, getTaskOptions().getIdentityFields(), Collections.singletonList((column)));
113 List<Set<Integer>> contribUniqueTransactionCombinations = identifyUniqueTransactions(contribIdsByTransaction, threshold);
114 List<Integer> contribUniqueIDs = extractIDsOfUniqueTransactions(contribUniqueTransactionCombinations);
115 columnsContributions.add(new TransactionUniquenessReportColumnContribution(
116 column,
117 contribUniqueTransactionCombinations.size(),
118 contribUniqueIDs.size()
119 ));
120 }
121 mapper.writeValue(output, new TransactionUniquenessReport(
122 totalIDs.size(),
123 dataset.getNumberOfRows(),
124 uniqueTransactionCombinations.size(),
125 uniqueIDs.size(),
126 columnsContributions
127 ));
128 } else {
129 mapper.writeValue(output, new TransactionUniquenessReport(
130 totalIDs.size(),
131 dataset.getNumberOfRows(),
132 uniqueTransactionCombinations.size(),
133 uniqueIDs.size(),
134 null
135 ));
136 }
137 }

139 private List<Integer> extractIDsOfUniqueTransactions(List<Set<Integer>> uniqueTransactionCombinations) {
140 return uniqueTransactionCombinations.stream()

146 private List<Set<Integer>> identifyUniqueTransactions(Map<Integer, Set<Integer>> transactionsById, int threshold) {
147 return transactionsById.values().stream()

159 private Map<Integer, Set<Integer>> groupTransactionIdByTargetValues(IPVDataset dataset, List<String> identityFields, List<String> externallyObservableFields) {
160 Stream<Tuple<Integer, Set<Integer>>> encodedTransactions = encodeTransactionAndId(dataset, identityFields, externallyObservableFields);
161
162 Map<Integer, Set<Integer>> groupedTransactionIDs = new HashMap<>();
163
164 encodedTransactions.forEach(
165 transaction -> groupedTransactionIDs.merge(
166 transaction.getFirst(),
167 transaction.getSecond(),
168 (set1, set2) -> {
169 set1.addAll(set2);
170 return set1;
171 })
172 );
173
174 return groupedTransactionIDs;
175 }
176
177 private Stream<Tuple<Integer, Set<Integer>>> encodeTransactionAndId(IPVDataset dataset, List<String> identityFields, List<String> externallyObservableFields) {
178 final List<Integer> encodedIdFields = mapFieldNamesToPositions(dataset.getSchema(), identityFields);
179 final List<Integer> encodedTargetFields = mapFieldNamesToPositions(dataset.getSchema(), externallyObservableFields);
180
181 return StreamSupport.stream(dataset.spliterator(), false).
182 map(row -> new Tuple<>(
183 encodeValues(row, encodedTargetFields),
184 new HashSet<>(Collections.singleton(
185 encodeValues(row, encodedIdFields)
186 ))
187 ));
188 }
189
190 private Integer encodeValues(List<String> row, List<Integer> fields) {
191 StringBuilder builder = new StringBuilder();
192 for (Integer field : fields) {
193 builder.append(row.get(field));
194 builder.append('#');
195 }
196
197 return builder.toString().hashCode();
198 }
199
200 private List<Integer> mapFieldNamesToPositions(IPVSchema schema, List<String> requiredFieldNames) {
201 List<? extends IPVSchemaField> schemaFields = schema.getFields();
202
203 return requiredFieldNames.stream().map(fieldName -> {
204 for (int i = 0; i < schemaFields.size(); ++i) {
205 if (schemaFields.get(i).getName().equals(fieldName))
206 return i;
207 }
208 throw new IllegalArgumentException("Unknown field " + fieldName);
209 }).collect(Collectors.toList());
210 }

212 private IPVDataset readInputDataset(InputStream inputStream) {
213 try (Reader reader = new InputStreamReader(inputStream)) {
214 switch (getInputFormat()) {
215 case CSV:
216 CSVDatasetOptions options = (CSVDatasetOptions) getInputOptions();
217
218 return IPVDataset.load(reader, options.isHasHeader(), options.getFieldDelimiter(), options.getQuoteChar(), options.isTrimFields());
219 case JSON:
220 return JSONIPVDataset.load(reader);
221
222 case DICOM:
223 case XLS:
224 case XLSX:
225 case XML:
226 case PDF:
227 case DOC:
228 case DOCX:
229 case PLAIN:
230 case FHIR_JSON:
231 case HL7:
232 case PARQUET:
233 case VCF:
234 case JDBC:
235 default:
236 throw new IllegalArgumentException("Format not supported (at the moment). Please contact support.");
237 }
238 } catch (IOException e) {
239 throw new RuntimeException("Format not supported at the moment", e);
240 }
241 }

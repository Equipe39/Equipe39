29 public static void mustBeTheSame(IPVDataset original, IPVDataset reloaded) {
30 if (original.getNumberOfRows() != reloaded.getNumberOfRows()) throw new RuntimeException("Size differ");
31 if (original.getNumberOfColumns() != reloaded.getNumberOfColumns()) throw new RuntimeException("Size differ");
32
33 int numberOfRows = original.getNumberOfRows();
34 int numberOfColumns = original.getNumberOfColumns();

37 for (int j = 0; j < numberOfColumns; j++) {
38 if (!reloaded.get(i, j).equals(original.get(i, j))) throw new RuntimeException("Value differ");

43 public static void validateIsKAnonymous(IPVDataset dataset, List<ColumnInformation> columnInformation, int k) {
44 List<Partition> partitions = PartitionUtils.createPartitions(dataset, columnInformation);
45 List<Integer> quasiColumns = AnonymizationUtils.getColumnsByType(columnInformation, ColumnType.QUASI);
46
47 for (Partition p : partitions) {

49
50 IPVDataset data = p.getMember();
51
52 String firstKey = AnonymizationUtils.generateEQKey(data.getRow(0), quasiColumns);
53 for (int i = 1; i < data.getNumberOfRows(); i++) {
54 String key = AnonymizationUtils.generateEQKey(data.getRow(i), quasiColumns);
55 if (!key.equals(firstKey)) throw new RuntimeException("Value differ");

152 private Map<String, Integer> getNodeFrequencies(NodeList nl) {
153 HashMap<String, Integer> nodesFrequencies = new HashMap<>();

164 private List<String> generatePaths(Document document, List<String> parts, String head) {
165 if (parts.isEmpty()) {
166 try {
167 NodeList nl = (NodeList) xPath.compile(head).evaluate(document, XPathConstants.NODESET);
168 if (nl.getLength() == 1) {
169 return Collections.singletonList(head);
170 } else {
171 List<String> paths = new ArrayList<>();
172 for (int i = 1; i <= nl.getLength(); i++) {
173 paths.add(head + "[" + i + "]");
174 }
175 return paths;
176 }
177 } catch (XPathExpressionException e) {
178 throw new RuntimeException(e.getMessage());
179 }
180 }
181
182 String part = parts.get(0);
183
184 List<String> paths = new ArrayList<>();
185
186 try {
187 NodeList nl = (NodeList) xPath.compile(head + "/" + part).evaluate(document, XPathConstants.NODESET);
188
189 Map<String, Integer> nodesFrequencies = getNodeFrequencies(nl);
190
191 HashMap<String, Integer> arraysCounters = new HashMap<>();
192 for (int i = 0; i < nl.getLength(); i++) {
193 String node = nl.item(i).getNodeName();
194 if (!isArray(node, nodesFrequencies)) {
195 paths.addAll(
196 generatePaths(
197 document,
198 parts.subList(1, parts.size()),
199 head + "/" + node
200 )
201 );
202 } else {
203 arraysCounters.merge(node, 1, (a, b) -> a + b);
204 paths.addAll(
205 generatePaths(
206 document,
207 parts.subList(1, parts.size()),
208 head + "/" + nl.item(i).getNodeName() + "[" + arraysCounters.get(node) + "]"
209 )
210 );
211 }
212 }
213 } catch (XPathExpressionException e) {
214 throw new RuntimeException(e.getMessage());
215 }
216
217 return paths;
218 }

97 @Override
98 public String mask(String identifier) {
99 Address randomAddress;
100
101 Address address = addressIdentifier.parseAddress(identifier);
102 if (address == null) {
103 address = new Address("", "", "", "", "", "");
104 randomAddress = new Address();
105 } else {
106 randomAddress = new Address(address);
107 }
108
109 if (address.isPOBox()) {
110 randomAddress.setPoBox(true);
111
112 if (this.getPseudorandom) {
113 String poBoxNumber = address.getPoBoxNumber();
114 randomAddress.setPoBoxNumber(Long.toString(Math.abs(HashUtils.longFromHash(poBoxNumber)) % 10000));
115 } else {
116 randomAddress.setPoBoxNumber(random.nextInt(10000) + "");
117 }
118 return randomAddress.toString();
119 }
120
121 if (this.randomizeNumber) {
122 if (this.getPseudorandom) {
123 String number = randomAddress.getNumber();
124 randomAddress.setNumber(Long.toString(Math.abs(HashUtils.longFromHash(number)) % 300));
125 } else {
126 randomAddress.setNumber(this.random.nextInt(300) + "");
127 }
128 }
129
130 if (this.randomizeCity) {
131 //psuedorandom is embedded into the provider itself, we have set the configuration accordingly
132 randomAddress.setCityOrState(cityMaskingProvider.mask(address.getCityOrState()));
133 }
134
135 if (this.randomizeCountry) {
136 //psuedorandom is embedded into the provider itself, we have set the configuration accordingly
137 randomAddress.setCountry(countryMaskingProvider.mask(address.getCountry()));
138 }
139
140 if (this.randomizeName) {
141 if (this.getPseudorandom) {
142 String sname = randomAddress.getName();
143 randomAddress.setName(streetNameManager.getPseudorandom(sname));
144 } else {
145 randomAddress.setName(streetNameManager.getRandomKey());
146 }
147 }
148
149 if (this.randomizeRoadType) {
150 RoadTypes[] roadTypes = RoadTypes.values();
151 int randomPosition;
152
153 if (this.getPseudorandom) {
154 randomPosition = (int) (Math.abs(HashUtils.longFromHash(identifier)) % roadTypes.length);
155 } else {
156 randomPosition = random.nextInt(roadTypes.length);
157 }
158
159 String randomRoadType = roadTypes[randomPosition].name();
160 randomAddress.setRoadType(randomRoadType);
161 }
162
163 if (this.randomizePostalCode) {
164
165 if (this.getPseudorandom) {
166 String postalCode = address.getPostalCode();
167 randomAddress.setPostalCode(postalCodeManager.getPseudorandom(postalCode));
168 } else if (this.nearestPostalCode) {
169 String postalCode = address.getPostalCode();
170 randomAddress.setPostalCode(postalCodeManager.getClosestPostalCode(postalCode, this.nearestPostalCodeK));
171 } else {
172 randomAddress.setPostalCode(postalCodeManager.getRandomKey());
173 }
174 }
175
176 return randomAddress.toString();
177 }

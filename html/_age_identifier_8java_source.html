141 public Age parseAge(String identifier) {
142 AgePortion yearPortion = MISSING_AGE_PORTION;
143 AgePortion monthPortion = MISSING_AGE_PORTION;
144 AgePortion weekPortion = MISSING_AGE_PORTION;
145 AgePortion daysPortion = MISSING_AGE_PORTION;
146
147 for (Pattern pattern : getPatterns()) {
148 Matcher matcher = pattern.matcher(identifier);
149 if (matcher.matches()) {
150 try {
151 int start = matcher.start("year");
152 int end = matcher.end("year");
153 yearPortion = new AgePortion(true, start, end, AgePortionFormat.NUMERICAL);
154 } catch (IllegalArgumentException e) {
155 }
156
157 try {
158 int start = matcher.start("month");

191 private Age tryWordParse(String identifier) {
192 AgePortion yearPortion = MISSING_AGE_PORTION;
193 AgePortion monthPortion = MISSING_AGE_PORTION;
194 AgePortion weekPortion = MISSING_AGE_PORTION;
195 AgePortion daysPortion = MISSING_AGE_PORTION;
196
197 String input = identifier.toLowerCase();
198
199 if ((input.startsWith("on his") || input.startsWith("on her")) && input.endsWith("birthday")) {
200 String middlePart = input.substring("on his".length(), input.length() - "birthday".length());
201 Long number = NumberUtils.createNumber(middlePart);
202
203 if (number != null) {
204 AgePortion portion = new AgePortion(true, "on his".length(), input.length() - "birthday".length(), AgePortionFormat.WORDS);
205 return new Age(portion, monthPortion, weekPortion, daysPortion);
206 }
207 }
208
209 for (String suffix : suffixes) {
210 if (input.endsWith(suffix)) {
211 input = input.substring(0, input.length() - suffix.length());
212 if (NumberUtils.createNumber(input) != null) {
213 AgePortion portion = new AgePortion(true, 0, input.trim().length(), AgePortionFormat.WORDS);
214
215 if (suffix.contains("years") || suffix.contains("yrs")) {
216 yearPortion = portion;
217 } else if (suffix.contains("months")) {
218 monthPortion = portion;
219 } else if (suffix.contains("days")) {
220 daysPortion = portion;
221 } else if (suffix.contains("weeks") || suffix.contains("wks")) {
222 weekPortion = portion;
223 }
224
225 return new Age(yearPortion, monthPortion, weekPortion, daysPortion);
226 }
227 }
228 }
229
230 return null;
231 }

259 @Override
260 public Tuple<Boolean, Tuple<Integer, Integer>> isOfThisTypeWithOffset(String identifier) {
261
262 identifier = identifier.toLowerCase();
263
264 if (this.matches(identifier)) {
265 return new Tuple<>(true, new Tuple<>(0, identifier.length()));
266 }
267
268 Tuple<Boolean, Tuple<Integer, Integer>> tryAuxResults = checkAuxiliaryPatterns(identifier);
269 if (tryAuxResults.getFirst()) {
270 return tryAuxResults;
271 }
272
273 if (tryWordMatch(identifier)) {
274 return new Tuple<>(true, new Tuple<>(0, identifier.length()));
275 }
276
277 return new Tuple<>(false, null);
278 }

280 private Tuple<Boolean, Tuple<Integer, Integer>> checkAuxiliaryPatterns(String identifier) {

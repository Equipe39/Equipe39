63 protected String mask(String identifier, Age age) {
64 AgePortion[] portions = new AgePortion[]{
65 age.getYearPortion(),
66 age.getMonthPortion(),
67 age.getWeeksPortion(),
68 age.getDaysPortion()
69 };
70
71 int[] upperBound = new int[]{100, 12, 48, 365};
72 int[] lowerBound = new int[]{1, 1, 0, 0};
73
74 StringBuilder builder = new StringBuilder();
75 int lastEnd = 0;
76
77 for (int i = 0; i < portions.length; i++) {
78 AgePortion portion = portions[i];
79 if (!portion.exists()) {
80 continue;
81 }
82
83 int start = portion.getStart();
84
85 builder.append(identifier, lastEnd, start);
86
87 if (this.redactNumbers) {
88 builder.append("XX");
89 } else if (this.randomNumbers) {
90 String randomYear = generateRandomNumber(lowerBound[i], upperBound[i], portion.getFormat());
91 builder.append(randomYear);
92 }
93
94 lastEnd = portion.getEnd();
95 }
96
97 builder.append(identifier.substring(lastEnd));
98
99 return builder.toString();
100 }
101
102 private String generateRandomNumber(int lower, int upper, AgePortionFormat format) {
103 int random = lower + this.random.nextInt(upper - lower);
104
105 if (format == AgePortionFormat.NUMERICAL) {
106 return "" + random;
107 }
108
109 return NumberUtils.createWords(random);
110 }

112 @Override
113 public String mask(String identifier) {
114 Age age = AGE_IDENTIFIER.parseAge(identifier);
115
116 if (age == null) {
117 switch (failMode) {
118 case FailMode.THROW_ERROR:
119 log.error("Invalid age");
120 throw new IllegalArgumentException("Invalid age");
121 case FailMode.RETURN_ORIGINAL:
122 return identifier;
123 case FailMode.RETURN_EMPTY:
124 default:
125 return "";
126 }
127 }
128
129 return mask(identifier, age);
130 }

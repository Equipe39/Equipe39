142 public static List<Partition> generatePartitionsByColumnIndex(IPVDataset anonymized, List<Integer> quasiColumns) {
143 Map<String, List<List<String>>> partitionsMap = new HashMap<>();
144 int n = anonymized.getNumberOfRows();
145
146 for (int i = 0; i < n; i++) {
147 List<String> row = anonymized.getRow(i);
148
149 StringBuilder builder = new StringBuilder();
150 try {
151 for (Integer quasiColumn : quasiColumns) {
152 int columnIndex = quasiColumn;
153 String value = row.get(columnIndex);
154 builder.append(value);
155 builder.append(":");
156 }
157 } catch (ArrayIndexOutOfBoundsException e) {
158 logger.error("Error accessing row {}", row);
159 throw e;
160 }
161
162 String key = builder.toString();
163 List<List<String>> member = partitionsMap.get(key);
164
165 if (member == null) {
166 partitionsMap.put(key, new ArrayList<>());
167 member = partitionsMap.get(key);
168 }
169
170 member.add(row);
171 }
172
173 List<Partition> partitions = new ArrayList<>();
174
175 for (List<List<String>> l : partitionsMap.values()) {
176 partitions.add(new InMemoryPartition(l));
177 }
178
179 return partitions;
180 }

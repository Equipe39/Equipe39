41 public Integer matchAnonymizedRow(List<String> anonymizedRow, Collection<LinkInfo> linkInformation, List<ColumnInformation> columnInformations) {
42 List<List<Set<Integer>>> matching = new ArrayList<>(linkInformation.size());
43
44 for (LinkInfo info : linkInformation) {
45 List<Set<Integer>> infoMatching = new ArrayList<>();
46
47 int sourceIndex = info.getSourceIndex();
48 int targetIndex = info.getTargetIndex();
49
50 String value = anonymizedRow.get(sourceIndex);
51
52 ColumnInformation columnInformation = columnInformations.get(sourceIndex);
53 if (columnInformation instanceof NumericalRange) {
54 String[] tokens = value.split("-");
55 Set<Integer> matchedRows;
56
57 if (tokens.length == 1) {
58 Double numericValue = Double.parseDouble(value);
59 matchedRows = datasetLinker.matchValue(numericValue, targetIndex);
60 } else {
61 Double minValue = Double.parseDouble(tokens[0]);
62 Double maxValue = Double.parseDouble(tokens[1]);
63 matchedRows = datasetLinker.matchValueRange(minValue, maxValue, targetIndex);
64 }
65
66 if (matchedRows == null) {
67 return 0;
68 } else {
69 infoMatching.add(matchedRows);
70 }
71 } else {
72 if (value.equals(info.getWildcharPattern())) {
73 continue;
74 }
75
76 if (!(columnInformation instanceof CategoricalInformation)) {
77 Set<Integer> matchedRows = datasetLinker.matchValue(value, targetIndex, info.isPrefixMatch());
78 if (matchedRows == null) {
79 return 0;
80 } else {
81 infoMatching.add(matchedRows);
82 }
83 } else {
84 CategoricalInformation categoricalInformation = (CategoricalInformation) columnInformation;
85 int level = categoricalInformation.getHierarchy().getNodeLevel(value);
86
87 if (level <= 0) {
88 Set<Integer> matched = datasetLinker.matchValue(value, targetIndex, false);
89 if (matched == null) {
90 return 0;
91 }
92 infoMatching.add(matched);
93 } else {
94 Set<String> leaves = categoricalInformation.getHierarchy().getNodeLeaves(value);
95
96 if (leaves == null) {
97 logger.info("null leaves for >" + value + "< source index is " + sourceIndex);
98 }
99
100 for (String leaf : Objects.requireNonNull(leaves)) {
101 Set<Integer> leafMatches = datasetLinker.matchValue(leaf, targetIndex, false);
102
103 if (leafMatches != null) {
104 infoMatching.add(leafMatches);
105 }
106 }
107
108 if (infoMatching.isEmpty()) {
109 return 0;
110 }
111 }
112 }
113 }
114
115 matching.add(infoMatching);
116 }
117
118 if (matching.isEmpty()) {
119 return datasetLinker.getNumberOfRows();
120 }
121
122 return intersectAll(matching);
123 }

125 private <T> Integer intersectAll(List<List<Set<T>>> setGroups) {
126 if (setGroups.size() == 1) {
127 int size = 0;
128
129 for (Set<T> setGroup : setGroups.get(0)) {
130 size += setGroup.size();
131 }
132
133 return size;
134 }
135
136 int result = 0;
137
138 setGroups.sort(new Comparator<List<Set<T>>>() {
139 private int sizeOf(List<Set<T>> sets) {
140 int size = 0;
141
142 for (Set<T> set : sets) {
143 size += set.size();
144 }
145
146 return size;
147 }
148
149 @Override
150 public int compare(List<Set<T>> o1, List<Set<T>> o2) {
151 return Integer.compare(sizeOf(o1), sizeOf(o2));
152 }
153 });
154
155 for (Set<T> setGroup : setGroups.get(0)) {
156 outer:
157 for (T entry : setGroup) {
158 inner:
159 for (int i = 1; i < setGroups.size(); ++i) {
160 for (Set<T> setGroupI : setGroups.get(i)) {
161 if (setGroupI.contains(entry)) {
162 continue inner;
163 }
164 }
165 continue outer;
166 }
167 result += 1;
168 }
169 }
170
171 return result;

174 public List<Integer> matchesPerRecord(IPVDataset source, Collection<LinkInfo> linkInformation) {
175 final List<Integer> matchResults = new ArrayList<>();

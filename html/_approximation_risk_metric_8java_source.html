97 private double calculateEQRisk(double fk, double pk) {
98 double k_risk = 1.0;
99 double qk = 1 - pk;
100
101 for (int i = 1; i <= 7; ++i) {
102 double v1 = factorial(i) * Math.pow(qk, i);
103
104 double p = 1.0;
105 for (int j = 1; j <= i; j++) {
106 p = p * (fk + j);
107 }
108
109 k_risk += v1 / p;
110 }
111
112 return (pk / fk) * k_risk;
113 }

115 public Double reportLocalP() {
116 double risk = 0.0;
117
118 for (int k = 0; k < F.size(); ++k) {
119 final int Fk = extractFk(k);
120 final double fk = f.get(k);
121
122 double pk = (double) f.get(k) / (double) Fk;
123
124 double k_risk = calculateEQRisk(fk, pk);
125
126 risk = FastMath.max(risk, k_risk);
127 }
128
129 return risk;
130 }
131
132 public double reportGlobalP() {
133 double risk = 0.0;
134
135 final double p = n / N;
136
137 for (int k = 0; k < F.size(); ++k) {
138 double k_risk = calculateEQRisk(f.get(k), p);
139 risk = FastMath.max(risk, k_risk);
140 }
141
142 return risk;
143 }

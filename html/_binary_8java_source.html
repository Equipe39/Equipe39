78 @Override
79 public String randomise(String value) {
80 boolean boolValue = value.equals(this.binaryValues.getFirst());
81 double u = rnd.nextDouble();
82
83 boolean noisyBoolValue;
84
85 if (u <= 1/(1 + Math.exp(this.epsilon))) {
86 noisyBoolValue = !boolValue;
87 } else {
88 noisyBoolValue = boolValue;
89 }
90
91 return (noisyBoolValue) ? this.binaryValues.getFirst() : this.binaryValues.getSecond();

51 @Override
52 public Collection<IPVVulnerability> apply(IPVDataset dataset) {
53
54 Collection<ItemSet> quasiIdentifiers = new ArrayList<>();
55
56 IPVGenerator generator = new LayerGenerator(dataset.getNumberOfColumns());
57
58 while (generator.hasNext()) {
59 ItemSet itemSet = generator.next();
60
61 if (this.maxLength > 0 && itemSet.size() > maxLength) {
62 break;
63 }
64
65 boolean needsToBeChecked = true;
66 for (ItemSet qi : quasiIdentifiers) {
67 if (qi.isSubSetOf(itemSet)) {
68 needsToBeChecked = false;
69 break;
70 }
71 }
72
73 if (needsToBeChecked) {
74 if (VerificationUtils.isQuasiIdentifier(itemSet, dataset, this.k)) {
75 quasiIdentifiers.add(itemSet);
76 }
77 } else if (!prune) {
78 quasiIdentifiers.add(itemSet);
79 }
80
81 }
82
83 Collection<IPVVulnerability> vulnerabilities = new ArrayList<>();
84 for (ItemSet itemSet : quasiIdentifiers) {
85 vulnerabilities.add(new IPVVulnerability(itemSet));
86 }
87
88 return vulnerabilities;
89
90 }

117 public static Set<Integer> buildExcludedFieldsList(String excludedFieldsString) {
118 if (Objects.isNull(excludedFieldsString) || excludedFieldsString.isEmpty()) {
119 return Collections.emptySet();
120 }
121
122 return Arrays.stream(excludedFieldsString.split(",")).
123 map(Integer::parseInt).
124 collect(Collectors.toSet());
125 }

127 public static Collection<IPVVulnerability> findVulnerabilities(Dataset<Row> dataset, Set<Integer> excludedFields, int k, int maxSize) {
128 final String[] columnNames = dataset.columns();
129 final LayerGenerator generator = new LayerGenerator(columnNames.length);
130
131 while (generator.hasNext()) {
132 final ItemSet itemSet = generator.next();
133
134 if (hasReachedMaxSize(maxSize, itemSet)) {
135 logger.info("Max size reached");
136 break;
137 }
138
139 logger.info("Testing" + itemSet);
140
141 if (containsExcludedField(itemSet, excludedFields)) {
142 logger.info("Skipping " + itemSet + " because contains excluded fields");
143 continue;
144 }
145
146 final Column[] selectionColumns = computeSelectColumn(itemSet, columnNames);
147
148 if (isQuasiIdentifier(dataset, selectionColumns, k)) {
149 logger.info("Banning " + itemSet);
150 generator.ban(itemSet);
151 }
152 }
153
154 return generator.getBanned().parallelStream().
155 map( itemSet -> new IPVVulnerability(itemSet, k)).
156 collect(Collectors.toList());
157 }
158
159 private static boolean hasReachedMaxSize(int maxSize, ItemSet itemSet) {
160 return maxSize > 0 && itemSet.size() > maxSize;
161 }
162
163 private static boolean isQuasiIdentifier(Dataset<Row> dataset, Column[] selectionColumns, int k) {
164 Dataset<Row> selection = dataset.
165 groupBy(selectionColumns).
166 agg(count(lit(1)).as(AGGREGATED_COUNT_NAME)).
167 filter(col(AGGREGATED_COUNT_NAME).lt(lit(k)));
168
169 return selection.count() > 0;
170 }
171
172 private static Column[] computeSelectColumn(ItemSet itemSet, String[] columnNames) {
173 final Column[] selectColumns = new Column[itemSet.size()];
174
175 int i = 0;
176 for (int item : itemSet.getItems()) {
177 selectColumns[i++] = col(columnNames[item]);
178 }
179
180 return selectColumns;
181 }
182
183 private static boolean containsExcludedField(ItemSet itemSet, Set<Integer> excludedFields) {
184 for (final int item : itemSet.getItems()) {
185 if (excludedFields.contains(item)) return true;
186 }
187
188 return false;
189 }

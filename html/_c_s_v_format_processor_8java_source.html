48 @Override
49 protected Iterable<Record> extractRecords(InputStream dataset, DatasetOptions datasetOptions, int firstN) {
50 if (!(datasetOptions instanceof CSVDatasetOptions))
51 throw new IllegalArgumentException("Dataset masking options not consistent with the format processor: CSV");
52
53 final CSVDatasetOptions csvOptions = (CSVDatasetOptions) datasetOptions;
54
55 final MappingIterator<String[]> reader;
56 try {
57 reader = mapper.readerFor(String[].class)
58 .with(
59 CsvSchema.emptySchema()
60 .withColumnSeparator(csvOptions.getFieldDelimiter())
61 .withQuoteChar(csvOptions.getQuoteChar())
62 )
63 .readValues(dataset);
64 } catch (IOException e) {
65 logger.error("Unable to read values from input stream", e);
66
67 throw new RuntimeException(e);
68 }
69
70 return () -> new Iterator<>() {
71 int seenSoFar;
72 int readSoFar;
73
74 Map<String, Integer> fields = null;
75
76 CSVRecord nextRecord = null;
77
78 @Override
79 public boolean hasNext() {
80 while (null == nextRecord && reader.hasNext()) {
81 if (firstN > 0 && readSoFar >= firstN) {
82 return false;
83 }
84
85 String[] values = processValues(reader.next(), csvOptions.isTrimFields());
86
87 seenSoFar += 1;
88
89 if (0 == values.length) {
90 logger.debug("Skipping record {} because it contains no fields (corrupt row?)", seenSoFar);
91 continue;
92 }
93 if (fields == null && 1 == values.length && values[0].trim().equals("")) {
94 logger.debug("Skipping record {} because it contains only one empty field (corrupt row?)", seenSoFar);
95 continue;
96 }
97 if (fields != null && fields.size() != values.length) {
98 logger.debug("Skipping record {} because it contains {} field(s), but schema contains {} header(s) (corrupt row?)", seenSoFar, values.length, fields.size());
99 continue;
100 }
101
102 readSoFar += 1;
103
104 if (null == fields) { // is first read?
105 if (csvOptions.isHasHeader()) {
106 fields = buildFieldsMap(values);
107 readSoFar -= 1; // header does not count as record
108 } else {
109 fields = buildFieldsMap(generateColumnNames(values.length));
110 }
111 nextRecord = new CSVRecord(values, fields, csvOptions, csvOptions.isHasHeader());
112 } else {
113 nextRecord = new CSVRecord(values, fields, csvOptions, false);
114 }
115 }
116
117 return nextRecord != null;
118 }

120 private String[] processValues(String[] values, boolean trimFields) {
121 if (trimFields) {
122 for (int i = 0; i < values.length; ++i) {
123 values[i] = values[i].trim();

125 }
126 return values;

129 @Override

166 @Override
167 public Map<IPVVulnerability, List<Integer>> identifyVulnerabilitiesStream(InputStream input, IPVAlgorithm algorithm, DataTypeFormat inputFormatType, DatasetOptions datasetOptions, boolean isFullReport, int kValue) throws IOException {
168 CSVDatasetOptions csvDatasetOptions = (CSVDatasetOptions) datasetOptions;
169 IPVDataset dataset = IPVDataset.load(input, csvDatasetOptions.isHasHeader(), csvDatasetOptions.getFieldDelimiter(), csvDatasetOptions.getQuoteChar(), csvDatasetOptions.isTrimFields());
170
171 Collection<IPVVulnerability> vulnerabilities = algorithm.apply(dataset);
172
173 Map<IPVVulnerability, List<Integer>> results = new HashMap<>();
174
175 if (isFullReport && algorithm instanceof WithRowExtractor) {
176 for (IPVVulnerability vulnerability : vulnerabilities) {
177 Integer[] rowIDs = extractRowIds(vulnerability.getItemSet(), kValue, csvDatasetOptions.isHasHeader(), dataset);
178 results.put(vulnerability, Arrays.asList(rowIDs));
179 }
180 } else {
181 for (IPVVulnerability vulnerability : vulnerabilities) {
182 results.put(vulnerability, null);
183 }
184 }
185
186 return results;
187 }

189 private static Set<Integer> joinRowSets(Map<String, Set<Integer>> valueMap, int k) {
190 Set<Integer> jointRowSets = new LinkedHashSet<>();
191
192 for (Set<Integer> value : valueMap.values()) {
193 if (value.size() < k) {
194 jointRowSets.addAll(value);
195 }
196 }
197
198 return jointRowSets;
199 }

201 private static Integer[] extractRowIds(ItemSet itemSet, int k, boolean hasHeader, IPVDataset dataset) {
202 Integer[] rows = joinRowSets(VerificationUtils.buildValueMap(itemSet, dataset), k).toArray(new Integer[0]);
203 Arrays.sort(rows);
204
205 if (hasHeader) {
206 for (int i = 0; i < rows.length; i++) {
207 rows[i]++;
208 }
209 }
210
211 return rows;

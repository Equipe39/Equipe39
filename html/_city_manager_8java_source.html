90 private static double cityDistance(City city1, City city2) {
91 return Math.sqrt(
92 Math.pow((city1.getLocation().getLatitude() - city2.getLocation().getLatitude()), 2) +
93 Math.pow((city1.getLocation().getLongitude() - city2.getLocation().getLongitude()), 2));

149 // NOTE: benchmarked before fix for cartesian. In repeat testing, consistently faster than Nearest v2.
150 // Benchmark times: 6022, 5810, 6256
151 private void precomputeNearest1() {
152 for (String key : cityListMap.keySet()) {
153 final List<City> cityList = cityListMap.get(key);
154 final List<KDTree.CartesianPoint> cityListPoints = cityList.stream().map(
155 (c) -> new KDTree.CartesianPoint(c.getLocation().getLatitude(), c.getLocation().getLongitude())).collect(Collectors.toList());
156
157 for (int i = 0; i < cityList.size(); i++) {
158 final City city = cityList.get(i);
159 final KDTree.CartesianPoint cityPoint = cityListPoints.get(i);
160
161 final List<CityDistancer> otherCities = new ArrayList<>(cityList.size());
162 for (int j = 0; j < cityList.size(); j++) {
163 // Original includes self as well as other cities... not filtering self
164 otherCities.add(new CityDistancer(cityPoint, cityList.get(j), cityListPoints.get(j)));
165 }
166 Collections.sort(otherCities);
167
168 city.setNeighbors(otherCities.stream().map(CityDistancer::getCity).collect(Collectors.toList()));
169 }
170 }
171 }

194 public String getClosestCity(String city, int k) {
195 String key = city.toUpperCase();
196 City lookup = getKey(key);
197
198 if (lookup == null) {
199 return getRandomKey();
200 }
201
202 List<City> neighbors = lookup.getNeighbors();
203 if (neighbors == null) {
204 return getRandomKey(lookup.getNameCountryCode());
205 }
206
207 if (k > neighbors.size()) {
208 k = neighbors.size();
209 }
210
211 return (neighbors.get(random.nextInt(k))).getName();

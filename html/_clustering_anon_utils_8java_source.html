43 public static String calculateCommonAncestor(Set<String> values, GeneralizationHierarchy hierarchy) {
44
45 if (values.size() == 1) {
46 return values.iterator().next().toUpperCase();
47 }
48
49 List<Map<String, Integer>> counters = new ArrayList<>();
50 for (int i = 0; i < hierarchy.getHeight(); i++) {
51 counters.add(new HashMap<>());
52 }
53
54 for (String value : values) {
55 int currentLevel = hierarchy.getNodeLevel(value);
56
57 if (currentLevel == -1) {
58 break;
59 }
60
61 for (int i = currentLevel; i < hierarchy.getHeight(); i++) {
62 String ancestor = hierarchy.encode(value, i - currentLevel, true);
63
64 Integer counter = counters.get(i).get(ancestor);
65 if (counter == null) {
66 counter = 0;
67 }
68
69 counters.get(i).put(ancestor, counter + 1);
70 }
71 }
72
73 for (Map<String, Integer> levelMap : counters) {
74 for (Map.Entry<String, Integer> entry : levelMap.entrySet()) {
75 Integer counter = entry.getValue();
76 if (counter == values.size()) {
77 return entry.getKey();
78 }
79 }
80 }
81
82 return hierarchy.getTopTerm();
83
84 }

86 public static List<List<String>> calculateCategoricalCentroids(
87 List<List<Set<String>>> categoricalVariables, List<ColumnInformation> columnInformationList) {
88
89 List<List<String>> centroids = new ArrayList<>();
90
91 for (List<Set<String>> clusterList : categoricalVariables) {
92 List<String> clusterCentroids = new ArrayList<>();
93
94 for (int i = 0; i < clusterList.size(); i++) {
95 Set<String> values = clusterList.get(i);
96 ColumnInformation columnInformation = columnInformationList.get(i);
97
98 if (values.size() == 0 || !columnInformation.isCategorical()) {
99 clusterCentroids.add(null);
100 } else {
101 GeneralizationHierarchy hierarchy = ((CategoricalInformation) columnInformation).getHierarchy();
102 clusterCentroids.add(calculateCommonAncestor(values, hierarchy));
103 }
104
105 }
106
107 centroids.add(clusterCentroids);
108 }
109
110
111 return centroids;
112 }

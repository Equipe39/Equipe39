103 public static Dataset<Row> computeConfusionMatrix(final Dataset<Row> dataset, List<OutlierRemovalFilter> filters) {
104 final Column[] selectionColumns = new Column[filters.size()];
105
106 Dataset<Row> operationalDataset = dataset;
107
108 int i = 0;
109 for (OutlierRemovalFilter filter : filters) {
110 String newColumn = UUID.randomUUID().toString();
111 selectionColumns[i++] = col(newColumn);
112
113 operationalDataset = augmentDatasetWithFilter(operationalDataset, filter, newColumn);
114 }
115
116 return operationalDataset.select(selectionColumns);
117 }

119 private static Dataset<Row> augmentDatasetWithFilter(Dataset<Row> dataset, OutlierRemovalFilter filter, String columnName) {
120 logger.info("Augmenting dataset with filter: " + filter);
121
122 List<ThresholdCondition> thresholds = filter.getThresholds();
123 if (thresholds.size() != 1) {
124 throw new UnsupportedOperationException("Each filter supports only one threshold in this installment");
125 }
126
127 if (filter.getIdentityFields().isEmpty()) {
128 return dataset.withColumn(columnName, dataset.col(thresholds.get(0).getColumnName()));
129 } else {
130 ThresholdCondition threshold = thresholds.get(0);
131 List<String> identityFields = filter.getIdentityFields();
132
133 RelationalGroupedDataset groupedDataset = dataset.groupBy(buildAggregationCondition(dataset, identityFields));
134
135 Dataset<Row> aggregatedDataset = groupedDataset.agg(buildAggregation(threshold).as(columnName));
136
137 return dataset.join(
138 aggregatedDataset,
139 buildJoinExpression(dataset, aggregatedDataset, identityFields),
140 "leftouter"
141 ).select(buildSelection(dataset, columnName));
142 }
143 }

152

158 private Tuple2<Dataset<Row>, Column> addAggregationConstraints(List<OutlierRemovalFilter> withAggregation, Dataset<Row> dataset) {
159 Column outlierCondition = null;
160 Dataset<Row> operationalDataset = dataset;
161
162 for (List<OutlierRemovalFilter> filtersWithSameAggregation : groupByIdentityFields(withAggregation)) {
163 List<String> identityFields = filtersWithSameAggregation.get(0).getIdentityFields();
164 RelationalGroupedDataset groupedDataset = dataset.groupBy(buildAggregationCondition(dataset, identityFields));
165
166 List<Column> aggregationExpressions = new ArrayList<>();
167
168 for (OutlierRemovalFilter filter : filtersWithSameAggregation) {
169 Tuple2<List<Column>, Column> response = addAggregationFilter(filter);
170
171 aggregationExpressions.addAll(response._1);
172
173 if (outlierCondition == null) {
174 outlierCondition = response._2;
175 } else {
176 outlierCondition = outlierCondition.or(response._2);
177 }
178 }
179
180 Dataset<Row> aggregatedDataset = groupedDataset.agg(aggregationExpressions.get(0), aggregationExpressions.subList(1, aggregationExpressions.size()).toArray(new Column[0]));
181 operationalDataset = operationalDataset.join(
182 aggregatedDataset,
183 buildJoinExpression(dataset, aggregatedDataset, identityFields),
184 "leftouter"
185 );
186 }
187
188 return new Tuple2<>(operationalDataset, outlierCondition);
189 }

191 private Collection<? extends List<OutlierRemovalFilter>> groupByIdentityFields(List<OutlierRemovalFilter> filters) {
192 Map<List<String>, List<OutlierRemovalFilter>> groups = new HashMap<>();
193
194 for (OutlierRemovalFilter filter : filters) {
195 List<String> ids = filter.getIdentityFields();
196 if (groups.containsKey(ids)) {
197 groups.get(ids).add(filter);
198 } else {
199 List<OutlierRemovalFilter> group = new ArrayList<>();
200 group.add(filter);
201 groups.put(ids, group);
202 }
203 }
204
205 return groups.values();

208 private Tuple2<List<Column>, Column> addAggregationFilter(OutlierRemovalFilter filter) {
209 Column filterCondition = null;
210 List<Column> aggregationCondition = new ArrayList<>();
211
212 for (ThresholdCondition threshold : filter.getThresholds()) {
213 String aggregationName = UUID.randomUUID().toString();
214
215 // update aggregation
216 aggregationCondition.add(buildAggregation(threshold).as(aggregationName));
217
218 // update filter
219 Column filterColumn = buildFilterCondition(threshold, aggregationName);
220
221 if (threshold.isTrueOnNull()) {
222 filterColumn = col(aggregationName).isNull().or(filterColumn);
223 } else {
224 filterColumn = col(aggregationName).isNotNull().and(filterColumn);
225 }
226
227 if (filterCondition == null) {
228 filterCondition = filterColumn;
229 } else {
230 filterCondition = filterCondition.and(filterColumn);
231 }
232 }
233
234 return new Tuple2<>(aggregationCondition, filterCondition);
235 }

262 Column joinExpression = null;

276 private Column buildFilterCondition(ThresholdCondition condition, String aggregationName) {
277 if (null == condition.getCondition()) return null;
278
279 Column column = col(aggregationName);
280 switch (condition.getCondition()) {
281 case LT:
282 return column.lt(condition.getValue());
283 case LEQ:
284 return column.leq(condition.getValue());
285 case GT:
286 return column.gt(condition.getValue());
287 case GEQ:
288 return column.geq(condition.getValue());
289 case EQ:
290 return column.equalTo(condition.getValue());
291 case NEQ:
292 return column.notEqual(condition.getValue());
293 default:
294 throw new IllegalArgumentException("Unknown condition");
295 }
296 }

306 private Column buildSimpleFiltering(Dataset<Row> dataset, List<ThresholdCondition> conditions) {
307 Column filter = null;
308
309 for (ThresholdCondition condition : conditions) {
310 Column original = dataset.col(condition.getColumnName());
311 Column column = original;
312
313 switch (condition.getCondition()) {
314 case LT:
315 column = column.lt(condition.getValue());
316 break;
317 case LEQ:
318 column = column.leq(condition.getValue());
319 break;
320 case GT:
321 column = column.gt(condition.getValue());
322 break;
323 case GEQ:
324 column = column.geq(condition.getValue());
325 break;
326 case EQ:
327 column = column.equalTo(condition.getValue());
328 break;
329 case NEQ:
330 column = column.notEqual(condition.getValue());
331 break;
332 }
333
334 if (condition.isTrueOnNull()) {
335 column = original.isNull().or(column);
336 } else {
337 column = original.isNotNull().and(column);
338 }
339
340 if (null == filter) {
341 filter = column;
342 } else {
343 filter = filter.and(column);
344 }
345 }
346
347 return filter;
348 }

357 private static Column buildAggregation(ThresholdCondition condition) {
358 String columnName = condition.getColumnName();
359 Column column = col(columnName);
360
361 boolean properAggregation = false;
362
363 switch (condition.getAggregationType()) {
364 case SUM:
365 column = sum(column);
366 properAggregation = true;
367 break;
368 case COUNT:
369 column = sum(lit(1L));
370 properAggregation = true;
371 break;
372 case MAX:
373 column = max(column);
374 properAggregation = true;
375 break;
376 case MIN:
377 column = min(column);
378 properAggregation = true;
379 break;
380 case NOT_AGGREGATED:
381 break;
382 default:
383 throw new IllegalArgumentException("Unknown aggregation type");
384 }
385
386 if (! properAggregation) throw new IllegalArgumentException("No aggregation values specified");
387
388 return column;
389 }

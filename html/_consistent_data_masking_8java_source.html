110 public static String mergeRecordObjectsByPath(Iterable<Tuple2<String, String>> maskedRecords,
111 DataTypeFormat inputFormatType, DatasetOptions datasetOptions, Map<String, Integer> fieldPaths)
112 throws IOException {
113
114 Record firstNode = null;
115

118 String path = t._2();

120 Record record = RecordFactory.parseString(r, inputFormatType, datasetOptions, fieldPaths, false);

122 if (firstNode == null) {
123 firstNode = record;
124 } else {
125 byte[] updatedValue = record.getFieldValue(path);

158 public static Dataset<Row> doGloballyConsistentMasking(Dataset<Row> dataset, Map<String, DataMaskingTarget> identifiedTypes,
159 final Map<String, MaskingProvider> maskingProviders,
160 final Set<String> fieldsInNamespace,
161 final Map<String, Integer> fieldPaths, final DataMaskingOptions maskingOptions) {
162 JavaRDD<String> maskedRdd = dataset.javaRDD().zipWithIndex()
163 .flatMap(s -> projectionMapForGlobalConsistency(s, fieldsInNamespace, maskingOptions).iterator())
164 .groupBy(s -> groupByPathForGlobalConsistency(s, maskingOptions, fieldPaths))
165 .map(s -> {
166 String key = s._1();
167 Iterable<Tuple3<Row, String, Long>> records = s._2();
168 String maskedKey = null;
169 List<Tuple2<Long, Tuple2<String, String>>> projectedRecords = new ArrayList<>();
170 for (Tuple3<Row, String, Long> r : records) {
171 Row record = r._1();
172 String path = r._2();
173 Long rowIndex = r._3();
174 // we mask the key based on the first field of the namespace
175 if (maskedKey == null) {
176 maskedKey = maskingProviders.get(path).mask(key);
177 }
178 String updatedRecord = updateKeyToRecord(maskedKey, record, maskingOptions, path, fieldPaths);
179 Tuple2<String, String> maskedRecord = new Tuple2<>(updatedRecord, path);
180 Tuple2<Long, Tuple2<String, String>> projectedRecord = new Tuple2<>(rowIndex, maskedRecord);
181 projectedRecords.add(projectedRecord);
182 }
183 return projectedRecords;
184 })
185 .flatMapToPair(List::iterator)
186 .groupByKey()
187 .map(longIterableTuple2 -> mergeMaskedPaths(longIterableTuple2, fieldsInNamespace, maskingOptions, fieldPaths));
188
189 JavaRDD<Row> rowRdd = maskedRdd.map(row -> ((CSVRecord) CSVRecord.fromString(row, maskingOptions.getDatasetOptions(), fieldPaths, false)))
190 .map(csvRecord -> {
191 List<Object> orderedValues = new ArrayList<>();
192
193 for (StructField field : dataset.schema().fields()) {
194 orderedValues.add(
195 csvRecord.getFieldValue(field.name())
196 );
197 }
198
199 return JavaConverters.asScalaIteratorConverter(orderedValues.iterator()).asScala().toSeq();
200 })
201 .map(Row::fromSeq);
202
203 return dataset.sparkSession().createDataFrame(rowRdd, dataset.schema()); // to be fixed!
204 }

206 public static Dataset<Row> doConsistentMasking(SparkSession sparkSession, Dataset<Row> dataset,
207 final Map<String, MaskingProvider> maskingProviders,
208 final String fieldName,
209 final Map<String, Integer> fieldPaths,
210 final DataMaskingOptions maskingOptions) {
211
212 JavaRDD<String> rdd = dataset.javaRDD().mapToPair(s -> createMapPairForSingleConsistency(s, maskingOptions, fieldName, fieldPaths))
213 .groupByKey()
214 .map((Function<Tuple2<String, Iterable<Row>>, List<String>>) tuple2 -> {
215 String key = tuple2._1();
216 Iterable<Row> records = tuple2._2();
217 return writeMaskedKeyForSingleConsistency(key, fieldName, maskingProviders, records, maskingOptions, fieldName, fieldPaths);
218 }).flatMap((FlatMapFunction<List<String>, String>) List::iterator);
219
220 return sparkSession.createDataset(rdd.rdd(), Encoders.STRING()).toDF();

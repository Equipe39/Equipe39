98 private Country[] getCountriesByDistance(final Country country, List<Location> countryList) {
99 Country[] countryArray = new Country[countryList.size()];
100 countryArray = countryList.toArray(countryArray);
101
102 Arrays.sort(countryArray, (o1, o2) -> {
103 double d1 = GeoUtils.latitudeLongitudeDistance(o1.getLocation(), country.getLocation());
104 double d2 = GeoUtils.latitudeLongitudeDistance(o2.getLocation(), country.getLocation());
105 return Double.compare(d1, d2);
106 });
107
108 return countryArray;
109 }

135 private String getPseudorandomElement(List<Location> keys, String key) {
136 long hash = Math.abs(HashUtils.longFromHash(key));
137
138 if (keys == null || keys.isEmpty()) {
139 return Long.toString(hash);
140 }
141
142 int position = (int) (hash % keys.size());
143 return ((Country) (keys.get(position))).getName();
144 }

171 private void readCountryList(Collection<ResourceEntry> entries) {
172 for (ResourceEntry entry : entries) {
173 try (InputStream inputStream = entry.createStream();
174 CSVParser reader = Readers.createCSVReaderFromStream(inputStream)) {
175 String locale = entry.getCountryCode();
176 for (CSVRecord record : reader) {
177 String countryName = record.get(0).toUpperCase();
178 String iso2Letter = record.get(1).toUpperCase();
179 String iso3Letter = record.get(2).toUpperCase();
180 String friendlyName = record.get(3).toUpperCase();
181 String continent = record.get(4);
182 Double latitude = Double.parseDouble(record.get(5));
183 Double longitude = Double.parseDouble(record.get(6));
184
185 /* TODO: temp fix until data is finished */
186 if (continent.equals("Unknown")) {
187 logger.info(String.format("Unknown continent for %s", countryName));
188 }
189
190 Country country = new Country(countryName, iso2Letter, iso3Letter, continent, latitude, longitude, locale);
191
192 addToCountryListMap(country, locale);
193 addToCountryListMap(country, getAllCountriesName());
194
195 addToCountryMap(country, countryName, CountryNameSpecification.NAME, locale);
196 addToCountryMap(country, iso2Letter, CountryNameSpecification.ISO2, locale);
197 addToCountryMap(country, iso3Letter, CountryNameSpecification.ISO3, locale);
198
199 addToCountryMap(country, countryName, CountryNameSpecification.NAME, getAllCountriesName());
200 addToCountryMap(country, iso2Letter, CountryNameSpecification.ISO2, getAllCountriesName());
201 addToCountryMap(country, iso3Letter, CountryNameSpecification.ISO3, getAllCountriesName());
202
203 if (!friendlyName.equals("")) {
204 addToCountryMap(country, friendlyName, CountryNameSpecification.NAME, locale);
205 addToCountryMap(country, friendlyName, CountryNameSpecification.NAME, getAllCountriesName());
206 }
207
208 }
209 } catch (IOException | NullPointerException e) {
210 logger.error("Error loading country list", e);
211 }
212 }

277 public String getClosestCountry(String originalCountry, int k) {
278
279 CountryNameSpecification spec = getSpecificationFromName(originalCountry);
280 if (spec == null) {
281 return getRandomKey(CountryNameSpecification.NAME);
282 }
283
284 MapWithRandomPick<String, Country> lookupMap = getSpecificationMap(spec);
285 if (lookupMap == null) {
286 return getRandomKey(spec);
287 }
288
289 String key = originalCountry.toUpperCase();
290 Country lookupResult = lookupMap.getMap().get(key);
291 if (lookupResult == null) {
292 return getRandomKey(spec);
293 }
294
295 Country[] neighbors = lookupResult.getNeighbors();
296 if (neighbors == null) {
297 return getRandomKey(spec);
298 }
299
300 if (k > neighbors.length) {
301 k = neighbors.length;
302 }
303
304 Country randomCountry = neighbors[random.nextInt(k)];
305 return randomCountry.getName(spec);
306 }

167 protected long computeOffendingRecords(Dataset<Row> dataset, List<String> vulnerability, Column k) {
168 Row[] rows = (Row[]) dataset.groupBy(buildGroupBy(vulnerability))
169 .agg(count(lit(1)).as(AGGREGATED_COUNT_NAME))
170 .filter(col(AGGREGATED_COUNT_NAME).lt(k))
171 .agg(sum(col(AGGREGATED_COUNT_NAME)).as(AGGREGATED_COUNT_NAME))
172 .collect();
173
174 Row actualRow = rows[0];
175 int idx = actualRow.fieldIndex(AGGREGATED_COUNT_NAME);
176
177 return actualRow.getLong(idx);
178 }

190 public Collection<List<String>> findVulnerabilities(final Dataset<Row> dataset, final Set<String> excludedFieldNames, final int k) {
191 int numberOfColumns = extractNumberOfColumns(dataset);
192
193 List<String> columnNames = Arrays.asList(dataset.columns());
194 Set<Integer> excludedFields = toIndex(columnNames, excludedFieldNames);
195 final List<Integer> uniques = uniqueColumns(dataset, k, excludedFields);
196 final List<Set<Integer>> knownNotSafe = new ArrayList<>();
197 final List<Set<Integer>> knownSafe = new ArrayList<>();
198
199 for (Integer uniq : uniques) {
200 knownNotSafe.add(Collections.singleton(uniq));
201 }
202
203 final List<Integer> validSeeds = findValidSeeds(numberOfColumns, uniques, excludedFields);
204
205 final Collection<Integer> attributesToCheck = new ArrayList<>(validSeeds);
206 for (final Integer validSeed : validSeeds) {
207 attributesToCheck.remove((Object) validSeed);
208 explorationPhase(dataset, Collections.singleton(validSeed), new ArrayList<>(attributesToCheck), 0, k, knownSafe, knownNotSafe);
209 }
210
211 return generatorFromNotSafe(knownNotSafe, numberOfColumns).getBanned().parallelStream().
212 map(itemSet -> {
213 List<String> vulnerableFields = new ArrayList<>(itemSet.size());
214
215 for (int item : itemSet.getItems()) {
216 vulnerableFields.add(columnNames.get(item));
217 }
218
219 return vulnerableFields;
220 }).collect(Collectors.toList());
221 }

272 private void explorationPhase(Dataset<Row> dataset, Set<Integer> baseItemset, Collection<Integer> availableAttributes, int currentLevel, int kValue, List<Set<Integer>> knownSafe, List<Set<Integer>> knownNotSafe) {
273 final Collection<Integer> remainingAttributes = new ArrayList<>(availableAttributes);
274 for (final Integer attribute : availableAttributes) {
275 final Set<Integer> itemSet = new HashSet<>(baseItemset);
276 itemSet.add(attribute);
277 remainingAttributes.remove((Object) attribute);
278
279 if (isKnownNotSafe(knownNotSafe, itemSet)) {
280// log.debug("{} is known not safe", itemSet);
281 continue;
282 }
283 if (isKnownSafe(knownSafe, itemSet)) {
284// log.debug("{} is knowwn safe", itemSet);
285 continue;
286 }
287
288 if (isNotUnique(dataset, kValue, toGroupByCondition(itemSet, dataset.columns()))) {
289 addWithMinimality(knownSafe, itemSet, this::testInLeft);
290 explorationPhase(dataset, itemSet, new ArrayList<>(remainingAttributes), currentLevel + 1, kValue, knownSafe, knownNotSafe);
291 } else {
292 addWithMinimality(knownNotSafe, itemSet, this::testInRight);
293 }
294 }
295 }

305 private void addWithMinimality(List<Set<Integer>> knownItemSets, Set<Integer> itemSet, BiPredicate<Set<Integer>, Set<Integer>> test) {
306 knownItemSets.removeIf(current -> test.test(current, itemSet));
307 knownItemSets.add(itemSet);
308 }

310 private boolean isKnownNotSafe(List<Set<Integer>> knownNotSafe, Set<Integer> itemSet) {
311 for (Set<Integer> known : knownNotSafe) {
312 if (testInRight(known, itemSet)) return true;
313 }
314 return false;
315 }

317 private boolean isKnownSafe(List<Set<Integer>> knownSafe, Set<Integer> itemSet) {
318 for (Set<Integer> known : knownSafe) {
319 if (testInLeft(known, itemSet)) return true;
320 }
321 return false;

328 private List<Integer> findValidSeeds(int numberOfColumns, List<Integer> uniques, Set<Integer> excludedFields) {
329 List<Integer> allColumns = new ArrayList<>(numberOfColumns - uniques.size() - excludedFields.size());
330
331 for (int columnId = 0; columnId < numberOfColumns; ++columnId) {
332 if (!uniques.contains(columnId) && !excludedFields.contains(columnId)) {
333 allColumns.add(columnId);
334 }
335 }
336
337 return allColumns;
338 }

340 private List<Integer> uniqueColumns(Dataset<Row> dataset, final int k, Set<Integer> excludedFields) {
341 String[] columnNames = dataset.columns();
342
343 List<Integer> columns = new ArrayList<>();
344
345 for (int columnId = 0; columnId < columnNames.length; ++columnId) {
346 if (excludedFields.contains(columnId)) continue;
347 String columnName = columnNames[columnId];
348
349 Dataset<Row> selection = dataset.
350 groupBy(columnName).
351 agg(count(lit(1)).as(AGGREGATED_COUNT_NAME)).
352 filter(col(AGGREGATED_COUNT_NAME).lt(lit(k)));
353
354 if (selection.count() != 0) {
355 columns.add(columnId);
356 }
357 }
358
359 return columns;
360 }

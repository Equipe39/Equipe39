61 private static Dataset<Row> handleConsistency(Dataset<Row> dataset, DataMaskingOptions maskingOptions,
62 ConfigurationManager configurationManager, Set<String> alreadyMaskedFields, Map<String, MaskingProvider> maskingProviders,
63 Map<String, Integer> fieldMap) {
64
65 final Map<String, Set<String>> consistentMaskingFields = analyzeConfiguration(maskingOptions, configurationManager);
66
67 if (consistentMaskingFields.isEmpty()) {
68 return dataset;
69 }
70
71 dataset = applyPreprocessing(dataset, maskingOptions.getInputFormat());
72
73 for (Map.Entry<String, Set<String>> entry : consistentMaskingFields.entrySet()) {
74 Set<String> fieldsInNamespace = entry.getValue();
75 alreadyMaskedFields.addAll(fieldsInNamespace);
76
77 if (fieldsInNamespace.size() == 1) {
78 dataset = ConsistentDataMasking.doConsistentMasking(dataset.sparkSession(), dataset,
79 maskingProviders, fieldsInNamespace.iterator().next(), fieldMap, maskingOptions);
80 } else {
81 dataset = ConsistentDataMasking.doGloballyConsistentMasking(dataset, maskingOptions.getToBeMasked(), maskingProviders,
82 fieldsInNamespace, fieldMap, maskingOptions);
83 }
84 }
85
86 dataset = applyPostprocessing(dataset, maskingOptions.getInputFormat());
87
88 return dataset;
89 }
96 public static Dataset<Row> run(final ConfigurationManager configurationManager,
97 final DataMaskingOptions maskingOptions, Dataset<Row> dataset) {
98 MaskingProviderFactory maskingProviderFactory = new MaskingProviderFactory(configurationManager,
99 maskingOptions.getToBeMasked());
100
101 final Map<String, MaskingProvider> maskingProviders = new HashMap<>();
102 for (String key : maskingOptions.getToBeMasked().keySet()) {
103 maskingProviders.put(key, maskingProviderFactory.get(key));
104 }
105
106 final List<String> fieldNames = SparkUtils.createFieldNames(dataset, maskingOptions.getInputFormat(), maskingOptions.getDatasetOptions());
107 final Map<String, Integer> fieldMap = RecordUtils.createFieldMap(dataset.schema());
108
109 final Set<String> alreadyMaskedFields = new HashSet<>();
110
111 dataset = handleConsistency(dataset, maskingOptions, configurationManager, alreadyMaskedFields, maskingProviders, fieldMap);
112
113 final DataTypeFormat inputFormat = maskingOptions.getInputFormat();
114 final FormatProcessor formatProcessor = FormatProcessorFactory.getProcessor(inputFormat);
115 final DatasetOptions datasetOptions = maskingOptions.getDatasetOptions();
116
117 final List<String> fieldTypes = RecordUtils.getFieldClasses(dataset.schema().fields());
118
119 JavaRDD<Row> rdd = dataset.toJavaRDD().map(value -> {
120 Record record = RecordUtils.createRecord(value, inputFormat, datasetOptions, fieldNames, fieldMap, fieldTypes, false);
121 Record maskedRecord = formatProcessor.maskRecord(record, maskingProviderFactory, alreadyMaskedFields, maskingOptions);
122
123 if (inputFormat == DataTypeFormat.CSV || inputFormat == DataTypeFormat.PARQUET) {
124 return RowFactory.create(((RowRecord) maskedRecord).getValues());
125 } else {
126 return RowFactory.create(Collections.singletonList(maskedRecord.toString()).toArray());
127 }
128 });
129
130 return dataset.sparkSession().createDataFrame(rdd, dataset.schema());
131 }

133 public static Map<String,Set<String>> analyzeConfiguration(DataMaskingOptions maskingOptions, ConfigurationManager configurationManager) {
134
135 Map<String, Set<String>> consistentFields = new HashMap<>();
136
137 Set<String> fields = maskingOptions.getToBeMasked().keySet();
138
139 for(String field: fields) {
140 MaskingConfiguration fieldConfiguration = configurationManager.getFieldConfiguration(field);
141 boolean isConsistent = fieldConfiguration.getBooleanValue("persistence.export");
142 String consistentType = fieldConfiguration.getStringValue("persistence.type");
143
144 if (isConsistent && consistentType.equals("memory")) {
145 String ns = fieldConfiguration.getStringValue("persistence.namespace");
146
147 if (!consistentFields.containsKey(ns)) {
148 consistentFields.put(ns, new HashSet<>());
149 }
150
151 consistentFields.get(ns).add(field);
152 }
153
154 }
155
156 return consistentFields;
157 }

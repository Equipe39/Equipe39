136 private static boolean hasNoCyclicDependencies(Map<String, FieldRelationship> predefinedRelationships) {
137 for (String fieldName : predefinedRelationships.keySet()) {
138 if (hasCyclicDependencies(fieldName, predefinedRelationships)) {
139 return false;
140 }
141 }

143 return true;

146 private static boolean hasCyclicDependencies(String fieldName, Map<String, FieldRelationship> predefinedRelationships) {
147 FieldRelationship fieldRelationship = predefinedRelationships.get(fieldName);
148 Queue<String> dependencies = new LinkedList<>();
149
150 for (RelationshipOperand operand : fieldRelationship.getOperands()) {
151 dependencies.add(operand.getName());
152 }
153
154 Set<String> visited = new HashSet<>();
155 visited.add(fieldName);
156
157 while (!dependencies.isEmpty()) {
158 String dependency = dependencies.poll();
159 FieldRelationship f = predefinedRelationships.get(dependency);
160
161 if (f == null) {
162 continue;
163 }
164
165 for (RelationshipOperand operand : f.getOperands()) {
166 String operandName = operand.getName();
167 if (visited.contains(operandName)) {
168 return true;
169 }
170
171 dependencies.add(operand.getName());
172 }
173
174 visited.add(dependency);
175
176 }
177
178 return false;
179 }

201 private Map<String, FieldRelationship> predefinedRelationshipsFromJSON(JsonNode node) throws IOException {
202 Map<String, FieldRelationship> predefinedRelationships = new HashMap<>();
203
204 if (node != null) {
205 Iterator<JsonNode> iterator = node.elements();

207 while (iterator.hasNext()) {
208 JsonNode element = iterator.next();
209 FieldRelationship fieldRelationship = JsonUtils.MAPPER.readValue(element.toString(), FieldRelationship.class);
210 String fieldName = fieldRelationship.getFieldName();
211 predefinedRelationships.put(fieldName, fieldRelationship);

219 private Set<DataTypeFormat> findSupportedFormats(Map<DataTypeFormat, DataFormatProperties> formatPropertiesMap) {

231 private void validate(Map<DataTypeFormat, DataFormatProperties> formatPropertiesMap) {
232 Set<DataTypeFormat> formatsThatSupportMasking = findSupportedFormats(formatPropertiesMap);
233
234 if (this.inputFormat == null) {
235 throw new RuntimeException("unknown input format. Supported ones are: " + StringUtils.join(formatsThatSupportMasking, ","));
236 }
237
238 if (this.outputFormat == null) {
239 throw new RuntimeException("unknown output format. Supported ones are: " + StringUtils.join(DataTypeFormat.values(), ","));
240 }
241
242 DataFormatProperties formatProperties = formatPropertiesMap.get(this.inputFormat);
243
244 if (formatProperties == null || !formatProperties.supportsMasking()) {
245 throw new RuntimeException("input format does not support masking. Formats that support masking are: "
246 + StringUtils.join(formatsThatSupportMasking, ","));
247 }
248
249 if (!formatProperties.getValidOutputFormats().contains(this.outputFormat)) {
250 throw new RuntimeException("input format does not support: " + this.outputFormat.name() + " as output. Valid output formats: "
251 + StringUtils.join(formatProperties.getValidOutputFormats(), ","));
252 }
253 }

255 private Map<String, DataMaskingTarget> preserveBackCompatibility(Map<String, JsonNode> toBeMasked) {
256 return toBeMasked.entrySet().stream().map(entry -> {
257 JsonNode fieldValue = entry.getValue();
258 String fieldName = entry.getKey();
259
260 final DataMaskingTarget target;
261 if (fieldValue.isObject()) {
262 String providerTypeValue = fieldValue.get("providerType").asText();
263 String targetPath = fieldValue.get("targetPath").asText();
264 target = new DataMaskingTarget(ProviderType.valueOf(providerTypeValue), targetPath);
265 } else if (fieldValue.isTextual()) {
266 String providerTypeValue = fieldValue.asText();
267 target = new DataMaskingTarget(ProviderType.valueOf(providerTypeValue), fieldName);
268 } else {
269 throw new RuntimeException("Invalid type for the entry " + fieldName + " of toBeMasked");
270 }
271 return new AbstractMap.SimpleEntry<>(fieldName, target);
272 }).collect(Collectors.toMap(
273 Map.Entry::getKey,
274 Map.Entry::getValue
275 ));
276 }

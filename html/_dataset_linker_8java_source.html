49 private void buildTargetIndex(InputStream target, Collection<LinkInfo> linkInfos) throws IOException {
50 logger.debug("Building target index for {}", linkInfos);
51
52 int nCols = Integer.MIN_VALUE;
53
54 for (LinkInfo info : linkInfos) {
55 if (info.getTargetIndex() > nCols) nCols = info.getTargetIndex();

60 logger.debug("Minimum number of required columns: {}", nCols);

62 targetIndices = (Map<String, Set<Integer>>[]) new Map[nCols];
63 targetIndicesNumerical = (TreeMap<Double, Set<Integer>>[]) new TreeMap[nCols];
64
65 for (LinkInfo info : linkInfos) {
66 if (info.isNumerical()) {
67 targetIndicesNumerical[info.getTargetIndex()] = new TreeMap<>();
68 } else {
69 targetIndices[info.getTargetIndex()] = new HashMap<>();

95 } else {

167 private <T> Set<T> intersect(Set<T> first, Set<T> second) {
168 Set<T> small;
169 Set<T> large;
170
171 if (first.size() < second.size()) {
172 small = first;
173 large = second;
174 } else {
175 small = second;
176 large = first;
177 }
178
179 Set<T> result = new HashSet<>(small.size());
180
181 for (T entry : small) {
182 if (large.contains(entry)) {
183 result.add(entry);
184 }
185 }
186
187 return result;
188 }

190 private <T> Integer intersectAll(List<Set<T>> sets) {
191 if (sets.size() == 1) {
192 return sets.get(0).size();
193 }
194
195 Collections.sort(sets, new Comparator<Set<T>>() {
196 @Override
197 public int compare(Set<T> o1, Set<T> o2) {
198 return Integer.compare(o1.size(), o2.size());
199 }
200 });
201
202 int results = 0;
203
204 outer:
205 for (T entry : sets.get(0)) {
206 for (int i = 1; i < sets.size(); i++) {
207 Set<T> setI = sets.get(i);
208 if (!setI.contains(entry)) {
209 continue outer;
210 }
211 }
212
213 results += 1;
214 }
215
216 return results;
217 }

219 public Integer matchRow(List<String> sourceRow, Collection<LinkInfo> linkInformation) {
220 List<Set<Integer>> toBeLinked = new ArrayList<>();
221
222 for (LinkInfo info : linkInformation) {
223 int sourceIndex = info.getSourceIndex();
224 int targetIndex = info.getTargetIndex();
225
226 String value = sourceRow.get(sourceIndex);
227 Set<Integer> targetRows;
228
229 if (!info.isNumerical()) {
230 String key = createKey(value);
231
232 if (key.equals(info.getWildcharPattern())) {
233 continue;
234 }
235
236 targetRows = matchValue(value, targetIndex, info.isPrefixMatch());
237 } else {
238 Double numericValue = Double.parseDouble(value);
239 targetRows = matchValue(numericValue, targetIndex);
240 }
241
242 if (targetRows == null) {
243 return 0;
244 }
245
246 toBeLinked.add(targetRows);
247 }
248
249 if (toBeLinked.isEmpty()) {
250 return null;
251 }
252
253 return intersectAll(toBeLinked);
254 }

101 public void estimateLinkingRiskForAnonymized(Dataset<Row> dataset,
102 DatasetLinkingOptions configuration, String outputPath) {
103
104 final List<BasicColumnInformation> basicColumnInformation = configuration.getColumnInformation();
105
106 final Column[] groupByCondition = extractGroupByCondition(basicColumnInformation, dataset, false);
107 Dataset<Row> classes = extractEquivalenceClasses(dataset, groupByCondition, COUNT_COLUMN);
108
109 final Column[] groupByConditionKB = extractGroupByCondition(basicColumnInformation, this.knowledgeBase, true);
110 Dataset<Row> knowledgeBaseClasses;
111
112 if (configuration.isTargetAggregated()) {
113 knowledgeBaseClasses = this.knowledgeBase.withColumnRenamed(configuration.getTargetCountColumn(), COUNT_DEMOGRAPHIC);
114 }
115 else {
116 knowledgeBaseClasses = extractEquivalenceClasses(this.knowledgeBase, groupByConditionKB, COUNT_DEMOGRAPHIC);
117 }
118
119 logger.info("Extracted equivalence classes: #" + classes.count());
120
121 List<List<Integer>> generalizationLevels = filterDistinctGeneralizationLevels(dataset, basicColumnInformation);
122
123 logger.info("Extracted generalization levels: " + generalizationLevels.size());
124
125 final int anonymizedColumnsCount = countAnonymisedColumns(basicColumnInformation);
126
127 logger.info("Anonymized columns count " + anonymizedColumnsCount);
128
129 generalizationLevels.forEach( generalizationLevel -> {
130 Dataset<Row> glDataset = extractDataset(classes, generalizationLevel, anonymizedColumnsCount , basicColumnInformation);
131
132 Dataset<Row> modifiedCensus = generalizeKnowledgeBase(knowledgeBaseClasses, generalizationLevel, basicColumnInformation);
133
134 final Column joinCondition = extractJoinCondition(basicColumnInformation, glDataset, modifiedCensus);
135 Dataset<Row> joint = modifiedCensus.join(glDataset, joinCondition);
136
137 final Column[] groupByConditionInner = extractGroupByCondition(basicColumnInformation, glDataset, false);
138 final Column[] groupByConditionForRisk = extractRiskGroupByCondition(groupByConditionInner);
139
140 Dataset<Row> aggregated = joint.groupBy(groupByConditionForRisk).agg(sum(col(COUNT_DEMOGRAPHIC)).alias(COUNT_DEMOGRAPHIC));;
141 Dataset<Row> risk = aggregated.select(col(COUNT_COLUMN), lit(1d).divide(col(COUNT_DEMOGRAPHIC)).alias(RISK_LABEL));
142
143 risk.select(COUNT_COLUMN, RISK_LABEL).write().mode(SaveMode.Append).csv(outputPath);
144 });
145
146 }

156 private static Dataset<Row> generalizeKnowledgeBase(
157 Dataset<Row> knowledgeBase,
158 List<Integer> generalizationLevel,
159 List<BasicColumnInformation> basicColumnInformation
160 ) {
161 final String[] names = knowledgeBase.columns();
162 JavaRDD<Object[]> rdd = knowledgeBase.rdd().toJavaRDD().map(row -> {
163 Object[] o = new Object[row.length()];
164
165 for (int i = 0; i < o.length; ++i) {
166 o[i] = row.get(i);
167 }
168
169 return o;
170 });
171
172 int quasiIndex = 0;
173
174 for (int i = 0; i < basicColumnInformation.size(); ++i) {
175 final BasicColumnInformation bci = basicColumnInformation.get(i);
176
177 if (!bci.isAnonymised()) {
178 continue;
179 }
180
181 final int level = generalizationLevel.get(quasiIndex);
182 quasiIndex++;
183
184 if (0 == level) continue;
185
186 final String target = bci.getTarget();
187
188 final GeneralizationHierarchy hierarchy = bci.getHierarchy();
189
190 final int position = ArrayUtils.indexOf(names, target);
191
192 if (position < 0) continue;
193
194 rdd = rdd.map(row -> {
195 if (position < row.length) {
196 Object rowValue = row[position];
197
198 if (Objects.nonNull(rowValue)) {
199 String originalValue = rowValue.toString();
200 String encodedValue = hierarchy.encode(originalValue, level, false);
201
202 row[position] = encodedValue;
203 }
204 } else {
205 logger.error(String.format("SOMETHING WENT WRONG! %s: %d >= %d", target, position, row.length));
206 }
207
208 return row;
209 });
210 }
211
212 return knowledgeBase.sparkSession().createDataFrame(rdd.map(RowFactory::create), knowledgeBase.schema());
213 }

215 protected static Dataset<Row> extractDataset(
216 Dataset<Row> dataset,
217 List<Integer> generalizationLevel,
218 int generalizedColumnsCount,
219 List<BasicColumnInformation> basicColumnInformation)
220 {
221
222 if (generalizedColumnsCount == 0) {
223 return dataset;
224 }
225
226 return dataset.filter((Row row) -> {
227 List<Integer> levels = new ArrayList<>();
228
229 for(BasicColumnInformation bci: basicColumnInformation) {
230 if (!bci.isAnonymised()) {
231 continue;
232 }
233
234 String name = bci.getName();
235 final GeneralizationHierarchy hierarchy = bci.getHierarchy();
236 String value = row.getString(row.fieldIndex(name));
237
238 int level = hierarchy.getNodeLevel(value);
239 levels.add(level);
240 }
241
242 return levels.equals(generalizationLevel);
243 });
244
245 }

247 protected static List<List<Integer>> filterDistinctGeneralizationLevels(Dataset<Row> anonymisedDataset,
248 List<BasicColumnInformation> basicColumnInformation) {
249 final int anonymizedColumnsCount = countAnonymisedColumns(basicColumnInformation);
250
251 if (anonymizedColumnsCount == 0) {
252 List<Integer> zeros = new ArrayList<>(basicColumnInformation.size());
253 for(int i = 0; i < basicColumnInformation.size(); i++) {
254 zeros.add(0);
255 }
256 return Arrays.asList(zeros);
257 }
258
259 List<String> uniqueLevelsAsString = anonymisedDataset.javaRDD().map(row -> {
260 StringBuilder builder = new StringBuilder();
261
262 for(BasicColumnInformation bci: basicColumnInformation) {
263 if (!bci.isAnonymised()) {
264 continue;
265 }
266
267 String name = bci.getName();
268 final GeneralizationHierarchy hierarchy = bci.getHierarchy();
269 String value = row.getString(row.fieldIndex(name));
270
271 int level = hierarchy.getNodeLevel(value);
272
273 builder.append(level).append(":");
274 }
275
276 return builder.toString();
277 }).distinct().collect();
278
279 List<List<Integer>> distinctLevels = new ArrayList<>();
280
281 for(String levelS: uniqueLevelsAsString) {
282 List<Integer> levels = convertToList(levelS);
283 distinctLevels.add(levels);
284 }
285
286 return distinctLevels;
287 }

299 private Column extractJoinCondition(List<BasicColumnInformation> basicColumnInformation, Dataset<Row> dataset, Dataset<Row> kb) {
300 return basicColumnInformation.stream().map( bci -> {
301 String name = bci.getName();
302 String target = bci.getTarget();
303
304 if (null == target) return null;
305
306 return kb.col(target).equalTo(dataset.col(name));
307 }).filter(Objects::nonNull).reduce(Column::and).orElseThrow(RuntimeException::new);
308 }

311 private Column[] extractGroupByCondition(List<BasicColumnInformation> basicColumnInformation, Dataset<Row> dataset, boolean useTargetName) {
312
313 Column[] condition = new Column[basicColumnInformation.size()];
314
315 for(int i = 0; i < basicColumnInformation.size(); i++) {
316 BasicColumnInformation bci = basicColumnInformation.get(i);
317 String name = useTargetName ? bci.getTarget() : bci.getName();
318 condition[i] = dataset.col(name);
319 }
320
321 return condition;
322 }

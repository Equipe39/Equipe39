38 public static long daysBetween(Date startDate, Date endDate) {
39 Calendar date = Calendar.getInstance();
40 date.setTime(startDate);

42 long daysBetween = 0;

57 public static Map<String, Date> createOPM(String startDate, String endDate,
58 String dateFormatPattern, int daysBudget) throws ParseException {
59
60 Map<String, Date> map = new HashMap<>();
61 SecureRandom random = new SecureRandom();
62
63 DateFormat df = new SimpleDateFormat(dateFormatPattern);
64
65 Date start = df.parse(startDate);
66 Date end = df.parse(endDate);
67
68 Calendar cal = Calendar.getInstance();
69 cal.setTime(start);
70
71 long originalSpaceSize = daysBetween(start, end);
72 long mappedSpaceSize = originalSpaceSize + daysBudget;
73
74 int maxStep = (int) Math.floor((double) mappedSpaceSize / (double) originalSpaceSize);
75
76 Date mappedValue = addDays(start, -daysBudget);
77
78 cal.setTime(start);
79 while (cal.getTime().before(end)) {
80 String key = df.format(cal.getTime());
81
82 int nextStep = 1 + random.nextInt(maxStep);
83
84 mappedValue = addDays(mappedValue, nextStep);
85 map.put(key, mappedValue);
86
87 cal.add(Calendar.DATE, 1);
88 }
89
90 return map;
91 }

93 public static Map<String, String> loadMapping(InputStream is) {
94 Map<String, String> map = new HashMap<>();
95
96 try (CSVParser reader = Readers.createCSVReaderFromStream(is)) {
97 for (CSVRecord line : reader) {
98 String key = line.get(0);

100 map.put(key, value);
101 }
102
103 is.close();
104 } catch (IOException e) {
105 e.printStackTrace();
106 }

31public class DateYYYYMMDDHierarchy implements GeneralizationHierarchy {
32 private static final String TOP_TERM = "*";
33 private static final int TOP_LEVEL = 3;
34 private static final int YEAR_LEVEL = 2;
35 private static final int MONTH_LEVEL = 1;
36 private static final int DAY_LEVEL = 0;

38 @Override

43 @Override
44 public long getTotalLeaves() {
45 return (

48 * 31L) // DAYS (UPPER BOUND)
49 );

52 @Override

57 @Override
58 public Set<String> getNodeLeaves(String value) {
59 String[] parts = value.split("-");
60
61 switch (parts.length) {
62 case 1:
63 return enumerateDaysInYear(parts[0]);
64 case 2:
65 return enumerateDaysInYearMonth(parts[0], parts[1]);
66 case 3:
67 return Collections.singleton(value);
68 default:
69 throw new IllegalArgumentException();
70 }
71 }

73 private Set<String> enumerateDaysInYearMonth(String year, String month) {
74 LocalDate from = LocalDate.of(
75 Year.parse(year).getValue(),
76 Month.of(Integer.parseInt(month, 10)),
77 1
78 );
79
80 return enumerateDays(from, from.plus(1, ChronoUnit.MONTHS));

83 private Set<String> enumerateDaysInYear(String year) {

89

93 private Set<String> enumerateDays(LocalDate from, LocalDate to) {
94 Set<String> days = new HashSet<>();
95
96 for (; from.isBefore(to); from = from.plus(1, ChronoUnit.DAYS)) {
97 days.add(from.toString());
98 }
99
100 return days;
101 }

103 @Override
104 public int getNodeLevel(String value) {
105 if (value.equals(TOP_TERM)) return TOP_LEVEL;
106
107 String[] parts = value.split("-");
108
109 switch (parts.length) {
110 case 1:
111 return YEAR_LEVEL;
112 case 2:
113 return MONTH_LEVEL;
114 case 3:
115 return DAY_LEVEL;
116
117 default:
118 throw new RuntimeException("Unknown: " + value);
119 }
120 }

122 @Override

127 @Override
128 public String encode(String value, int level, boolean randomizeOnFail) {
129 if (TOP_LEVEL == level) return getTopTerm();
130 if (DAY_LEVEL == level) return value;
131
132 String[] parts = value.split("-");
133
134 if (3 < parts.length) {
135 if (randomizeOnFail) {
136 return getTopTerm();
137 }
138 }
139
140 switch (level) {
141 case MONTH_LEVEL:
142 if (2 <= parts.length) {
143 return parts[0] + '-' + parts[1];
144 }
145 case YEAR_LEVEL:
146 if (1 <= parts.length) {
147 return parts[0];
148 }
149 default:
150 if (randomizeOnFail) {
151 return getTopTerm();
152 }
153 throw new IllegalArgumentException();
154 }

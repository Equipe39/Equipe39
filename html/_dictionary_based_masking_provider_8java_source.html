52 private List<String> loadTermsFromFile(String dictionaryFilename) throws IOException {
53 List<String> terms = new ArrayList<>();
54
55 try (BufferedReader br = new BufferedReader(new FileReader(dictionaryFilename))) {
56 String line;
57 while ((line = br.readLine()) != null) {
58 terms.add(line.trim());
59 }
60 }
61
62 return terms;
63 }
64
65 public DictionaryBasedMaskingProvider(SecureRandom random, MaskingConfiguration maskingConfiguration) {
66 this.random = random;
67 String dictionaryFilename = maskingConfiguration.getStringValue("dictionaryBased.mask.filename");
68 try {
69 this.terms = loadTermsFromFile(dictionaryFilename);
70 } catch (IOException e) {
71 String msg = "unable to load dictionary file: " + dictionaryFilename;
72 log.error(msg);
73 throw new RuntimeException(msg);
74 }
75 }

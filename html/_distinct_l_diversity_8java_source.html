48 private boolean checkDistinctLDiversity(Partition partition, List<Integer> sensitiveColumns) {
49 for (Integer sensitiveColumn : sensitiveColumns) {
50 Set<String> uniqueValues = new HashSet<>();
51
52 IPVDataset members = partition.getMember();
53 int numberOfRows = members.getNumberOfRows();
54
55 for (int i = 0; i < numberOfRows; i++) {
56 String value = members.get(i, sensitiveColumn);
57 uniqueValues.add(value.toUpperCase());
58
59 //break early if we have l-represented values
60 if (uniqueValues.size() >= l) {
61 break;
62 }
63 }
64
65 if (uniqueValues.size() < this.l) {
66 return false;
67 }
68 }
69
70 return true;
71 }
72
73 @Override
74 public boolean check(PrivacyMetric metric) {
75 LDiversityMetric lDiversityMetric = (LDiversityMetric) metric;
76
77 List<Histogram> histograms = lDiversityMetric.getHistograms();
78
79 for (Histogram histogram : histograms) {
80 if (histogram.size() < this.l) {
81 return false;
82 }
83 }
84
85 return true;
86 }

88 @Override

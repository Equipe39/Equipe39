128 private static JavaRDD<Row> extractDistribution(Dataset<Row> dataset, DistributionExtractorOptions configuration) {
129 if (containsCustomExtraction(configuration)) {
130 return executeCustomDistributionExtraction(dataset, configuration);
131 } else {
132 final Column[] idColumns = buildSelect(configuration.getIdentityFields());
133
134 final Dataset<Row> histograms;
135 if (configuration.getIdentityFields().isEmpty()) {
136 histograms = computeFieldSelection(dataset, buildSelectCondition(configuration.getThresholds()));
137 } else {
138 List<DistributionExtractorOptions.ProjectedExpression> conditions = configuration.getThresholds();
139 Column firstAggregation = buildAggregation(conditions.get(0));
140 Column[] subsequentAggregations = buildAggregations(conditions.subList(1, conditions.size()));
141 histograms = removeIdentity(configuration.getIdentityFields(), computeAggregation(dataset, idColumns, firstAggregation, subsequentAggregations));
142 }
143
144 String valueColumnName = extractDistributionColumn(configuration.getThresholds());
145
146 return histograms.where(col(valueColumnName).isNotNull()).javaRDD();
147 }
148 }

198 private static List<String> applyBinningIfRequired(JavaDoubleRDD data, double min, double max, DistributionExtractorOptions.BinningCondition options) {
199 switch (options.getType()) {
200 case SIZE:
201 return extractBinsBySize(data, min, max, options.getBinSize(), options.isReturnMidPoint());
202 case NUMBER:
203 return extractBins(data, min, max, options.getBinNumber(), options.isReturnMidPoint());
204 case NONE:
205 return extractActualValues(data);
206 }
207 throw new IllegalArgumentException("Unknown binning type");
208 }

210 private static List<String> extractBinsBySize(JavaDoubleRDD data, double m, double M, double binSize, boolean returnMidpoint) {
211 JavaRDD<Tuple2<double[], Long>> bins = data.mapToPair(value -> {
212 double scaledValue = value - m;
213 return new Tuple2<>((int) Math.floor(scaledValue / binSize), 1L);
214 }).reduceByKey(Long::sum).map(createRanges(binSize, m, M));
215
216 if (returnMidpoint) {
217 return bins.map(tuple -> "" + (tuple._1()[0] + tuple._1()[1])/2.0 + "," + tuple._2()).collect();
218 }
219
220 return bins.map(tuple -> "" + tuple._1()[0] + "," + tuple._1()[1] + "," + tuple._2()).collect();
221 }

227 public static List<Tuple<Double,Double>> calculateCDF(List<Tuple<Double,Double>> numbersForCDF) {
228 List<Tuple<Double, Double>> cdf = new ArrayList<>(numbersForCDF.size());
229
230 numbersForCDF.sort(Comparator.comparing(Tuple::getFirst));
231
232 double totalValues = 0.0;
233 for(Tuple<Double, Double> t: numbersForCDF) {
234 totalValues += t.getSecond();
235 }
236
237 double sum = 0.0;
238
239 for(Tuple<Double, Double> t: numbersForCDF) {
240 cdf.add(new Tuple<>(t.getFirst(), (sum + t.getSecond()) / totalValues));
241 sum += t.getSecond();
242 }
243
244 return cdf;
245 }

267 private static List<String> extractBins(JavaDoubleRDD numericData, double m, double M, double numberOfBins, boolean returnMidpoint) {
268 final double range = M - m;
269 final double binSize = range / numberOfBins;
270
271 JavaRDD<Tuple2<double[], Long>> bins = numericData.mapToPair(value -> {
272 double scaledValue = value - m;
273 return new Tuple2<>((int) Math.floor(scaledValue / binSize), 1L);
274 }).reduceByKey(Long::sum).map(createRanges(binSize, m, M));
275
276 if (returnMidpoint) {
277 return bins.map(tuple -> "" + (tuple._1()[0] + tuple._1()[1])/2.0 + "," + tuple._2()).collect();
278 }
279
280 return bins.map(tuple -> "" + tuple._1()[0] + "," + tuple._1()[1] + "," + tuple._2()).collect();
281 }

283 private static Function<Tuple2<Integer, Long>, Tuple2<double[], Long>> createRanges(final double binSize, final double min, double max) {
284 return idCount -> {
285 int id = idCount._1();
286 double[] ranges = new double[2];
287
288 ranges[0] = (id * binSize) + min;
289 ranges[1] = ((id + 1) * binSize) + min;
290
291 return new Tuple2<>(ranges, idCount._2());
292 };
293 }

323 private static Column buildAggregation(DistributionExtractorOptions.ProjectedExpression condition) {
324 String columnName = condition.getColumnName();
325 Column column = col(columnName);
326
327 int properAggregation = 0;
328
329 switch (condition.getAggregationType()) {
330 case SUM:
331 column = sum(column);
332 properAggregation += 1;
333 break;
334 case COUNT:
335 column = sum(lit(1L));
336 properAggregation += 1;
337 break;
338 case MAX:
339 column = max(column);
340 properAggregation += 1;
341 break;
342 case MIN:
343 column = min(column);
344 properAggregation += 1;
345 break;
346 case NOT_AGGREGATED:
347 break;
348 default:
349 throw new IllegalArgumentException("Unknown aggregation type");
350 }
351
352 if (0 == properAggregation) throw new IllegalArgumentException("No aggregation values specified");
353
354 return column.as(columnName);
355 }

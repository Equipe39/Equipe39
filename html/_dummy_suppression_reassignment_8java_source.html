67 private int checkNonConformingClusters(List<KMeansCluster> clusters) {
68 int nonConforming = 0;
69
70 for (KMeansCluster cluster : clusters) {
71 Partition partition = cluster.getOriginalData();
72
73 boolean conforms = AnonymizationUtils.checkPrivacyConstraints(privacyConstraints, partition, this.sensitiveColumns);
74 partition.setAnonymous(conforms);
75
76 if (!conforms) {
77 nonConforming++;
78 }
79 }
80
81 return nonConforming;

91 private List<KMeansCluster> suppressAndMerge(List<KMeansCluster> clusters) {
92
93 List<KMeansCluster> candidateClusters = new ArrayList<>(clusters);
94
95 int nonConforming = checkNonConformingClusters(candidateClusters);
96
97 while (nonConforming > 0) {
98 List<Integer> toRemove = new ArrayList<>();
99
100 for (int i = (candidateClusters.size() - 1); i >= 0; i--) {
101 KMeansCluster cluster = candidateClusters.get(i);
102
103 if (cluster.getOriginalData().isAnonymous()) {
104 continue;
105 }
106
107 int numberOfRecords = cluster.getOriginalData().size();
108
109 double suppressionAfterRemoval = (100.0 * (numberOfRecords + suppressedRows)) / (double) this.original.getNumberOfRows();
110
111 if (suppressionAfterRemoval <= this.maximumSuppressionRate) {
112 this.suppressedRows += numberOfRecords;
113 } else {
114 int index = KMeans.getNearestPointIndex(cluster.getCenter(), candidateClusters, i);
115
116 if (index == Integer.MAX_VALUE) {
117 throw new RuntimeException("we could not merge with any cluster");
118 }
119
120 KMeansCluster targetCluster = candidateClusters.get(index);
121 mergeClusters(cluster, targetCluster);
122 }
123
124 toRemove.add(i);
125 }
126
127 for (Integer c : toRemove) {
128 candidateClusters.remove(c.intValue());
129 }
130
131 nonConforming = checkNonConformingClusters(candidateClusters);
132 }
133
134 return candidateClusters;
135 }

138 private List<String> buildAnonymizedRow(List<String> originalRow, List<String> categoricalCentroids, List<Double> clusterCenter) {
139 List<String> anonymizedRow = new ArrayList<>();
140
141 int quasiIndex = 0;
142
143 for (int i = 0; i < columnInformationList.size(); i++) {
144 ColumnInformation columnInformation = this.columnInformationList.get(i);
145
146 if (columnInformation.getColumnType() != ColumnType.QUASI) {
147 anonymizedRow.add(originalRow.get(i));
148 } else {
149 boolean isCategorical = columnInformation.isCategorical();
150
151 if (isCategorical) {
152 anonymizedRow.add(categoricalCentroids.get(quasiIndex));
153 } else {
154 anonymizedRow.add(clusterCenter.get(quasiIndex).toString());
155 }
156
157 quasiIndex++;
158 }
159 }
160
161 return anonymizedRow;
162 }

164 private List<String> createCategoricalCentroids(Partition partition) {
165
166 List<Integer> categoricalColumns = AnonymizationUtils.getColumnsByType(this.columnInformationList, ColumnType.QUASI);
167 categoricalColumns = categoricalColumns.stream().filter(x -> this.columnInformationList.get(x).isCategorical()).collect(Collectors.toList());
168
169 if (categoricalColumns.isEmpty()) {
170 return Collections.emptyList();
171 }
172
173 List<Set<String>> values = new ArrayList<>();
174 List<GeneralizationHierarchy> hierarchies = new ArrayList<>();
175
176 for (Integer categoricalColumn : categoricalColumns) {
177 values.add(new HashSet<>());
178 CategoricalInformation categoricalInformation = (CategoricalInformation) this.columnInformationList.get(categoricalColumn);
179 hierarchies.add(categoricalInformation.getHierarchy());
180 }
181
182 for (List<String> originalRow : partition.getMember()) {
183 for (int i = 0; i < categoricalColumns.size(); i++) {
184 Integer column = categoricalColumns.get(i);
185 String value = originalRow.get(column);
186
187 values.get(i).add(value);
188 }
189 }
190
191 List<String> centroids = new ArrayList<>();
192
193 for (int i = 0; i < categoricalColumns.size(); i++) {
194 GeneralizationHierarchy hierarchy = hierarchies.get(i);
195 centroids.add(ClusteringAnonUtils.calculateCommonAncestor(values.get(i), hierarchy));
196 }
197
198 return centroids;
199 }

201 private IPVDataset createAnonymized(List<KMeansCluster> clusters) {
202 IPVDataset anonymized = new IPVDataset(original.getNumberOfColumns());
203
204 for (KMeansCluster cluster : clusters) {
205 Partition partition = cluster.getOriginalData();
206
207 if (partition.size() == 0) {
208 continue;
209 }
210
211 List<Double> clusterCenter = cluster.getCenter();
212
213 List<String> categoricalCentroids = createCategoricalCentroids(partition);
214
215 for (List<String> originalRow : partition.getMember()) {
216 anonymized.addRow(buildAnonymizedRow(originalRow, categoricalCentroids, clusterCenter));
217 }
218 }
219
220 return anonymized;
221 }

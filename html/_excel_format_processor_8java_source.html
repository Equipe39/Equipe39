80 private static void inspectSheet(Sheet sheet, Map<String, List<IdentifiedType>> allTypes, Collection<Identifier> identifiers) {
81
82 DataFormatter formatter = new DataFormatter();
83 String sheetName = sheet.getSheetName();
84
85 for (Row row : sheet) {
86 for (Cell cell : row) {
87 String value = formatter.formatCellValue(cell);
88
89 if (value == null || value.isEmpty()) {
90 continue;
91 }
92
93 CellReference cellRef = new CellReference(row.getRowNum(), cell.getColumnIndex());
94 String cellName = "/" + sheetName + "/" + cellRef.formatAsString();
95
96 for (Identifier identifier : identifiers) {
97 if (identifier.isOfThisType(value)) {
98 String type = identifier.getType().getName();
99
100 Collection<IdentifiedType> columnTypes = allTypes.get(cellName);
101 if (columnTypes == null) {
102 allTypes.put(cellName, new ArrayList<>());
103 columnTypes = allTypes.get(cellName);
104 }
105
106 columnTypes.add(new IdentifiedType(type, 1L));
107
108 }
109 }
110 }
111 }

114 @Override
115 public IdentificationReport identifyTypesStream(InputStream input, DataTypeFormat inputFormatType,
116 DatasetOptions datasetOptions, Collection<Identifier> identifiers,
117 int firstN) throws IOException {
118
119 final Map<String, List<IdentifiedType>> allTypes = new HashMap<>();
120
121 Workbook wb;
122
123 if (inputFormatType == DataTypeFormat.XLSX) {
124 wb = new XSSFWorkbook(input);
125 } else {
126 wb = new HSSFWorkbook(input);
127 }
128
129 int numberOfSheets = wb.getNumberOfSheets();
130
131 for (int i = 0; i < numberOfSheets; i++) {
132 Sheet sheet = wb.getSheetAt(i);
133 inspectSheet(sheet, allTypes, identifiers);
134 }
135
136 wb.close();
137
138
139 Map<String, IdentifiedType> bestTypes = IdentifierUtils.getIdentifiedType(allTypes, 1L, IdentificationConfiguration.DEFAULT);
140
141 return new IdentificationReport(
142 allTypes,
143 bestTypes,
144 1L
145 );

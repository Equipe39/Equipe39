189 @Test
190 public void testXLSRanges() throws Exception {
191 try (InputStream inputStream = ExcelMaskingProviderTest.class.getResourceAsStream("/sampleXLS.xls")) {
192 byte[] inputBytes = IOUtils.toByteArray(inputStream);
193
194 String rangePath = "/d/$D$7:$E$10";
195 String targetPath = "/d/$D$7:$D$10";
196
197 List<Tuple<String, String>> toValidateList = Arrays.asList(
198 new Tuple<>("/d/$D$7", "foo"),
199 new Tuple<>("/d/$D$8", "goo"),
200 new Tuple<>("/d/$D$9", "hoo"),
201 new Tuple<>("/d/$D$10", "joo"),
202 new Tuple<>("/d/$E$7", "foo"),
203 new Tuple<>("/d/$E$8", "goo"),
204 new Tuple<>("/d/$E$9", "hoo"),
205 new Tuple<>("/d/$E$10", "joo")
206 );
207
208 Map<String, DataMaskingTarget> maskingProviders = new HashMap<>();
209 maskingProviders.put(rangePath, new DataMaskingTarget(ProviderType.HASH, targetPath));
210
211 ExcelMaskingProvider excelMaskingProvider = new ExcelMaskingProvider(new DefaultMaskingConfiguration(),
212 DataTypeFormat.XLS, maskingProviders, new MaskingProviderFactory(new ConfigurationManager(), maskingProviders));
213
214 MaskingProvider hashMaskingProvider = new HashMaskingProvider();
215
216 byte[] maskedData = excelMaskingProvider.mask(inputBytes);
217
218 for (Tuple<String, String> toValidate : toValidateList) {
219 String path = toValidate.getFirst();
220 String originalValue = toValidate.getSecond();
221
222 assertEquals(originalValue, ExcelUtils.getValue(inputBytes, DataTypeFormat.XLS, path));
223
224 String maskedValue = ExcelUtils.getValue(maskedData, DataTypeFormat.XLS, path);
225 assertEquals(hashMaskingProvider.mask(originalValue), maskedValue);
226 }
227
228 try (OutputStream os = new FileOutputStream("/tmp/masked.xls")) {
229 IOUtils.write(maskedData, os);
230 }
231 }
232 }

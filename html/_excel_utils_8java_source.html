38 public static String getValue(byte[] inputBytes, DataTypeFormat inputFormatType, String path) throws IOException {
39 try (Workbook wb = readWorkBook(inputFormatType, inputBytes)) {
40 String[] parts = path.split("/");
41 String workbookName = parts[1];
42 String cellReference = parts[2];
43
44 Sheet sheet = wb.getSheet(workbookName);
45 if (sheet == null) {
46 return null;
47 }
48
49 CellReference cellReference1 = new CellReference(cellReference);
50
51 Row row = sheet.getRow(cellReference1.getRow());
52 if (row == null) {
53 return null;
54 }
55
56 Cell toRetrieve = row.getCell(cellReference1.getCol());
57
58 return new DataFormatter().formatCellValue(toRetrieve);
59 }
60 }
61
62 private static Workbook readWorkBook(DataTypeFormat inputFormatType, byte[] inputBytes) throws IOException {
63 InputStream input = new ByteArrayInputStream(inputBytes);
64
65 if (inputFormatType == DataTypeFormat.XLSX) {
66 return new XSSFWorkbook(input);
67 } else {
68 return new HSSFWorkbook(input);
69 }
70 }

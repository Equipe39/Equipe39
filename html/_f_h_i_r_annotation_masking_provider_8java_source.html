69 public JsonNode mask(JsonNode node) {
70 try {
71 FHIRAnnotation obj = JsonUtils.MAPPER.treeToValue(node, FHIRAnnotation.class);
72 FHIRAnnotation maskedObj = mask(obj);
73 return JsonUtils.MAPPER.valueToTree(maskedObj);
74 } catch (Exception e) {
75 return NullNode.getInstance();
76 }
77 }
78
79 public FHIRAnnotation mask(FHIRAnnotation annotation) {
80 if (annotation == null) {
81 return null;
82 }
83
84 if (this.removeExtensions) {
85 annotation.setExtension(null);
86 }
87
88 if (this.removeAuthorString) {
89 annotation.setAuthorString(null);
90 }
91
92 if (this.removeText) {
93 annotation.setText(null);
94 }
95
96 if (this.maskAuthorReference && !isAlreadyMasked(AUTHORREFERENCE_FIELD_PATH)) {
97 FHIRReference reference = annotation.getAuthorReference();
98 if (reference != null) {
99 annotation.setAuthorReference(this.authorReferenceMaskingProvider.mask(reference));
100 }
101 }
102
103 if (this.maskTime && !isAlreadyMasked(TIME_FIELD_PATH)) {
104 String time = annotation.getTime();
105 if (time != null) {
106 annotation.setTime(this.timeMaskingProvider.mask(time));
107 }
108 }
109
110 return annotation;
111 }

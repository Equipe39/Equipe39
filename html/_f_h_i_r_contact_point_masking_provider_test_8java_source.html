41 @Test
42 public void testEmail() throws Exception {
43 String cpJSON = "{\n" +
44 " \"system\": \"email\",\n" +
45 " \"value\": \"p.heuvel@gmail.com\",\n" +
46 " \"use\": \"home\"\n" +
47 " }";
48
49 ObjectMapper objectMapper = new ObjectMapper();
50 FHIRContactPoint contactPoint = objectMapper.readValue(cpJSON, FHIRContactPoint.class);
51
52 String value = contactPoint.getValue();
53
54 MaskingConfiguration maskingConfiguration = new DefaultMaskingConfiguration();
55 maskingConfiguration.setValue("fhir.contactPoint.removeSystem", false);
56
57 FHIRContactPointMaskingProvider contactPointMaskingProvider = new FHIRContactPointMaskingProvider(
58 maskingConfiguration, new HashSet<String>(), "/contact", this.factory);
59 FHIRContactPoint maskedContactPoint = contactPointMaskingProvider.mask(contactPoint);
60
61 String maskedValue = maskedContactPoint.getValue();
62
63 assertNotEquals(maskedValue, value);
64 assertEquals("email", maskedContactPoint.getSystem());
65 assertEquals("home", maskedContactPoint.getUse());
66 assertTrue(new EmailIdentifier().isOfThisType(maskedValue));
67 }

100 private List<FHIRResourceMaskingAction> buildMaskingActions(FHIRResourceMaskingConfiguration resourceConfiguration,
101 MaskingConfiguration maskingConfiguration) {
102 List<FHIRResourceMaskingAction> maskingActions = new ArrayList<>();
103
104 List<FHIRResourceField> fields = resourceConfiguration.getFields();
105
106 String basePath = resourceConfiguration.getBasePath();
107
108 for (FHIRResourceField field : fields) {
109
110 String path = field.getPath();
111 String fullPath = basePath + path;
112
113 MaskingProvider maskingProvider = null;
114 AbstractComplexMaskingProvider abstractComplexMaskingProvider = null;
115 boolean isDelete = false;
116
117 String fhirType = field.getFhirType();
118
119 if (fhirType.equals("Delete")) {
120 isDelete = true;
121 } else if (fhirType.startsWith("FHIR_")) {
122 abstractComplexMaskingProvider = createFHIRMaskingProvider(fhirType, fullPath, maskingConfiguration);
123 } else {
124 MaskingConfiguration nameMaskingConfiguration = getConfigurationForSubfield(fullPath, maskingConfiguration);
125 maskingProvider = this.factory.get(ProviderType.valueOf(fhirType), nameMaskingConfiguration);
126 }
127
128 maskingActions.add(new FHIRResourceMaskingAction(fullPath, path, maskingProvider,
129 abstractComplexMaskingProvider, isDelete));
130 }
131
132 return maskingActions;
133 }

145 private void maskFinalPathComplex(JsonNode node, JsonNode valueNode, String path,
146 AbstractComplexMaskingProvider<JsonNode> abstractComplexMaskingProvider) {
147 if (valueNode == null) {
148 return;
149 }
150
151 if (valueNode.isObject()) {
152 JsonNode maskedValueNode = abstractComplexMaskingProvider.mask(valueNode);
153 ((ObjectNode) node).set(path, maskedValueNode);
154 } else if (valueNode.isArray()) {
155 Iterator<JsonNode> items = valueNode.elements();
156 ArrayNode arrayNode = new ArrayNode(JsonNodeFactory.instance);
157
158 while (items.hasNext()) {
159 JsonNode item = items.next();
160 JsonNode maskedItem = abstractComplexMaskingProvider.mask(item);
161 arrayNode.add(maskedItem);
162 }
163
164 ((ObjectNode) node).set(path, arrayNode);
165 }
166 }

168 private ValueNode getValueNode(JsonNode originalValueNode, String maskedValue) {
169 switch (originalValueNode.getNodeType()) {
170 case NUMBER:
171 if (originalValueNode.isInt()) {
172 return new IntNode(Integer.parseInt(maskedValue));
173
174 } else if (originalValueNode.isFloat()) {
175 return new FloatNode(Float.parseFloat(maskedValue));
176 } else {
177 return new DoubleNode(Double.parseDouble(maskedValue));
178 }
179 case BOOLEAN:
180 return BooleanNode.valueOf(Boolean.parseBoolean(maskedValue));
181 case BINARY:
182 return new BinaryNode(maskedValue.getBytes());
183 case STRING:
184 default:
185 return new TextNode(maskedValue);
186 }
187 }

193 private void maskFinalPathSimple(JsonNode node, JsonNode valueNode, String path, MaskingProvider maskingProvider) {
194
195 if (valueNode == null) {
196 return;
197 }
198
199 if (valueNode.isObject()) {
200 return;
201 }
202
203 if (valueNode.isArray()) {
204 Iterator<JsonNode> items = valueNode.elements();
205 ArrayNode arrayNode = new ArrayNode(JsonNodeFactory.instance);
206
207 while (items.hasNext()) {
208 JsonNode item = items.next();
209
210 if (item.isNull() || item.isObject() || item.isArray()) {
211 arrayNode.add(item);
212 continue;
213 }
214
215 if (!item.isNull()) {
216 String maskedItem = maskingProvider.mask(item.asText());
217 arrayNode.add(getValueNode(item, maskedItem));
218 }
219 }
220
221 ((ObjectNode) node).set(path, arrayNode);
222 } else {
223 if (!valueNode.isNull()) {
224 String value = valueNode.asText();
225 String maskedValue = maskingProvider.mask(value);
226 putValue(node, valueNode, path, maskedValue);
227 }
228 }

231 private void maskNode(JsonNode rootNode, JsonNode node, String[] paths, int pathIndex, FHIRResourceMaskingAction maskingAction) {
232 String path = paths[pathIndex];
233
234 if (pathIndex < (paths.length - 1)) {
235 JsonNode subNode = node.get(path);
236
237 if (subNode == null) {
238 return;
239 }
240
241 if (subNode.isArray()) {
242 Iterator<JsonNode> items = subNode.elements();
243 while (items.hasNext()) {
244 maskNode(rootNode, items.next(), paths, pathIndex + 1, maskingAction);
245 }
246 } else if (subNode.isObject()) {
247 maskNode(rootNode, subNode, paths, pathIndex + 1, maskingAction);
248 }
249 } else {
250
251 if (maskingAction.isDelete()) {
252 ((ObjectNode) node).set(path, NullNode.getInstance());
253 return;
254 }
255
256 if (maskingAction.getAbstractComplexMaskingProvider() != null) {
257 AbstractComplexMaskingProvider<JsonNode> abstractComplexMaskingProvider = maskingAction.getAbstractComplexMaskingProvider();
258 JsonNode valueNode = node.get(path);
259 maskFinalPathComplex(node, valueNode, path, abstractComplexMaskingProvider);
260 } else {
261 MaskingProvider maskingProvider = maskingAction.getMaskingProvider();
262 JsonNode valueNode = node.get(path);
263 maskFinalPathSimple(node, valueNode, path, maskingProvider);
264 }
265 }

269 public JsonNode mask(JsonNode node) {
270 this.fhirBaseDomainResourceMaskingProvider.mask(node);
271
272 for (FHIRResourceMaskingAction maskingAction : this.maskingActionList) {
273 if (!isAlreadyMasked(maskingAction.getFullPath())) {
274 String[] paths = maskingAction.getPaths();
275 maskNode(node, node, paths, 0, maskingAction);
276 }
277 }
278
279
280 return node;
281 }

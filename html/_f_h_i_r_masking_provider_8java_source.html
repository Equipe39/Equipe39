70 public static Collection<String> loadRulesForResource(String resourceName, Map<String, DataMaskingTarget> toBeMasked) {
71 List<String> rules = new ArrayList<>();
72
73 String prefix = resourceName + ".";
74
75 for (Map.Entry<String, DataMaskingTarget> entry : toBeMasked.entrySet()) {
76 if (entry.getKey().startsWith(prefix)) {
77 rules.add(entry.getValue().getProviderType().name());
78 }
79 }
80
81 return rules;
82 }

128 private JsonNode maskResource(JsonNode resource) {
129 if (resource == null || resource.isNull() || !resource.isObject()) {
130 return resource;
131 }
132
133 JsonNode resourceNode = resource.get("resourceType");
134
135 if (resourceNode == null || resourceNode.isNull() || !resourceNode.isTextual()) {
136 return resource;
137 }
138
139 String resourceType = resourceNode.asText();
140
141 String resourceTypeName = resourceType.toLowerCase();
142 FHIRGenericMaskingProvider maskingProvider = maskingProviderMap.get(resourceTypeName);
143
144 if (maskingProvider != null) {
145 return maskingProvider.mask(resource);
146 }
147
148 return resource;
149 }

152 The masking provider works on FHIR objects that look like this:
153 {
154 "resourceType":"ContactPoint",
155 "system":"email",
156 "value":"p.heuvel@gmail.com",
157 "use":"home"

162 public String mask(JsonNode resource) {
163 JsonNode maskedValue = maskResource(resource);
164
165 if (maskedValue == null) {
166 return null;
167 }
168
169 return maskedValue.toString();
170 }

172 public String mask(String identifier) {
173
174 try {
175 JsonNode resource = JsonUtils.MAPPER.readTree(identifier);
176
177 String maskedResource = mask(resource);
178 if (maskedResource == null) {
179 return identifier;
180 }
181
182 return maskedResource;

184 } catch (IOException e) {

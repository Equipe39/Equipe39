67 @Override
68 public JsonNode mask(JsonNode node) {
69 try {
70 FHIRPeriod obj = JsonUtils.MAPPER.treeToValue(node, FHIRPeriod.class);
71 FHIRPeriod maskedObj = mask(obj);
72 return JsonUtils.MAPPER.valueToTree(maskedObj);
73 } catch (Exception e) {

78 public FHIRPeriod mask(FHIRPeriod period) {
79 if (period == null) {
80 return null;
81 }
82
83 if (this.removeEnd) {
84 period.setEnd(null);
85 } else if (this.maskEnd && !isAlreadyMasked(END_PATH)) {
86 String end = period.getEnd();
87 if (end != null) {
88 period.setEnd(endMaskingProvider.mask(end));
89 }
90
91 }
92
93 if (this.removeStart) {
94 period.setStart(null);
95 } else if (this.maskStart && !isAlreadyMasked(START_PATH)) {
96 String start = period.getStart();
97 if (start != null) {
98 period.setStart(startMaskingProvider.mask(start));
99 }
100 }
101
102 return period;
103 }

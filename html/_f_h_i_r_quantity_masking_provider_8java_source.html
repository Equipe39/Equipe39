63 @Override

74 public FHIRQuantity mask(FHIRQuantity quantity) {
75 if (quantity == null) {
76 return null;
77 }
78
79 if (this.maskValue && !isAlreadyMasked(VALUE_PATH)) {
80 float value = quantity.getValue();
81 quantity.setValue(Float.parseFloat(valueMaskingProvider.mask(Float.toString(value))));
82 }
83
84 if (this.maskSystem && !isAlreadyMasked(SYSTEM_PATH)) {
85 String system = quantity.getSystem();
86 if (system != null) {
87 quantity.setSystem(systemMaskingProvider.mask(system));
88 }
89 }
90
91 if (this.maskCode && !isAlreadyMasked(CODE_PATH)) {
92 String code = quantity.getCode();
93 if (code != null) {
94 quantity.setCode(codeMaskingProvider.mask(code));
95 }
96 }
97
98 return quantity;
99 }

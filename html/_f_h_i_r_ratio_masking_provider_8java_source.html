46 this.NUMERATOR_PATH = fieldPath + "/numerator";

58 @Override
59 public JsonNode mask(JsonNode node) {
60 try {
61 FHIRRatio obj = JsonUtils.MAPPER.treeToValue(node, FHIRRatio.class);
62 FHIRRatio maskedObj = mask(obj);
63 return JsonUtils.MAPPER.valueToTree(maskedObj);
64 } catch (Exception e) {
65 return NullNode.getInstance();
66 }
67 }
68
69 public FHIRRatio mask(FHIRRatio ratio) {
70 if (this.deleteDenominator) {
71 ratio.setDenominator(null);
72 } else if (this.maskDenominator && !isAlreadyMasked(DENOMINATOR_PATH)) {
73 ratio.setDenominator(denominatorMaskingProvider.mask(ratio.getDenominator()));
74 }
75
76
77 if (this.deleteNumerator) {
78 ratio.setNumerator(null);
79 } else if (this.maskNumerator && !isAlreadyMasked(NUMERATOR_PATH)) {
80 ratio.setNumerator(numeratorMaskingProvider.mask(ratio.getNumerator()));
81 }
82
83 return ratio;
84 }

65 @Override

76 private boolean matchPrefix(String value) {
77 int slashIndex = value.lastIndexOf('/');
78 if (slashIndex == -1) {
79 return false;
80 }
81
82 String prefix = value.substring(0, slashIndex);
83
84 return this.maskReferenceExcludePrefixList.contains(prefix.toUpperCase());
85 }
86
87 public FHIRReference mask(FHIRReference reference) {
88 if (reference == null) {
89 return null;
90 }
91
92 if (this.removeDisplay) {
93 reference.setDisplay(null);
94 } else if (this.maskDisplay && !isAlreadyMasked(DISPLAY_FIELD_PATH)) {
95 String display = reference.getDisplay();
96 if (display != null) {
97 reference.setDisplay(displayMaskingProvider.mask(display));
98 }
99 }
100
101 if (this.removeExtension) {
102 reference.setExtension(null);
103 }
104
105 if (this.maskReference && !isAlreadyMasked(REFERENCE_FIELD_PATH)) {
106 String refValue = reference.getReference();
107 if (refValue != null) {
108 boolean maskingRequired = this.maskReferenceExcludePrefixList.isEmpty() || !matchPrefix(refValue);
109
110 if (maskingRequired) {
111 String maskedReference = FHIRMaskingUtils.maskResourceId(refValue, this.preserveReferencePrefix, referenceMaskingProvider);
112 reference.setReference(maskedReference);
113 }
114 }
115 }
116
117 return reference;
118 }

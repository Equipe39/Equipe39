59 @Override
60 public JsonNode mask(JsonNode node) {
61 try {
62 FHIRTiming obj = JsonUtils.MAPPER.treeToValue(node, FHIRTiming.class);
63 FHIRTiming maskedObj = mask(obj);
64 return JsonUtils.MAPPER.valueToTree(maskedObj);
65 } catch (Exception e) {
66 return NullNode.getInstance();
67 }

70 private Collection<String> maskEvents(Collection<String> events) {
71 Collection<String> maskedEvents = new ArrayList<>();
72
73 for (String event : events) {
74 if (event != null) {
75 maskedEvents.add(eventMaskingProvider.mask(event));
76 }
77 }
78
79 return maskedEvents;
80 }
81
82 public FHIRTiming mask(FHIRTiming timing) {
83
84 if (this.maskEvent && !isAlreadyMasked(EVENT_PATH)) {
85 Collection<String> events = timing.getEvent();
86 if (events != null) {
87 timing.setEvent(maskEvents(events));
88 }
89 }
90
91 if (this.maskCode && !isAlreadyMasked(CODE_PATH)) {
92 timing.setCode(codeMaskingProvider.mask(timing.getCode()));
93 }
94
95 return timing;
96 }

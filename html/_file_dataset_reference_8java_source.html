102 @Override
103 public Dataset<Row> readDataset(SparkSession sparkSession, String input) {
104 DataFrameReader reader = sparkSession.read();
105
106 if (basePath != null) {
107 reader = reader.option("basePath", basePath);
108 }
109
110 switch (this.format) {
111 case CSV:
112 CSVDatasetOptions csvDatasetOptions = (CSVDatasetOptions) this.options;
113 reader = reader
114 .option("sep", Character.toString(csvDatasetOptions.getFieldDelimiter()))
115 .option("quote", Character.toString(csvDatasetOptions.getQuoteChar()))
116 .option("header", csvDatasetOptions.isHasHeader());
117
118 if (csvDatasetOptions.isTrimFields()) {
119 reader = reader
120 .option("ignoreLeadingWhiteSpace", true)
121 .option("ignoreTrailingWhiteSpace", true);
122 }
123
124 Dataset<Row> dataset = reader.csv(input);
125
126 if (csvDatasetOptions.isHasHeader()) {
127 return dataset;
128 } else {
129 return dataset.toDF(CSVFormatProcessor.generateColumnNames(dataset.schema().fields().length));
130 }
131 case PARQUET:
132 return reader.parquet(input);
133 default:
134 return reader.text(input);
135 }
136 }

138 public void writeDataset(Dataset<Row> outputDataset, String path) {
139 // Prepare partitions
140 List<String> partitions;
141 if (this.partitionBy != null && !this.partitionBy.isEmpty()) {
142 partitions = Arrays.asList(this.partitionBy.split(","));
143 } else {
144 partitions = Collections.emptyList();
145 }
146
147 Export.doExport(outputDataset, format, buildFinalPath(path), partitions, this.append);
148 }
149
150 private String buildFinalPath(String path) {
151 if (null != datasourceURL) {
152 if (path.startsWith("/")) {
153 path = path.substring(1);
154 }
155 if (datasourceURL.endsWith("/")) {
156 return datasourceURL + path;
157 }
158 return datasourceURL + "/" + path;
159 }
160 return path;
161 }

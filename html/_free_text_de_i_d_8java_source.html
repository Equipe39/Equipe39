88 @Override
89 public void processFile(InputStream input, OutputStream output) throws MisconfigurationException {
90 if (taskOptions.getAnnotate()) {
91 try (PrintStream printStream = new PrintStream(output)) {
92 processor.applyFunction(input, printStream, annotator, null, entity -> "[[" + getEntityType(entity) + "]" + entity.getText() + "]");
93 }
94 } else {
95 processor.maskDocument(input, output, maskingProvider);
96 }
97 }

63 private static GeneralizationHierarchy getDefaultHierarchy(String hierarchyType, JsonNode hierarchyConfig) {
64 switch (hierarchyType) {
65 case "COUNTRY":
66 return CountryHierarchy.getInstance();
67 case "CITY":
68 return CityHierarchy.getInstance();
69 case "GENDER":
70 return GenderHierarchy.getInstance();
71 case "RACE":
72 return RaceHierarchy.getInstance();
73 case "MARITAL_STATUS":
74 return MaritalStatusHierarchy.getInstance();
75 case "YOB":
76 return YOBHierarchy.getInstance();
77 case "ICDV9":
78 return ICDv9Hierarchy.getInstance();
79 case "ZIPCODE":
80 return ZIPCodeHierarchy.getInstance();
81 case "ZIPCODE_MATERIALIZED":
82 return ZIPCodeMaterializedHierarchy.getInstance();
83 case "ZIPCODE_COMPUTATIONAL":
84 return new ZIPCodeCompBasedHierarchy();
85 case "HEIGHT":
86 return HeightHierarchy.getInstance();
87 case "RELIGION":
88 return ReligionHierarchy.getInstance();
89 case "DATE-YYYY-MM-DD":
90 return new DateYYYYMMDDHierarchy();
91 case "DATE":
92 if (hierarchyConfig != null && hierarchyConfig.has("format") && !hierarchyConfig.get("format").isNull())
93 return new DateHierarchy(hierarchyConfig.get("format").asText());
94 return new DateHierarchy();
95 case "LAT_LON":
96 return new LatitudeLongitudeHierarchy();
97 default:
98 return constructHierarchy(hierarchyType);
99 }
100 }

102 private static GeneralizationHierarchy constructHierarchy(String hierarchyType) {
103 try {
104 Class<? extends GeneralizationHierarchy> hierarchyClass = (Class<? extends GeneralizationHierarchy>) Class.forName(hierarchyType);
105
106 return hierarchyClass.getConstructor().newInstance();
107 } catch (ClassNotFoundException e) {
108 logger.error("Unable to instantiate hierarchy class {}", hierarchyType);
109 throw new RuntimeException(e);
110 } catch (InstantiationException | InvocationTargetException | IllegalAccessException e) {
111 logger.error("Unable to create instance of hierarchy {}", hierarchyType);
112 throw new RuntimeException(e);
113 } catch (NoSuchMethodException e) {
114 logger.error("Unable to retrieve empty constructor for hierarchy {}", hierarchyType);
115 throw new RuntimeException(e);
116 }
117 }

119 public static GeneralizationHierarchy buildHierarchy(JsonNode hierarchySpec) {
120 MaterializedHierarchy hierarchy = new MaterializedHierarchy();
121
122 hierarchySpec.get("terms").forEach(hierarchyPath -> {
123 List<String> pathTerms = new ArrayList<>();
124
125 hierarchyPath.elements().forEachRemaining(term -> pathTerms.add(term.asText()));
126
127 hierarchy.add(pathTerms);
128 });
129
130 return hierarchy;
131 }

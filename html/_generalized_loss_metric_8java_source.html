38 private double getLossNumerical(String anonymized, NumericalRange columnInformation) {
39 Double lowest = columnInformation.getLow();
40 Double highest = columnInformation.getHigh();
41
42 String[] tokens = anonymized.split("-");
43
44 if (tokens.length == 1) {
45 throw new RuntimeException("not implemented yet");
46 }
47
48 Long low = Long.valueOf(tokens[0]);
49 Long high = Long.valueOf(tokens[1]);
50
51 return ((double) (high - low)) / (highest - lowest);
52 }

54 private double getLossCategorical(String anonymized, CategoricalInformation columnInformation) {
55 long hierarchyLeaves = columnInformation.getHierarchy().getTotalLeaves();
56
57 int leaves = columnInformation.getHierarchy().leavesForNode(anonymized);
58
59 if (leaves == 0) {
60 return 0.0;
61 }
62
63 return ((double) (leaves - 1)) / ((double) (hierarchyLeaves - 1));
64 }

144 public double report() {
145
146 int numberOfColumns = original.getNumberOfColumns();
147 List<Double> lossPerColumn = new ArrayList<>(numberOfColumns);
148
149 for (int k = 0; k < numberOfColumns; k++) {
150 lossPerColumn.add(0.0);
151 }
152
153 int anonymizedNumberOfRows = anonymized.getNumberOfRows();
154
155 for (int i = 0; i < anonymizedNumberOfRows; i++) {
156
157 for (int j = 0; j < numberOfColumns; j++) {
158
159 ColumnInformation columnInformation = columnInformationList.get(j);
160 if (columnInformation.getColumnType() != ColumnType.QUASI) {
161 continue;
162 }
163
164 double loss = getLoss(anonymized.get(i, j), columnInformation);
165 lossPerColumn.set(j, lossPerColumn.get(j) + loss);
166 }
167 }
168
169 double lm = 0.0;
170
171 if (anonymized.getNumberOfRows() < original.getNumberOfRows()) {
172 int diff = original.getNumberOfRows() - anonymized.getNumberOfRows();
173
174 for (int j = 0; j < numberOfColumns; j++) {
175 ColumnInformation columnInformation = columnInformationList.get(j);
176 if (columnInformation.getColumnType() != ColumnType.QUASI) {
177 continue;
178 }
179
180 lossPerColumn.set(j, lossPerColumn.get(j) + (double) diff);
181 }
182 }
183
184 for (int k = 0; k < numberOfColumns; k++) {
185 double weight = columnInformationList.get(k).getWeight();
186 double averageColumnLoss = weight * lossPerColumn.get(k) / ((double) original.getNumberOfRows());
187 lm += averageColumnLoss;
188 }
189
190
191 return lm;
192 }

194 private InformationLossResult reportForColumn(int columnIndex) {
195
196 ColumnInformation columnInformation = columnInformationList.get(columnIndex);
197 if (columnInformation.getColumnType() != ColumnType.QUASI) {
198 return null;
199 }
200
201 double lossPerColumn = 0.0;
202
203 int anonymizedNumberOfRows = anonymized.getNumberOfRows();
204 double weight = columnInformationList.get(columnIndex).getWeight();
205
206 for (int i = 0; i < anonymizedNumberOfRows; i++) {
207 double loss = getLoss(anonymized.get(i, columnIndex), columnInformation);
208 lossPerColumn += loss;
209 }
210
211 if (anonymized.getNumberOfRows() < original.getNumberOfRows()) {
212 int diff = original.getNumberOfRows() - anonymized.getNumberOfRows();
213 lossPerColumn += weight * diff;
214 }
215
216 double averageColumnLoss = weight * lossPerColumn / ((double) original.getNumberOfRows());
217
218 return new InformationLossResult(averageColumnLoss, 0.0, 1.0);
219 }

28public class Geometric implements DPMechanism {
29 protected Random rnd = new SecureRandom();
30 protected double shape;
31 protected double epsilon;
32 protected double lowerBound;
33 protected double upperBound;
34
35 @Override
36 public void setOptions(AnonymizationAlgorithmOptions options) {
37 if (!(options instanceof DifferentialPrivacyMechanismOptions)) throw new IllegalArgumentException("Expecting instance of DifferentialPrivacyMechanismOptions");
38
39 this.epsilon = ((DifferentialPrivacyMechanismOptions) options).getEpsilon();
40
41 if (this.epsilon < 0.0) {
42 throw new RuntimeException("Epsilon parameter must be positive");
43 }
44 List<Double> bounds = ((DifferentialPrivacyMechanismOptions) options).getBounds();
45 this.lowerBound = bounds.get(0);
46 this.upperBound = bounds.get(1);
47
48 if (this.lowerBound > this.upperBound) {
49 throw new RuntimeException("Upper bound must be greater than lower bound");
50 }
51
52 this.shape = (this.upperBound - this.lowerBound)/this.epsilon;
53 }
54
55 @Override
56 public void analyseForParams(List<Partition> equivalenceClasses, int columnIndex) {
57 this.lowerBound = Double.POSITIVE_INFINITY;
58 this.upperBound = Double.NEGATIVE_INFINITY;
59
60 for (Partition partition : equivalenceClasses) {
61 for (List<String> row : partition.getMember()) {
62 double value = Double.parseDouble(row.get(columnIndex));
63
64 if (value < this.lowerBound) {
65 this.lowerBound = Math.floor(value);
66 }
67
68 if (value > this.upperBound) {
69 this.upperBound = Math.ceil(value);
70 }
71 }
72 }
73
74 double diameter = this.upperBound - this.lowerBound;
75
76 this.shape = diameter/this.epsilon;
77 }
78
79 @Override
80 public double randomise(double value) {
81 double u = this.rnd.nextDouble() - 0.5;
82 double sgn = (u < 0) ? -1 : 1;
83 u *= sgn * (Math.exp(1 / this.shape) + 1) / (Math.exp(1 / this.shape) - 1);
84
85 double cumProb = -0.5;
86 int i = -1;
87
88 while (u > cumProb) {
89 i += 1;
90 cumProb += Math.exp(- i / this.shape);
91 }
92
93 return value + (sgn * i);
94 }

96 @Override
97 public String getName() {
98 return "Native Geometric mechanism";
99 }

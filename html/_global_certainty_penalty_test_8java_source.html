63 Mondrian mondrian = new Mondrian();
64
65 List<PrivacyConstraint> privacyConstraints = new ArrayList<>();

67
68 mondrian.initialize(dataset, columnInformationList, privacyConstraints, new MondrianOptions()); //dataset, 2, 1, columnInformationList);
69

71
72 InformationMetric globalCertaintyPenalty = new GlobalCertaintyPenalty().initialize(dataset, anonymizedDataset,
73 mondrian.getOriginalPartitions(), mondrian.getAnonymizedPartitions(), columnInformationList, null);
74
75 //partition 1: NCP0 = 4/9, NCP1 = 9/10
76 //partition 2: NCP0 = 4/9, NCP1 = 7/10
77 //NCP(p1) = 1 * 4/9 + 1 * 9/10 = 4/9 + 9/10 = 1.34
78 //NCP(p2) = 1 * 4/9 + 1 * 7/10 = 4/9 + 7/10 = 1.14
79 //GCP = (2 * NCP(p1) + 3 * NCP(p2)) / 2 * 5 = (2* 1.34 + 3 * 1.14) / 10 = 6.10 / 10 = 0.61
80
81 Double gcp = globalCertaintyPenalty.report();
82// System.out.println("GCP = " + gcp);
83
84 double ncpP1 = 4.0/9.0 + 9.0/10.0;
85 double ncpP2 = 4.0/9.0 + 7.0/10.0;
86 assertEquals(gcp, ((2 * ncpP1 + 3 * ncpP2) / (2.0 * 5.0)));

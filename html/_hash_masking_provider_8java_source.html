70 public static String bytesToHex(byte[] bytes) {
71 char[] hexChars = new char[bytes.length * 2];
72 for (int j = 0; j < bytes.length; j++) {
73 int v = bytes[j] & 0xFF;
74 hexChars[j * 2] = hexArray[v >>> 4];
75 hexChars[j * 2 + 1] = hexArray[v & 0x0F];

36public class HolidayIdentifier extends AbstractIdentifier {
37 private final Set<String> terms;
38 private final int minimumLength;
39 private final int maximumLength;
40
41 public HolidayIdentifier() {
42 terms = populateTerms();
43
44 int min = Integer.MAX_VALUE;
45 int max = Integer.MIN_VALUE;
46
47 for (String key : terms) {
48 min = Math.min(min, key.length());
49 max = Math.max(max, key.length());
50 }
51
52 this.minimumLength = min;
53 this.maximumLength = max;
54 }
55
56 private Set<String> populateTerms() {
57 try (InputStream inputStream = HolidayIdentifier.class.getResourceAsStream("/holidays.csv")) {
58 CsvMapper mapper = new CsvMapper().enable(CsvParser.Feature.WRAP_AS_ARRAY);
59
60 MappingIterator<String[]> termsIterator = mapper.readerFor(
61 String[].class
62 ).with(CsvSchema.emptySchema().withoutHeader()).readValues(inputStream);
63
64 return StreamSupport.stream(
65 Spliterators.spliteratorUnknownSize(
66 termsIterator,
67 Spliterator.ORDERED
68 ), true).
69 map(s -> s[0].trim()).
70 filter(((Predicate<String>)String::isEmpty).negate()).
71 collect(Collectors.toSet());
72 } catch (IOException e) {
73 throw new RuntimeException("Unable to load the processed terms", e);
74 }
75 }

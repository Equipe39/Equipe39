88 private String ipv4mask(String identifier) {
89 String[] parts = identifier.split("\\.");
90 String[] maskedParts = new String[4];
91
92 if (this.preservedPrefixes >= 4) {
93 return identifier;
94 }
95
96 for (int i = 0; i < parts.length; i++) {
97 if (i < this.preservedPrefixes) {
98 maskedParts[i] = parts[i];
99 } else {
100 maskedParts[i] = randomSubnet();
101 }
102 }
103
104 return StringUtils.join(maskedParts, '.');
105 }
106
107 private String ipv6mask(String identifier) {
108
109 String suffix = "";
110
111 int idx = identifier.indexOf('%');
112 if (idx > 0) {
113 suffix = identifier.substring(idx);
114 identifier = identifier.substring(0, idx);
115 }
116
117 String[] parts = identifier.split(":", -1);
118 String[] maskedParts = new String[parts.length];
119
120 for (int i = 0; i < parts.length; i++) {
121 String part = parts[i];
122
123 if (part.isEmpty()) {
124 maskedParts[i] = "";
125 } else if (ipAddressIdentifier.isIPv4(part)) {
126 maskedParts[i] = ipv4mask(part);
127 } else {
128 maskedParts[i] = RandomGenerators.randomHexSequence(Math.max(part.length() / 2, 1));
129 }
130
131 }
132
133 return StringUtils.join(maskedParts, ':');
134 }

153 @Override
154 public String mask(String identifier) {
155 if (ipAddressIdentifier.isIPv4(identifier)) {
156 return ipv4mask(identifier);
157 } else if (ipAddressIdentifier.isIPv6(identifier)) {
158 return ipv6mask(identifier);
159 }
160
161 switch (failMode) {
162 case FailMode.RETURN_ORIGINAL:
163 return identifier;
164 case FailMode.THROW_ERROR:
165 logger.error("invalid IP address");
166 throw new IllegalArgumentException("invalid IP address");
167 case FailMode.RETURN_EMPTY:
168 return "";
169 case FailMode.GENERATE_RANDOM:
170 default:
171 return String.format("%d.%d.%d.%d", random.nextInt(255), random.nextInt(255), random.nextInt(255), random.nextInt(255));
172 }

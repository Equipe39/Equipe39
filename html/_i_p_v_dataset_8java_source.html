152 public static IPVDataset load(Reader reader, final boolean hasHeader, Character fieldDelimiter, Character quoteCharacter, boolean trimFields) throws IOException {
153 CsvMapper mapper = new CsvMapper();
154 CsvSchema schema = CsvSchema.emptySchema().withColumnSeparator(fieldDelimiter).withQuoteChar(quoteCharacter).withSkipFirstDataRow(false);
155 mapper.enable(CsvParser.Feature.WRAP_AS_ARRAY);
156 if (trimFields) {
157 mapper.enable(CsvParser.Feature.TRIM_SPACES);
158 }
159
160 MappingIterator<String[]> it = mapper.readerFor(String[].class).with(schema).readValues(reader);
161
162 List<String> header = null;
163 List<List<String>> values = new ArrayList<>();
164 int numberOfFields = -1;
165 boolean skipFirst = hasHeader;
166
167 while (it.hasNext()) {
168 List<String> csvRecord = Arrays.asList(it.next());
169
170 if (skipFirst) {
171 header = csvRecord;
172 skipFirst = false;
173 continue;
174 }
175 if (csvRecord.isEmpty()) {
176 continue;
177 }
178 if (numberOfFields == -1) {
179 numberOfFields = csvRecord.size();
180 }
181 if (csvRecord.size() == numberOfFields) {
182 values.add(csvRecord);
183 } else {
184 logger.warn("Record has a different size than what expected: {} instead of {}", csvRecord.size(), numberOfFields);
185 }
186 }
187
188 return new IPVDataset(
189 values,
190 hasHeader ? generateSchemaWithColumnNames(header) : generateSchemaWithoutColumnNames(numberOfFields),
191 null != header
192 );
193 }

247 public void toCSV(CSVDatasetOptions options, Appendable writer) {
248 CSVFormat format = CSVFormat.DEFAULT
249 .withRecordSeparator('\n')
250 .withDelimiter(options.getFieldDelimiter())
251 .withQuote(options.getQuoteChar())
252 .withTrim(options.isTrimFields())
253 .withHeader().withSkipHeaderRecord(!options.isHasHeader());
254
255 if (options.isHasHeader()) {
256 format = format.withHeader(schema.getFields().stream().map(IPVSchemaField::getName).toArray(String[]::new));
257 }
258
259 try (CSVPrinter printer = new CSVPrinter(writer, format)) {
260 printer.printRecords(this);
261 } catch (IOException e) {
262 logger.error("Error creating writer", e);
263 throw new RuntimeException(e);
264 }

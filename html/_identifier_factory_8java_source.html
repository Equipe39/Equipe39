157 public static IdentifierFactory initializeIdentifiers(JsonNode identifiers) {
158 IdentifierFactory factory = new IdentifierFactory(Collections.emptySet());
159
160 if (identifiers.isArray()) {
161 identifiers.elements().forEachRemaining(
162 identifier -> {
163 if (null == identifier || identifier.isNull()) return;
164
165 if (identifier.isTextual()) {
166 try {
167 factory.registerIdentifier((Class<? extends Identifier>) Class.forName(identifier.asText()));
168 } catch (ClassNotFoundException | InstantiationException | IllegalAccessException e) {
169 logger.error("Error initializing class", e);
170 }
171 } else if (identifier.isObject()) {
172 final PluggableIdentifierType type = PluggableIdentifierType.valueOf(identifier.get("type").asText());
173
174 switch (type) {
175 case DICTIONARY:
176 factory.registerIdentifier(buildDictionaryBasedUserDefinedIdentifier(identifier));
177 break;
178 case REGEX:
179 factory.registerIdentifier(buildRegexBasedUserDefinedIdentifier(identifier));
180 break;
181 default:
182 logger.error("Unknown type: " + type);
183 throw new RuntimeException("Unknown type " + type);
184 }
185 } else {
186 logger.error("Not supported node type: " + identifier.getNodeType().toString());
187 throw new RuntimeException("Not supported node type");
188 }
189 }
190 );
191 } else if (identifiers.isObject()) {
192 JsonNode resourceTypeNode = identifiers.get("resourceType");
193 if (resourceTypeNode == null) {
194 throw new RuntimeException("Missing resourceType from identifiers");
195 }
196
197 if (identifiers.has("path")) {
198 logger.info("Load identifiers from a path: " + identifiers.get("path").toString());
199 readClassNamesFromFile(
200 identifiers.get("path").asText(), ResourceEntryType.valueOf(resourceTypeNode.asText())).forEach(
201 identifierClassName -> {
202 try {
203 factory.registerIdentifier((Class<? extends Identifier>) Class.forName(identifierClassName));
204 } catch (ClassNotFoundException | InstantiationException | IllegalAccessException e) {
205 logger.error("Error initializing class", e);
206 }
207 });
208 } else {
209 if (!identifiers.has("paths")) {
210 throw new RuntimeException("Missing path from identifiers");
211 }
212
213 JsonNode pathsNode = identifiers.get("paths");
214
215 pathsNode.elements().forEachRemaining(pathNode -> readClassNamesFromFile(
216 pathNode.asText(), ResourceEntryType.valueOf(resourceTypeNode.asText())).forEach(
217 identifierClassName -> {
218 try {
219 factory.registerIdentifier((Class<? extends Identifier>) Class.forName(identifierClassName));
220 } catch (ClassNotFoundException | InstantiationException | IllegalAccessException e) {
221 logger.error("Error initializing class", e);
222 }
223 }
224 ));
225 }
226 }
227
228 return factory;
229 }

232 ResourceEntry resourceEntry = new ResourceEntry(

239 Properties properties = new Properties();

242 Set<String> identifiers = new HashSet<>(properties.size());
243 for (Enumeration<?> names = properties.propertyNames(); names.hasMoreElements(); ) {

284 private static List<String> extractRegularExpressions(JsonNode spec) {
285 List<String> expressions = new ArrayList<>();
286
287 if (spec.has("regex")) {
288 spec.get("regex").elements().forEachRemaining(e -> {
289 if (null != e && e.isTextual()) {
290 expressions.add(e.asText().trim());
291 }
292 });
293 } else if (spec.has("paths")) {
294 spec.get("paths").elements().forEachRemaining(
295 path -> expressions.addAll(readStringsFromFile(path.asText()))
296 );
297 } else {
298 throw new RuntimeException("No specification available");
299 }
300
301 return expressions;
302 }

304 private static Identifier buildDictionaryBasedUserDefinedIdentifier(JsonNode spec) {
305 boolean isPosIndependent = !spec.has("isPosIndependent") || spec.get("isPosIndependent").asBoolean();
306 boolean ignoreCase = !spec.has("ignoreCase") || spec.get("ignoreCase").asBoolean();
307
308 return new PluggableLookupIdentifier(
309 spec.get("providerType").asText(),
310 Collections.emptyList(),
311 buildTermsList(spec),
312 ignoreCase,
313 ValueClass.TEXT,
314 isPosIndependent
315 );

318 private static Collection<String> buildTermsList(JsonNode spec) {
319 Set<String> terms = new HashSet<>();
320
321 if (spec.has("terms")) {
322 spec.get("terms").elements().forEachRemaining(term -> terms.add(term.asText().trim()));
323 }
324 if (spec.has("paths")) {
325 spec.get("paths").elements().forEachRemaining(
326 path -> terms.addAll(readStringsFromFile(path.asText()))
327 );
328 }
329 if (terms.isEmpty()) throw new RuntimeException("No terms specified");
330
331 return terms;
332 }

334 private static Collection<? extends String> readStringsFromFile(String path) {
335 Set<String> terms = new HashSet<>();
336 for (File file : populateFilesToProcess(path)) {
337 try {
338 terms.addAll(
339 Files.lines(file.toPath()).map(String::trim).collect(Collectors.toList())
340 );
341 } catch (IOException e) {
342 throw new RuntimeException(e);
343 }
344 }
345
346 return terms;
347 }

349 private static File[] populateFilesToProcess(String fileName) {
350 final File fileToProcess = new File(fileName);
351
352 if (!fileToProcess.exists()) throw new RuntimeException(fileToProcess.getAbsoluteFile() + " does not exist");
353
354 if (fileToProcess.isDirectory()) {
355 return Objects.requireNonNull(fileToProcess.listFiles());
356 } else {
357 return new File[]{fileToProcess};
358 }

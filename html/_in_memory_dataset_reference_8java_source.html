38public class InMemoryDatasetReference extends DatasetReference {

43 public InMemoryDatasetReference(List<List<String>> data, List<String> columnNames) {
44 if (data != null && columnNames != null) {
45 if (!data.isEmpty()) {
46 for (List<String> row : data) {
47 if (row.size() != columnNames.size())
48 throw new IllegalArgumentException("data rows and columnNames must have same size");
49 }
50 }
51
52 this.data = data;
53 this.columnNames = columnNames;
54 } else {
55 throw new IllegalArgumentException("Data or column names are undefined");
56 }
57 }
58
59 public InMemoryDatasetReference() {
60 this(Collections.emptyList(), Collections.emptyList());
61 }

63 @Override
64 public Dataset<Row> readDataset(SparkSession sparkSession, String inputReference) {
65 if (null == this.dataset) {
66 this.dataset = createDataset(sparkSession);
67 }
68
69 return dataset;
70 }

72 private Dataset<Row> createDataset(SparkSession sparkSession) {
73 return sparkSession.createDataFrame(
74 createData(),
75 createSchema()
76 );
77 }
78
79 private StructType createSchema() {
80 if (null == this.columnNames) throw new IllegalArgumentException();
81
82 return new StructType(this.columnNames.stream().map(name -> new StructField(name, DataTypes.StringType, false, Metadata.empty())).toArray(StructField[]::new));
83 }
84
85 private List<Row> createData() {
86 if (this.data == null) throw new IllegalArgumentException();
87
88 return this.data.stream().map(dataPoint -> RowFactory.create(dataPoint.toArray())).collect(Collectors.toList());
89 }
90
91 @Override
92 public void writeDataset(Dataset<Row> outputDataset, String path) {
93 this.dataset = outputDataset;
94 }

96 @Override
97 public String toString() {
98 if (this.dataset != null) {
99 return this.dataset.toString();
100 } else {
101 if (this.data != null && this.columnNames != null) {
102 Stream<String> headers = Stream.of(String.join(",", columnNames));
103 Stream<String> dataRows = this.data.stream().map(row -> String.join(",", row));
104
105 return Stream.concat(headers, dataRows).collect(Collectors.joining("\n"));
106 }
107 }
108 return "InMemoryDatasetReference[NotInitialized]";
109 }

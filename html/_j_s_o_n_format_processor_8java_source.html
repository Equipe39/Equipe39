45 private static Map<ProviderType, Long> identifyJSONElement(JsonNode node) {
46 if (node == null || node.isNull()) {
47 Map<ProviderType, Long> results = new HashMap<>();
48 results.put(ProviderType.EMPTY, 1L);
49 return results;
50 }
51
52 if (node.isObject()) {
53 Map<ProviderType, Long> results = new HashMap<>();
54 results.put(ProviderType.UNKNOWN, 1L);
55 return results;
56 }
57
58 if (node.isArray()) {
59 return identifyJSONArrayElement(node);
60 }
61
62 if (node.isNumber() || node.isTextual()) {
63 return IdentifierUtils.identifySingleValue(node.asText());
64 }
65
66 if (node.isBoolean()) {
67 Map<ProviderType, Long> results = new HashMap<>();
68 results.put(ProviderType.BOOLEAN, 1L);
69 return results;
70 }
71
72 return null;
73 }
74
75 private static Map<ProviderType, Long> identifyListOfElements(Iterator<JsonNode> iterator) {
76 Map<ProviderType, Long> results = new HashMap<>();
77
78 while (iterator.hasNext()) {
79 JsonNode element = iterator.next();
80 Map<ProviderType, Long> elementResults = identifyJSONElement(element);
81
82 if (elementResults == null) {
83 continue;
84 }
85
86 for (Map.Entry<ProviderType, Long> entry : elementResults.entrySet()) {
87 ProviderType providerType = entry.getKey();
88 Long counter = results.get(providerType);
89
90 if (counter == null) {
91 counter = 1L;
92 } else {
93 counter += 1;
94 }
95
96 results.put(providerType, counter);
97 }
98 }
99
100 return results;
101 }

127 private MappingIterator<JsonNode> createIterators(InputStream dataset, DatasetOptions datasetOptions) throws IOException {
128 final JsonParser parser = JsonUtils.MAPPER.getFactory().createParser(dataset);
129
130 if (null != datasetOptions) {
131 if (!(datasetOptions instanceof JSONDatasetOptions)) {
132 throw new IllegalArgumentException("Dataset masking options not consistent with the format processor: JSON");
133 }
134 }
135
136 return JsonUtils.MAPPER.readerFor(JsonNode.class).readValues(parser);
137 }

139 @Override
140 // NOTE: assumes that we consider the last field, excluding * and such
141 protected String extractFieldName(String fieldName) {
142 final String[] parts = fieldName.split("/");
143
144 for (int i = parts.length - 1; i >= 0; --i) {
145 final String part = parts[i].trim();
146
147 if (part.isEmpty()) continue;
148 if ("*".equals(part)) continue;
149
150 return part;
151 }
152 return "";
153 }

47 public JsonNode update(JsonNode obj, JsonNode value) {
48 List<String> list = Arrays.asList(pattern.split("/"));
49 JsonNode node;
50 if (list.size() > 2) {
51 // because of "" before the first / and the second one will be the field we want to modify
52 String newPattern = StringUtils.join(list.subList(0, list.size() - 1), "/");
53
54 node = obj.at(newPattern);
55 } else {
56 node = obj.at("");
57 }
58
59 if (node.isArray()) {
60 ((ArrayNode) node).set(Integer.parseInt(list.get(list.size() - 1), 10), value);
61 } else if (node.isObject()) {
62 ((ObjectNode) node).set(list.get(list.size() - 1), value);
63 }
64
65 return obj;
66 }

67 private static ValueNode createNodeValue(JsonNode node) {
68 JsonNodeType type = node.getNodeType();
69 switch (type) {
70 case STRING:
71 return new TextNode(node.asText());
72 case NUMBER:
73 if (node.isDouble()) {
74 return new DoubleNode(node.asDouble());
75 } else if (node.isInt()) {
76 return new IntNode(node.asInt());
77 } else if (node.isLong()) {
78 return new LongNode(node.asLong());
79 }
80 break;
81 case BOOLEAN:
82 return BooleanNode.valueOf(node.asBoolean());
83 default:
84 logger.info("Unexpected value: {}", type);
85 }
86
87 return NullNode.getInstance();
88 }

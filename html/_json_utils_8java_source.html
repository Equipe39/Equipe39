55 private static void mergeMaps(Map<String, List<JsonNode>> a, Map<String, List<JsonNode>> b) {

78 private static Map<String, List<JsonNode>> traverseObject(JsonNode node, String parentPath) {
79 Map<String, List<JsonNode>> pathMap = new HashMap<>();
80
81 if (node.isObject()) {
82 Iterator<String> iterator = node.fieldNames();
83 while (iterator.hasNext()) {
84 String key = iterator.next();
85 JsonNode value = node.get(key);
86 Map<String, List<JsonNode>> innerPaths = traverseObject(value, parentPath + "/" + key);
87 mergeMaps(pathMap, innerPaths);
88 }
89 } else {
90 String newParent = parentPath;
91
92 if (node.isArray()) {
93 for (int i = 0; i < node.size(); ++i) {
94 JsonNode n = node.get(i);
95 if (n.isObject()) {
96 Map<String, List<JsonNode>> innerPaths = traverseObject(n, newParent);
97 mergeMaps(pathMap, innerPaths);
98 }
99 }
100 }
101
102 if (parentPath.isEmpty()) {
103 newParent += "/";
104 }
105
106 addToMap(pathMap, newParent, node);
107 }
108
109 return pathMap;
110
111 }

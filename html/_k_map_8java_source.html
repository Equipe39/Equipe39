89 private Partition anonymizePartition(Partition eqClass, List<LatticeNode> nodes, int suppressionBudget) {
90 if (eqClass.size() == 0) {
91 return new InMemoryPartition(Collections.emptyList());
92 }
93
94 boolean constraintMatch = AnonymizationUtils.checkPrivacyConstraints(this.privacyConstraints, eqClass, null);
95
96 if (constraintMatch) {
97 Partition cp = copyPartition(eqClass);
98 cp.setAnonymous(true);
99 return cp;
100 }
101
102 if (suppressionBudget >= eqClass.size()) {
103 return null;
104 }
105
106 for (LatticeNode node : nodes) {
107 int[] level = node.getValues();
108 IPVDataset anonymized = DatasetGeneralizer.generalize(eqClass.getMember(), this.columnInformationList, level);
109 Partition p = new InMemoryPartition(anonymized);
110
111 if (AnonymizationUtils.checkPrivacyConstraints(this.privacyConstraints, p, null)) {
112 p.setAnonymous(true);
113 return p;
114 }
115 }
116
117 throw new RuntimeException("unreachable");
118 }

124 private List<LatticeNode> calculateNodes() {
125
126 List<ColumnInformation> quasiColumnInformationList = this.columnInformationList.stream().filter(columnInformation -> {
127 return columnInformation.getColumnType() == ColumnType.QUASI;
128 }).collect(Collectors.toList());
129
130 int[] maxLevel = new int[quasiColumnInformationList.size()];
131
132 for (int i = 0; i < quasiColumnInformationList.size(); i++) {
133 CategoricalInformation categoricalInformation = (CategoricalInformation) quasiColumnInformationList.get(i);
134 int maximumLevel = categoricalInformation.getMaximumLevel();
135 maxLevel[i] = maximumLevel >= 0 ? maximumLevel : categoricalInformation.getHierarchy().getHeight();
136 }
137
138 int[][] levels = new int[quasiColumnInformationList.size()][];
139
140 for (int i = 0; i < quasiColumnInformationList.size(); i++) {
141 int maxCurrentLevel = maxLevel[i];
142 levels[i] = new int[maxCurrentLevel];
143 for (int k = 0; k < maxCurrentLevel; k++) {
144 levels[i][k] = k;
145 }
146 }
147
148 List<LatticeNode> nodes = new ArrayList<>();
149
150 List<List<Integer>> productResult = Lattice.calculateProduct(levels);
151 for (List<Integer> l : productResult) {
152 nodes.add(new LatticeNode(l));
153 }
154
155 nodes.sort(Comparator.comparingInt(LatticeNode::sum));
156 return nodes;
157
158 }

160 @Override
161 public IPVDataset apply() {
162 List<Integer> matchColumns = AnonymizationUtils.getColumnsByType(this.columnInformationList, ColumnType.QUASI);
163
164 int suppressionBudget = (int) Math.ceil((this.suppressionRate / 100.0) * this.originalDataset.getNumberOfRows());
165 this.suppressedRows = 0;
166
167 final List<Partition> partitions = AnonymizationUtils.generatePartitionsByColumnIndex(this.originalDataset, matchColumns);
168
169 this.anonymizedPartitions = new ArrayList<>(partitions.size());
170
171 List<LatticeNode> possibleNodes = calculateNodes();
172
173 this.originalPartitions = partitions;
174
175 for (Partition partition : partitions) {
176 Partition anonymizedPartition = anonymizePartition(partition, possibleNodes, suppressionBudget);
177
178 if (anonymizedPartition == null) {
179 suppressionBudget -= partition.size();
180 this.suppressedRows += partition.size();
181 Partition cp = copyPartition(partition);
182 cp.setAnonymous(false);
183 this.anonymizedPartitions.add(cp);
184 continue;
185 }
186
187 this.anonymizedPartitions.add(anonymizedPartition);
188 }
189
190 return createDataset(this.anonymizedPartitions, this.originalDataset.getNumberOfColumns());
191 }

193 private IPVDataset createDataset(List<Partition> anonymizedPartitions, int numberOfColumns) {
194 IPVDataset dataset = new IPVDataset(numberOfColumns);
195
196 for (Partition partition : anonymizedPartitions) {
197 if (!partition.isAnonymous()) {
198 continue;
199 }
200
201 IPVDataset members = partition.getMember();
202
203 for (int i = 0; i < members.getNumberOfRows(); i++) {
204 dataset.addRow(members.getRow(i));
205 }
206 }
207
208 return dataset;
209 }

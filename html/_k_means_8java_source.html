40 private List<MutableTuple<Double, Double>> findMinMax() {
41 List<MutableTuple<Double, Double>> minMaxValues = new ArrayList<>(this.columnsToCluster.size());
42
43 for (Integer column : columnsToCluster) {
44 minMaxValues.add(new MutableTuple<>(Double.MAX_VALUE, Double.MIN_VALUE));
45 }
46
47 for (List<String> row : this.dataset) {
48 for (int i = 0; i < columnsToCluster.size(); i++) {
49 Integer column = columnsToCluster.get(i);
50
51 double value = getValue(column, row);
52
53 MutableTuple<Double, Double> minMaxForColumn = minMaxValues.get(i);
54 minMaxForColumn.setFirst(Math.min(value, minMaxForColumn.getFirst()));
55 minMaxForColumn.setSecond(Math.max(value, minMaxForColumn.getSecond()));
56 }
57 }
58
59 return minMaxValues;
60 }

62 private List<KMeansCluster> initializeRandomCenters(int n) {
63 List<KMeansCluster> centers = new ArrayList<>(n);
64
65 List<MutableTuple<Double, Double>> minMaxValues = findMinMax();
66
67 for (int i = 0; i < n; i++) {
68
69 List<Double> meanPoint = new ArrayList<>();
70
71 for (MutableTuple<Double, Double> minMax : minMaxValues) {
72 double min = minMax.getFirst();
73 double max = minMax.getSecond();
74
75 double rnd = min + Math.random() * (max - min);
76 meanPoint.add(rnd);
77 }
78
79 centers.add(new KMeansCluster(meanPoint));
80 }
81
82 return centers;
83 }

85 private static double euclideanDistance(List<Double> row, List<Double> center) {
86 double distance = 0.0;
87
88 for (int i = 0; i < row.size(); i++) {
89 distance += Math.pow(row.get(i) - center.get(i), 2);
90 }
91
92 return Math.sqrt(distance);
93 }

99 static int getNearestPointIndex(List<Double> row, List<KMeansCluster> clusters, int excludeIndex) {
100 double minDistance = Double.MAX_VALUE;
101 int minIndex = Integer.MAX_VALUE;
102
103 for (int i = 0; i < clusters.size(); i++) {
104 KMeansCluster cluster = clusters.get(i);
105 List<Double> center = cluster.getCenter();
106
107 if (center == null) {
108 continue;
109 }
110
111 double distance = euclideanDistance(row, center);
112
113 if (distance < minDistance && i != excludeIndex) {
114 minDistance = distance;
115 minIndex = i;
116 }
117 }
118
119 return minIndex;
120 }

142 private boolean haveClustersConverged(List<KMeansCluster> oldCenters, List<KMeansCluster> newCenters) {
143 double accumDist = 0;
144
145 for (int i = 0; i < oldCenters.size(); i++) {
146 List<Double> oldCenter = oldCenters.get(i).getCenter();
147 List<Double> newCenter = newCenters.get(i).getCenter();
148
149 if (oldCenter == null && newCenter == null) {
150 continue;
151 }
152
153 if (oldCenter == null || newCenter == null) {
154 return false;
155 }
156
157 double dist = euclideanDistance(oldCenter, newCenter);
158 accumDist += dist;
159 }
160
161 return accumDist == 0;
162 }

164 private List<KMeansCluster> applyAlgorithm(List<KMeansCluster> centers) {
165
166 boolean converged;
167 int iterations = 0;
168
169 do {
170 List<KMeansCluster> newCenters = getNewCenters(centers);
171 converged = haveClustersConverged(centers, newCenters);
172 centers = newCenters;
173 iterations++;
174 } while (!converged && iterations < maxIterations);
175
176 return centers;
177 }

179 private double getValue(Integer column, List<String> row) {
180 ColumnInformation columnInfo = this.columnInformation.get(column);
181 String value = row.get(column);
182
183 if (columnInfo.isCategorical()) {
184 MaterializedHierarchy hierarchy = (MaterializedHierarchy) ((CategoricalInformation) columnInfo).getHierarchy();
185 Integer index = hierarchy.getIndex(value);
186 return (double) index;
187 } else {
188 return Double.parseDouble(value);
189 }

220 private List<KMeansCluster> assignToPartitions(List<KMeansCluster> clusters) {
221 int numOfRows = this.datasetNormalizedValues.size();
222
223 for (KMeansCluster cluster : clusters) {
224 cluster.initializePartition(this.dataset.getNumberOfColumns());
225 }
226
227 for (int i = 0; i < numOfRows; i++) {
228 List<Double> normalized = this.datasetNormalizedValues.get(i);
229 int index = getNearestPointIndex(normalized, clusters);
230
231 List<String> datasetRow = this.dataset.getRow(i);
232 clusters.get(index).addOriginalRow(datasetRow);
233 }
234
235 return clusters;
236 }

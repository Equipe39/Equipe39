74 public void computeCenter() {
75 if (values.size() == 0) {
76 return;
77 }
78
79 List<Double> newCenter = values.get(0);
80
81 for (int i = 1; i < values.size(); i++) {
82 List<Double> row = values.get(i);
83
84 for (int j = 0; j < row.size(); j++) {
85 newCenter.set(j, row.get(j) + newCenter.get(j));
86 }
87 }
88
89 newCenter.replaceAll(aDouble -> aDouble / (double) values.size());
90
91 this.center = newCenter;

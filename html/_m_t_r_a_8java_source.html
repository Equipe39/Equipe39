76 @Override
77 public Collection<IPVVulnerability> apply(final IPVDataset dataset) {
78 final IPVGenerator generator = new LayerGenerator(dataset.getNumberOfColumns());
79
80 final ConcurrentLinkedQueue<MTRATask> tasks = new ConcurrentLinkedQueue<>();
81 final ConcurrentLinkedQueue<MTRAResult> results = new ConcurrentLinkedQueue<>();
82
83 int parallelism = (int) (nThread * multiplier);
84 for (int i = 0; generator.hasNext() && i < parallelism; ++i) {
85 tasks.add(new MTRATask(generator.next()));
86 }
87
88 // init executors
89 final Thread[] executors = new Thread[nThread];
90 int counter = 0;
91 for (int i = 0; i < nThread; ++i) {
92 executors[i] = new Thread(() -> {
93 while (true) {
94 MTRATask task = tasks.poll();
95 MTRAResult result;
96
97 if (null == task) {
98 continue;
99 }
100
101 if (null == task.getItemSet()) {
102 results.add(new MTRAResult(null, -1));
103 return; // termination condition
104 }
105
106 final Map<String, MutableInteger> counters = new HashMap<>();
107
108 // process data
109 for (int row = 0; row < dataset.getNumberOfRows(); ++row) {
110 StringBuilder builder = new StringBuilder();
111 for (Integer column : task.getItemSet().getItems()) {
112 builder.append(',');
113 builder.append(dataset.get(row, column));
114 }
115 final String key = builder.toString();
116
117 MutableInteger counterValue = counters.get(key);
118
119 if (null != counterValue) {
120 counterValue.set(counterValue.get() + 1);
121 } else {
122 counters.put(key, new MutableInteger(1));
123 }
124 }
125
126 int k = Integer.MAX_VALUE;
127
128 for (Map.Entry<String, MutableInteger> counter1 : counters.entrySet()) {
129 k = Math.min(k, counter1.getValue().get());
130 }
131
132 result = new MTRAResult(task, k);
133
134 results.add(result);
135 }
136 });
137 counter += 1;
138 executors[i].start();
139 }
140
141 boolean terminated = false;
142
143 final List<IPVVulnerability> vulnerabilities = new LinkedList<>();
144
145 while (0 < counter) {
146 MTRAResult result = results.poll();
147
148 if (null == result) {
149 continue;
150 }
151
152 if (null == result.getTask()) {
153 counter -= 1;
154 continue;
155 }
156
157 vulnerabilities.add(new IPVVulnerability(result.getTask().getItemSet(), result.getkValue()));
158
159 if (generator.hasNext()) {
160 final MTRATask task = new MTRATask(generator.next());
161
162 tasks.add(task);
163 } else if (!terminated) {
164 for (int i = 0; i < nThread; ++i) {
165 tasks.add(new MTRATask(null));
166 }
167 terminated = true;
168 }
169 }
170
171
172 return vulnerabilities;
173 }

155 private String createNamespaceID(MaskingConfiguration configuration) {
156 String namespace = configuration.getStringValue("persistence.namespace");
157 String type = configuration.getStringValue(PERSISTENCE_TYPE_CONF_NAME);
158
159 if (!type.equals("file")) {
160 return namespace;
161 }

166 private MaskingProvider getPersistentMaskingProvider(MaskingConfiguration configuration, ProviderType providerType) {
167 String namespace = configuration.getStringValue("persistence.namespace");
168
169 if (namespace == null) {
170 throw new IllegalArgumentException("persistence.namespace is not defined");
171 }
172
173 String namespaceID = createNamespaceID(configuration);
174
175 if (!globalPersistent.containsKey(namespaceID)) {
176 logger.info("Persistent provider for " + namespaceID + " does not exist. Creating..");
177 globalPersistent.put(namespaceID, createPersistentWrapper(getProviderFromType(providerType, configuration), configuration));
178 }
179
180 return globalPersistent.get(namespaceID);
181 }
182
183 private MaskingProvider createPersistentWrapper(MaskingProvider providerFromType, MaskingConfiguration configuration) {
184 final AbstractPersistentMaskingProvider persistentMaskingProvider;
185 switch (AbstractPersistentMaskingProvider.PersistencyType.valueOf(configuration.getStringValue(PERSISTENCE_TYPE_CONF_NAME).toUpperCase())) {
186 case MEMORY:
187 persistentMaskingProvider = new LocallyPersistentMaskingProvider(providerFromType, configuration);
188 break;
189 case FILE:
190 persistentMaskingProvider = new FileBackedPersistentMaskingProvider(providerFromType, configuration);
191 break;
192 case DATABASE:
193 persistentMaskingProvider = new DBPersistentMaskingProvider(providerFromType, configuration);
194 break;
195 case CAUSAL:
196 persistentMaskingProvider = new CausalOrderingConsistentMaskingProvider(providerFromType, configuration);
197 break;
198 default:
199 throw new MisconfigurationException("Unknown persistence type: " + configuration.getStringValue(PERSISTENCE_TYPE_CONF_NAME));
200 }
201
202 return persistentMaskingProvider;
203 }

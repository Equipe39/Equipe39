84 @Override
85 public Tuple<Boolean, Tuple<Integer, Integer>> isOfThisTypeWithOffset(String data) {
86 Tuple<Boolean, Tuple<Integer, Integer>> response = patterns.stream()
87 .map(
88 pattern -> pattern.matcher(data)
89 )
90 .filter(Matcher::matches)
91 .map(matcher -> new Tuple<>(true, new Tuple<>(matcher.start(1), matcher.end(1) - matcher.start(1))))
92 .reduce((a, b) -> a).orElse(new Tuple<>(false, null));
93
94 if (response.getFirst() && !containsDigit(extractText(data, response.getSecond()))) {
95 return new Tuple<>(false, null);
96 }
97
98 return response;
99 }

101 private boolean containsDigit(String text) {
102 for (int i = 0; i < text.length(); ++i) {

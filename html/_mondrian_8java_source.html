92 private static List<String> calculateCommonAncestors(MondrianPartition partition, List<ColumnInformation> columnInformationList) {
93
94 List<String> commonAncestors = new ArrayList<>();
95
96 for (int columnIndex = 0; columnIndex < columnInformationList.size(); columnIndex++) {
97 ColumnInformation columnInformation = columnInformationList.get(columnIndex);
98 if (columnInformation.getColumnType() != ColumnType.QUASI || !columnInformation.isCategorical()) {
99 commonAncestors.add(null);
100 continue;
101 }
102
103 CategoricalInformation categoricalInformation = (CategoricalInformation) columnInformation;
104 String commonAncestor = findCommonAncestor(partition.getMember().getValues(), columnIndex,
105 (MaterializedHierarchy) categoricalInformation.getHierarchy());
106 commonAncestors.add(commonAncestor);
107 }
108
109 return commonAncestors;
110 }

112 public static Partition anonymizePartition(MondrianPartition partition,
113 List<Integer> quasiColumns, List<Integer> nonQuasiColumns,
114 List<ColumnInformation> columnInformationList, CategoricalSplitStrategy categoricalSplitStrategy) {
115 List<List<String>> anonymizedValues = new ArrayList<>();
116
117 IPVDataset member = partition.getMember();
118 List<String> middle = partition.getMiddle();
119
120 List<String> commonAncestors = null;
121
122 if (categoricalSplitStrategy == CategoricalSplitStrategy.ORDER_BASED) {
123 commonAncestors = calculateCommonAncestors(partition, columnInformationList);
124 }
125
126 for (int i = 0; i < partition.size(); i++) {
127 List<String> row = member.getRow(i);
128 List<String> anonymizedRow = new ArrayList<>(row.size());
129
130 for (int k = 0; k < row.size(); k++) {
131 anonymizedRow.add(null);
132 }
133
134 for (int quasiCol : quasiColumns) {
135 ColumnInformation columnInformation = columnInformationList.get(quasiCol);
136 if (!columnInformation.isCategorical() || categoricalSplitStrategy == CategoricalSplitStrategy.HIERARCHY_BASED) {
137 anonymizedRow.set(quasiCol, middle.get(quasiCol));
138 } else {
139 assert commonAncestors != null;
140 anonymizedRow.set(quasiCol, commonAncestors.get(quasiCol));
141 }
142 }
143
144 for (Integer k : nonQuasiColumns) {
145 anonymizedRow.set(k, row.get(k));
146 }
147
148 anonymizedValues.add(anonymizedRow);
149 }
150
151 return new InMemoryPartition(anonymizedValues);
152 }

155 private IPVDataset getAnonymizedDataset() {
156 List<List<String>> values = new ArrayList<>();
157
158 for (Partition p : partitions) {
159 MondrianPartition partition = (MondrianPartition) p;
160
161 Partition anonP = anonymizePartition(partition, this.quasiColumns, this.nonQuasiColumns,
162 this.columnInformationList, this.categoricalSplitStrategy);
163 values.addAll(anonP.getMember().getValues());
164 this.anonymizedPartitions.add(anonP);
165
166 p.setAnonymous(true);
167 anonP.setAnonymous(true);
168 }
169
170 return new IPVDataset(values, dataset.getSchema(), dataset.hasColumnNames());
171 }

173 private void anonymize(MondrianPartition partition, int level) {
174
175 if (!partition.isSplittable()) {
176 partitions.add(partition);
177 return;
178 }
179
180 int dim = partition.chooseDimension();
181
182 List<MondrianPartition> subPartitions = partition.split(dim, level);
183
184 if (subPartitions.size() == 0) {
185 partition.disallow(dim);
186 anonymize(partition, level + 1);
187 } else {
188 for (MondrianPartition p : subPartitions) {
189 anonymize(p, level + 1);
190 }
191 }

195 private double calculateCardinality(IPVDataset dataset, int columnIndex) {
196 int numberOfRows = dataset.getNumberOfRows();
197 Set<String> values = new HashSet<>();
198
199 for (int i = 0; i < numberOfRows; i++) {
200 String value = dataset.get(i, columnIndex);
201 values.add(value.toLowerCase());
202 }
203
204 return values.size();

207 @Override
208 public IPVDataset apply() {
209 if (dataset == null || dataset.getNumberOfRows() == 0) {
210 return dataset;
211 }
212
213 List<String> middle = new ArrayList<>();
214 List<Interval> width = new ArrayList<>();
215
216 for (int i = 0; i < dataset.getNumberOfColumns(); i++) {
217 ColumnInformation columnInformation = columnInformationList.get(i);
218
219 if (columnInformation.getColumnType() == ColumnType.QUASI) {
220 if (!columnInformation.isCategorical()) {
221 NumericalRange numericalInformation = (NumericalRange) columnInformation;
222 middle.add(MondrianPartition.generateMiddleKey(numericalInformation.getLow(), numericalInformation.getHigh()));
223 width.add(new Interval(numericalInformation.getLow(), numericalInformation.getHigh()));
224 } else {
225 CategoricalInformation categoricalInformation = (CategoricalInformation) columnInformation;
226 String topTerm = categoricalInformation.getHierarchy().getTopTerm();
227 middle.add(topTerm.toUpperCase());
228 width.add(new Interval(0d, calculateCardinality(dataset, i)));
229 }
230 } else {
231 middle.add(null);
232 width.add(null);
233 }
234 }
235
236 anonymize(new MondrianPartition(dataset, middle, width, columnInformationList, this.privacyConstraints, this.categoricalSplitStrategy), 0);
237
238 return getAnonymizedDataset();
239 }

249 private List<ColumnInformation> buildColumnInformationList(IPVDataset dataset, Collection<IPVVulnerability> vulnerabilities,
250 Collection<String> sensitiveFields, Map<String, ProviderType> fieldTypes) {
251 List<ColumnInformation> columnInformationList = new ArrayList<>(dataset.getNumberOfColumns());
252
253 IPVSchema schema = dataset.getSchema();
254 List<? extends IPVSchemaField> fields = schema.getFields();
255
256 IPVVulnerability vulnerability = AnonymizationUtils.mergeVulnerabilities(vulnerabilities);
257
258 final boolean isForLinking = false;
259
260 for (int i = 0; i < dataset.getNumberOfColumns(); i++) {
261 String fieldName = fields.get(i).getName();
262
263 if (sensitiveFields.contains(fieldName)) {
264 columnInformationList.add(new SensitiveColumnInformation(isForLinking));
265 continue;
266 }
267
268 boolean isQuasi = vulnerability.contains(i);
269 if (!isQuasi) {
270 columnInformationList.add(new DefaultColumnInformation(isForLinking));
271 continue;
272 }
273
274 ProviderType fieldType = fieldTypes.get(fieldName);
275 if (fieldType == null) {
276 columnInformationList.add(ColumnInformationGenerator.generateCategoricalFromData(dataset, i, ColumnType.QUASI));
277 continue;
278 }
279
280 if (fieldType.getTypeClass() == TypeClass.NUMERICAL) {
281 NumericalRange numericalRange = ColumnInformationGenerator.generateNumericalRange(dataset, i, ColumnType.QUASI);
282 columnInformationList.add(numericalRange);
283 } else {
284 GeneralizationHierarchy hierarchy = GeneralizationHierarchyFactory.getDefaultHierarchy(fieldType);
285 if (hierarchy == null) {
286 columnInformationList.add(ColumnInformationGenerator.generateCategoricalFromData(dataset, i, ColumnType.QUASI));
287 } else {
288 columnInformationList.add(new CategoricalInformation(hierarchy, ColumnType.QUASI));
289 }
290 }
291 }
292
293 return columnInformationList;
294 }

303 @Override
304 public AnonymizationAlgorithm initialize(IPVDataset dataset, List<ColumnInformation> columnInformationList,
305 List<PrivacyConstraint> privacyConstraints, AnonymizationAlgorithmOptions options) {
306
307 if (columnInformationList.size() != dataset.getNumberOfColumns())
308 throw new IllegalArgumentException("Number of column information not matching number of columns of the dataset");
309
310 this.categoricalSplitStrategy = CategoricalSplitStrategy.ORDER_BASED;
311
312 if (options != null) {
313 if (!(options instanceof MondrianOptions)) {
314 throw new IllegalArgumentException("Expecting instance of OLAOptions");
315 }
316
317 MondrianOptions mondrianOptions = (MondrianOptions) options;
318 this.categoricalSplitStrategy = mondrianOptions.getCategoricalSplitStrategy();
319 }
320
321 this.dataset = dataset;
322 this.quasiColumns = AnonymizationUtils.getColumnsByType(columnInformationList, ColumnType.QUASI);
323 this.nonQuasiColumns = getNonQuasiColumns(dataset.getNumberOfColumns(), this.quasiColumns);
324 this.columnInformationList = columnInformationList;
325 this.privacyConstraints = privacyConstraints;
326 this.anonymizedPartitions = new ArrayList<>();
327
328 AnonymizationUtils.initializeConstraints(dataset, columnInformationList, privacyConstraints);
329
330 initialize();
331
332 return this;

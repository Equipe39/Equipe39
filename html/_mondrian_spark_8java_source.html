45 if(!partition.isSplittable()) {

133 public static JavaRDD<String> run(InputStream confStream, JavaRDD<String> inputRDD) throws IOException, MisconfigurationException {
134 AnonymizationOptions anonymizationOptions = new ObjectMapper().readValue(confStream, AnonymizationOptions.class);
135
136 JavaRDD<String[]> inputParsed = inputRDD.map((Function<String, String[]>) s -> {
137 CSVParser parser = CSVParser.parse(s, CSVFormat.RFC4180.withDelimiter(','));
138 CSVRecord csvRecord = parser.getRecords().get(0);
139
140 String[] results = new String[csvRecord.size()];
141
142 for(int i = 0; i < csvRecord.size(); i++) {
143 results[i] = csvRecord.get(i);
144 }
145
146 return results;
147 });
148
149 List<String> middle = new ArrayList<>();
150 List<Interval> width = new ArrayList<>();
151
152 List<ColumnInformation> columnInformationList = anonymizationOptions.getColumnInformation();
153
154 for(int i = 0; i < columnInformationList.size(); i++) {
155 ColumnInformation columnInformation = columnInformationList.get(i);
156
157 if (columnInformation.getColumnType() == ColumnType.QUASI) {
158 if (!columnInformation.isCategorical()) {
159 Tuple3<Double, Double, Double> numericalInformation = MondrianSparkUtils.findMedian(
160 MondrianSparkUtils.extractValues(inputParsed, i, true));
161
162 double low = numericalInformation._1();
163 double max = numericalInformation._2();
164
165 columnInformationList.set(i, new NumericalRange(Arrays.asList(low, max), ColumnType.QUASI));
166
167 middle.add(MondrianPartition.generateMiddleKey(low, max));
168 width.add(new Interval(low, max, numericalInformation._3()));
169 } else {
170 CategoricalInformation categoricalInformation = (CategoricalInformation) columnInformation;
171 String topTerm = categoricalInformation.getHierarchy().getTopTerm();
172 middle.add(topTerm.toUpperCase());
173 width.add(new Interval(0d, MondrianSparkUtils.calculateCardinality(inputParsed, i)));
174 }
175 }
176 else {
177 middle.add(null);
178 width.add(null);
179 }
180 }
181
182 CategoricalSplitStrategy categoricalSplitStrategy = CategoricalSplitStrategy.ORDER_BASED;
183 List<MondrianSparkPartition> finalPartitions = new ArrayList<>();
184 anonymize(new MondrianSparkPartition(inputParsed, middle, width, anonymizationOptions.getColumnInformation(),
185 anonymizationOptions.getPrivacyConstraints(), categoricalSplitStrategy), 0, finalPartitions);
186
187 return getAnonymizedDataset(finalPartitions, anonymizationOptions.getColumnInformation(), categoricalSplitStrategy);
188 }

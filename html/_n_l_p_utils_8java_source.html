86 public static final Function<IdentifiedEntity, String> TAG_FUNCTION_WITH_CACHE = (entity) -> {
87 final String type = entity.getType().iterator().next().getType();

89 final CacheEntry typeCache = cache.computeIfAbsent(type, k -> new CacheEntry());

106 public static String applyFunction(String inputText,
107 List<IdentifiedEntity> identifiedEntities,
108 Function<IdentifiedEntity, String> function) {
109 identifiedEntities.sort(Comparator.comparingInt(IdentifiedEntity::getEnd));
110 identifiedEntities.sort(Comparator.comparingInt(IdentifiedEntity::getStart));
111
112 StringBuilder builder = new StringBuilder();
113 int lastOffset = 0;
114
115 for (IdentifiedEntity entity: identifiedEntities) {
116 builder.append(inputText, lastOffset, entity.getStart());
117
118 builder.append(function.apply(entity));
119
120 lastOffset = entity.getEnd();
121 }
122
123 builder.append(inputText.substring(lastOffset));
124
125 return builder.toString();

139 private static String maskTextForIdentifiedEntity(final String text, final IdentifiedEntity identifiedEntity, final MaskingProviderFactory maskingProviderFactory) {
140 final String identifiedType = identifiedEntity.getType().iterator().next().getType();
141
142 final ProviderType providerType = ProviderType.valueOf(identifiedType);
143 final MaskingConfiguration maskingConfiguration = maskingProviderFactory.getConfigurationForField(identifiedType);
144 final MaskingProvider maskingProvider = maskingProviderFactory.get(providerType, maskingConfiguration);
145
146 final String maskedText = maskingProvider.mask(identifiedEntity.getText());
147
148 return text.substring(0, identifiedEntity.getStart()) + maskedText + text.substring(identifiedEntity.getEnd());
149 }

151 public static List<IdentifiedEntity> maskAnnotatedTokens(List<IdentifiedEntity> annotatedTokens, MaskingProviderFactory maskingProviderFactory) {
152
153 List<IdentifiedEntity> maskedTokens = new ArrayList<>();
154
155 for(IdentifiedEntity annotatedToken: annotatedTokens) {
156 Set<IdentifiedEntityType> types = annotatedToken.getType();
157
158 if (types == null || types.isEmpty()) {
159 maskedTokens.add(annotatedToken);
160 } else {
161 ProviderType providerType = ProviderType.valueOf(types.iterator().next().getType());
162 String fieldName = providerType.name();
163 MaskingProvider maskingProvider = maskingProviderFactory.get(fieldName, providerType);
164 maskedTokens.add(new IdentifiedEntity(maskingProvider.mask(annotatedToken.getText()), annotatedToken.getStart(), annotatedToken.getEnd(),
165 types, Collections.singleton(PartOfSpeechType.UNKNOWN)));
166 }
167 }
168
169 return maskedTokens;
170 }

172 public static String maskAnnotatedText(String originalText, boolean keepTags, MaskingProviderFactory maskingProviderFactory) {
173 String pattern = "<ProviderType:";
174 String closingPattern = "</ProviderType>";
175
176 int fromIndex = 0;
177
178 StringBuilder builder = new StringBuilder();
179
180 int index = originalText.indexOf(pattern, fromIndex);
181 if (index == -1) {
182 return originalText;
183 }
184
185 while (index != -1) {
186 builder.append(originalText, fromIndex, index);
187
188 int closingBracket = originalText.indexOf('>', index);
189 if (closingBracket == -1) {
190 builder.append(originalText.substring(index));
191 break;
192 }
193
194 String providerTypeName = originalText.substring(index + pattern.length(), closingBracket);
195
196 int closingPatternIndex = originalText.indexOf(closingPattern, closingBracket);
197 if (closingPatternIndex == -1) {
198 builder.append(originalText.substring(index));
199 break;
200 }
201
202 String value = originalText.substring(closingBracket + 1, closingPatternIndex);
203
204 MaskingProvider maskingProvider = maskingProviderFactory.get(providerTypeName, ProviderType.valueOf(providerTypeName));
205
206 if (keepTags) {
207 builder.append(pattern).append(providerTypeName).append('>');
208 }
209
210 builder.append(maskingProvider.mask(value));
211
212 if (keepTags) {
213 builder.append(closingPattern);
214 }
215
216 fromIndex = closingPatternIndex + closingPattern.length();
217 index = originalText.indexOf(pattern, fromIndex);
218
219 if (index == -1) {
220 builder.append(originalText.substring(fromIndex));
221 }
222 }
223
224 return builder.toString();
225 }

106 @Test
107 public void testMasking() throws Exception {
108 DefaultMaskingConfiguration maskingConfiguration = new DefaultMaskingConfiguration();
109
110 MaskingProviderFactory maskingProviderFactory = new MaskingProviderFactory(new ConfigurationManager(maskingConfiguration), null);
111
112 try (InputStream inputStream = NLPUtilsTest.class.getResourceAsStream("/opennlp.json")) {
113 NLPAnnotator identifier = new OpenNLPAnnotator(JsonUtils.MAPPER.readTree(inputStream));
114
115 String note1 = "Peter was not home at present. His wife picked up the phone and took a message for him to return my call so that I could talk to him about his healthcare management. I will try to reach out to Peter again next week if no call returned prior. Peter appears to work so I might have to try him at a later hour";
116 List<IdentifiedEntity> identifiedEntities = identifier.identify(note1, Language.UNKNOWN);
117
118 int[] maskOK = new int[identifiedEntities.size()];
119
120 for (int j = 0; j < 100; j++) {
121 //System.out.println("--- masked -----");
122 List<IdentifiedEntity> maskedTokens = NLPUtils.maskAnnotatedTokens(identifiedEntities, maskingProviderFactory);
123
124 for (int i = 0; i < identifiedEntities.size(); i++) {
125 IdentifiedEntity original = identifiedEntities.get(i);
126 IdentifiedEntity masked = maskedTokens.get(i);
127
128 if (original.getType() == null) {
129 assertEquals(original.getText(), masked.getText());
130 } else {
131 if (!original.getText().equals(masked.getText())) {
132 maskOK[i]++;
133 }
134 }
135 }
136 }
137
138 for (int value : maskOK) {
139 assertTrue(value > 0);
140 }
141 }
142 }

92 public List<String> fileContentsAsList(InputStream inputStream) throws Exception{
93 List<String> lines = new ArrayList<>();
94
95 BufferedReader reader = new BufferedReader(
96 new InputStreamReader(inputStream));
97
98 String line = reader.readLine();
99 while (line != null) {
100 lines.add(line);
101 line = reader.readLine();
102 }
103 reader.close();
104
105 return lines;
106 }

75 public OpenNLPPOSTagger(JsonNode configuration) {
76 this.defaultLanguage = Language.valueOf(configuration.get("defaultLanguage").asText());
77 this.sentenceDetector = new HashMap<>();
78 this.tagger = new HashMap<>();
79 this.tokenizer = new HashMap<>();
80
81 Iterator<Map.Entry<String, JsonNode>> iterator = configuration.get("configuration").fields();
82 while (iterator.hasNext()) {
83 Map.Entry<String, JsonNode> entry = iterator.next();
84 JsonNode localConfiguration = entry.getValue();
85 Language language = Language.valueOf(entry.getKey());
86
87 if (localConfiguration.has("sentenceFinder")) {
88 this.sentenceDetector.put(language, new SentenceDetectorME(loadSentenceModel(localConfiguration.get("sentenceFinder").asText())));
89 }
90 this.tokenizer.put(language, new TokenizerME(loadTokenizerModel(localConfiguration.get("tokenizer").asText())));
91 this.tagger.put(language, new POSTaggerME(loadPOSModel(localConfiguration.get("tagger").asText())));
92 }
93 }

104 @Override
105 public List<IdentifiedEntity> identify(String text, Language language) {
106 List<IdentifiedEntity> results = new ArrayList<>();
107
108 if (language == Language.UNKNOWN) {
109 language = this.defaultLanguage;
110 }
111
112 final SentenceDetectorME sentenceDetector = this.sentenceDetector.get(language);
113
114 final Span[] sentenceSpans;
115 if (sentenceDetector == null) {
116 logger.info("Sentence detection for language {} is missing", language.name());
117 sentenceSpans = new Span[] {
118 new Span(0, text.length())
119 };
120 } else {
121 synchronized (this.sentenceDetector) {
122 sentenceSpans = sentenceDetector.sentPosDetect(text);
123 }
124 }
125
126 for (Span sentenceSpan : sentenceSpans) {
127 String sentence = text.substring(sentenceSpan.getStart(), sentenceSpan.getEnd());
128
129 final String[] tokens;
130 final Span[] tokenPositionsWithinSentence;
131
132 synchronized (tokenizer) {
133 tokens = tokenizer.get(language).tokenize(sentence);
134 tokenPositionsWithinSentence = tokenizer.get(language).tokenizePos(sentence);
135 }
136 final String[] tags;
137
138 synchronized (tagger) {
139 tags = tagger.get(language).tag(tokens);
140 }
141
142 int sentenceStart = sentenceSpan.getStart();
143
144 for(int i = 0; i < tokens.length; i++) {
145 int beginningOfFirstWord = tokenPositionsWithinSentence[i].getStart();
146 int endOfLastWord = tokenPositionsWithinSentence[i].getEnd();
147
148 int startOffsetInDoc = sentenceStart + beginningOfFirstWord;
149 int endOffsetInDoc = sentenceStart + endOfLastWord;
150
151 String nameInDocument = text.substring(startOffsetInDoc, endOffsetInDoc);
152
153 IdentifiedEntity entity = new IdentifiedEntity(nameInDocument, startOffsetInDoc, endOffsetInDoc,
154 Collections.emptySet(),
155 Collections.singleton(PartOfSpeechType.valueOf(tags[i])));
156
157 results.add(entity);
158 }
159 }
160
161 return results;
162 }

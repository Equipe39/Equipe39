86 public PhoneNumber getPhoneNumber(String identifier) {
87 Matcher matcher = getMatchingPattern(identifier);
88 if (matcher == null) {
89 if (msisdnManager.isValidUSNumber(identifier)) {
90 return parseUSNumber(identifier);

93 return null;
94 } else {
95 if (identifier.length() < 6) {

102 if (null == countryCode) countryCode = "1";

104 String separator = matcher.group("separator");
105 if (null == separator) separator = "";
106
107 String areaCode = matcher.group("areaCode");

109
110 return new PhoneNumber((null == prefix ? "+" : prefix), countryCode, separator, number, areaCode, null != prefix);

120 private Matcher getMatchingPattern(String data) {
121 if (data.length() < 6) {
122 return null;
123 }
124
125 for (Pattern pattern : phonePatterns) {

127 if (!m.matches()) {

133
134 return null;

152 @Override
153 public Tuple<Boolean, Tuple<Integer, Integer>> isOfThisTypeWithOffset(String data) {
154
155 if (data.length() > 64) {
156 return new Tuple<>(false, null);
157 }
158
159 final long numberOfDigits = countDigits(data);
160
161 if (numberOfDigits < 7 || numberOfDigits > 15) {
162 return new Tuple<>(false, null);
163 }
164
165 return Arrays.stream(phonePatterns).map(
166 pattern -> {
167 return pattern.matcher(data);
168 }
169 ).filter(Matcher::matches).
170 filter(this::limitLength).
171 map(
172 matcher -> new Tuple<>(true, new Tuple<>(matcher.start(1), matcher.end(1) - matcher.start(1)))
173 ).reduce((a, b) -> a).orElse(new Tuple<>(false, null));
174 }

176 private boolean limitLength(Matcher matcher) {

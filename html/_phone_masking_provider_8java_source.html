91 private String oldMask(String identifier) {
92 PhoneNumber phoneNumber = phoneIdentifier.getPhoneNumber(identifier);
93 if (phoneNumber == null) {
94 return generateRandomPhoneNumber();
95 }
96
97 String countryCode = null;
98 if (this.preserveCountryCode) {
99 countryCode = phoneNumber.getCountryCode();
100 } else {
101 countryCode = msisdnManager.getRandomCountryCode();
102 }
103
104 String areaCode;
105 if (this.preserveAreaCode) {
106 areaCode = phoneNumber.getAreaCode();
107 } else {
108 StringCharacterIterator acIterator = new StringCharacterIterator(phoneNumber.getAreaCode());
109 StringBuilder randomAreaCode = new StringBuilder();
110
111 for (char c = acIterator.first(); c != CharacterIterator.DONE; c = acIterator.next()) {
112 randomAreaCode.append(RandomGenerators.randomDigit());
113 }
114
115 areaCode = randomAreaCode.toString();
116 }
117
118 StringBuilder builder = new StringBuilder();
119 StringCharacterIterator iterator = new StringCharacterIterator(phoneNumber.getNumber());
120
121 for (char c = iterator.first(); CharacterIterator.DONE != c; c = iterator.next()) {
122 if (Character.isDigit(c)) {
123 c = (char) ('0' + random.nextInt(10));
124 }
125 builder.append(c);
126 }
127
128 if (phoneNumber.isHasPrefix()) {
129 return phoneNumber.getPrefix() + countryCode +
130 phoneNumber.getSeparator() +
131 areaCode +
132 builder;
133 }
134
135 return areaCode + builder;
136 }

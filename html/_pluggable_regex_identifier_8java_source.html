101 public Tuple<Boolean, Tuple<Integer, Integer>> isOfThisTypeWithOffset(String data) {
102 if (!quickCheck(data)) {
103 return new Tuple<>(false, null);
104 }
105
106 for (Pattern p : this.patterns) {
107 Matcher matcher = p.matcher(data);
108
109 if (!matcher.matches()) {
110 continue;
111 }
112
113 if (matcher.groupCount() == 0) {
114 return new Tuple<>(true, new Tuple<>(0, data.length()));
115 }
116
117 int begin = matcher.start(1);
118 int end = matcher.end(1);
119
120 return new Tuple<>(true, new Tuple<>(begin, end - begin));
121 }
122
123 return new Tuple<>(false, null);
124 }

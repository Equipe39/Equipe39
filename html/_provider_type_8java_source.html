424 public static ProviderType registerType(String typeName) {
425 ProviderType providerType = valueOf(typeName);

433 private static synchronized int insertType(ProviderType type) {
434 if (!registeredTypes.containsKey(type.name())) {
435 registeredTypes.put(type.name(), type);
436 return registeredTypes.size();
437 }
438
439 return registeredTypes.get(type.name()).getId();

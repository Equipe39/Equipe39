52 private boolean checkHistogram(Histogram<String> histogram) {
53 List<Long> frequencies = new ArrayList<>();
54
55 for (String key : histogram.keySet()) {
56 Long counter = histogram.get(key);
57 frequencies.add(counter);
58 }
59
60 Collections.sort(frequencies);
61
62 double threshold = 0;
63 for (int i = frequencies.size() - this.l; i >= 0; i--) { // minSize=(int)l;
64 threshold += frequencies.get(i);
65 }
66
67 threshold *= this.c;
68
69 return frequencies.get(frequencies.size() - 1) < threshold;
70 }
71
72 private boolean checkRecursiveCLDiversity(Partition partition, int column) {
73 if (partition.size() < this.l) {
74 return false;
75 }
76
77 Histogram histogram = Histogram.createHistogram(partition.getMember(), column);
78 return checkHistogram(histogram);
79 }

81 @Override

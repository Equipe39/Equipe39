44 @Override

46
47 if (this.quasiSameAsLinking) {
48 List<String> row = partition.getMember().getRow(0);
49 Integer matchedRows = anonymizedDatasetLinker.matchAnonymizedRow(row, this.linkInformation, this.columnInformation);
50
51 if (matchedRows == 0) {
52 return true;
53 }
54
55 double risk = 1.0 / (double) matchedRows;
56
57 return (!(risk > riskThreshold));
58 }
59
60 List<Integer> matches = anonymizedDatasetLinker.matchesPerRecord(partition, this.linkInformation, this.columnInformation);
61
62 if (matches.isEmpty()) {
63 return true;
64 }
65
66 for (Integer match : matches) {
67 if (match == 0) {
68 continue;
69 }
70
71 double risk = 1.0 / (double) match;
72 if (risk > this.riskThreshold) {
73 return false;
74 }
75 }
76
77 return true;
78 }

125 private boolean checkIfQuasiAndLinkAreTheSame(List<ColumnInformation> columnInformation, Collection<LinkInfo> linkInformation) {
126
127 for (int i = 0; i < columnInformation.size(); ++i) {
128 if (columnInformation.get(i).getColumnType() != ColumnType.QUASI) {
129 continue;
130 }
131
132 boolean match = false;
133
134 for (LinkInfo info : linkInformation) {
135 if (info.getSourceIndex() == i) {
136 match = true;
137 break;
138 }
139 }
140
141 if (!match) {
142 return false;
143 }
144 }
145
146 return true;

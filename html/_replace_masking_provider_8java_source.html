140 private String mask(String identifier, boolean checkSetConstrains) {
141 if (checkSetConstrains && shouldValueBePreserved(identifier)) {
142 return identifier;
143 }
144
145 if (this.replaceMode == ReplaceMode.WITH_SET) {
146 int position = random.nextInt(replacementValues.size());
147
148 return replacementValues.get(position);
149 }
150
151 int identifierLength = identifier.length();
152 if (offset > identifierLength) {
153 return "";
154 }
155
156 StringBuilder builder = new StringBuilder();
157
158 if (prefix != null) {
159 builder.append(prefix);
160 }
161
162 int stop = offset + preservedCharacters;
163 if (stop > identifierLength) {
164 stop = identifierLength;
165 }
166
167 String maskedValue = identifier.substring(this.offset, stop);
168 if (this.replaceMode == ReplaceMode.WITH_PARTIAL) {
169 builder.append(maskedValue);
170 return builder.toString();
171 }
172
173 if (this.offset > 0) {
174 if (this.replaceMode == ReplaceMode.WITH_ASTERISKS) {
175 builder.append(String.valueOf(asterisk).repeat(this.offset));
176 } else {
177 if (this.replaceMode == ReplaceMode.WITH_RANDOM) {
178 builder.append(RandomGenerators.randomReplacement(identifier.substring(0, this.offset)));
179 } else if (this.replaceMode == ReplaceMode.WITH_DETERMINISTIC) {
180 builder.append(RandomGenerators.deterministicReplacement(identifier.substring(0, this.offset)));
181 }
182 }
183 }
184
185 builder.append(maskedValue);
186
187 if (stop < identifierLength) {
188 if (this.replaceMode == ReplaceMode.WITH_ASTERISKS) {
189 builder.append(String.valueOf(asterisk).repeat(identifierLength - stop));
190 } else if (this.replaceMode == ReplaceMode.WITH_RANDOM) {
191 builder.append(RandomGenerators.randomReplacement(identifier.substring(stop, identifierLength)));
192 } else if (this.replaceMode == ReplaceMode.WITH_DETERMINISTIC) {
193 builder.append(RandomGenerators.deterministicReplacement(identifier.substring(stop, identifierLength)));
194 }
195 }
196
197 return builder.toString();
198 }

208 private boolean shouldValueBePreserved(String value) {
209 if (replaceOnValueInSet && !testValues.contains(value)) {

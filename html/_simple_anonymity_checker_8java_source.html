38 private boolean checkConstraints(Partition partition) {
39 for (PrivacyConstraint privacyConstraint : privacyConstraints) {
40 if (!privacyConstraint.check(partition, sensitiveColumns)) {
41 return false;
42 }
43 }

48 public double calculateSuppressionRate(LatticeNode node) {
49 List<Partition> partitions;
50
51 int contentRequirements = this.privacyConstraintsContentRequirements;
52
53 if (contentRequirements == ContentRequirements.NONE) {
54 Map<String, Integer> eqCounters = DatasetGeneralizer.generalizeCSVAndCountEQ(dataset, columnInformationList, node.getValues());
55
56 partitions = new ArrayList<>();
57 for (Integer eqSize : eqCounters.values()) {
58 partitions.add(new VirtualPartition(eqSize));
59 }
60 } else {
61 Collection<Partition> partitionCollection =
62 DatasetGeneralizer.generalizeAndPartition(dataset, columnInformationList, node.getValues(), contentRequirements);
63 partitions = new ArrayList<>(partitionCollection);
64 }
65
66 double suppressedRows = 0.0d;
67
68 for (Partition partition : partitions) {
69 if (!checkConstraints(partition)) {
70 suppressedRows += partition.size();
71 partition.setAnonymous(false);
72 } else {
73 partition.setAnonymous(true);
74 }
75 }
76
77 /* DM has not content requirements so we are fine */
78 InformationMetric metric = new DiscernibilityStar(); /*TODO: replace with NUE */
79 metric.initialize(dataset, null, partitions, null, this.columnInformationList, null);
80 node.setInformationLoss(metric.report());
81
82 return 100.0 * (suppressedRows / (double) dataset.getNumberOfRows());
83 }

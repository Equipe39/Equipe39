118 @Test
119 public void testFilterDataset() {
120 List<String> testData = Arrays.asList(
121 "aaaaa",
122 "bbbbb",
123 "ccccc"
124 );
125 try (JavaSparkContext context = new JavaSparkContext(spark.sparkContext())) {
126 JavaRDD<Row> rdd = context.parallelize(testData).map(value -> new GenericRow(new Object[]{value}));
127
128 Dataset<Row> dataset = spark.createDataset(rdd.rdd(), RowEncoder.apply(new StructType(new StructField[]{
129 new StructField("column", DataTypes.StringType, false, Metadata.empty())
130 })));
131
132 assertThat(SparkUtils.filterOnFieldValue(dataset, "column", "a{5}").count(), is(1L));
133 assertThat(SparkUtils.filterOnFieldValue(dataset, "column", "[ab]{5}").count(), is(2L));
134 }

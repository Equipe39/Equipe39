50 private void generateRandomDataset(Appendable out, int numberOfRows, List<ColumnInformation> columnInformationList) throws IOException {
51 try (CSVPrinter printer = new CSVPrinter(out, CSVFormat.RFC4180);) {
52 List<List<String>> leaves = new ArrayList<>();
53
54 for (ColumnInformation columnInformation : columnInformationList) {
55 ColumnType columnType = columnInformation.getColumnType();
56 if (columnType == ColumnType.QUASI) {
57 CategoricalInformation categoricalInformation = (CategoricalInformation) columnInformation;
58 leaves.add(getLeaves(((MaterializedHierarchy) categoricalInformation.getHierarchy()).getTerms()));
59 } else {
60 List<String> dummy = new ArrayList<>();
61 dummy.add("foo");
62 leaves.add(dummy);
63 }
64 }
65
66 SecureRandom random = new SecureRandom();
67
68 for (int i = 0; i < numberOfRows; i++) {
69 List<String> row = new ArrayList<>();
70
71 for (List<String> leavesList : leaves) {
72 int index = random.nextInt(leavesList.size());
73 row.add(leavesList.get(index));
74 }
75
76 printer.printRecord(row);
77 }
78 }
79 }

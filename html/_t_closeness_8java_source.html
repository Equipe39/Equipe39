49 public static double orderBasedDistance(List<Double> partitionValues, List<Double> totalOrdered,
50 Histogram<Double> totalHistogram, long totalCount) {
51
52 double sum = 0.0;
53 double distance = 0.0;
54
55 Histogram<Double> partitionHistogram = Histogram.createHistogram(partitionValues);
56 long partitionCount = partitionValues.size();
57
58 for (Double q : totalOrdered) {
59
60 double qFreq = (double) totalHistogram.get(q) / (double) totalCount;
61 double pFreq = (double) partitionHistogram.getOrDefault(q, 0L) / (double) partitionCount;
62
63 sum += (pFreq - qFreq);
64 distance += Math.abs(sum);
65 }
66
67 return distance / (double) (totalOrdered.size() - 1);
68 }

70 public static double equalDistance(List<String> partitionValues, Histogram<String> totalHistogram, long totalCount) {
71
72 double sum = 0.0;
73
74 Histogram<String> partitionHistogram = Histogram.createHistogram(partitionValues);
75 int partitionSize = partitionValues.size();
76
77 for (Map.Entry<String, Long> entry : totalHistogram.entrySet()) {
78 String key = entry.getKey();
79 Long count = entry.getValue();
80 double qFreq = (double) count / (double) totalCount;
81 double pFreq = (double) partitionHistogram.getOrDefault(key, 0L) / (double) partitionSize;
82
83 sum += Math.abs(pFreq - qFreq);
84 }
85
86 return sum / 2.0;
87 }

118 private void initializeCategoricalInformation(IPVDataset dataset, List<ColumnInformation> columnInformationList) {
119
120 for (int i = 0; i < columnInformationList.size(); i++) {
121 ColumnInformation columnInformation = columnInformationList.get(i);
122
123 if (columnInformation instanceof CategoricalInformation && columnInformation.getColumnType() == ColumnType.SENSITIVE) {
124 List<String> values = new ArrayList<>();
125
126 for (int row = 0; row < dataset.getNumberOfRows(); row++) {
127 values.add(dataset.get(row, i));

138 private void initializeNumericalInformation(IPVDataset dataset, List<ColumnInformation> columnInformationList) {

143 if (columnInformation instanceof NumericalRange && columnInformation.getColumnType() == ColumnType.SENSITIVE) {
144 List<Double> values = new ArrayList<>();
145
146 for (int row = 0; row < dataset.getNumberOfRows(); row++) {

169 private boolean checkNumerical(IPVDataset members, Integer column) {
170 List<Double> partitionValues = new ArrayList<>();
171
172 for (int i = 0; i < members.getNumberOfRows(); i++) {
173 Double value = Double.parseDouble(members.get(i, column));
174 partitionValues.add(value);
175 }
176
177 List<Double> totalOrdered = this.totalOrdered.get(column);
178 Histogram<Double> totalHistogram = this.histograms.get(column);
179
180 double distance = orderBasedDistance(partitionValues, totalOrdered, totalHistogram, this.totalCount);
181 return distance <= this.t;
182 }

184 private boolean checkCategorical(IPVDataset members, Integer column) {
185 List<String> partitionValues = new ArrayList<>();
186
187 for (int i = 0; i < members.getNumberOfRows(); i++) {
188 partitionValues.add(members.get(i, column));
189 }
190
191 Histogram<String> totalHistogram = this.categoricalHistograms.get(column);
192
193 double distance = equalDistance(partitionValues, totalHistogram, this.totalCount);

197 private boolean checkPartition(Partition partition, Integer column) {
198 IPVDataset members = partition.getMember();
199 ColumnInformation columnInformation = this.columnInformationList.get(column);
200
201 if (columnInformation instanceof NumericalRange) {
202 return checkNumerical(members, column);
203 } else {
204 return checkCategorical(members, column);

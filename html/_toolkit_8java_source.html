103 protected static void processTasks(File input, File output, TaskToExecute taskToExecute) {
104 if (!input.exists()) {
105 logger.error("File not found: {}", input);
106 throw new RuntimeException("File not found: "+ input);
107 }
108
109 if (input.isDirectory()) {
110 if (!output.isDirectory()) {
111 logger.error("Input is a directory, output must be a directory as well");
112 throw new RuntimeException("Input is a directory, output must be a directory as well");
113 }
114
115 processDirectory(input, output, taskToExecute);
116 } else {
117 if (output.isDirectory()) {
118 output = buildOutputFile(input, output, taskToExecute);
119 }
120 checkParentOfOutputExists(output.getAbsoluteFile().getParentFile());
121 processFile(input, output, taskToExecute);
122 }
123 }

137 private static void processDirectory(File inputDirectory, File outputDirectory, TaskToExecute taskToExecute) {
138 Iterable<File> listOfFiles = listValidFiles(inputDirectory, taskToExecute.getExtension());
139
140 if (!outputDirectory.exists()) {
141 if (!outputDirectory.mkdirs()) {
142 throw new RuntimeException("Unable to create output directory");
143 }
144 } else {
145 if (!outputDirectory.isDirectory()) {
146 throw new IllegalArgumentException("Output destination is not a directory");
147 }
148 }
149
150 for (File inputFile : listOfFiles) {
151 if (!inputFile.isHidden()) {
152 if (inputFile.isFile()) {
153 File outputFile = buildOutputFile(inputFile, outputDirectory, taskToExecute);
154
155 logger.trace("Processing {} with {} as output", inputFile.getName(), outputFile.getName());
156
157 processFile(inputFile, outputFile, taskToExecute);
158 } else {
159 processDirectory(inputFile, new File(outputDirectory, inputFile.getName()), taskToExecute);
160 }
161 } else {
162 logger.trace("Skipping hidden file {}", inputFile.getName());
163 }
164 }

175 private static Iterable<File> listValidFiles(File directory, String extension) {
176 if (null != extension) {
177 return Arrays.asList(
178 Objects.requireNonNull(directory.listFiles(
179 pathname -> pathname.isDirectory() ||
180 (pathname.isFile() && pathname.getName().endsWith(extension))
181 ))
182 );
183 }
184 return Arrays.asList(Objects.requireNonNull(directory.listFiles()));
185 }

203 private static ObjectMapper buildMapper(String configurationFile) {
204 final String extension = extractFileExtension(configurationFile);
205
206 if (extension.equalsIgnoreCase("json")) {
207 return new ObjectMapper();
208 } else if (extension.equalsIgnoreCase("yaml") || extension.equalsIgnoreCase("yml")) {
209 return new ObjectMapper(new YAMLFactory());
210 }
211
212 throw new IllegalArgumentException("Unknown extension " + configurationFile);
213 }

215 private static String extractFileExtension(String configurationFile) {
216 return configurationFile.substring(configurationFile.lastIndexOf(".") + 1);

219 private static Map<String, Object> buildTaskOptions(String additionalArguments) {
220 if (null == additionalArguments || additionalArguments.isEmpty()) return Collections.emptyMap();
221
222 return Arrays.stream(additionalArguments.split(";"))
223 .map(String::trim)
224 .filter(((Predicate<String>)String::isEmpty).negate())
225 .filter(((Predicate<String>) line -> line.startsWith("#")).negate())
226 .map( option -> option.split("\\s*=\\s*"))
227 .filter( optionArgument -> 2 == optionArgument.length)
228 .collect(
229 Collectors.toMap(
230 optionArgument -> optionArgument[0],
231 optionArgument -> optionArgument[1]
232 )
233 );
234 }

240 private static Options buildOptions() {
241 Options options = new Options();
242
243 Arrays.stream(CommandLineOptions.values()).map(
244 commandLineOption -> Option.builder(commandLineOption.shortOption)
245 .longOpt(commandLineOption.longOption)
246 .required(commandLineOption.required)
247 .desc(commandLineOption.desc)
248 .hasArg(commandLineOption.hasArgs)
249 .build()
250 ).forEach(options::addOption);
251
252 return options;
253 }

81 private String randomizeUsernamePassword(String userInfo) {
82 final String[] parts = userInfo.split(":");
83
84 if (parts.length == 0) {
85 return RandomGenerators.randomUIDGenerator(8);
86 }
87
88 final String username = RandomGenerators.randomUIDGenerator(parts[0].length());
89
90 if (parts.length == 2) {
91 final String password = RandomGenerators.randomUIDGenerator(8 + random.nextInt(8));
92
93 return username + ":" + password;
94 }
95
96 return username;
97 }

99 private String randomizeHostname(String hostname) {
100 /*
101 host can be expressed as a host name or a literal IP address. If IPv6 literal
102 address is used, it should be enclosed in square brackets ('[' and ']'), as specified by RFC 2732;
103 */
104 if (hostname == null || hostname.length() == 0) {
105 return hostname;
106 }
107
108 if (hostname.charAt(0) == '[' && hostname.charAt(hostname.length() - 1) == ']') {
109 hostname = hostname.substring(1, hostname.length() - 1);
110 return String.format("[%s]", RandomGenerators.randomHostnameGenerator(hostname, 0));
111 }
112
113 return RandomGenerators.randomHostnameGenerator(hostname, preserveDomains);
114 }

134 private String maskQuery(String query) {
135 if (query.equals("/") || query.equals("/?") || query.length() == 0) {
136 return query;
137 }
138
139 Collection<Identifier> identifiers = IdentifierFactory.defaultIdentifiers();
140
141 String[] tokens = query.split("&");
142 String[] maskedTokens = new String[tokens.length];
143
144 for (int i = 0; i < tokens.length; i++) {
145 String token = tokens[i];
146 if (i == 0) {
147 token = token.substring(1);
148 }
149
150 String[] parts = token.split("=");
151 if (parts.length == 1 || parts[1].equals("")) {
152 maskedTokens[i] = token;
153 continue;
154 }
155
156 for (Identifier identifier : identifiers) {
157 if (identifier.isOfThisType(parts[1])) {
158 MaskingProvider maskingProvider = getMaskingProvider(identifier.getType());
159 parts[1] = maskingProvider.mask(parts[1]);
160 break;
161 }
162 }
163
164 maskedTokens[i] = parts[0] + "=" + parts[1];
165 }
166
167 return "?" + StringUtils.join(maskedTokens, '&');
168 }

170 private String buildURLString(String protocol, String host, int port, String file, String userInfo) {
171 StringBuilder builder = new StringBuilder(protocol);
172 builder.append("://");
173
174 if (userInfo != null) {
175 builder.append(userInfo);
176 builder.append("@");
177 }
178
179 if (host != null) {
180 builder.append(host);
181 }
182
183 builder.append(":");
184 builder.append(port);
185 if (!removeQuery) {
186 if (file != null) {
187 if (!maskQuery) {
188 builder.append(file);
189 } else {
190 builder.append(maskQuery(file));
191 }
192 }
193 }
194 return builder.toString();
195 }

197 private String maskURL(URL url) {
198 String protocol = url.getProtocol();
199 int port = url.getPort();
200 String host = url.getHost();
201 String filename = url.getFile();
202
203 if (port == -1) {
204 port = url.getDefaultPort();
205 }
206
207 if (this.randomizePort) {
208 port = randomizePort(port);
209 }
210
211 host = randomizeHostname(host);
212
213 String userInfo = url.getUserInfo();
214 if (userInfo != null && this.maskUsernamePassword) {
215 userInfo = randomizeUsernamePassword(userInfo);
216 }
217
218 return buildURLString(protocol, host, port, filename, userInfo);

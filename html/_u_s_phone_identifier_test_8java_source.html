106 "1-123-123-1234"
107 };
108
109 USPhoneIdentifier identifier = new USPhoneIdentifier();
110
111 for (int i = 0; i < validPhones.length; ++i) {
112 String number = validPhones[i];
113 String validationValue = phones[i];
114
115 Tuple<Boolean, Tuple<Integer, Integer>> response = identifier.isOfThisTypeWithOffset(number);
116
117 assertTrue(response.getFirst(), number);
118
119 Tuple<Integer, Integer> range = response.getSecond();
120
121 assertNotNull(range, number);
122
123 assertThat(number,range.getFirst() + range.getSecond(), lessThanOrEqualTo(number.length()));
124
125 String extractedString = number.substring(range.getFirst(), range.getFirst() + range.getSecond());
126
127 assertEquals(validationValue, extractedString, number);
128 }

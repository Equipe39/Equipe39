65 private Map<? extends String, ? extends String> readWMIList(Collection<ResourceEntry> entries) {
66 Map<String, String> names = new HashMap<>();
67
68 for (ResourceEntry entry : entries) {
69 InputStream inputStream = entry.createStream();
70 try (CSVParser reader = Readers.createCSVReaderFromStream(inputStream)) {
71 for (CSVRecord line : reader) {
72 names.put(line.get(0).toUpperCase(), line.get(1));
73 }
74 inputStream.close();
75 } catch (IOException | NullPointerException e) {
76 e.printStackTrace();
77 }

83 @Override
84 public boolean isValidKey(String data) {
85
86 /* All standards for VIN are 17-digit format
87 First 3 digits is WMI - World manufacturer identifier
88 Digits 4-9 are the vehicle description section
89 Digits 10-17 is the vehicle identifier section
90 */
91 if (data.length() != 17) {
92 return false;
93 }
94
95 /* In 1981, the National Highway Traffic Safety Administration of the United States
96 standardized the format.[1] It required all over-the-road-vehicles sold to contain a 17-character VIN,
97 which does not include the letters I (i), O (o), or Q (q) (to avoid confusion with numerals 1 and 0).
98 */
99 //data = data.toUpperCase();
100 for (int i = 0; i < data.length(); i++) {
101 char c = data.charAt(i);
102
103 /* VIN is composed only of digits and letters */
104 if (!Character.isDigit(c) && !Character.isLetter(c)) {
105 return false;
106 }
107 if (c == 'I' || c == 'i' || c == 'O' || c == 'o' || c == 'Q' || c == 'q') {
108 return false;
109 }
110 }
111
112 /* check if the WMI is one of the known */
113 String wmi = data.substring(0, 3);
114 return isValidWMI(wmi);
115 }

122 private String randomUIDGenerator(int length, char[] excludedCharacters) {
123 StringBuilder builder = new StringBuilder();
124 char nextRandom;
125
126 for (int i = 0; i < length; i++) {
127 if (random.nextBoolean()) {
128 nextRandom = (char) ('a' + random.nextInt(26));
129 } else {
130 nextRandom = (char) ('0' + random.nextInt(10));
131 }
132
133 if (excludedCharacters != null) {
134 if (ArrayUtils.contains(excludedCharacters, nextRandom)) {
135 i--;
136 continue;
137 }
138 }
139
140 builder.append(nextRandom);
141 }
142
143 return builder.toString();
144
145 }

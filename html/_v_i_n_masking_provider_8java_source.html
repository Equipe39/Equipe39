72 private String randomVIN() {

79 @Override
80 public String mask(String identifier) {
81 if (!vinIdentifier.isOfThisType(identifier)) {
82 return randomVIN();
83 }
84
85 String wmi = identifier.substring(0, 3);
86 String vds = null;
87 String vis = RandomGenerators.randomUIDGeneratorWithInclusions(8, allowedCharacters);
88
89 if (!this.preserveWMI) {
90 wmi = vinManager.getRandomWMI(wmi);
91 }
92
93 if (!this.preserveVDS) {
94 vds = RandomGenerators.randomUIDGeneratorWithInclusions(6, allowedCharacters);
95 } else {
96 vds = identifier.substring(3, 9);
97 }
98

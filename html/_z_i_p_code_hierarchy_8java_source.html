58 @Override
59 public Set<String> getNodeLeaves(String value) {
60 int level = getNodeLevel(value);
61
62 if (level == 0) { //this is a leaf in the tree
63 return Collections.singleton(value);
64 }
65
66 //we calculate how many ZIP codes are in the leaves
67 //based on the level. If the value is 1234*, then the
68 //level is 1, so we have 10^(1) = 10 remaining codes
69 int remainingZIPCodes = (int) Math.pow(10, level);
70
71 // if the value is 1234*, then the prefix we preserve is 1234
72 String prefix = value.substring(0, 5 - level);
73
74 Set<String> leaves = new HashSet<>();
75
76 for (int i = 0; i < remainingZIPCodes; i++) {
77 //we left pad so the final value will be a 5-digit code
78 leaves.add(prefix + StringUtils.leftPad(i + "", level, '0'));
79 }
80
81 return leaves;
82 }

84 @Override
85 public int getNodeLevel(String value) {
86 int level = 0;
87
88 for (int i = (value.length() - 1); i >= 0; i--) {
89 if (value.charAt(i) == '*') {
90 level++;
91 }
92 }
93
94 return level;
95 }

98 public String getTopTerm() {
99 return topTerm;

102 @Override
103 public String encode(String value, int level, boolean randomizeOnFail) {
104 if (level <= 0) {
105 return value;
106 }
107
108 if (level >= this.height) {
109 return this.topTerm;
110 }
111
112 if (value.length() != 5) {
113 return topTerm;
114 }
115
116 StringBuilder prefix = new StringBuilder(value.substring(0, value.length() - level));
117
118 for (int i = 0; i < level; i++) {
119 prefix.append("*");
120 }
121
122 return prefix.toString();
123 }

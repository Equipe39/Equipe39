52 private void addToPrefixMap(String code, String countryCode, Integer population) {
53 Map<String, ZIPCode> map = zipCodeMapThreeDigits.get(countryCode);
54
55 if (map == null) {
56 zipCodeMapThreeDigits.put(countryCode, new HashMap<>());
57 map = zipCodeMapThreeDigits.get(countryCode);
58 }
59
60 String key = code.toUpperCase();
61 if (key.length() > this.prefixDigits) {
62 key = key.substring(0, this.prefixDigits);
63 }
64
65 ZIPCode zipCode = map.get(key);
66 if (zipCode == null) {
67 map.put(key, new ZIPCode(key, population));
68 } else {
69 map.put(key, new ZIPCode(key, population + zipCode.getPopulation()));
70 }
71 }

73 private void buildPrefixMap(Collection<ResourceEntry> entries) {
74 Map<String, Map<String, ZIPCode>> codes = new HashMap<>();
75
76 for (ResourceEntry entry : entries) {
77 InputStream inputStream = entry.createStream();
78 String countryCode = entry.getCountryCode();
79
80 try (CSVParser reader = Readers.createCSVReaderFromStream(inputStream)) {
81 for (CSVRecord line : reader) {
82 String code = line.get(0);
83 String populationString = line.get(1);
84
85 Integer population = Integer.valueOf(populationString);
86 ZIPCode zipCode = new ZIPCode(code, population);
87
88 String key = code.toUpperCase();
89 addToPrefixMap(code, countryCode, population);
90 }
91 inputStream.close();
92 } catch (IOException | NullPointerException e) {
93 e.printStackTrace();

162 public IPVDataset sampleDataset(IPVDataset original, int sampleSize) {
163 List<List<String>> values = new ArrayList<>();
164
165 original.forEach(values::add);
166
167 // shuffle list
168 Collections.shuffle(values);
169
170 return new IPVDataset(
171 values.subList(0, sampleSize),
172 original.getSchema(),
173 original.hasColumnNames()
174 );
175 }

177 public IPVDataset generateRandomData(Map<Integer, Integer> eqSizes) {
178
179 int index = 0;
180 List<List<String>> rows = new ArrayList<>();
181
182 for(Map.Entry<Integer, Integer> entry: eqSizes.entrySet()) {
183 Integer eqSize = entry.getKey();
184 Integer counter = entry.getValue();
185
186 for(int i = 0; i < counter; i++) {
187
188 for(int j = 0; j < eqSize; j++) {
189 rows.add(Collections.singletonList(Integer.toString(index)));
190 }
191
192 index++;
193 }
194 }
195
196 return new IPVDataset(
197 rows, IPVDataset.generateSchemaWithoutColumnNames(1),
198 false
199 );
200 }

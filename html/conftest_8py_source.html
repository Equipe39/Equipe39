51def _compare_dirs(dir1, dir2, comparer):
52 """
53 Compare two directories recursively. Files in each directory are
54 assumed to be equal if their names and contents are equal.
55
56 @param dir1: First directory path
57 @param dir2: Second directory path
58
59 @return: True if the directory trees are the same and
60 there were no errors while accessing the directories or files,
61 False otherwise.
62 """
63 result = True
64
65 dirs_cmp = filecmp.dircmp(dir1, dir2)
66
67 if len(dirs_cmp.left_only):
68 for name in dirs_cmp.left_only:
69 LOGGER.error("Unexpected file %s found in %s", name, dirs_cmp.left)
70 result = False
71
72 if len(dirs_cmp.right_only):
73 for name in dirs_cmp.right_only:
74 LOGGER.error("Expected file %s missing from %s", name, dirs_cmp.left)
75 result = False
76
77 if len(dirs_cmp.funny_files):
78 for name in dirs_cmp.funny_files:
79 LOGGER.error("Unexpected funny file %s found in %s", name, dirs_cmp.left)
80 result = False
81
82 for file in dirs_cmp.common_files:
83 dir1_file = os.path.join(dir1, file)
84 dir2_file = os.path.join(dir2, file)
85
86 if not comparer(dir1_file, dir2_file):
87 result = False
88
89 for common_dir in dirs_cmp.common_dirs:
90 new_dir1 = os.path.join(dir1, common_dir)
91 new_dir2 = os.path.join(dir2, common_dir)
92 if not _compare_dirs(new_dir1, new_dir2):
93 result = False
94
95 return result

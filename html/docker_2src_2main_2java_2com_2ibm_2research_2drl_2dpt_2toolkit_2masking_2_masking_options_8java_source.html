52 private Map<String, DataMaskingTarget> buildToBeMasked(Map<String, JsonNode> toBeMaskedNodes) {
53 return toBeMaskedNodes.entrySet().stream().map(entry ->
54 (Map.Entry<String, DataMaskingTarget>) new AbstractMap.SimpleEntry<>(
55 entry.getKey(),
56 convertValue(entry.getKey(), entry.getValue())
57 )).collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue));

60 private DataMaskingTarget convertValue(String fieldReference, JsonNode target) {
61 if (target.isTextual()) {
62 return new DataMaskingTarget(ProviderType.valueOf(target.asText()), fieldReference);
63 } else if (target.isObject() && target.has("providerType") && target.has("targetPath")) {
64 final JsonNode providerTypeNode = target.get("providerType");
65
66 if (providerTypeNode.isTextual()) {
67 final String providerType = providerTypeNode.asText();
68
69
70 final JsonNode targetPathNode = target.get("targetPath");
71 if (targetPathNode.isTextual()) {
72 final String targetPath = targetPathNode.asText();
73
74 return new DataMaskingTarget(ProviderType.valueOf(providerType), targetPath);
75 }
76 }
77 }
78 throw new IllegalArgumentException("Unable to deserialize " + target);
79 }

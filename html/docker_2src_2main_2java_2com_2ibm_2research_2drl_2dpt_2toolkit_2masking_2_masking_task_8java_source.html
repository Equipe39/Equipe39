118 private static Map<ProviderType, Class<? extends MaskingProvider>> loadUserDefinedMaskingProviders(String filename) throws IOException {
119 final Map<ProviderType, Class<? extends MaskingProvider>> userDefinedMaskingProviders = new HashMap<>();
120
121 try (BufferedReader reader = new BufferedReader(new FileReader(filename))) {
122 for (String line = reader.readLine(); null != line; line = reader.readLine()) {
123 final String[] parts = line.split(":");
124 final ProviderType providerType = ProviderType.valueOf(parts[0].trim());
125
126 final Class<? extends MaskingProvider> maskingProvider = (Class<? extends MaskingProvider>) Class.forName(parts[1].trim());
127
128 userDefinedMaskingProviders.put(providerType, maskingProvider);
129 }
130 } catch (FileNotFoundException e) {
131 throw new IllegalArgumentException(filename + " not found", e);
132 } catch (IOException e) {
133 throw new IOException("Error reading user defined masking providers from " + filename, e);
134 } catch (ClassNotFoundException e) {
135 throw new IllegalArgumentException("Class not found " + e.getMessage(), e);
136 }
137 return userDefinedMaskingProviders;
138 }

140 private void maskDataset(InputStream dataset, OutputStream output) throws IOException {
141 final Set<String> alreadyMaskedFields = new HashSet<>();
142
143 final Map<ProviderType, Class<? extends MaskingProvider>> userDefinedTypes;
144 if (null != this.getTaskOptions().getMaskingProviders() && !this.getTaskOptions().getMaskingProviders().isEmpty()) {
145 userDefinedTypes = loadUserDefinedMaskingProviders(this.getTaskOptions().getMaskingProviders());
146 } else {
147 userDefinedTypes = Collections.emptyMap();
148 }
149
150 FormatProcessor formatProcessor = FormatProcessorFactory.getProcessor(dataMaskingOptions.getInputFormat());
151 formatProcessor.maskStream(dataset, output, this.maskingProviderFactory, dataMaskingOptions, alreadyMaskedFields, userDefinedTypes);
152 }

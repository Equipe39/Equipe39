90 public double report() {
91 double value = 0.0;
92
93 int totalRows = this.original.getNumberOfRows();
94
95 for (Partition p : partitions) {
96 int pSize = p.size();
97
98 if (p.isAnonymous()) {
99 value += Math.pow(pSize, 2);
100 } else {
101 value += pSize * totalRows;
102 }
103 }
104
105 return value;
106 }

108 @Override

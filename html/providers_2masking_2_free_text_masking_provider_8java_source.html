66 @Override
67 public String mask(String text) {
68 try {
69 final List<IdentifiedEntity> identifiedEntities = identifyEntities(text);
70 final List<IdentifiedEntity> maskedEntities = maskIdentifiedEntities(identifiedEntities);
71
72 return NLPUtils.applyFunction(text, maskedEntities, NLPUtils.IDENTITY_FUNCTION);
73 } catch (IOException e) {
74 logger.error("Problem when masking text");
75 throw new RuntimeException(e);
76 }
77 }

79 private List<IdentifiedEntity> maskIdentifiedEntities(List<IdentifiedEntity> identifiedEntities) {
80 return identifiedEntities.stream()
81 .map(
82 entity -> {
83 String type = entity.getType().iterator().next().getType();
84 if (toBeMasked.containsKey(type)) {
85 String maskedValue = factory.get(type, toBeMasked.get(type).getProviderType()).mask(entity.getText());
86
87 return new IdentifiedEntity(
88 truncateOrPad(maskedValue, entity.getEnd() - entity.getStart()),
89 entity.getStart(), entity.getEnd(), entity.getType(), entity.getPos());
90 }
91 return entity;
92 }
93 )
94 .collect(Collectors.toList());
95 }

97 private String truncateOrPad(String maskedValue, int length) {
98 StringBuilder builder = new StringBuilder();
99
100 builder.append(maskedValue, 0, Math.min(length, maskedValue.length()));
101
102 builder.append(" ".repeat(length - builder.length()));
103
104 return builder.toString();
105 }

111 protected List<IdentifiedEntity> grep(String sourceTokens, String targetValue, String separator,
112 boolean ignoreCase,
113 boolean findAnywhere,
114 String tokenType) {
115 if (sourceTokens == null || sourceTokens.isEmpty()) {
116 return Collections.emptyList();
117 }
118
119 Set<String> lookupValues = new HashSet<>();
120
121 String[] tokens = sourceTokens.split(separator);
122 Collections.addAll(lookupValues, tokens);
123
124 if (lookupValues.isEmpty()) {
125 return Collections.emptyList();
126 }
127
128 int patternFlags = ignoreCase ? Pattern.CASE_INSENSITIVE : 0;
129
130 Set<Pattern> patterns = new HashSet<>();
131 for(String lookupValue: lookupValues) {
132 if (findAnywhere) {
133 patterns.add(Pattern.compile(lookupValue, patternFlags));
134 } else {
135 patterns.add(Pattern.compile("\\b" + lookupValue + "\\b", patternFlags));
136 }
137 }
138
139 List<IdentifiedEntity> results = new ArrayList<>();
140
141 for (Pattern pattern: patterns) {
142 Matcher matcher = pattern.matcher(targetValue);
143 while (matcher.find()) {
144 int start = matcher.start();
145 int end = matcher.end();
146
147 String text = targetValue.substring(start, end);
148
149 IdentifiedEntity entity = new IdentifiedEntity(text, start, end,
150 Collections.singleton(new IdentifiedEntityType(tokenType, tokenType, "__LOOKUP_TOKENS__")),
151 Collections.singleton(PartOfSpeechType.UNKNOWN));
152
153 results.add(entity);
154 }
155 }
156
157 results.sort(Comparator.comparingInt(IdentifiedEntity::getStart));
158
159 return results;
160 }

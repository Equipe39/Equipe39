81 @Override
82 public Tuple2<String, Long> call(String s) throws Exception {
83 String key = "";
84 CSVRecord record = CSVParser.parse(s, CSVFormat.RFC4180).getRecords().get(0);
85
86 for(Integer column: quasiColumns) {
87 key += record.get(column) + ":";
88 }
89
90 return new Tuple2<String, Long>(key, 1L);
91 }
92 }).keys().count();
93
94 double aecs = total_records/(double)equivalence_classes;
95
96 if (!normalized) {
97 return aecs;
98 }
99 else {
100 return aecs / (double)this.k;

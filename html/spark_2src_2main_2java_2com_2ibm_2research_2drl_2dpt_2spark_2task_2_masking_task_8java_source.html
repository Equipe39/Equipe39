72 @Override
73 public Dataset<Row> process(Dataset<Row> dataset) {
74 final MaskingProviderFactory factory = buildMaskingProviderFactory();
75
76 final Queue<String> fieldsToMask = new ArrayDeque<>(this.taskOptions.getToBeMasked().keySet());
77
78 final String prefix = findPrefix(dataset.columns());
79
80 dataset = expandDatasetWithFieldsToPreserve(dataset, prefix);
81
82 final Collection<String> fieldsToSuppress = getFieldsToSuppress(this.taskOptions.getToBeMasked());
83 final Set<String> alreadyMaskedFields = new HashSet<>();
84
85 while (! fieldsToMask.isEmpty()) {
86 String fieldName = Objects.requireNonNull(fieldsToMask.poll());
87
88 if (requiresOtherField(fieldName, alreadyMaskedFields)) {
89 logger.trace("Delaying processing of " + fieldName);
90
91 fieldsToMask.add(fieldName);
92
93 continue;
94 }
95
96 dataset = maskField(dataset, fieldName, factory, prefix);
97
98 alreadyMaskedFields.add(fieldName);
99 }
100
101 return suppressFields(dataset, fieldsToSuppress, prefix);
102 }

104 private Dataset<Row> expandDatasetWithFieldsToPreserve(Dataset<Row> dataset, String prefix) {
105 final Collection<String> requiresOriginal = findFieldsMaskedButAlsoRequiredAsOriginal(dataset.columns(), this.taskOptions.getToBeMasked(), this.taskOptions.getPredefinedRelationships());
106
107 for (String requiredOriginal : requiresOriginal) {
108 dataset = dataset.withColumn(prefix + requiredOriginal, dataset.col(requiredOriginal));
109 }
110
111 return dataset;
112 }

114 private Collection<String> findFieldsMaskedButAlsoRequiredAsOriginal(String[] columnNames, Map<String, DataMaskingTarget> columnToBeMasked, Map<String, FieldRelationship> relationships) {
115 final Set<String> operands = relationships.values().stream()
116 .map(FieldRelationship::getOperands)
117 .flatMap(Arrays::stream)
118 .map(RelationshipOperand::getName)
119 .collect(Collectors.toSet());
120
121 return Arrays.stream(columnNames).filter(columnToBeMasked::containsKey)
122 .filter(operands::contains)
123 .collect(Collectors.toList());
124 }

126 private String findPrefix(String[] columnNames) {
127 Set<String> names = new HashSet<>(Arrays.asList(columnNames));
128
129 for (String prefix : MaskingTask.PREFIX) {
130 boolean good = true;
131
132 for (String columnName : columnNames) {
133 if (columnName.startsWith(prefix) || names.contains(prefix + columnName)) {
134 good = false;
135 break;
136 }
137 }
138
139 if (good) {
140 return prefix;
141 }
142 }
143
144 throw new RuntimeException("Unable to find a good candidate as prefix");
145 }

152 private Dataset<Row> maskField(Dataset<Row> dataset, String fieldName, MaskingProviderFactory factory, String prefix) {
153 final DataMaskingTarget target = this.taskOptions.getToBeMasked().get(fieldName);
154 final MaskingProvider provider = factory.get(fieldName, target.getProviderType());
155 final DataType targetDataType = retrieveDatatype(dataset.schema(), fieldName);
156 final FieldRelationship relationship = taskOptions.getPredefinedRelationships().get(fieldName);
157
158 if (relationship == null) {
159 UDF1<String, String> mask = provider::mask;
160
161 return dataset.withColumn(target.getTargetPath(),
162 udf(mask, DataTypes.StringType).apply(
163 dataset.col(fieldName).cast(DataTypes.StringType)
164 ).cast(targetDataType));
165 } else {
166 String operandFieldName = relationship.getOperands()[0].getName();
167 String operandFieldPreservedValueName = prefix + operandFieldName;
168
169 switch (relationship.getRelationshipType()) {
170 case KEY:
171 UDF2<String, String, String> keyedUDF = provider::maskWithKey;
172
173 return dataset.withColumn(target.getTargetPath(),
174 udf(keyedUDF, DataTypes.StringType).apply(
175 dataset.col(fieldName).cast(DataTypes.StringType), dataset.col(operandFieldPreservedValueName).cast(DataTypes.StringType)
176 ).cast(targetDataType));
177 case DISTANCE:
178 UDF3<String, String, String, String> distanceUDF = provider::maskDistance;
179 return dataset.withColumn(target.getTargetPath(),
180 udf(distanceUDF, DataTypes.StringType).apply(
181 dataset.col(fieldName).cast(DataTypes.StringType),
182 dataset.col(operandFieldPreservedValueName).cast(DataTypes.StringType),
183 dataset.col(operandFieldName).cast(DataTypes.StringType)
184 ).cast(targetDataType));
185 case EQUALS:
186 UDF2<String, String, String> equalUDF = provider::maskEqual;
187 return dataset.withColumn(target.getTargetPath(),
188 udf(equalUDF, DataTypes.StringType).apply(
189 dataset.col(fieldName).cast(DataTypes.StringType), dataset.col(operandFieldName).cast(DataTypes.StringType)
190 ).cast(targetDataType));
191 case GREATER:
192 UDF3<String, String, String, String> greaterUDF = provider::maskGreater;
193 return dataset.withColumn(target.getTargetPath(),
194 udf(greaterUDF, DataTypes.StringType).apply(
195 dataset.col(fieldName).cast(DataTypes.StringType),
196 dataset.col(operandFieldName).cast(DataTypes.StringType),
197 dataset.col(operandFieldPreservedValueName).cast(DataTypes.StringType)
198 ).cast(targetDataType));
199 case LESS:
200 UDF3<String, String, String, String> lesserUDF = provider::maskLess;
201 return dataset.withColumn(target.getTargetPath(),
202 udf(lesserUDF, DataTypes.StringType).apply(
203 dataset.col(fieldName).cast(DataTypes.StringType),
204 dataset.col(operandFieldName).cast(DataTypes.StringType),
205 dataset.col(operandFieldPreservedValueName).cast(DataTypes.StringType)
206 ).cast(targetDataType));
207 case LINKED:
208 UDF3<String, String, ProviderType, String> linkedUDF = provider::maskLinked;
209 return dataset.withColumn(target.getTargetPath(),
210 udf(linkedUDF, DataTypes.StringType).apply(
211 dataset.col(fieldName).cast(DataTypes.StringType),
212 dataset.col(operandFieldName).cast(DataTypes.StringType),
213 lit(relationship.getOperands()[0].getType())
214 ).cast(targetDataType));
215 case RATIO:
216 UDF3<String, String, String, String> ratioUDF = provider::maskWithRatio;
217 return dataset.withColumn(target.getTargetPath(),
218 udf(ratioUDF, DataTypes.StringType).apply(
219 dataset.col(fieldName).cast(DataTypes.StringType),
220 dataset.col(operandFieldName).cast(DataTypes.StringType),
221 dataset.col(operandFieldPreservedValueName).cast(DataTypes.StringType)
222 ).cast(targetDataType));
223 case GREP_AND_MASK: // TODO: needs to be revisited
224 case SUM:
225 case SUM_APPROXIMATE:
226 case PRODUCT:
227 default:
228 throw new UnsupportedOperationException();
229 }
230 }
231 }

239 private Dataset<Row> suppressFields(Dataset<Row> dataset, Collection<String> fieldsToSuppress, String prefix) {
240 return dataset.select(
241 findRemainingColumnNames(dataset.columns(), fieldsToSuppress, prefix)
242 );
243 }

245 private Column[] findRemainingColumnNames(String[] columns, Collection<String> fieldsToSuppress, String prefix) {
246 return Arrays.stream(columns).filter(
247 ((Predicate<String>) fieldsToSuppress::contains).or(v -> v.startsWith(prefix)).negate()
248 ).map(functions::col).toArray(Column[]::new);
249 }

251 private Collection<String> getFieldsToSuppress(Map<String, DataMaskingTarget> maskingTargets) {
252 List<String> fieldsToSuppress = new ArrayList<>();
253 for (Map.Entry<String, DataMaskingTarget> toBeMasked : maskingTargets.entrySet()) {
254 if (toBeMasked.getValue().getProviderType().equals(ProviderType.SUPPRESS_FIELD)) {
255 fieldsToSuppress.add(toBeMasked.getKey());
256 }
257 }
258 return fieldsToSuppress;
259 }

261 private boolean requiresOtherField(String fieldsToMask, Collection<String> alreadyMaskedFields) {
262 FieldRelationship fieldRelationship = this.taskOptions.getPredefinedRelationships().get(fieldsToMask);
263
264 if (fieldRelationship != null) {
265 for (RelationshipOperand operand : fieldRelationship.getOperands()) {
266 if (!alreadyMaskedFields.contains(operand.getName())) {
267 return true;
268 }
269 }
270 }
271
272 return false;
273 }

73 @Override
74 public String mask(String text) {
75 try {
76 final List<IdentifiedEntity> identifiedEntities = identifyEntities(text);
77 final List<IdentifiedEntity> maskedEntities = maskIdentifiedEntities(identifiedEntities);
78
79 return NLPUtils.applyFunction(text, maskedEntities, NLPUtils.IDENTITY_FUNCTION);
80 } catch (IOException e) {
81 logger.error("Problem when masking text");
82 throw new RuntimeException(e);

87 @Override
88 public String maskGrepAndMask(String text, List<String> targetTokens) {
89 List<IdentifiedEntity> greppedEntities = new ArrayList<>();
90
91 for(String targetText : targetTokens) {
92 greppedEntities.addAll(grep(targetText, text,
93 this.lookupTokensSeparator, this.lookupTokensIgnoreCase, this.lookupTokensFindAnywhere, this.lookupTokensType));
94 }
95
96 try {
97 final List<IdentifiedEntity> identifiedEntities = identifyEntities(text);
98 identifiedEntities.addAll(greppedEntities);
99
100 final List<IdentifiedEntity> maskedEntities = maskIdentifiedEntities(annotator.merge(identifiedEntities));
101
102 return NLPUtils.applyFunction(text, maskedEntities, NLPUtils.IDENTITY_FUNCTION);
103 } catch (IOException e) {
104 e.printStackTrace();
105 throw new RuntimeException(e.getMessage());
106 }
107 }

109 private List<IdentifiedEntity> maskIdentifiedEntities(List<IdentifiedEntity> identifiedEntities) {
110 return identifiedEntities.stream()
111 .map(
112 entity -> {
113 String type = entity.getType().iterator().next().getType();
114 if (factory.getToBeMasked().containsKey(type)) {
115 String maskedValue = factory.get(type, factory.getToBeMasked().get(type).getProviderType()).mask(entity.getText());
116
117 return new IdentifiedEntity(
118 truncateOrPad(maskedValue, entity.getEnd() - entity.getStart()),
119 entity.getStart(), entity.getEnd(), entity.getType(), entity.getPos());
120 }
121 return entity;
122 }
123 )
124 .collect(Collectors.toList());
125 }

127 private String truncateOrPad(String maskedValue, int length) {
128 StringBuilder builder = new StringBuilder();
129
130 builder.append(maskedValue, 0, Math.min(length, maskedValue.length()));
131
132 builder.append(" ".repeat(length - builder.length()));
133
134 return builder.toString();
135 }

141 protected List<IdentifiedEntity> grep(String sourceTokens, String targetValue, String separator,
142 boolean ignoreCase,
143 boolean findAnywhere,
144 String tokenType) {
145 if (sourceTokens == null || sourceTokens.isEmpty()) {
146 return Collections.emptyList();
147 }
148
149 Set<String> lookupValues = new HashSet<>();
150
151 String[] tokens = sourceTokens.split(separator);
152 Collections.addAll(lookupValues, tokens);
153
154 if (lookupValues.isEmpty()) {
155 return Collections.emptyList();
156 }
157
158 int patternFlags = ignoreCase ? Pattern.CASE_INSENSITIVE : 0;
159
160 Set<Pattern> patterns = new HashSet<>();
161 for(String lookupValue: lookupValues) {
162 if (findAnywhere) {
163 patterns.add(Pattern.compile(lookupValue, patternFlags));
164 } else {
165 patterns.add(Pattern.compile("\\b" + lookupValue + "\\b", patternFlags));
166 }
167 }
168
169 List<IdentifiedEntity> results = new ArrayList<>();
170
171 for (Pattern pattern: patterns) {
172 Matcher matcher = pattern.matcher(targetValue);
173 while (matcher.find()) {
174 int start = matcher.start();
175 int end = matcher.end();
176
177 String text = targetValue.substring(start, end);
178
179 IdentifiedEntity entity = new IdentifiedEntity(text, start, end,
180 Collections.singleton(new IdentifiedEntityType(tokenType, tokenType, "__LOOKUP_TOKENS__")),
181 Collections.singleton(PartOfSpeechType.UNKNOWN));
182
183 results.add(entity);
184 }
185 }
186
187 results.sort(Comparator.comparingInt(IdentifiedEntity::getStart));
188
189 return results;
190 }

77 @Override

79 int index = this.fieldMap.get(s);
80 String type = this.fieldTypes.get(index);
81
82 if (bytes == null || bytes.length == 0) {
83 this.values[index] = null;
84 return;
85 }
86
87 String v = new String(bytes);
88 Object toSet;
89
90 switch (type) {
91 case "StringType":
92 toSet = v;
93 break;
94 case "BinaryType":
95 case "ByteType":
96 toSet = bytes[0];
97 break;
98 case "BooleanType":
99 toSet = Boolean.parseBoolean(v);
100 break;
101 case "DecimalType":
102 toSet = new BigDecimal(v);
103 break;
104 case "IntegerType":
105 toSet = Integer.parseInt(getIntegerPart(v));
106 break;
107 case "LongType":
108 toSet = Long.parseLong(getIntegerPart(v));
109 break;
110 case "FloatType":
111 toSet = Float.parseFloat(v);
112 break;
113 case "DoubleType":
114 toSet = Double.parseDouble(v);
115 break;
116 case "ShortType":
117 toSet = Short.parseShort(getIntegerPart(v));
118 break;
119 case "TimestampType":
120 toSet = Timestamp.valueOf(v);
121 break;
122 case "ObjectType":
123 default:
124 toSet = values[index];
125 break;
126
127 }
128
129 this.values[index] = toSet;
130 }

41 private double getLossCategorical(String value, ColumnInformation columnInformation) {
42 CategoricalInformation categoricalInformation = (CategoricalInformation)columnInformation;
43
44 int level = categoricalInformation.getHierarchy().getNodeLevel(value);
45
46 if (level == 0) {
47 return 0.0;
48 }
49
50 return (level + 1.0) / (double) categoricalInformation.getHierarchy().getHeight();
51 }

88 @Override
89 public Double report() {
90 final List<Integer> quasiColumns = AnonymizationUtils.getColumnsByType(this.columnInformationList, ColumnType.QUASI);
91
92 final DoubleAccumulator cells = this.anonymized.rdd().sparkContext().doubleAccumulator();
93
94 double precision = this.anonymized.map((Function<String, Double>) s -> {
95 CSVRecord record = CSVParser.parse(s, CSVFormat.RFC4180).getRecords().get(0);
96
97 double precision1 = 0.0;
98
99 for(int j: quasiColumns) {
100 ColumnInformation columnInformation = columnInformationList.get(j);
101 if (!columnInformation.isCategorical()) {
102 continue;
103 }
104
105 cells.add(1.0);
106
107 double loss = getLossCategorical(record.get(j), columnInformation) * columnInformation.getWeight();
108 precision1 += loss;
109 }
110
111 return precision1;
112 }).reduce((Function2<Double, Double, Double>) (aDouble, aDouble2) -> aDouble + aDouble2);
113
114 return (precision) / cells.value();
115
116 }

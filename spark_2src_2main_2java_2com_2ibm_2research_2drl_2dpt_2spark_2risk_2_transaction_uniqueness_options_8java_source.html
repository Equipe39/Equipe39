94 public TransactionUniquenessOptions(JsonNode configuration) {
95 validateField(configuration, "factor", JsonNodeType.NUMBER);
96 validateField(configuration, "threshold", JsonNodeType.NUMBER);
97 validateField(configuration, "inputFormat", JsonNodeType.STRING);
98 validateField(configuration, "idColumn", JsonNodeType.ARRAY);

100 validateField(configuration, "joinRequired", JsonNodeType.BOOLEAN);

102 this.factor = configuration.get("factor").asInt();
103 this.threshold = configuration.get("threshold").asInt();
104 this.idColumns = readArray(configuration.get("idColumn"));

106 if (this.idColumns.length == 0) {

116 this.inputFormat = DataTypeFormat.valueOf(configuration.get("inputFormat").asText());
117 this.datasetOptions = createDatasetOptions(configuration, this.inputFormat);

135 this.joinInformation = new JoinInformation(joinInformation.get("rightTable").asText(),
136 joinInformation.get("leftColumn").asText(), joinInformation.get("rightColumn").asText(),
137 rightTableInputFormat, rightTableDatasetOptions);
